

#include <iostream>
#include <fstream>
#include <vector>
using namespace std;

bool isPrime(int n){
	for( int i = 2;i*i<=n;i++)
		if(n%i==0) return false;
	return true;
}
void goldbach(vector <int> p, int n){
	bool flg =false;
	int min,gap=n;
	for( int i=0; i<p.size();i++){
		for(int j=i;j<p.size();j++){ 
			if(p[i] + p[j] == n && p[j]-p[i] < gap){
				gap = p[j]-p[i];
				min =p[i];
			}
		}
	}
	cout<<min<<" "<<n-min<<endl;
}
int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");

	instream>>no_cases;

	while(no_cases--){
		vector <int> prime;
		prime.push_back(2);
		int n;
		instream>>n;
		for (int i=3;i<n-1;i+=2){
			if( isPrime(i)) prime.push_back(i);	
		}
		goldbach(prime,n);
	}
	instream.close();
	return 0;
}
