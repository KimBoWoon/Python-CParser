void permuteString(char *str, int strlength, int getnum)
{
	if (strlength <= 0) return;

	int dividenum = factorial[strlength - 1];

	int pos = getnum / dividenum - (getnum % dividenum == 0 ? 1 : 0);

	std::cout << str[pos];
	str[pos] = '-';
	char *newstr = new char[strlength - 1];
	int count = 0;
	for (int i = 0; i < strlength; i++)
	{
		if (str[i] == '-') continue;

		newstr[count++] = str[i];
	}
	permuteString(newstr, strlength - 1, getnum % dividenum + (getnum % dividenum == 0 ? dividenum : 0));
	delete[]newstr;
}
