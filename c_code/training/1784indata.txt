#include <iostream>
#include <fstream>
#include <cstdlib>
#include <string>
#include <stack>
using namespace std;

int main()
{
	ifstream inStream;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}
	int gatsu = 0;	
	int galhogatsu = 0;
	inStream >> gatsu;

	for (int i = 0; i < gatsu; i++)
	{
		string galho;
		stack <char> s;
		inStream >> galho;
		galhogatsu = galho.size();
		int result = 0;
		for (int i = 0; i < galhogatsu; i++)
		{
			if (galho[i] == '(')
			{
				s.push('(');
			}

			else if (galho[i] == ')')
			{
				if (s.empty() == false)
					s.pop();
				else
				{
					result = 1;
					break;
				}
			}
		}

		if (s.empty() == false)
		{
			result = 1;
		}
		if (result == 1)
		{
			cout << "NO" << endl;
		}
		else if (result == 0)
		{
			cout << "YES" << endl;
		}
	}
	inStream.close();
	return 0;
}
