

#define _CRT_SECURE_NO_WARNINGS

#include <fstream>
#include <iostream>

using namespace std;

int PalinDrome(char c[], int first, int last);
int convertToN(int num, int targetNum);

char str[100];

int main(void) {

	ifstream dataFile("input.txt");

	int caseCount;
	dataFile >> caseCount;

	for (int i = 0; i < caseCount; i++) {
		int num;
		dataFile >> num;

		int flag = 0;

		for (int j = 2; j <= 64; j++) {
			if ( convertToN(num, j) == 1) {
				flag = 1;
				cout << "1" << endl;
				break;
			}
		}

		if (flag == 0) {
			cout << "0" << endl;
		}
	}

	getchar();

	dataFile.close();

	return 0;
}

int convertToN(int num, int targetNum) {
	int i = 0;
	int div = num / targetNum;
	int res = num % targetNum;

	str[i] = res+48;
	while (1){
		if (div == 0) {
			break;
		}
		else {
			i++;
			res = div % targetNum;
			div = div / targetNum;
			str[i] = res+48;
		}
	}

	if (PalinDrome(str, 0, i) == 1) {
		return 1;
	}
	else {
		return 0;
	}
}

int PalinDrome(char c[], int first, int last) {

	if (last <= first) {
		return 1;
	}
	else if (c[first] != c[last]) {
		return 0;
	}
	else {
		return PalinDrome(c, first + 1, last - 1);
	}
}
