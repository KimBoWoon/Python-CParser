
#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){

	if(b==0)	return a;
	else		return gcd(b,a%b);
}

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file = fopen("input.txt","r");
	if(in_file==NULL)	exit(1);

	fscanf(in_file,"%d",&num_cases);
	for(i=0; i<num_cases; i++){

		int val1, val2;

		fscanf(in_file, "%d %d", &val1, &val2);

		printf("%d\n",gcd(val1, val2));
	}
	fclose(in_file);
}
