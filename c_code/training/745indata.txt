




#include <iostream>
#include <string>
#include <fstream>

int palindrome(std::string str, int x, int y);

int main() {
	std::ifstream inFile;
	inFile.open("input.txt");

	int testCaseNum;

	inFile >> testCaseNum;

	for (int i = 0; i < testCaseNum; i++) {
		std::string str = "";
		inFile >> str;
		std::cout<<palindrome(str, 0, str.size() - 1)<<std::endl;
	}
}

int palindrome(std::string str, int x, int y) {
	
	if (x >= y)
		return 1;
	
	else if (str[x] != str[y])
		return 0;
	
	else if (str[x] == str[y])
		return palindrome(str, x + 1, y - 1);
}
