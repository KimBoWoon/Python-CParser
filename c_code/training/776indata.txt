#include <iostream>
#include <fstream>
#include <cstdlib>
#include<string>
#include<cstring>

using namespace std;

int checkPalindrome(char str[], int first, int last);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		char line[1001];

		inStream >> line;

		cout << checkPalindrome(line, 0, (strlen(line) - 1)) << endl;


	}

	inStream.close();
	return 0;
}

int checkPalindrome(char str[], int first, int last) { 
	if (last <= first) 
		return 1; 
	
	else if (str[first] != str[last]) 
		return 0;
	
	else 
		return checkPalindrome(str, first + 1, last - 1);
}
