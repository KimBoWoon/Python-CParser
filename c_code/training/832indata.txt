#include <stdio.h>
#include <stdlib.h>
int recursiveFibonacci(int num);

void main()
{
	FILE *f;
	int count;		
	int i;			
	int number;		

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);	

	for(i=0; i<count; i++)
	{
		int result;		
		fscanf(f, "%d", &number);	
		result = recursiveFibonacci(number);	
		printf("%d\n", result);	
	}
}
int recursiveFibonacci(int num)
{
	if(num == 0)
		return 0;
	else if(num == 1)
		return 1;
	else
		return recursiveFibonacci(num-1)+recursiveFibonacci(num-2);
}
