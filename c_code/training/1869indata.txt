#include<stdio.h>
#include<string.h>

char *result[2] = {"NO", "YES"};
char pareList[50];
int len;

char* parenthesis_check(int index, int left, int right)
{
	if(index==len){ 
		if(left == right)	return result[1];
		else				return result[0];
	}else{
		if(right > left) return result[0];
		else{
			if(pareList[index] == '(')	left += 1;
			else						right += 1;

			parenthesis_check(index+1, left, right);
		}
	}
	
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testNum;
	int i;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i)
	{
		fscanf(inFile, "%s", &pareList);
		len = strlen(pareList);
		printf("%s\n", parenthesis_check(0, 0, 0));
	}
	fclose(inFile);
}
