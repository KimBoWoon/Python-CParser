#include <iostream>
#include <fstream>
#include <cstdlib>
#include <string>

using namespace std;

int Sosu(int num1)
{
	int result = 1;
	int namerji = 0;
	for (int i = 2; i < num1; i++)
	{
		namerji = num1%i;	
		if (namerji == 0)
		{
			result = 0;
			break;
		}
	}
	return result;
}


int main()
{
	int gatsu = 0;	
	int goldbach = 0; 
	int arr[500] = { 0, };	

	ifstream inStream;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> gatsu;
	for (int i = 0; i < gatsu; i++)
	{
		inStream >> goldbach;
		int num1 = 0;	
		int num2 = 0;	

		int ban = goldbach / 2;
		int count = 0;

		for (int j = 2; j <= ban; j++)
		{
			Sosu(j);
			if (Sosu(j) == 1)	
			{
				Sosu(goldbach - j);

				if (Sosu(goldbach - j) == 1)	
				{
					arr[count] = j;
					count++;
				}
			}
		}
		num1 = arr[count-1];	
		num2 = goldbach - arr[count-1];

		cout << num1 << " " << num2 << endl;
	}
	inStream.close();
	return 0;
}
