#include <stdio.h>

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int H = 0; 
	int W = 0; 
	int N = 0; 
	int roomNumber = 0; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d %d %d", &H, &W, &N); 

		if (N%H == 0)
			roomNumber = H * 100 + N / H;
		else
			roomNumber = N%H * 100 + N / H + 1;

		printf("%d\n", roomNumber);

	}

	fclose(f);
	return 0;
}
