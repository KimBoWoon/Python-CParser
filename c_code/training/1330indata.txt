#include <iostream>
#include <fstream>

void henry(int mol, int deno, int unit_deno);
int gcd(int num1, int num2);

int main(void){

	std::ifstream inFile("input.txt");

	if (inFile.fail()){
		std::cout << "not file" << std::endl;
		return 0;
	}

	int testCases = 0;
	inFile >> testCases;

	int mol = 0;
	int deno = 0;

	for (int i = 0; i < testCases; i++){
		inFile >> mol >> deno;

		henry(mol, deno, deno / mol);

	}
	return 0;
}

void henry(int mol, int deno, int unit_deno){

	int gcd_number = gcd(mol, deno);
	mol = mol / gcd_number;
	deno = deno / gcd_number;

	if (mol == 1){
		std::cout << deno << std::endl;
		return;
	}
	if (mol * unit_deno > deno){
		henry((mol * unit_deno) - deno, deno * unit_deno, unit_deno + 1);
	}
	else{
		henry(mol, deno, unit_deno + 1);
	}
}

int gcd(int num1, int num2){

	if (num1 < 0 || num2 < 0){
		return -1;
	}

	if (num2 == 0){
		int temp = num1;
		num1 = num2;
		num2 = temp;
	}

	int r = 0;
	r = num1 % num2;
	num1 = num2;
	num2 = r;

	if (r == 0){
		return num1;
	}
	else{
		return gcd(num1, num2);
	}
}
