#include <iostream>
#include <fstream>
#include <vector>
using namespace std;

class Ducci {
	int* data;

	Ducci(int size) {
		data = new int[size];
	}
};
int abs(int num) {
	if (num < 0)	return num * -1;
	return num;
}
int main() {
	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;

	while (cases--) {
		int size;
		file >> size;
		
		int data[20];
		for (int i = 0; i < size; i++)	file >> data[i];

		int i;
		int temp[20];
		for (i = 0; i < 1000; i++) {
			
			int sum = 0;
			for (int j = 0; j < size; j++) {
				int start = j, end = j + 1;
				if (j == size - 1)	end = 0;

				temp[j] = abs(data[end] - data[start]);
				sum += abs(data[end] - data[start]);
			}
			if (sum == 0)	break;
			for (int j = 0; j < size; j++)
				data[j] = temp[j];
		}
		if (i == 1000)	cout << "LOOP" << endl;
		else			cout << "ZERO" << endl;

	}


	return 1;
}
