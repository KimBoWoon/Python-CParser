
#include <iostream>
#include <fstream>

void getPrimeNumber(int num){
	int *n;
	n = new int[num];

	for (int i = 2; i <= num; i++) {
		n[i] = i;
	}

	for (int i = 2; i <= num; i++) {
		if (n[i] == 0)
			continue;
		for (int j = i + i; j <= num; j += i){
			n[j] = 0;
		}
	}

	if (n[num] == 0) std::cout << 0 << std::endl;
	else std::cout << 1 << std::endl;
	
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase,num;
	if (fin.is_open() == false)  return -1; 
	
	fin >> numcase;
	for (int i = 0; i < numcase; i++) {
		fin >> num;
		getPrimeNumber(num);
	}

	fin.close();	
}
