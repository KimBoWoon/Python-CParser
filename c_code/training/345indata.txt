#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

char map[100][100+1];
int room[100];
int idx=-1;
int m,n;

void searchRoom(int x, int y){
	if( map[x][y] == '.' ){
		map[x][y] = '-'; room[idx]+=1;
		if(x+1<n) searchRoom(x+1,y);
		if(y+1<m) searchRoom(x,y+1);
		if(x-1>=0) searchRoom(x-1,y);
		if(y-1>=0) searchRoom(x,y-1);
	}

}

int compare(const void *a , const void *b){
        return ( *(int*)a < *(int*)b );
}


int main(void){
	FILE* file;
	file = fopen("input.txt","r");

	int cases;
	fscanf(file, "%d", &cases);
	while(cases--){
		
		fscanf(file, "%d%d",  &m,&n);
		for(int i=0; i<n; i++){
			fscanf(file, "%s", map[i]);	
		}
		for(int i=1; i<n; i++){
			for(int j=1; j<m; j++){
				if(map[i][j] == '.' ){
					idx += 1;
					searchRoom(i,j);
				}
			}
		}
		printf("%d\n",idx+1);
		qsort(&room,idx+1, sizeof(room[0]),compare); 
		for(int i=0; i<idx+1; i++){
			printf("%d ", room[i]);
			room[i]=0;
		}
		puts("");
		idx=-1;
	}
	return 0;
}
