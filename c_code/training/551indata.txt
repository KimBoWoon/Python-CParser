
#include <stdio.h>

int factorial(int n){
    if( n < 2 ) return 1;
    else return factorial(n-1)*n;
}

void swap(char *a, char *b){
    char temp = *a;
    *a = *b;
    *b = temp;
}

void permute(FILE *fin){
    int N, K;
    int n, k = 0;
    int i, j, index = 0;

    char checked[13] = {"abcdefghijkl"};
        
    fscanf(fin, "%d %d", &N, &K);
        
    k = K-1;
    n = N-1;
    
    while(k>0){    
        printf("%c",checked[k/factorial(n)]);   
        
        for(i=k/factorial(n); i<12; i++)
            swap(&checked[i], &checked[i+1]);   
        k = k%factorial(n--);
        index++;
    }
    
    for(i=0; i<N-index; i++)        
        printf("%c", checked[i]);       
    printf("\n");
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        permute(fin);        
    
    fclose(fin);            
    
    return 0;           
}
