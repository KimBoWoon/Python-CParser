#include <iostream>
#include <fstream>

using namespace std;
int main()
{
	ifstream FILE("input.txt");
	int number = 0, testCase = 0,i=0;

	FILE >> testCase;
	while (testCase--)
	{
		FILE >> number;
		if (number==2)
			cout << "1" << endl;
		for (i = 2; i <= number / 2;i++)
		{
			if (number%i == 0)
			{
				cout << "0" << endl;
				break;
			}
		}
		if (number%i != 0)
			cout << "1" << endl;
	}
}
