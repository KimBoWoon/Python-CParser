

#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int facto(int n);

int main()
{
	int numOfCase = 0;	
	int n; 

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> n;

		cout << facto(n) << endl;

	}
	input.close();
	return 0;
}

int facto(int n){
	if (n == 0){
		return 1;
	}
	else{
		return (facto(n - 1)*n);
	}
}
