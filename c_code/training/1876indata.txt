#include <stdio.h>

int main(void)
{
	FILE *file;
	int cases;
	char data[51]={0};
	int i,cnt;
	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);

	while(cases--){
		fscanf(file,"%s",&data);
		i=0,cnt=0;
		while(data[i]!=0){
			if(data[i]=='(') cnt++;
			else if(data[i]==')') cnt--;
			i++;
			if(cnt==-1) break;
		}
		if(cnt==0) printf("YES\n");
		else printf("NO\n");
	}
	fclose(file);
}
