#include <stdio.h>
#include <stdlib.h>

int fib(int n){
	if (n <= 1) 
		return n;
	else		
		return fib(n-1)+fib(n-2);
}

int main(void)
{
	FILE *file;
	int cases,num;
	
	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);
	while(cases--){
		fscanf(file,"%d",&num);
		printf("%d\n",fib(num));
	}
	fclose(file);
}
