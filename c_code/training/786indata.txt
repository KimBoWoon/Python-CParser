#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int Palindrome(char str[],int first,int last){
	if(last <= first)
		return 1;
	else if(str[first]!=str[last])
		return 0;
	else
		return Palindrome(str,first+1,last-1); 
}

int main(void)
{
	FILE *file;
	int cases;
	char str[1000];
	
	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);

	while(cases--){
		fscanf(file,"%s",&str);
		printf("%d\n",Palindrome(str,0,strlen(str)-1));
	}
	
	fclose(file);
}
