

#include <cstdio>
#include <cstdlib>

int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;

	readStream = fopen("input.txt", "r");	

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		exit(1);
	}
	int testCases = 0, 
		triangleNumber = 0, 
		i = 0, j = 0, k = 0, sum = 0;

	bool isTriangleNumber = false;

	short* triangleNumberArray = 0x00L;
	
	triangleNumberArray = (short*)malloc(sizeof(short) * 1001);

	if(triangleNumberArray == 0x00L){
		fprintf(stderr, "Don`t allocate the memory.\n");
		exit(1);
	}

	for(i = 1; i < 1001; ++i)
		triangleNumberArray[i] = ((i+1)*i)/2;

	fscanf(readStream, "%d", &testCases);

	while(testCases--){

		fscanf(readStream, "%d", &triangleNumber);
		isTriangleNumber = false;
		for(i = 1; i < 1001; ++i){
			if(isTriangleNumber || (triangleNumberArray[i] > triangleNumber)) break;
			for(j = 1; j < 100l; ++j){
				if(isTriangleNumber || (triangleNumberArray[j] > triangleNumber)) break;
				for(k = 1; k < 1001; ++k){
					sum = triangleNumberArray[i] + triangleNumberArray[j] + triangleNumberArray[k];
					if(sum > triangleNumber) break;
					if(sum == triangleNumber){
						isTriangleNumber = true;
						break;
					}
				}
			}
		}
		(isTriangleNumber) ? printf("1\n") : printf("0\n");
	}
	if(triangleNumberArray != 0x00L) free(triangleNumberArray);
	fclose(readStream);
}
