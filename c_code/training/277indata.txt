




#include<iostream>
#include<fstream>

void add(int a, int b, int c, int d);
int gcd(int a, int b);

int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){

		int a,b,c,d;

		is>>a>>b>>c>>d;

		add(a,b,c,d);
	}

}

void add(int a, int b, int c, int d){
	
	
	int bunmo = b*d/gcd(b,d);
	int bunja = bunmo/b*a + bunmo/d*c;

	int gcd_ = gcd(bunmo, bunja);

	std::cout<<bunja/gcd_<<' '<<bunmo/gcd_<<std::endl;


}

int gcd(int a, int b){

	if(b == 0)
		return a;

	else
		return gcd(b, a%b);


}
