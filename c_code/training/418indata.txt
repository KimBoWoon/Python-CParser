

#include <iostream>
#include <fstream>
using namespace std;
int euclid(int a, int b);

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases = 0;
	inFile >> testCases;
	int a, b = 0; 
	for (int i = 0; i < testCases; i++){
		inFile >> a >> b;
		cout << euclid(a,b) << endl;
	}
	return 0;
}
int euclid(int a, int b){
	int AmodB = 0;
	if (b == 0){
		return a;
	}
	else{
		AmodB = a%b;
		return euclid(b, AmodB);
	}
}
