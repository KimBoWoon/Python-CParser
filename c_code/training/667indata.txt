#include <iostream>
#include <fstream>
using namespace std;
int factorial(int n)
{
	if(n==0)
		return 1;
	else
		return n*factorial(n-1);
}
int main()
{
	int t,n;
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>t;
	for(int i=0; i<t; i++)
	{
		inStream>>n;
		cout<<factorial(n)<<endl;
	}
}
