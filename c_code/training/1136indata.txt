

#include <stdio.h>
#include <stdlib.h>

int TriangleNumber(int n);
int IsSumOfExactlyThreeTriangleNumbers(int numData);

void main(void)
{
	FILE *inFile;
	int i;
	int numCases, numData;

	inFile = fopen("input.txt", "r"); 

	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &numCases); 

	for (i = 0; i<numCases; i++)
	{
		fscanf(inFile, "%d", &numData); 

		printf("%d\n", IsSumOfExactlyThreeTriangleNumbers(numData)); 
	}

	fclose(inFile); 
}


int TriangleNumber(int n) 
{
	return n * (n + 1) / 2;
}


int IsSumOfExactlyThreeTriangleNumbers(int numData)
{
	
	int i, j, k;

	for (i = 1; i < 45; i++)
		for (j = 1; j < 45; j++)
			for (k = 1; k < 45; k++)
				if (numData == TriangleNumber(i) + TriangleNumber(j) + TriangleNumber(k))
					return 1;
	return 0;
}
