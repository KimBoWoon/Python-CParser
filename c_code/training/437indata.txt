  



#include <iostream>
#include <fstream>
using namespace std;
int gcd(int a, int b);
int main()
{
	ifstream inFile("input.txt");  
	int no_cases=0,num=0;
	inFile>> no_cases;
	for(int l =0; l < no_cases; l++){
		int a=0, b=0;
		inFile>>a>>b; 
		cout<<gcd(a,b)<<endl; 
	}

	return 0;
}
int gcd(int a, int b)
{
	if(b==0)
		return a;
	else
	
		return gcd(b, a%b);
	
}
