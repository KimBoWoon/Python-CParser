
#include <iostream>
#include <fstream>
#include <cmath>

class primeNumber
{
private:
	int num;
	bool isItPrime;
public:
	primeNumber(const int& _num):num(_num)
	{
		num = _num;
		isItPrime = isItPrimeNumber();
	}
	bool isItPrimeNumber()
	{
		for (int i = 2 ; i <= sqrt(num); i ++)
			if ( num % i == 0)
				return false;
		return true;
	}
	bool getIsItPrime(){	return isItPrime;	}

};

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if ( input.fail() )
		std::cout << "err" << std::endl;

	int caseNum;
	input >> caseNum;
	for ( int i = 0 ; i < caseNum ; i ++)
	{
		int num;
		input >> num;
		primeNumber pnum = primeNumber(num);
		std::cout << (pnum.getIsItPrime() ? "1" : "0") << std::endl;
	}
	std::ofstream output;
	output.open("test.txt");


	return 0;
}
