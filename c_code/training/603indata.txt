




#include<stdio.h>
#include<stdlib.h>


int f(int k)
{
	if(k <= 1)
		return 1;
	else
		return k*f(k-1);
}


int main()
{
	FILE *inFile;
	int i;
	int numCases;
	int n;

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		fscanf(inFile, "%d", &n);

		printf("%d\n", f(n));

	}

	fclose(inFile);
}
