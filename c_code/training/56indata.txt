
#include <iostream>
#include <fstream>
using  namespace std;
int main(){
	ifstream in;
	in.open("input.txt");
	int n,number,dNum;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> number;
		for (int j = 2; j <= number; j++){
			dNum = number;
			if (dNum%j == 0){
				if (j == number){
					cout << "1" << endl;
					break;
				}
				cout << "0" << endl;
				break;
			}
			else if (j == number)
				cout << "1" << endl;
		}
	}
	return 0;
}
