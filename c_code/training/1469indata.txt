#include <iostream>
#include <fstream>
using namespace std;

int main ()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase;i++)
	{
		int height,width,person,num=1;
		instream >>height>>width>>person;

		while(person>height)
		{
			num++;
			person-=height;
		}
		cout<<person;
		if(num<10)
			cout<<'0';
		cout<<num<<endl;
	}
}
