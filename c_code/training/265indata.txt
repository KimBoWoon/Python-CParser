
#include<fstream>
#include<iostream>

using namespace std;

void add(int *a, int *b, int *c, int *d){
	int t = *b**d;
	int t2 = *a**d + *c**b;
	*a = t2;
	*b = t;
}

int gcd(int a, int b){
	if (b == 0) return a;
	else gcd(b, a%b);
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in)return 1;

	int n;
	in >> n;
	for (int i = 0; i < n; i++){
		int a, b, c, d;
		in >> a >> b >> c >> d;
		add(&a, &b, &c, &d);
		
		cout << a / gcd(a, b) << " " << b / gcd(a, b) << endl;;
	}
	in.close();
	return 0;
}
