


#include <stdio.h>
#include <stdlib.h>

int prime(int number);
void main()
{
	FILE *infile;
	infile = fopen("input.txt", "r"); 
	
	int test_number;	
	int i;				
	int number;

	fscanf(infile, "%d", &test_number);
	for (i = 0; i < test_number; i++)
	{
		fscanf(infile, "%d", &number);
		printf("%d\n",prime(number));
	}
	fclose(infile);
}

int prime(int number)
{
	int i;
	for (i = 2; i*i <= number; i++)
	{
		if (number % i == 0)
			return 0;
	}
	return 1;

}
