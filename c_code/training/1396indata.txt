#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream inStream;
	int gatsu=0;	

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"wrong\n";
		exit(1);
	}
	inStream>>gatsu;	
	int chung = 0;		
	int sero = 0;		
	int saram = 0;		

	for(int i=0;i<gatsu;i++)
	{
		int apjariSutja = 0;	
		int ditjariSutja = 1;	
		int sum = 0;
		
		inStream>>chung;
		inStream>>sero;
		inStream>>saram;

		ditjariSutja = ditjariSutja + (saram/chung);
		saram = saram%chung;
		if(saram!=0)
		{
			apjariSutja = saram;
		}
		else
		{
			apjariSutja = chung;
			ditjariSutja = ditjariSutja-1;
		}

		sum = (apjariSutja*100)+(ditjariSutja);
		
		cout<<sum<<endl;

	}

	return 0;
}
