

#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>
int gcd(int n1,int n2);
int main(void)
{
	FILE *in_file;
	int no_cases, num1,num2;
	int i;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   


	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d", &num1, &num2);
		{
			printf("%d\n", gcd(num1,num2));  
		}
	}
	fclose(in_file);  
}

int gcd(int n1, int n2)
{
	if (n2 == 0)
		return n1;
	else if (n2 > 0)
	{
		return gcd(n2, n1%n2);
	}
}
