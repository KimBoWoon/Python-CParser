

#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)

int fib(int n);

int fib(int n)
{
	if (n <= 1) 
		return n;
	else 
		return fib(n - 1) + fib(n - 2);
}

int main()
{
	FILE *in_file;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	int n, TestCases, i = 0;

	fscanf(in_file, "%d", &TestCases);

	for (i = 0; i < TestCases; i++){

		fscanf(in_file, "%d", &n);
		printf("%d\n", fib(n));
	}


}
