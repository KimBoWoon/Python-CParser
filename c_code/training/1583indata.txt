#include <iostream>
#include <fstream>
using namespace std;

int p[500][500], m[500];

int minSum(int n, const int d[]){

	int left, right;

	for (int i = 0; i < n; i++){
		for (int j = i; j >= 0; j--){
			if (i != j)
			{
				p[j][i] = 9999999999;
				for (int k = 0; k < i-j; k++){
					if (p[j][i] > m[i + 1] - m[j] + p[j][j + k] + p[j + k + 1][i])
						p[j][i] = m[i + 1] - m[j] + p[j][j + k] + p[j + k + 1][i];
				}
			}
		}
	}

	return p[0][n - 1];
}

int main(){
	int caseNum;
	ifstream inputFile;
	inputFile.open("input.txt");

	inputFile >> caseNum;

	for (int i = 0; i < caseNum; i++){
		int testNum;
		int* arr;
		inputFile >> testNum;

		arr = new int[testNum];
		for (int j = 1; j <= testNum; j++){
			inputFile >> arr[j];
			m[j] = m[j - 1] + arr[j];
		}

		cout << minSum(testNum, arr) << endl;

	}
	return 0;
}
