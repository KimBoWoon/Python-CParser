
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int odd[10000000];
int compare(const void *first, const void *second) {
	if (*(int*)first > *(int*)second)
		return 1;
	else if (*(int*)first < *(int*)second)
		return -1;
	else
		return 0;
}

int main(void) {
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;
	int cnt = 0;
	for (int i = 0; i < testcases; i++) {
		int num, target;
		in >> num >> target;
		int targetNum = 200000000;
		int diffNum;

		for (int j = 0; j < num; j++) {
			in >> odd[j];
		}

		qsort(odd, num, sizeof(int), compare);

		int startPoint = 0;
		int endPoint = num-1;
		while (startPoint<endPoint) {
			diffNum = target -(odd[startPoint] + odd[endPoint]);
			if (diffNum >= 0) {
				startPoint++;
			}
			else {
				endPoint--;
			}

			diffNum = abs(diffNum);
			if (diffNum < targetNum) {
				cnt = 0;
				cnt++;
				targetNum = diffNum;
			}
			else if (diffNum == targetNum) {
				cnt++;
				targetNum = diffNum;
			}
		}
		cout << cnt << endl;
		cnt = 0;
	}
}
