#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b)
{
	return (b==0)? b:gcd(b,a%b);
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in>>t;
	while(t--)
	{
		int a,b,x;
		in>>a>>b;
		while(a>=1)
		{
			if(b%a!=0) x = (int)b/a+1;
			else x = b/a;
			a = a*x-b;
			b = b*x;
		}
		cout<<x<<endl;
	}
}
