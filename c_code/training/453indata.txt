#include <iostream>
#include <fstream>
#include <cstring>

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail())
	{
		std::cerr << "file open err" ;
		return 1;
	}

	int caseNum;
	input >> caseNum;
	for (int i = 0; i < caseNum; i++)
	{
		int a, b;
		input >> a >> b;
		std::cout << gcd(a,b) << std::endl;
	}
	
	return 0;
}
