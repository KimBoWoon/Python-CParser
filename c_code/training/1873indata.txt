#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;


int parent(const char* data, int index) {
	if (data[index] == 'e')	return 0;

	if (data[index] == '(') {
		int num = parent(data, index + 1);
		if (num == 0)	return -11111;
		return parent(data, index + 1) + 1;
	}
	else if (data[index] == ')')		return parent(data, index + 1) - 1;
}
int main() {
	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;
	while (cases--) {
		char data[100];
		file >> data;
		int len = strlen(data);
		data[len] = 'e';
		data[len + 1] = '\0';

		if (parent(data, 0) == 0)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	
	return -1;
}
