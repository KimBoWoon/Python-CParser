

#include <iostream>
#include <fstream>
using namespace std;

int mc(int n){
	if (n > 100)
		return (n - 10);
	else
		return mc(mc(n + 11));
}
int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;
	for (int i = 0; i < testcases; i++){
		int value;
		in >> value;
		cout << mc(value) << endl;
	}
}
