

#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>
int McCarthy(int n);
int main(void)
{
	FILE *in_file;
	int no_cases, num;
	int i;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d", &num);
		{
			printf("%d\n", McCarthy(num));  
		}
	}
	fclose(in_file);  
}

int McCarthy(int n)
{
	if (n > 100)
		return n - 10;
	else if (n <= 100)
		return McCarthy(McCarthy(n + 11));
}
