#include<stdio.h>

int Fibonacci(int n)
{
	if(n==0)		return 0;
	else if(n==1)	return 1;
	return Fibonacci(n-1)+Fibonacci(n-2);
}

void main()
{
	FILE*inFile = fopen("input.txt", "r");
	int testNum, num;
	int i;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		fscanf(inFile, "%d", &num);
		printf("%d\n", Fibonacci(num));
	}

	fclose(inFile);
}
