#include <cstdio>
#include <algorithm>

const int INF = 987654321;
int arr[1000000];

int main() {
	freopen("input.txt", "r", stdin);

	int nCase;
	scanf("%d", &nCase);

	while (nCase--) {
		int n, k;
		scanf("%d %d", &n, &k);

		for (int i = 0; i < n; i++)
			scanf("%d", arr + i);

		std::sort(arr, arr + n);

		int left = 0, right = n - 1;
		int minGap = INF, cnt = 0, sum = 0;

		while (left < right) {
			sum = arr[left] + arr[right];
			
			if (abs(sum - k) < minGap) {
				minGap = abs(sum - k);
				cnt = 1;
			}
			else if (abs(sum - k) == minGap) {
				cnt++;
			}

			if (sum > k) --right;
			else if (sum < k) ++left;
			else {
				++left; --right;
			}
		}

		printf("%d\n", cnt);
	}

	return 0;
}
