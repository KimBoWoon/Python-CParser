#include <stdio.h>
#include <stdlib.h>

int file_sum[505][505] = {0,};	

int main(void) 
{
	int numcase;
	int n; 
	int i, j, k;
	int i_temp;					
	int file[505][505] = {0,};	
	int min = 2000000001;		

	FILE * in_file;
	in_file = fopen("input.txt", "r");

	fscanf(in_file, "%d", &numcase);

	while (numcase--) 
	{
		fscanf(in_file, "%d", &n);

		for (i=0; i<n; i++)		
		{
			fscanf(in_file, "%d", &file[i][i]);
		}
		for (i=1; i<n; i++)		
		{
			for (j=i-1; j>=0; j--) 
			{
				file[j][i] = file[i][i] + file[j][i-1];
			}
		}

		for (i=1; i<n; i++)		
		{
			for (j=i-1; j>=0; j--) 
			{
				i_temp = i-1;
				for (k=i; k>j; k--) 
				{
					if (min > file_sum[k][i] + file_sum[j][i_temp])            
					{   
						min = file_sum[k][i] + file_sum[j][i_temp];
					}
					i_temp--;
				}
				file_sum[j][i] = file[j][i] + min;
				min = 2000000001;
			}
		}
		printf("%d\n", file_sum[0][n-1]);	
	}
	return 0;
}
