#include<iostream>
#include<fstream>
#include<vector>
using namespace std;
vector<int> v;
int Count = 0;
int state = 0;

int main() {
	ifstream fin;
	fin.open("input.txt");
	int num,size;
	vector<int> v;
	int Count = 0;
	fin >> num;
	for (int cnt = 0;cnt < num;cnt++) {
		fin >> size;
		for (int cnt1 = 0;cnt1 < size;cnt1++) {
			int a;
			fin >> a;
			v.push_back(a);
		}
		while (Count != 100) {
			int result = 0;
			for (int i = 0;i<v.size();i++) {
				result += v[i];
			}
			if (result == 0) {	
				cout << "ZERO" << endl;
				break;
			}
			else {
				vector<int> tmp;
				for (int i = 0;i < v.size() - 1;i++) {	
					tmp.push_back(v[i] - v[i + 1]);
				}
				tmp.push_back(v[v.size() - 1] - v[0]);
				for (int i = 0;i < v.size();i++) {
					if (tmp[i] < 0) tmp[i] = -tmp[i];
				}
				for (int i = 0;i < v.size();i++) {
					v[i] = tmp[i];
				}
			}
			Count++;
		}
		if (Count == 100) cout << "LOOP" << endl;
		v.clear();
		Count = 0;	
	}

}
