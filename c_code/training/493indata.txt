#include<stdio.h>

int Euclid(int B, int S)
{
	if(S==0)		return B;
	return Euclid(S, B%S);
}

void main()
{
	FILE*inFile = fopen("input.txt", "r");
	int testNum, num1, num2;
	int i;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		fscanf(inFile, "%d %d", &num1, &num2);
		if(num1>=num2)
			printf("%d\n", Euclid(num1, num2));
		else
			printf("%d\n", Euclid(num2, num1));

	}

	fclose(inFile);
}
