#include <stdio.h>
#include <stdlib.h>



int mcCarthy(int n) {
if (n > 100) return n - 10;
else return mcCarthy(mcCarthy(n+11));
}
 
void main(void) {
FILE *in_file;
int i, no_cases;

in_file = fopen("input.txt","r");
if(in_file == NULL) exit(1);

fscanf(in_file, "%d", &no_cases);

for (i=0; i<no_cases; i++) {
int n;
 
fscanf (in_file, "%d", &n);
 
printf("%ld\n", mcCarthy(n));
}
fclose(in_file);
}
