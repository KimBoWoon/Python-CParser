
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main(){
	ifstream inStream;
	int numTestCases;

	inStream.open("input.txt");
	if(inStream.fail()){
		cerr<<"input.txt파일을 열 수 없음.\n";
		exit(1);
	}

	inStream>>numTestCases; 
	for(int i=0; i<numTestCases; i++) 
	{
		short height,wide; 
		int n; 

		inStream>>height>>wide; 
		inStream>>n; 
		
		for(short k=1; k<=wide; k++){ 
			for(short j=1; j<=height; j++){ 
				n--;
				if(n == 0)
					cout<<(j*100)+k<<endl;
			}
		}

	}
	inStream.close();
}
