
#include <fstream>
#include <iostream>
using namespace std;

int isPrime(int num){
	
	for (int i = 2; i*i <= num; i++){
		if (num % i == 0) return 0;
	}
	return 1;

}

int main(){

	int caseNum = 0, num = 0;
	ifstream inputFile("input.txt");
	inputFile >> caseNum;
	for (int i = 0; i < caseNum; i++){
		inputFile >> num;

		cout << isPrime(num) << endl;
	}

	return 0;
}
