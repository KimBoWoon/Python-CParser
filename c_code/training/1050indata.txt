
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}

	int testCases = 0, numCases = 0;
	int integerArray[15],
		copyArray[15];

	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		fscanf(readStream, "%d", &numCases);
		for(int i = 0; i < numCases; ++i)
			fscanf(readStream, "%d", &integerArray[i]);

		bool isSequence = false;

		int loopCount = 0;
		while(1){
			if(loopCount > 1000) break;
			int count = 0;
			for(int i = 0; i < numCases; ++i){
				copyArray[i] = abs(integerArray[i] - integerArray[(i+1)%numCases]);
				if(copyArray[i] == 0) ++count;
			}
			if(count == numCases){ 
				isSequence = true;
				break;
			}
			for(int i = 0; i < numCases; ++i)
				integerArray[i] = copyArray[i];
			++loopCount;
		}
		printf("%s\n", (isSequence) ? "ZERO" : "LOOP");
	}
	fclose(readStream);
}
