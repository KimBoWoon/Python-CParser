#include <stdio.h>

int check(char *arr){

 int i, open=0, close=0;




 for(i=0;arr[i]!='\0';i++){

  if(arr[i]=='(')

   open++;

  else if(arr[i]==')')

   close++;




  if(open<close){

   printf("NO\n");

   return;

  }

 }

 if(open==close&&open!=0){

  printf("YES\n");

  return;

 }

 else{

  printf("NO\n");

  return;

 }

}

int main(){

	FILE *in_file;
	int i, num;

	char buf[51]={'\0'};



	in_file=fopen("input.txt","r");
	fscanf(in_file,"%d",&num);

 for(i=0;i<num;i++){

  fscanf(in_file,"%s", buf);

  check(buf);

 }

}
