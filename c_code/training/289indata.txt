#include <stdio.h>
#include <stdlib.h>

double Fraction(int a, int b,int c, int d);
int gcd(int a, int b);

void main()
{
	FILE *file;
	int i=0;
	int t, n1,n2,n3,n4;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d%d%d%d", &n1,&n2,&n3,&n4);
		Fraction(n1,n2,n3,n4);
	}
	fclose(file);
}

double Fraction(int a, int b,int c, int d)
{
	int re1,re2,re3,re4;
	if(b==d){
		re1= a+c;
		re2=b;
	}
	else{
		re1=(a*d)+(b*c);
		re2=b*d;
	}
	
	if(gcd(re1,re2)!=0){
		re3=re1/gcd(re1,re2);
		re4=re2/gcd(re1,re2);
	}
		printf("%d %d\n",re3,re4);
}
int gcd(int a, int b) {

  if(b==0)
	  return a;
  else
	  return gcd(b, a%b);
}
