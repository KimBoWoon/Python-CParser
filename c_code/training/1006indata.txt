#include <iostream>
#include <string>
#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>

using namespace std;

int main() {
	int _cases;
	fstream inStream;
	inStream.open("input.txt");
	inStream >> _cases;

	for (int i = 0; i < _cases; i++) {
		int arr[1001] = { 999 };
		int temp[1001] = { 999 };
		int	nums;
		inStream >> nums;
		for (int j = 0; j < nums; j++){
			inStream >> arr[j];
		}
		int count = 0;
		int k = 0;
		while(1){
			if (k%nums == nums- 1)
				temp[k% nums] = abs(arr[k%nums] - arr[0]);	
			else
				temp[k%nums] = abs(arr[k%nums] - arr[k%nums + 1]);

			if (k%nums == nums -1)
			for (int j = 0; j < nums; j++){
				arr[j] = temp[j]; 
			}
			k++;
			if (k > 1000){
				for (int j = 0; j < nums; j++){
					if (arr[j] == 0) count++;
				}
				if (count == nums) cout << "ZERO" << endl;
				else cout << "LOOP" << endl;
				break;
			}
		}
	}
	inStream.close();		
	return 0;
}
