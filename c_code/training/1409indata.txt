

#include <stdio.h>
#include <stdlib.h>

int hotel_room(int h,int w, int n){
	return ( ( n - 1 ) % h + 1) * 100 + ( ( n - 1 ) / h ) + 1;
}
void main(){
	
	FILE *in_file;
	int no_cases;

	in_file = fopen("input.txt", "r");

	if(in_file == NULL) exit(1);

	fscanf(in_file,"%d", &no_cases);

	while(no_cases--){
		int w,h,n;
		fscanf(in_file,"%d %d %d", &w, &h, &n);
		printf("%d\n",hotel_room(w,h,n));
	}
	fclose(in_file);
}
