





#include <iostream>
#include <fstream>
#include <vector>
#include <cmath>


bool primeNumber(int n);
void sum(std::vector<int> vec, int n);

int main() {
	std::ifstream inFile;
	inFile.open("input.txt");

	int testCaseNum;

	inFile >> testCaseNum;

	for (int i = 0; i < testCaseNum; i++) {
		int n;
		std::vector<int> vec;

		inFile >> n;

		for (int x = 2; x < n; x++) {
			if (primeNumber(x))
				vec.push_back(x);
		}
		sum(vec, n);
	}
}

bool primeNumber(int n) {
	for(int i = 2; i * i <= n; i++) {
		if (n % i == 0)
			return false;
	}
	return true;
}

void sum(std::vector<int> vec, int n) {
	int n1 = 0;
	int n2 = 0;
	int minNum = 1001;

	for (size_t x = 0; x < vec.size(); x++) {
		for (size_t y = 0; y < vec.size(); y++) {
			if (vec[x] + vec[y] == n) {
				if (vec[x] - vec[y] < minNum) {
					n1 = vec[x];
					n2 = vec[y];
					minNum = std::abs((double)vec[x] - (double)vec[y]);
				}
			}
		}
	}
	std::cout << n1 << ' ' << n2 << std::endl;
}
