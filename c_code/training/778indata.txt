
#include<iostream>
#include<fstream>
#include<cstdlib>
#include <cstring>
using namespace std;

int Palindrome(char str[], int f, int l) {
	if (l <= f) return 1;
	else if (str[f] != str[l]) return 0; 
	else return Palindrome(str, f+1, l-1); 
}

int main(void){
	ifstream infile;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error"<<endl;
		exit(1);
	}
	int num;
	int first;
	int last;
	char str[1000]="";
	
	infile >> num;
	for(int i=0;i<num;i++){
		infile >> str;
		first = 0;
		last =  strlen(str)-1;

		cout<<Palindrome(str, first, last)<<endl;
	}
	return 0;
}
