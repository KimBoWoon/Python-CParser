#include <stdio.h>
#include <stdlib.h>

int f(int n){
	if (n <= 1) 
		return 1;
	else		
		return n*f(n-1);
}

int main(void)
{
	FILE *file;
	int cases,num;
	
	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);

	while(cases--){
		fscanf(file,"%d",&num);
		printf("%d\n",f(num));
	}
	fclose(file);
}
