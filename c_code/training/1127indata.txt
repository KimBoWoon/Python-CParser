#include<iostream>
#include<fstream>

using namespace std;

void makeDataBase(int dataBase[]){
	int increaseRange;
	
	dataBase[0] = 1;
	for (int i = 1; i < 100; i++){
		dataBase[i] = dataBase[i - 1] + i + 1;
	}
}

int main(){
	ifstream instream;
	instream.open("input.txt");

	int testCase;
	int dataBase[100], keyNum;
	bool result = false;

	instream >> testCase;

	makeDataBase(dataBase);

	for (int i = 0; i < testCase; i++){
		instream >> keyNum;
		result = false;

		for (int j = 0; j < 100; j++){
			if (result == true) break;
			for (int k = 0; k < 100; k++){
				if (result == true) break;
				for (int o = 0; o < 100; o++){
					if (keyNum == dataBase[j] + dataBase[k] + dataBase[o]){
						result = true;
						break;
					}
				}
			}
		}

		cout << result << endl;
	}
}
