void find_small_x(int a , int b)
{
	int x;
	x= 1; 
	while(x++)
	{
		if(a*x >= b && (a*x - b) != x)
			break;
		else
			continue;
	}

	
	a = a*x - b;
	b = b*x;
	if( a != 0)
	find_small_x(a, b);
	else
		cout<< x << endl;

}
