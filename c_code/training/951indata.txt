#include <cstdio>

using namespace std;

int arr[1001];
int sum[1001][1001];
int d[1001][1001];

int min(int a, int b) {
	return (a > b) ? b : a;
}
int main(void) {
	freopen("input.txt", "r", stdin);
	int t;
	scanf("%d", &t);

	while (t--) {
		int n;
		scanf("%d", &n);

		for (int i = 0; i < n; i++) {
			int inp;
			scanf("%d", &arr[i]);

			sum[i][0] = arr[i];
			d[i][i] = arr[i];
		}

		for (int i = 1; i < n; i++) {
			for (int j = 0; j < n - i; j++) {
				sum[j][i] = sum[j + 1][i - 1] + arr[j];
			}
		}

		for (int i = 1; i < n; i++) {
			for (int j = 0; j < n - i; j++) {
				int t = j, p = j + i;

				d[t][p] = sum[t][p - t] - min(d[t + 1][p], d[t][p - 1]);
			}
		}

		printf("%d\n", d[0][n - 1]);
	}

	return 0;
}
