


#include<stdio.h>
#include<stdlib.h>

#define abs(x) ( (x < 0) ? (-x) : (x) )     

int length;
int *ducci = NULL;

void ducciSequence()
{
	int i;
	int sum = 0;
	int loop = 1000;

	while(loop--)                           
	{
		int sub;
		int tmp = ducci[0];
		for(i=0; i<length-1; i++)
		{
			sub = ducci[i] - ducci[i+1];
			ducci[i] = abs(sub);
		}
		sub = ducci[length-1] - tmp;
		ducci[length-1] = abs(sub);
	}
	
	for(i=0; i<length; i++)
	{
		sum += ducci[i];
	}

	if(sum == 0)            
	{
		printf("ZERO\n");
	}
	else
	{
		printf("LOOP\n");
	}
}

int main(void)
{
	FILE *fp = NULL;
	int testCase;

	fp = fopen("input.txt", "r");               
	fscanf(fp, "%d", &testCase);                
	while(testCase--)
	{
		int i;
		fscanf(fp, "%d", &length);              
		
		ducci = (int *) malloc (length * sizeof(int));  
		for(i=0; i<length; i++)
		{
			fscanf(fp, "%d", &ducci[i]);        
		}

		ducciSequence();                        
		free(ducci);                            
	}

	fclose(fp);                                 
	return 0;
}
