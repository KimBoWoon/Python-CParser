


#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b)
{
	if(b==0)
	{
		return a;
	}
	else
	{
		return gcd(b,a%b);
	}
}

int main()
{
	fstream in;
	in.open("input.txt");
	int numcases;
	in>>numcases;
	for(int i=0; i<numcases; i++){
		int n,m,k;
		in>>n>>m;
		k=gcd(n,m);
		cout<<k<<endl;
	}
}
