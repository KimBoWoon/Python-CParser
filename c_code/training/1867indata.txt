#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>

using namespace std;

void cal(fstream &infile); 

int main()
{
	int nocases;
	char abc[50];
	fstream infile;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" <<endl;
		exit(1);
	}

	infile >> nocases;
	infile.getline(abc,50);
	for(int i = 0; i<nocases; i++)
	{
		cal(infile);
	}

	return 0;
}

void cal(fstream &infile)
{
	char str[50];
	int length;
	int count = 0;

	infile.getline(str,50);
	
	length = strlen(str);

	if(str[0] == '(' && str[length-1] == ')')
	{
		for(int i = 1; i<length-1; i++)
		{
			if(str[i] == '(')
			{
				count ++;
			}

			else if(str[i] == ')')
			{
				count --;
			}
		}

		if(count == 0)
		{
			cout << "YES" << endl;
		}

		else
			cout << "NO" << endl;
	}

	else
		cout << "NO" << endl;
}
