#include <cstdio>
#include <cstring>
#include <climits>

int dp[502][502];
int sum[502];
int n;

int main() {
	freopen("input.txt", "r", stdin);
	int nCase;
	scanf("%d", &nCase);

	while (nCase--) {
		memset(dp, 0, sizeof(dp));
		memset(sum, 0, sizeof(sum));

		scanf("%d", &n);
		for (int i = 1, file = 0; i <= n; ++i) {
			scanf("%d", &file);
			sum[i] = sum[i - 1] + file;
		}

		for (int i = n - 1; i >= 1; --i) {
			for (int j = i + 1, min = INT_MAX; j <= n; ++j) {
				for (int k = i; k < j; ++k) {
					if (min > dp[i][k] + dp[k + 1][j]) {
						min = dp[i][k] + dp[k + 1][j];
					}
				}
				dp[i][j] = min += sum[j] - sum[i - 1];
			}
		}

		printf("%d\n", dp[1][n]);
	}

	return 0;
}
