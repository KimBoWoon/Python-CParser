#include <iostream>
#include <fstream>
#include <cmath>

int ducci(int inputArr[],int num1, int num2,int inputNum)
{
	if (num2 >= inputNum)
		num2 = 0;
	if (inputArr[num1] > inputArr[num2])
		return inputArr[num1] - inputArr[num2];
	else
		return inputArr[num2] - inputArr[num1];
}

int checkArr(int arr1[], int arr2[],int inputNum)
{
	int sw = 0;
	for (int i = 0; i < inputNum; i++)
		if (arr2[i] != 0)
		{
			sw = 1;
			break;
		}
	if (sw == 0)
		return 0;

	for (int i = 0; i < inputNum; i++)
		if (arr1[i] != arr2[i])
		{
			sw = -1;
			break;
		}
	return sw;
}

int main()
{
	std::ifstream file("input.txt");
	int testCase = 0;
	file >> testCase;
	while (testCase--)
	{
		int count = 0;
		int inputNum = 0;
		file >> inputNum;
		int *inputArr = new int[inputNum];
		int ducciArr[1001][16];
		bool sw = false;
		for (int i = 0; i < inputNum; i++)
			file >> inputArr[i];
		while (count<1001)
		{
			if (count == 0)
			{
				for (int i = 0; i < inputNum; i++)
					ducciArr[count][i] = ducci(inputArr, i, i + 1, inputNum);
			}
			else
			{
				for (int i = 0; i < inputNum; i++)
					ducciArr[count][i] = ducci(ducciArr[count-1], i, i + 1, inputNum);
				for (int i = 0; i < count; i++)
				{
					sw = checkArr(ducciArr[i], ducciArr[count], inputNum);
					if (sw)
					{
						sw = true;
						break;
					}
					if (sw == 0)
						break;
				}
				if (sw==0)
				{
					std::cout << "ZERO" << std::endl;
					break;
				}
			}
			count++;
		}
		if (sw!=0)
			std::cout << "LOOP" << std::endl;
	}
}
