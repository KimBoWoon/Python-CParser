
#include <iostream>
#include <fstream>

using namespace std;

class Hotel_Mannager   
{
	int H;      
	int W;		
	public:
		Hotel_Mannager(int H,int W) 
		{
			this->H=H;
			this->W=W;
		}
		int assign_roomNum(int N); 
		~Hotel_Mannager(){}
		
};

int Hotel_Mannager::assign_roomNum(int N)
{
	int roomNum;
	roomNum= (N%H) ? (N%H)*100 + N/H +1 : H*100 + N/H ;
	return roomNum;
}

int main()
{
	ifstream file;

	file.open("input.txt");

	if(file.is_open()==false) 
		cout<<"입력파일이 없습니다."<<endl;

	else
	{	
		int case_num;
		file>>case_num;

		for(int i=0; i<case_num; i++)
		{	
			int H,W,N;
			file>>H>>W>>N;
			Hotel_Mannager hotel(H,W);
			cout<<hotel.assign_roomNum(N)<<endl;
		}
	}
}
