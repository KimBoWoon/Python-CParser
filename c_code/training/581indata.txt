#include <cstdio>

using namespace std;

char result[13];
int facto[13] = { 0, 1 };

void shiftStore(int n, int idx, char * store) {
	for (int i = idx; i < n - 1; i++) {
		store[i] = store[i + 1];
	}
}

void getPrmtByOrder(int n, int k, char * store, int rptr = 0) {
	if (n == 1 && k == 1) {
		result[rptr] = store[0];
		return;
	}
	else {
		int sptr = 0;
		while (facto[n - 1] < k) {
			sptr++;
			k -= facto[n - 1];
		}
		result[rptr++] = store[sptr];
		shiftStore(n, sptr, store);
		getPrmtByOrder(n - 1, k, store, rptr);
	}
}

int main(void) {
	FILE * fp = fopen("input.txt", "r");

	int nCase; 
	fscanf(fp, "%d", &nCase);

	for (int i = 2; i <= 12; i++) {
		facto[i] = facto[i - 1] * i;
	}

	while (nCase--) {
		char store[13] = "abcdefghijkl";
		int n, k;
		fscanf(fp, "%d %d", &n, &k);
		
		getPrmtByOrder(n, k, store);
		printf("%s\n", result);
	}

	fclose(fp);
	return 0;
}
