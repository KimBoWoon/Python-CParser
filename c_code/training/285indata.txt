int gcd(int n1, int n2) { 
	if (n2 == 0)
		return n1;
	else if (n1 == 0)
		return n2;

	else
		return gcd(n2, n1%n2);
}

int main() {
	FILE *fp;
	int i;
	int numcases;

	fp = fopen("input.txt", "r"); 
	if (fp == NULL) exit(1);

	fscanf(fp, "%d", &numcases);

	for (i = 0; i<numcases; i++) {
		int n1, n2, n3, n4;
		int result = 0;
		int a1, a2; 
		int i;

		fscanf(fp, "%d %d %d %d", &n1, &n2,&n3,&n4);

		a1 = (n1*n4)+(n2*n3);  a2 = n2*n4;

		result = gcd(a1, a2);



		printf("%d %d\n", a1/result, a2/result);

	}
	fclose(fp);
}
