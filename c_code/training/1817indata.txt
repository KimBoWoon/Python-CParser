#include<iostream>
#include<fstream>
#include<string>

using namespace std;

string data;

void reculsive(int i, int cnt){
	if (cnt < 0){
		cout << "NO" << endl;
		return;
	}
	else if (i == data.length()){
		if (cnt == 0)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;

		return;
	}

	if (data[i] == '(')
		reculsive(i + 1,cnt + 1);
	else
		reculsive(i + 1,cnt - 1);

	return;
}

int main(){

	int T;
	ifstream file;
	file.open("input.txt");

	file >> T;
	while (T--){
		file >> data;
		reculsive(0,0);
	}

	return 0;
}
