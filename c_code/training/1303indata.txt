

#include <stdio.h>
#include <stdlib.h>

int maxFrac(int a, int b);
int gcd(int a, int b);

static int value;

void main(void) {
	FILE *input;
	int no_cases;
	int i;
	int a, b;

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d%d", &a, &b);
		printf("%d\n", maxFrac(a, b));
	}

	fclose(input);
}

int maxFrac(int a, int b) {
	int no_gcd;
	no_gcd = gcd(a, b);
	if (a / no_gcd == 1) {
		value = b / no_gcd;
	} else {
		maxFrac(a * (b / a + 1) - b, b * (b / a + 1));
	}
	return value;
}

int gcd(int a, int b) {
	int temp;
	if (a < b) {
		temp = a;
		a = b;
		b = temp;
	}

	if (b == 0) {
		return a;
	} else
		return gcd(b, a % b);
}
