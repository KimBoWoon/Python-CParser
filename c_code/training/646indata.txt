
#include <iostream>
#include <fstream>

int Factorial(int num) {

	if (num <= 0) {
		return 1;
	}
	return num * Factorial(num - 1);
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num;
	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {
		fin >> num;
		std::cout << Factorial(num) << std::endl;
	}

	fin.close();
}
