


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int Facto(int val);

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value;				

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value;
		cout << Facto(value) << endl;
	}

	inStream.close();
	return 0;
}

int Facto(int val)
{
	if (val <= 1)
	{
		return 1;
	}
	else
	{
		return val * Facto(val - 1);
	}
}
