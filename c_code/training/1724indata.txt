#include <stdio.h>
 
int func(int N, int I){
    int a[21], i = 0, j;
 
    while (N > 0){
        a[i++] = N%I;
        N /= I;
         }
 
    for (j = 0; i - j - 1 >= j; j++){
        if (a[j] != a[i - j - 1]) return 0;
          }
    return 1;
}
 
int main (void)
{
	FILE* file;
    int T, N, i;
 
	file = fopen("input.txt", "r");

    fscanf(file, "%d", &T);
 
    while (T--){
        fscanf(file, "%d", &N);
 
        for (i = 2; i <= 64; i++){
            if (func(N, i)){
                printf("1");
				  puts("");
                break;
            		    }
        	  }
 
        if (i > 64){ 
			printf ("0");
			puts("");
		  }
    	}
}
