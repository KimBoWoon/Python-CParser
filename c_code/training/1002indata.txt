
#include <iostream>
#include <fstream>
using namespace std;

int *newArr;

bool isZero(int* arr, int size){
	for (int i = 0; i < size; i++){
		if (arr[i] != 0)
			return false;
	}
	return true;
}

int isLoop(int* arr, int index, int size){
	if (isZero(arr, size)) return 0;
	if (index == 1000) return 1;

	newArr = new int[size];
	for (int i = 0; i < size; i++){
		if (i != size - 1){
			if (arr[i] - arr[i + 1] >= 0) newArr[i] = arr[i] - arr[i + 1];
			else newArr[i] = arr[i + 1] - arr[i];
		}
		else{
			if (arr[i] - arr[0] >= 0) newArr[i] = arr[i] - arr[0];
			else newArr[i] = arr[0] - arr[i];
		}
	}
	delete arr;
	arr = newArr;
	isLoop(arr, index + 1, size);
}


int main(){

	int caseNum = 0;

	ifstream inputFile("input.txt");

	inputFile >> caseNum;

	for (int i = 0; i < caseNum; i++){


		int testNum = 0;
		int* arr;
		inputFile >> testNum;

		arr = new int[testNum];

		for (int i = 0; i < testNum; i++)
			inputFile >> arr[i];

		if (isLoop(arr, 1, testNum))
			cout << "LOOP" << endl;
		else
			cout << "ZERO" << endl;



	}

	return 0;
}
