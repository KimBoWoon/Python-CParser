
#include<iostream>
#include<fstream>
using namespace std;

int fibonacci(int num);

int main()
{
	int count = 0;

	ifstream file;
	file.open("input.txt");
	file >> count;

	int num = 0;
	for (int i = 0; i < count; i++)
	{
		file >> num;
		cout << fibonacci(num) << endl;
	}

	return 0;
}

int fibonacci(int num)
{
	if (num == 0)
		return 0;
	else if (num == 1)
		return 1;
	else
		return fibonacci(num - 2) + fibonacci(num - 1);
}
