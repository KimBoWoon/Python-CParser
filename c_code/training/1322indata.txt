
#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b){                
	if(b==0) return a;      
	else gcd(b, a%b);       
}

void henry(int a, int b){
	int up, down, G, X=0;

	if(a==1) X = b; 
	else X = ( b / a ) + 1; 

	up = a*X - 1*b;               
	down = b*X;

	G = gcd(up, down); 
	up/=G;                       
	down/=G;

	if(up == 0){                  
		cout<<X<<endl;
		return;                   
	}else                         
		henry(up, down);
}

int main(){
	ifstream in;
	in.open("input.txt");

	int t;
	in>>t;
	while(t--){
		int a, b;
		in>>a>>b;
		henry(a, b); 
	}
	in.close();
	return 0;
}
