

#include <stdio.h>
#include <stdlib.h>

int M_function(int n) 
{
	if(n > 100)
		return n-10;
	else 
		M_function(M_function(n+11));
}


void main()
{
	FILE *file;
	int i;
	int test;
	int n;

	file = fopen("input.txt", "r"); 
	if(file==NULL) exit(1);

	fscanf(file, "%d", &test); 
	for(i=0; i<test; i++)
	{
		fscanf(file, "%d", &n);
		printf("%d\n", M_function(n));
	}
	fclose(file); 
}
