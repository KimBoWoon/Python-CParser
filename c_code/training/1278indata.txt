#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int p(int n);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		
		int num;
		inStream >> num;

		int a = num / 2;

		while(p(a) != 1 || p(num - a) != 1){
			--a;
		}

		cout << a << " " << num - a << endl;
	}

	inStream.close();
	return 0;
}

int p(int n){
	if (n == 2)	
		return 1;

	else if (n % 2 == 0)	
		return 0;

	else{	
		int check = 0;

		for (int j = 3; j < n; j++)	
		if (n % j == 0)
			++check;

		if (check == 0)	
			return 1;

		else 
			return 0;
	}
}
