


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

void Hotel(int h, int w, int n);		

int main()
{
	ifstream inStream;
	int numTestCases;		
	int H;					
	int W;					
	int N;					

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> H;		
		inStream >> W;		
		inStream >> N;		
		Hotel(H, W, N);
	}

	inStream.close();
	return 0;
}

void Hotel(int h, int w, int n)
{
	int y = (n - 1) % h + 1;	
	int x = (n - 1) / h + 1;	

	cout << y;

	if (x < 10)
		cout << '0';

	cout << x << endl;
}
