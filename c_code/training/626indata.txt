
#include <stdio.h>

int getFactorial(int num);

void main() {
	int numTestCases = 0;
	int i = 0;
	int num = 0;
	FILE *f;
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d", &num);
		printf("%d\n", getFactorial(num));
	}
}

int getFactorial(int num) {
	if (num == 0)
		return 1;
	else {
		return num * getFactorial(num - 1);
	}
}
