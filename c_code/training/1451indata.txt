
#include <iostream>
#include <fstream>

void getHotel(int H, int W, int N) {

	int YY = 0,XX = 0;

	if (H*W < N) {
		std::cout << "불가능 합니다."<<std::endl;
	}

	if (N % H == 0) {
		YY = H;
		XX = N / H;
	}
	else {
		YY = N % H;
		XX = N / H + 1;
	}
	if(XX <10) std::cout << YY <<"0"<< XX << std::endl;
	else	std::cout << YY << XX << std::endl;	
}

int main() {
	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num1, num2, num3;
	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {
		fin >> num1 >> num2 >> num3;
		getHotel(num1, num2, num3);		
	}
	fin.close();
}
