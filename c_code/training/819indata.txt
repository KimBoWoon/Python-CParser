


#include <iostream>
#include <cstdlib>
#include <fstream>
using namespace std;

int fibo(int n)
{
	if (n <= 0)
		return 0;
	else if (n == 1)
		return 1;
	else
		return fibo(n - 1) + fibo(n - 2);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> num;
		cout << fibo(num) << endl;
	}
	inStream.close();
	return 0;
}
