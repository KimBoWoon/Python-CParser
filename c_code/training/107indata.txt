


#include<stdio.h>

int mcCarthy(int _n)    
{
	if(_n > 100)
	{
		return (_n-10);
	}
	else
	{
		return mcCarthy(mcCarthy(_n+11));
	}
}

int main(void)
{
	FILE *fp;
	int testCase;

	fp = fopen("input.txt", "r");   
	fscanf(fp, "%d", &testCase);    
	while(testCase--)
	{
		int n, result;
		fscanf(fp, "%d", &n);       
		result = mcCarthy(n);       
		printf("%d\n", result);     
	}

	fclose(fp);                     
	return 0;
}
