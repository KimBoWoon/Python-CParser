#include <iostream>
#include <fstream>
using namespace std;

long mecasi(int num){
    if(num > 100)   return num - 10;
    
    return mecasi(mecasi(num + 11));
}

int main(){
    ifstream file;
    file.open("input.txt");
    
    int cases;
    file >> cases;
    while(cases--){
        int num;
        file >> num;
        cout << mecasi(num) << endl;
    }
    return 1;
}
