#include<iostream>
#include<fstream>
using namespace std;

int main(){
	ifstream fin;
	fin.open("input.txt");
	if (!fin) {
		cout << "file error" << endl;
		return 0;
	}
	int num,H, W, N,room;
	fin >> num;
	for (int cnt = 0; cnt < num; cnt++){
		fin >> H>>W>>N;
		if (N > H*W) {
			cout << "방이 꽉찻습니다." << endl;
			return 0;
		}
		if (0 == N % H){		
			room = H * 100 + (N / H);
		}
		else room = (N - ((N / H)*H)) * 100 + ((N/H)+1);	
		cout << room<<endl;
	}
}
