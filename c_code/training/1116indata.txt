


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value;				

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	int Tri[45] = { 0 };		
	int	Val[1001] = { 0 };		

	for (int i = 1; i < 45; i++)
	{
		Tri[i] = i*(i + 1) / 2;		
	}

	for (int i = 1; i < 45; i++)	
	{
		for (int j = i; j < 45; j++)
		{
			for (int k = j; k < 45; k++)
			{
				if (Tri[i] + Tri[j] + Tri[k] < 1001)
				{
					Val[Tri[i] + Tri[j] + Tri[k]] = 1;
				}
			}
		}
	}

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value;
		cout << Val[value] << endl;
	}

	inStream.close();
	return 0;
}
