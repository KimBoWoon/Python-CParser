


#include <iostream>
#include <cstdlib>
#include <fstream>
#include <string>
#include <cstring>
using namespace std;
int enumer = 0;
int id = 0;
string VPSNotice(char arr[], int len)
{
	if (arr[id] == '(')
		enumer++;
	else if (arr[id] == ')')
		enumer--;
	id += 1; 
	if (enumer == 0 && len <= 1) 
		return "YES";
	else if (enumer < 0 || len <= 1)
		return "NO";
	else 
		return VPSNotice(arr, --len);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	char cstr[51];
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		enumer = 0;
		id = 0;
		inStream >> cstr;
		if (cstr[0] == ')')
		{
			cout << "NO" << endl;
			continue;
		}
		cout << VPSNotice(cstr, strlen(cstr)) << endl;
	}
	inStream.close();
	return 0;
}
