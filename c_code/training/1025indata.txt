void printResult(int* numArray, int numSize)
{
	int* newArray=new int[numSize];
	int start, end;
	int result;
	for (int j = 0; j < 1000; j++){
		for (int k = 0; k < numSize; k++)
		{
			if (k + 1 == numSize) {
				end = 0;
			}
			else end = k + 1;

			newArray[k] = abs(numArray[k] - numArray[end]);
		}
		for (int k = 0; k < numSize; k++)
		{
			numArray[k] = newArray[k];
		}
		for (int k = 0; k < numSize; k++)
		{
			if (numArray[k] == 0) result = 1;
			else{
				result = 0;
				break;
			}
		}
		if (result == 1){
			cout << "ZERO" << endl;
			return;
		}
	}
	cout << "LOOP" << endl;
	
}
