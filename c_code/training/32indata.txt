
#include <iostream>
#include <fstream>

using namespace std;

int prime(int num)     
{
	for(int j=2; j*j<=num; j++) 
		if(num%j==0) return 0; 

	return 1;
}

int main()
{
	ifstream file;

	file.open("input.txt");

	if(file.is_open()==false) 
		cout<<"입력파일이 없습니다."<<endl;

	else
	{	
		int case_num;
		file>>case_num;

		for(int i=0; i<case_num; i++)
		{	
			int num;
			file>>num;
			cout<<prime(num)<<endl;
		}
	}
}
