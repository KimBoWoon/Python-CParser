
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int checkPalindrome(char str[], int first, int last);

int main()
{
	int count = 0;

	ifstream file;
	file.open("input.txt");
	file >> count;

	char str[1000];

	for (int i = 0; i < count; i++)
	{
		file >> str;

		int strLength = 0;
		while (str[strLength] != '\0') 
			strLength++;

		cout << checkPalindrome(str, 0, strLength-1) << endl;
	}
}

int checkPalindrome(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first + 1, last - 1);
}
