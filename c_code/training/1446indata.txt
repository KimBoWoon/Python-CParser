
#include <stdio.h>
#include <stdlib.h>

void main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
		
	int testCases = 0,
		numberOfFloors = 0,
	 	numberOfRooms = 0,
		guestNumber = 0;

	readStream = fopen("input.txt", "r");
	
	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		exit(1);
	}
	
	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		fscanf(readStream, "%d %d %d", &numberOfFloors, &numberOfRooms, &guestNumber);
		(guestNumber % numberOfFloors) ? printf("%d%02d\n", (guestNumber % numberOfFloors), (guestNumber / numberOfFloors + 1)) : printf("%d%02d\n", (numberOfFloors), (guestNumber / numberOfFloors));
	}
	fclose(readStream);
}
