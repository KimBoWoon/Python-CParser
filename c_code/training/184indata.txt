#include <stdio.h>
#include <stdlib.h>

int McCarthy(int num);

void main()
{
	FILE *file;
	int i=0;
	int t, n;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d", &n);
		printf("%d\n",McCarthy(n)); 
	}
	fclose(file);
}
int McCarthy(int num)
{
	int ans=0;
	if(num>100) 
		ans = num-10;
	else
		return McCarthy(McCarthy(num+11)); 
}
