

#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int fibo(int num){
	if (num == 0)	return 0;
	else if (num == 1)  return 1;
	else if (num > 1)   return fibo(num - 1) + fibo(num - 2);
}
int main(){
	ifstream instream("input.txt");
	int numtestcases, numdata;
	instream >> numtestcases;
	for (int i = 0; i < numtestcases; i++){
		instream >> numdata;
		cout << fibo(numdata) << endl;
	}
	instream.close();
}
