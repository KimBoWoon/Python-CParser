#include <iostream>
#include <fstream>

int mcCarthy(int num);

int main(void){

	std::ifstream inFile("input.txt");
	if (inFile.fail()){
		std::cout << "not file" << std::endl;
		return 0;
	}

	int testCases = 0;
	inFile >> testCases;

	int num = 0;

	for (int i = 0; i < testCases; i++){
		inFile >> num;

		std::cout << mcCarthy(num) << std::endl;
	}
	return 0;
}

int mcCarthy(int num){

	if (num > 100){
		return num - 10;
	}
	else{
		mcCarthy(mcCarthy(num + 11));
	}
}
