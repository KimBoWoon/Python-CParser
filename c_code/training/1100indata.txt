
#include<iostream>
#include<fstream>

using namespace std;

int main(){

	ifstream f; 
	f.open("input.txt");

	int numCount = 0, obNum = 0; 
	int numArray[45]; 
	for (int i = 1; i < 45; i++){ 
		numArray[i] = (i)*(i + 1) / 2;
	}
	f >> numCount;

	for (int i = 0; i < numCount; i++){
		f >> obNum;

		for (int j = 1; j < 45; j++){ 
			if (numArray[j] > obNum){
				cout << "0" << endl;
				break;
			}
			for (int k = 1; k < 45; k++){
				if (numArray[j] + numArray[k]> obNum){
					break;
				}
				for (int l = 1; l < 45; l++){
					if (numArray[j] + numArray[k] + numArray[l] == obNum){
						cout << "1" << endl;
						j = 45; 
					}
					else if (numArray[j] + numArray[k] + numArray[l] > obNum){ 
						break;
					}
				}
			}
		}
	}
	return 0;
}
