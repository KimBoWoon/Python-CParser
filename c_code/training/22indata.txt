#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int itIs(int x);

int main()
{
	ifstream inStream;
	int numTestCases;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int ans = 0;
		int x = 0;
		inStream >> x;
		ans = itIs(x);
		cout << ans << endl;
	}
	inStream.close();
}

int itIs(int x)
{
	int i;
	if (x < 2)
	{
		return 0;
	}
	for (i = 2; i < x; i++)
	{
		if (x % i == 0)
			return 0;
	}
	return 1;
}
