#include <stdio.h>
#include <stdlib.h>

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file, "%d", &num_cases);

	for(i=0; i<num_cases; i++){

		int height, width, num;
		int key_m, key_n;
		int room_f, room_r;


		fscanf(in_file, "%d", &height);
		fscanf(in_file, "%d", &width);
		fscanf(in_file, "%d", &num);

		key_m = num/height;	
		key_n = num%height;	

		if(key_n==0){

			room_f = height;
			room_r = key_m;	

		}

		else if(key_n != 0){

			room_f = key_n;		
			room_r = key_m+1;	

		}

		if(room_r< 10){

			printf("%d0%d\n",room_f, room_r);

		}

		else if(9< room_r){

			printf("%d%d\n",room_f,room_r);

		}
	}
	fclose(in_file);
}
