

#include<iostream> 
#include<string> 
#include<stack> 
#include<algorithm> 
#include<fstream>
using namespace std;

void emtpyStack(stack<char> &st)
{
	while (!st.empty())
	{
		st.pop();
	}
}

int main()
{
	ifstream f;
	f.open("input.txt");
	stack<char> st;
	string s;
	int n;
	f >> n;
	for (int i = 0; i < n; ++i)
	{
		bool flag = true;
		f >> s;
		for (int j = 0; j < s.length(); ++j)
		{
			if (j == 0 && s[j] == ')')
			{
				cout << "NO" << endl;
				flag = false;
				break;
			}
			else if (j == 0 && s[j] == '(') st.push(s[j]);
			else if (j > 0)
			{
				if (s[j] == '(') st.push(s[j]);
				else if (s[j] == ')')
				{
					if (st.empty()){
						cout << "NO" << endl;
						flag = false;
						break;
					}
					else if (st.top() == '(')
					{
						st.pop(); continue;
					}
				}
			}
		}

		if (flag)
		{
			if (st.empty())
			{
				cout << "YES" << endl;
				emtpyStack(st);
			}
			else
			{
				cout << "NO" << endl;
				emtpyStack(st);
			}
		}
	}
	return 0;
}
