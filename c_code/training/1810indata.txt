


#include<iostream>
#include<fstream>
#include<cstdlib>
#include<string>
using namespace std;

void paren(string stringArray);
int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	if (inStream.fail()) exit(1);
	int numTest;
	inStream >> numTest;
	string stringArray;
	for (int i = 0; i < numTest; i++)
	{

		inStream >> stringArray;
		paren(stringArray);
	}
}

void paren(string stringArray)
{
	int length= stringArray.length();
	int rightCount = 0;
	int leftCount = 0;
	int result = 1;
	for (int k = 0; k < length; k++)
	{
		if (stringArray[k] == '(') rightCount++;
		if (stringArray[k] == ')') leftCount++;
		if (rightCount < leftCount){
			result = 0;
		}
	}
	if (rightCount != leftCount) result = 0;

	if (result==0) cout << "NO" << endl;
	else cout << "YES" << endl;
}
