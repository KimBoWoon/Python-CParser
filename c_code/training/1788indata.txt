#include <iostream>
#include <string>
#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>

using namespace std;
char str[50];
int stat = 0;
int rec(int num){
	if (stat < 0) return -1;
	if (str[num] == '('){
		stat++;
		return 1 + rec(num + 1);
	}
	if (str[num] == ')'){
		stat--;
		return -1 + rec(num + 1);
	}
	else return 0;
}

int main() {
	int _cases;
	fstream inStream;

	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1;
	}

	inStream >> _cases;

	for (int i = 0; i < _cases; i++) {
		stat = 0;
		inStream >> str;
		if (rec(0) == 0) cout << "YES" << endl;
		else cout << "NO" << endl;

	}
	inStream.close();		
	return 0;
}
