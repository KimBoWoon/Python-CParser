#include<iostream>
#include<fstream>
using namespace std;

int Pibonacci(int n){
	if (n == 0)		return 0;
	else if (n == 1)	return 1;
	else return Pibonacci(n - 1) + Pibonacci(n - 2);
}

int main(){
	ifstream inStream;
	int numCases;
	int number;

	inStream.open("input.txt");

	inStream >> numCases;

	for (int i = 0; i<numCases; i++){
		inStream >> number;

		cout << Pibonacci(number) << endl;
	}

	inStream.close();
}
