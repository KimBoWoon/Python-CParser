

#include<iostream>
#include<fstream>
#include<cstring>
using namespace std;

bool check(char* arr,int idx,int vps, int size) 
{
	if(vps<0) 
	{
		cout<<"NO"<<endl;
			return 0;
	}
	if(idx==size && vps>0) 
	{	
		cout<<"NO"<<endl;
			return 0;
	}

	else if(arr[idx]=='(') 
	{	
		idx++;
		vps++;
		return check(arr,idx,vps,size); 
	}
	else if(arr[idx]==')') 
	{
		idx++;
		vps--;
		return check(arr,idx,vps,size);	
	}
	if(vps==0 && idx==size) 
	{
		cout<<"YES"<<endl;
			return 1;
	}
}

int main()
{
	ifstream instream("input.txt",ios::in);
	int testCase;
	instream>>testCase;
	char VPS[50];
	
	for(int i=0;i<testCase;i++)
	{
		instream>>VPS;
		int temp =strlen(VPS); 
		int num=0;
		int vps=0;
		
		check(VPS,num,vps,temp);
	}
}
