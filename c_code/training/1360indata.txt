void henry(int a, int b){
	 int i=2, temp;
	 for(i=2;  ; i++){
		 if(a * i > b ){
			 a = a * i-b;
		     b = b * i;
						 }
		if(a==1)
			break;
					
		temp = f(a,b);
		a /= temp;
		b /= temp; 
		}
	 printf("%d\n",b);
}
void main(){

a:

int f(int a,int b){
	int temp;

	for( ; ; ){                         
		if(b==0)                     
			break;
		else
			temp = a%b;
			a = b;
			b=temp;
	}

}
