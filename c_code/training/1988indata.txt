#include <iostream>
#include <fstream>
using namespace std;

int partition(int *set,int left, int right){
	int pivot=set[left];
	int bottom=left+1;
	int top=right;
	int temp;

	while(bottom<=top){
		while(pivot>set[bottom]) bottom++;
		while(pivot<set[top]) top--;
		if(bottom<=top) {
			temp=set[bottom];
			set[bottom]=set[top];
			set[top]=temp;
		} 
	}
	temp=set[left];
	set[left]=set[top];
	set[top]=temp;

	return top;
}

void qSort(int *set,int left,int right){
	if(left<=right){
		int pivot=partition(set,left,right);
		qSort(set,left,pivot-1);
		qSort(set,pivot+1,right);
	}
}

int main(){
	ifstream in;
	in.open("input.txt");
	int t;
	in>>t;

	while(t--){
		int length,K;
		in>>length>>K;

		int min,num;
		int j=1,cnt=1;
		int p=length-1;

		int *set = new int[length]; 
		for(int i=0; i<length; i++) in>>set[i]; 

		qSort(set, 0, p);  
		min = (set[0] + set[length-1]); 

		if(min >= K) min = min-K;
		else min = K-min;

		while(j != p){  
			num = (set[j] + set[p]);

			if(num >= K) num = num-K;
			else num = K-num;

			if(min > num){
				min=num;
				cnt=0;
				cnt++;
			}else if(min == num)
				cnt++;

			if((set[j] + set[p]) > K) p--;
			else j++;
		}
		cout<<cnt<<endl; 
		delete set;  
	}
	in.close();
	return 0;
}
