
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 15

int ducci(int arr[], int size, int cnt){
    int i, zero = 0;
    int newArr[MAX_SIZE] = {0};
    
    for(i=0; i<size; i++)
        if(arr[i] == 0)
            zero++;
    if(zero == size) return 1;      
    if(cnt >= 1000) return 0;       
    for(i=0; i<size; i++)
        newArr[i] = abs(arr[i]+arr[(i+1)%size]);
    return ducci(newArr, size, ++cnt);
}

void ducciSequence(FILE *fin){
    int arr[MAX_SIZE] = {0};
    int i, size=0;
    
    fscanf(fin, "%d", &size);
    for(i=0; i<size; i++)
        fscanf(fin, "%d", &arr[i]);
    
    if(ducci(arr, size, 0))
        printf("ZERO\n");
    else
        printf("LOOP\n");
}
    
int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        ducciSequence(fin);        
    
    fclose(fin);            
    
    return 0;           
}
