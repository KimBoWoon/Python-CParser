#include <iostream>
#include <cstdlib>
#include <fstream>

using namespace std;
std::ifstream inStream;

int len(char str[]){
	int i = 0;

	while (*(str + i) != 0)
		i++;
	return i;
}

int check(char str[], int begin, int end){
	if (end <= begin) return 1;
	else if (str[begin] != str[end]) return 0;
	else return check(str, begin + 1, end - 1);
}

int main() {
	
	int cases;
	char strs[1001];
	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;
	
	for (int i = 0; i < cases; i++) {
		inStream >> strs;
		cout << check(strs, 0, len(strs) - 1) << endl;

	}
	inStream.close();		
	return 0;
}
