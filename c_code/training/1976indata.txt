

#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>
using namespace std;

int sortArr(const void *a, const void*b){
	return (*(int*)a - *(int*)b);
}

int twonumber(int *arr,int n,int k){ 
	int num1=0, num2=n-1, cnt=0;
	int ans=200000000;

	while (num2 >= 0 && num1 < n) {
		int sum = arr[num1] + arr[num2];
		int abssum = abs(k - sum); 

		if ((num1 < num2 && sum == k)||(num1 >= num2 && ans == abssum)) cnt++;

		if (num2 - num1 == 1 && cnt == 0) {
			ans = abs(sum-k);
			num1++,num2--,cnt++;
		}
		if (sum<= k) num1++;
		else if (sum>= k) num2--;
	}
	return cnt;
}

int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");

	instream>>no_cases;
	
	while(no_cases--){
		int n, k;
		instream>>n>>k;
		int *arr = new int[n];

		for(int i=0; i<n; i++) instream >> arr[i];
	
		qsort(arr,n, sizeof(int), sortArr);	
		cout<<twonumber(arr,n,k)<<endl;
	}
	instream.close();
}
