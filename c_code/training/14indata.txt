
#include<iostream>
#include<fstream>
#include<cmath>
using namespace std;

int main(){
	ifstream File;
	int numCase = 0;
	int num = 0;
	bool flag = true;

	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		flag = true;
		File >> num;
		for (int i = 2; i <= sqrt((double)num); i++){
			if (num % i == 0)
				flag = false;
		}
		if (flag)
			cout << '1' << endl;
		else
			cout << '0' << endl;
	}
	File.close();
	return 0;
}
