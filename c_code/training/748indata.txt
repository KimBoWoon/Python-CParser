
#include <cstdio>
#include <cstdlib>
#include <cstring>

int palindrome(char* pString, int begin, int end){
	
	if(end <= begin) return 1;
	else if(*(pString + begin) != *(pString + end)) return 0;
	else return palindrome(pString, begin + 1, end - 1);
}
int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}

	int testCases = 0;
	char stringArray[1000];

	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		fscanf(readStream, "%s", stringArray);
		printf("%d\n", palindrome(stringArray, 0, (strlen(stringArray)-1)));
	}
	fclose(readStream);
}
