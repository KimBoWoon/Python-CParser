#include <iostream>
#include <fstream>
using namespace std;

int GCD(int a, int b)
{
	if(b==0)
		return a;
	else if(b>0)
		return GCD(b,a%b);

}
int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream){
		cout << "파일을 열 수 없습니다."<<endl;
		return 0;
	}
	int test_number;
	instream >> test_number;
	for(int i=0; i<test_number; i++)
	{
		int A,B;
		instream >> A >> B;
		if(A>=B)
			cout << GCD(A,B) << endl;
		else
			cout << GCD(B,A) << endl;
	}

}
