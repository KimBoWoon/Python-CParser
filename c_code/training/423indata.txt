
#include <iostream>
#include <cstdlib>
#include <fstream>

int euc(int a, int b);

int main()
{
	std::fstream inputfile;
	int testcase;
	int a, b;

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> a >> b;
		std::cout << euc(a,b) << std::endl;
	}

	return 0;
}

int euc(int a, int b)
{
	if (b == 0)
		return a;
	else
		return euc(b, a%b);
}
