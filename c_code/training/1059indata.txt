

#pragma warning(disable:4996)

#include<stdio.h>
#include<stdlib.h>
#include <math.h>

void DucciSeq(int num, int* arr);

int main(void)
{
	int no_cases,i,j;
	int num = 0; 
	int arr[15] = { 0 };

	FILE *in_file;
	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   


	for ( i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d", &num);

		for (j = 0; j < num; j++) 
			fscanf(in_file, "%d", &arr[j]);

		DucciSeq(num, arr);
	}
	return 0;
}

void DucciSeq(int num, int* arr)
{
	int i, j;
	int numOfZero; 
	int temp; 

	for ( i = 0; i < 1000; i++)
	{
		numOfZero = 0;
		temp = arr[0];

		for ( j = 0; j < (num-1); j++)
		{
			arr[j] = abs(arr[j+1] - arr[j]);
			if (arr[j] == 0) numOfZero++;
		}
		arr[num-1] = abs (arr[num - 1] - temp);

		if ((numOfZero == (num-1)) && (arr[num-1]==0)) 
		{
			printf("ZERO\n");
			return;
		}
	}

	printf("LOOP\n");
}
