#include <cstdio>

int getLastHenryNum(int a, int b) {
	if (b % a == 0)
		return b / a;

	return getLastHenryNum(a * (b / a + 1) - b, b * (b / a + 1));
}

int main() {
	FILE * fp = fopen("input.txt", "r");
	int nCase;
	fscanf(fp, "%d", &nCase);

	while (nCase--) {
		int a, b;
		fscanf(fp, "%d %d", &a, &b);
		printf("%d\n", getLastHenryNum(a, b));
	}

	fclose(fp);
	return 0;
}
