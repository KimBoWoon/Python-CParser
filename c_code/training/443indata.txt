#include<iostream>
#include<fstream>

using namespace std;

int gcd(int a, int b){
	if (a == 0)
		return b;
	if (b == 0){
		return a;
	}
	return gcd(b, a%b);
}
int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;

	while (cases--){
		int a, b;
		inFile >> a >> b;
		cout << gcd(a, b) << endl;
	}
	return 1;
}
