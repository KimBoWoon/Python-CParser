

#include <iostream>
#include <fstream>
#include <cstdlib>
int cal(int x, int y); 
using namespace std;

int main()
{
	ifstream inStream;
	int numTestCases;

	inStream.open("input.txt"); 
	inStream >> numTestCases;
	
	for (int i = 0; i < numTestCases; i++) 
	{
		int temp, count = 0, sum, n; 
		inStream >> n;
		int turple[15];
		for (int i = 0; i < n; i++)
			inStream >> turple[i]; 
		
		while (count <= 1000) 
		{
			sum = 0;
			for (int i = 0; i < n; i++)
				sum += turple[i];

			temp = turple[0];
			for (int i = 0; i < n - 1; i++)
				turple[i] = cal(turple[i], turple[i + 1]); 
			turple[n - 1] = cal(turple[n - 1], temp); 

			if (sum == 0) 
			{
				cout << "ZERO" << endl;
				break;
			}
			count++;
		}
		if (count == 1001) 
			cout << "LOOP" << endl;
	}
	
	inStream.close();
	return 0;
}

int cal(int x, int y) 
{
	int T = 0;
	T = x - y;
	if (T < 0)
		T = -1 * T;
	return T;

}
