
#include<iostream>
#include<fstream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
	int factorial[13]={1,1};
	for(int i=2;i<=12;i++)
		factorial[i]=factorial[i-1]*i;
	ifstream fs;
	int n;
	fs.open("input.txt");
	fs>>n;
	while(n--)
	{
		string s="";
		int a,b,k=1;
		fs>>a>>b;
		b--;
		for(int i=0;i<a;i++) s+='a'+i;
		for(int i=1;i<a;i++)
		{
			int position=b,f=factorial[a-i];
			cout<< s[position/f];
			s.erase(s.begin()+position/f);
			b %= f;
		}
		cout<<s<<"\n";
	}
}
