
#include <stdio.h>
#include <stdlib.h>

int recur(int num);

void main(void)
{

	FILE *in_file;
	int no_cases;
	int i, data;
	in_file = fopen("input.txt","r");
	if(in_file == NULL) exit(1);
	fscanf(in_file,"%d",&no_cases);

	
	for(i=0; i<no_cases; i++){
		fscanf(in_file,"%d",&data);
		printf("%d\n",recur(data));
	}
}

int recur(int num)
{
	if(num==1||num==0)
		return 1;
	else
		return num*recur(num-1);
}
