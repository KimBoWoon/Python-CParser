#include <iostream>
#include <fstream>

using namespace std;



void VPS(char a);
int odd = 0 ;
int even = 0;

int main()
{
	ifstream infile;

	infile.open("input.txt");
	
	int numTestCases = 0;
	
	
	infile >> numTestCases;
	char a;
	infile.get(a);
	
	
	while(numTestCases--)
	{
		
		
		even = 0;
		odd = 0;
		int flag = 0;
		a = ' a';
		
		
		while(a  != '\n' && !infile.eof() ){
			
			infile.get(a);
		
		

			VPS(a);
			if(odd > even)
				flag++;
			
		
			
		}
		
		if( infile.eof())
			odd--;
		if(even != odd || flag > 0)
		{	cout << "NO" <<endl;
			
		}
		else if( even == odd )
			{cout << "YES" <<endl;
			
		}
		
		
	}
	infile.close();
	return 0;
}
void VPS(char a)
{
	
	if(a == '(')
		even++;
	else if( a == ')')
		odd++;
	
	
}
