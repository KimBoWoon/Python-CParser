
#include<iostream>
#include<fstream>
#include<string>
using namespace std;
bool isPalindrome(string s,int x,int y)
{
	if(y<=x)return true;
	if(s[x]!=s[y])return false;
	return isPalindrome(s,x+1,y-1);
}
int main()
{
	ifstream fs;
	int n;
	fs.open("input.txt");
	fs>>n;
	while(n--)
	{
		string s;
		fs>>s;
		cout<<isPalindrome(s,0,s.length()-1)<<"\n";
	}
}
