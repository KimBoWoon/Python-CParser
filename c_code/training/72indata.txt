
#include <stdio.h>
#include <stdlib.h>
#include <memory.h>
#include <math.h>

main()
{
    int i, case_num;
    int num, n;
    FILE *fp;
    fp= fopen("input.txt","r"); 
    if(fp==NULL) exit(1);
    fscanf(fp,"%d", &case_num); 
    for(i=0; i<case_num; i++) 
    {
        fscanf(fp,"%d", &num);
        
        n=2;
        while(1)
        {
            if(num % n==0 && num !=n)
            {
                printf("0\n");
                break;
            }
            else if(num == n)
        {
            printf("1\n");
            break;
        }
        n++;
        }
    }
}
