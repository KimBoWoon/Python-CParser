#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int Bunsu(int bunja, int bunmo)
{
	int tempchaedaegongyarksu = 0;
	if (bunmo == 0)
	{
		tempchaedaegongyarksu = bunja;
		return tempchaedaegongyarksu;
	}
	else
	{
		return Bunsu(bunmo, bunja%bunmo);
	}
}

int main()
{
	int gatsu = 0;	
	int bunja1 = 0;
	int bunmo1 = 0;
	int bunja2 = 0;
	int bunmo2 = 0;

	int bunja = 0;
	int bunmo = 0;

	ifstream inStream;
	
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> gatsu;

	for (int i = 0; i < gatsu; i++)
	{

		inStream >> bunja1;
		inStream >> bunmo1;
		inStream >> bunja2;
		inStream >> bunmo2;

		int tempbunja1 = bunja1;
		int tempbunmo1 = bunmo1;
		int tempbunja2 = bunja2;
		int tempbunmo2 = bunmo2;
		int chaedaegongyarksu = 0;

		bunja1 = (bunja1 *tempbunmo2);
		bunmo1 = (bunmo1 *tempbunmo2);
		bunja2 = (bunja2 *tempbunmo1);
		bunmo2 = (bunmo2 *tempbunmo1);

		bunja = bunja1 + bunja2;	
		bunmo = bunmo1;				
	

		chaedaegongyarksu = Bunsu(bunja, bunmo);	
		cout << bunja/chaedaegongyarksu << " " << bunmo/chaedaegongyarksu << endl;

	}

	inStream.close();
	return 0;
}
