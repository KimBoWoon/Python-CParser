#include<cmath>
#include<fstream>
#include<cstring>
using namespace std;

int Fibo(int num)
{
	if (num <= 1) 
		return num;
	else          
		return Fibo(num - 1) + Fibo(num - 2);

}


int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int testNum;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> testNum;
		cout << Fibo(testNum) << endl;
	}
	return 0;
}
