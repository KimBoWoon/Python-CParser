

#include <iostream>
#include <fstream>
#include <math.h>
#include <cstdlib>

using namespace std;


int main()
{
	ifstream infile;
	int testCase;
	int n; 
	int seq[16];
	int a=0;
	int step = 0;
	int temp=0;
	infile.open("input.txt");

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> n;
		a = 0;
		step = 0;
		for (int j = 0; j < n; j++)
			seq[j] = 0;

		for (int j = 0; j < n; j++)
			infile >> seq[j];

		while (1)
		{
			for (int j = 0; j < n; j++)
			{
				if (j == 0) temp = seq[j]; 
				if (j == n - 1)
				{
					seq[j] = abs(seq[j] - temp);
				}
				else
					seq[j] = abs(seq[j + 1] - seq[j]);

			}
			step++;
			for (int k = 0; k < n; k++) 
			{
				if (seq[k] == 0)
				{
					a++;
				}
				else
				{
					break;
				}
			}
			if (a == n || step>1000)
				break;
			a = 0;
		}

		if (a==n)
			cout << "ZERO" << endl;
		else
			cout << "LOOP" << endl;

	}

	infile.close();

	return 0;
}
