#include<iostream>
#include<fstream>
#include<cstring>
using namespace std;

int Palindrome(char str[], int first, int last){
	if (last <= first)	return 1;
	else if (str[first] != str[last])	return 0;
	else
		return Palindrome(str, first + 1, last - 1);
}

int main(){
	ifstream inStream;
	int numCases;
	char str[1000]; 

	inStream.open("input.txt");

	inStream >> numCases;

	for (int i = 0; i<numCases; i++){
		inStream >> str;

		int first = 0;
		int last = strlen(str) - 1;

		cout << Palindrome(str, first, last) << endl;
	}

	inStream.close();
}
