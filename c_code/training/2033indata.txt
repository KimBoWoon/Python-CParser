#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int compare (const void *first, const void *second)
{
    if (*(int*)first > *(int*)second)
        return 1;
    else if (*(int*)first < *(int*)second)
        return -1;
    else
        return 0;
}
 
int func(int *array, int size, int k)
{
	int start=0,end=size-1;
	int min=100000000;
	int result= 0;
	while(start<end)
	{
		if(abs(array[start]+array[end]-k)<min)
		{
			min=abs(array[start]+array[end]-k);
			result=1;
		}
		else if(abs(array[start]+array[end]-k)==min)
		{
			result++;
		}
		


		if(array[start]+array[end]>k)
		{
			end--;
		}
		else if(array[start]+array[end]<k)
		{
			start++;
		}
		else
		{
			start++;
			end--;
		}
	
	}

	return result;

}

void main()
{
	int datacase = 0;
	FILE* infile = fopen("input.txt","r");
	int i = 0;
	fscanf(infile,"%d",&datacase);

	for(i=0; i<datacase; i++)
	{
		int size = 0;
		int k = 0;
		int j = 0;
		int *array = NULL;
		int result = 0;
		fscanf(infile,"%d %d",&size,&k);
		array=(int*)malloc(sizeof(int)*size);
		for(; j<size; j++) fscanf(infile,"%d",array+j);
		qsort(array,size,sizeof(int),compare);
		result = func(array,size,k);
		printf("%d\n", result);
		free(array);
	}
	fclose(infile);
}
