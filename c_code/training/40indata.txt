#include <iostream>
#include <fstream>
#include <math.h>

bool is_prime(int num);

int main(void){

	std::ifstream inFile("input.txt");
	
	if (inFile.fail()){
		std::cout << "input file opening fail\n" << std::endl;
		return 0;
	}

	int testCases = 0;
	int num = 0;

	inFile >> testCases;
	
	for (int i = 0; i < testCases; i++){
		inFile >> num;
		std::cout << is_prime(num) << std::endl;
	}

	return 0;
}

bool is_prime(int num){

	for (int i = 2; i <= int(sqrt(num)); i++){

		if (num % i == 0){
			return false;
		}
	}
	return true;

}
