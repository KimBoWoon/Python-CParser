
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

int primenumber[1000];
void getprimenumber();
bool isgoldbach(int testnum);

int main()
{
	std::fstream infile;
	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << "input file errr";
		exit(1);
	}
	int testcase;
	int testnum;
	infile >> testcase;
	getprimenumber();
	for (int i = 0; i < testcase; i++)
	{
		infile>>testnum;
		isgoldbach(testnum);
	}
}

void getprimenumber()
{
	int i = 2;
	for (int i = 0; i < 1000; i++)
	{
		primenumber[i] = i;
	}
	for (i = 2; i <= 999; i++) {
		if (primenumber[i] == 0) 
			continue;
		for (int j = i + i; j <= 999; j += i) { 
			primenumber[j] = 0;
		}
	}
}
bool isgoldbach(const int testnum)
{
	int minusnum;
	int checknum;
	int currentpri1, currentpri2;
	int currentminus = 1000;
	for (int i = 2; i < 1000; i++)
	{
		if (primenumber[i] != 0)
		{
			minusnum = primenumber[i];
			checknum = primenumber[testnum - minusnum];
			if (checknum != 0)
			{
				if (currentminus > abs(minusnum - checknum))
				{
					currentminus = abs(minusnum - checknum);
					currentpri1 = minusnum;
					currentpri2 = checknum;
				}
			}
		}
		if (primenumber[i]!=0 && primenumber[i] > testnum) break;
	}
	std::cout << currentpri1 << " " << currentpri2 << std::endl;
	return true;
}
