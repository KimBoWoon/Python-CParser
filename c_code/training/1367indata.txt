
#include <stdio.h>

#define _CRT_SECURE_NO_WARNINGS

int henry(int a, int b);
int fraction_S(int a, int b, int c, int d);
int fraction_M(int a, int b, int c, int d);
int gcd(int a, int b);

int main()
{
	FILE *fp;
	int i, j;
	int a, b, x;
	int no_cases;

	fp = fopen("input.txt", "r");
	if (fp == NULL) exit(1);

	fscanf(fp, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		x = 0;

		fscanf(fp,"%d", &a);
		fscanf(fp,"%d", &b);

		henry(a, b);
	}
}

int henry(int a, int b) {
	int x;

	x = b / a + 1;

	if (a == 1) 
		printf("%d\n", b);
	else
		henry(fraction_S(a, b, -1, x), fraction_M(a, b, -1, x));
}

int fraction_S(int a, int b, int c, int d)
{
	int resultS, resultM;

	resultS = a*d + c*b;
	resultM = b*d;

	return resultS/gcd(resultS,resultM);
}

int fraction_M(int a, int b, int c, int d)
{
	int resultS, resultM;

	resultS = a*d + c*b;
	resultM = b*d;

	return resultM / gcd(resultS, resultM);
}

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
