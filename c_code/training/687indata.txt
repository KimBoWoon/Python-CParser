
#include <iostream>
#include <fstream>

const int MAX_SIZE = 102;

using namespace std;

int fact(int n) {
	if (n == 0) return 1;

	return n*fact(n - 1);
}
int main(void) {
	ifstream in("input.txt");

	int t;

	in >> t;

	for (int testcase = 0; testcase < t; testcase++) {
		int n, result = 0;


		in >> n;

		result = fact(n);


		cout << result << endl;
	}

	return 0;
}
