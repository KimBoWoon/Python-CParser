

#include <iostream>
#include <fstream>
#include <cstring>

using namespace std;

int palindrome(char[], int, int);

int main()
{
	ifstream infile;
	int testCase;
	char str[1001];
	infile.open("input.txt");
	
	infile >> testCase;

	while (testCase--)
	{
		infile >> str;
		cout << palindrome(str, 0, strlen(str) - 1) << endl;
	}

	infile.close();

	return 0;
}

int palindrome(char s[], int first, int last)
{
	if(last <= first)
		return 1;
	else if(s[first] != s[last])
		return 0;
	else
		return palindrome(s, first+1, last-1); 
	
}
