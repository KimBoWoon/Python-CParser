#include<stdio.h>
#include<stdlib.h>
int mac(int n){
		if( n > 100)           
			return n-10;
		else if (n <= 100)          
			return mac(mac(n+11));
}
void main()
{
	FILE *in_file;
	int i, numcases , n;

	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d", &numcases);

	for(i = 0; i < numcases; i++){          
		for(n = 1; n<= 1000; n++){          
			fscanf(in_file, "%d", &n);
			break;
		}	
		printf("%d\n", mac(n) );          
	}
	fclose(in_file);
}
