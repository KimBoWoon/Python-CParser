#include<iostream>
#include<fstream>
using namespace std;

int main()
{
	ifstream inStream;
	int testcase;

	inStream.open ("input.txt");
	if (inStream.fail())
	{
		cerr << "input file open failed\n";
		return 0;
	}

	inStream >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int value ;
		bool isPrime = true;

		inStream >> value;

		for (int div = 2; div*div <= value; div++)
		{
			if (value%div == 0)
				isPrime = false;
		}

		cout << isPrime << endl;
	}
}
