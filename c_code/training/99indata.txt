#include <stdio.h>
#include <stdlib.h>

int cal(int a); 

int main()
{
	FILE *infile;

	int nocases;
	int num;
	int i;

	infile = fopen("input.txt","r");

	if(infile == NULL)
		exit(1);
	

	fscanf(infile,"%d",&nocases);

	for(i=0; i<nocases; i++)
	{
		fscanf(infile,"%d",&num);

		cal(num);
	}

	return 0;
}

int cal(int a)
{
	int num=1;
	int j;

	for(j=2; j*j <= a; j++)
	{
		if(a%j == 0)
		{
			num = 0;
		}
	}

	printf("%d\n",num);

	return 0;
}
