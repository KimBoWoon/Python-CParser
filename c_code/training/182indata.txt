
#include <iostream>
#include <fstream>
using namespace std;

int mac(int n)
{
	if(n>100)
		return n-10;
	else
		return mac(mac(n+11));
}

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase;i++)
	{
		int num;
		instream >> num;

		cout<<mac(num)<<endl;;
	}
}
