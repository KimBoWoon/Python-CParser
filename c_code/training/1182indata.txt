#include <iostream>
#include <fstream>
#include <vector>
using namespace std;

int eureka(const int num){
    vector<int> tri;

    int t = 0;
    for(int i = 1;;i++){
        t += i;
        
        
        if(num == t)        break;
        else if(num < t)    break;
        
        tri.push_back(t);
    }
    
    
    for(int i = 0; i < tri.size(); i++){
        for(int j = 0; j < tri.size(); j++){
            for(int k = 0; k < tri.size(); k++){
                if(num == tri.at(i) + tri.at(j) + tri.at(k))    return 1;
            }
        }
    }
    return 0;
}

int main(){
    fstream file;
    file.open("input.txt");
    
    int cases;
    
    file >> cases;

    
    while(cases--){
        int num;
        file >> num;
        
        cout << eureka(num) << endl;
    }
    
    return 1;
}
