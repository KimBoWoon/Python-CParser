

#include <iostream>
#include <fstream>
#include <string>
using namespace std;
int parenthesis(char paren[52], int start, int end);
int flag = 0;						
char paren[52];
int main()
{
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases;
	int end = 0;								
	inFile >> testCases;
	inFile.ignore();					

	for (int i = 0; i < testCases; i++){
		end = 0;
		flag = 0;
		inFile.getline(paren, 52);	

		for (int j = 0; j < 50; j++){
			if (paren[j] != 0) end++;
			else break;
		}
		if (parenthesis(paren, 0, end) == 0){
			cout << "YES" << endl;
		}
		else
			cout << "NO" << endl;
	}
	return 0;
}

int parenthesis(char paren[52], int start, int end){ 
	if (flag < 0){						
		return -1;
	}
	if (start == end){					
		return flag;
	}
	else{
		if (paren[start] == '('){		
		flag++;
	}
		else if (paren[start] == ')'){
		flag--;
	}
		return parenthesis(paren, start + 1, end);
	}
}
