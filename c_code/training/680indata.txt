#include <iostream>
#include <fstream>
using namespace std;

long fac(int num){
    if(num == 0 || num == 1)    return 1;
    
    return num * fac(num - 1);
}

int main(){
    ifstream file;
    file.open("input.txt");
    
    int cases;
    file >> cases;
    
    while(cases--){
        int num;
        file >> num;
        cout << fac(num) << endl;
    }
    return 1;
}
