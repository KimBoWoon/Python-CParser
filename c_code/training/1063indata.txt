#include <stdio.h>
#include <stdlib.h>
#define MAX 20 
int numcases;
int i;                        
int a, b, c;                 
int num[MAX];
void main(){
	FILE *in_file;

	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d", &numcases);

	for(c = 0; c<numcases; c++){
		fscanf(in_file, "%d", &i);        

		for(a=0; a<i; a++){
			fscanf(in_file, "%d", &num[a]);                 
		}
		num[i] = num[0];
		for(a=0; a<1000; a++){
			for(b=0; b<i; b++){
				num[b] = num[b] - num[b+1];
				if(num[b] < 0)
					num[b] = -num[b];
			}
			num[i] = num[0];
		}

		for(b=0; b<i; b++){
			num[i+1] += num[b];
		}

		if(num[i+1] ==0) 
			printf("ZERO\n");                 
		else
			printf("LOOP\n");
	}

	fclose(in_file);
}
