

#include <iostream>
#include <fstream>
using namespace std;
void fraction(int a, int b, int c, int d);
int euclid(int a, int b);
int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){}
	int TestCases;
	inFile >> TestCases;
	for (int i = 0; i < TestCases; i++){
		int a,b,c,d = 0;
		inFile >> a >> b >> c >> d;
		fraction(a,b,c,d);
	}
	return 0;
}
void fraction(int a, int b, int c, int d){
	int son,mom,greatest = 0;
	son = a*d + b*c;
	mom = b*d;
	greatest = euclid(son, mom);

	son /= greatest;
	mom /= greatest;

	cout << son << " " << mom << endl;

}
int euclid(int a, int b){
	int AmodB = 0;
	if (b == 0){
		return a;
	}
	else{
		AmodB = a%b;
		return euclid(b, AmodB);
	}
}
