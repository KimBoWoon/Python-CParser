void baseConversion(int n, int base){

	static char baseTable[] = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ#@";

	if(n >= base)
		baseConversion(n/base,base);
	
	
	number2[k] = baseTable[n%base];
	k++;
	
	
}

char checkPalindrome(char str[], int first, int last){
	
	if(last <= first)
		return 1;
	else if(str[first] != str[last])
		return 0;
	else return checkPalindrome(str, first+1, last-1);


}
