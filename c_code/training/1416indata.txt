

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void main(void) {
	FILE *inFile;
	int caseNum;
	int width;
	int height;
	int order;

	int i;
	int j;

	int floorNumber;
	int roomNumber;

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &caseNum);

	for (i = 0; i < caseNum; i++) {
		fscanf(inFile, "%d %d %d", &height, &width, &order);
		floorNumber = order % height; 

		if (floorNumber == 0) {
			floorNumber = height;
			roomNumber = (order / height);			
		}
		else {
			roomNumber = (order / height) + 1; 
		}
		printf("%d\n", floorNumber * 100 + roomNumber);		
	}	
}
