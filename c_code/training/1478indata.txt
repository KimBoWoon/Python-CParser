#include <stdio.h>
#include <stdlib.h>

void printXX(int h, int w, int n)
{
	int result = 0;
	result = (n-1)/h+1;
 	
	if(result/10<1) 
	{
		printf("0");
		printf("%d",result);
	}
	else 
	{
		printf("%d",result);
	}
}
void printYY(int h, int n)
{
	int result = n%h;
	
	if(result==0) printf("%d", h);
	else printf("%d", result);
}
int main()
{
	int dataCase = 0;
   	int i = 0;

	fscanf(file,"%d",&dataCase);
	for (; i<dataCase; i++)
	{
		int h = 0;
		int w = 0;
 		int n = 0;
		fscanf(file,"%d",&h);

                printYY(h,n);
