
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>

int gcd(int n1,int n2){ 
	if(n2==0) 
		return n1;
	else if(n1==0)
		return n2;

	else
		return gcd(n2,n1%n2);
}

int main(){
	FILE *fp;
	int i;
	int numcases;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);
	
	for(i=0;i<numcases;i++){
		int n1,n2;
		int result=0;
		int i;

		fscanf(fp,"%d %d",&n1,&n2);
		result=gcd(n1,n2);

		printf("%d\n",result);
		
	}
	fclose(fp);
}
