

#include <iostream>
#include <fstream>
using namespace std;

int gcd(int n1,int n2)
{
	if(n2==0)
		return n1;
	else
		gcd(n2,n1%n2);
}

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >>testcase;

	for(int i=0;i<testcase;i++)
	{
		int mom,son;
		instream >>son>> mom;
		int x;
		for(x=2; ;x++)
		{
			if(son*x > mom) 
			{
				if(son==1){   
					 cout<<mom<<endl;
					 break;}
				son = son*x-mom;
				mom=x*mom;
	
				int div=gcd(son,mom);
				
				if(div>1) 
				{
					mom/=div; son/=div;
				}
			}
		}
	}
}
