

#pragma warning(disable : 4996) 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_SZIE 51

int IsValidParenthesis(char* str);		

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;
	char strData[MAX_SZIE];

	inFile = fopen("input.txt", "r"); 

	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &numCases); 
	for (i = 0; i<numCases; i++)
	{
		fscanf(inFile, "%s", &strData); 

		if (IsValidParenthesis(strData)) puts("YES");
		else puts("NO");
	}

	fclose(inFile); 
}

int IsValidParenthesis(char* str)
{
	int i, j, cnt;
	char buffer[MAX_SZIE];

	if (strlen(str) == 0) return 1;		
	else
	{
		for (i = 0; i < strlen(str) - 1; i++)
			if (str[i] == '(' && str[i + 1] == ')')
			{
				for (j = 0, cnt = 0; j < strlen(str); j++)
					if (j != i && j != (i + 1)) buffer[cnt++] = str[j];		
				buffer[cnt] = '\0';
				return IsValidParenthesis(buffer);		
			}
		return 0;
	}
}
