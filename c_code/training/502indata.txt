#include<stdio.h>
#include<stdlib.h>

int gcd(int a, int b)
{
    if (b == 0) 
        return a;
    else 
        return gcd(b, a%b);
}

void main(void)
{
    FILE *in_file;
    int i;
    int no_cases;
    
    in_file = fopen("input.txt","r");
    if(in_file == NULL) exit(1);
    
    fscanf(in_file,"%d",&no_cases);
    
    for(i=0; i<no_cases; i++)
    {
        int a,b;
        
        fscanf(in_file, "%d %d",&a,&b);
        gcd(a,b);
        
        printf("%d \n",gcd(a,b));
    }
    fclose(in_file);
}
