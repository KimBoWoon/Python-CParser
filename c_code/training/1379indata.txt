

#include <stdio.h>

int main(void)
{
	FILE *f = fopen("input.txt", "r");

	}

	int testCase = 0; 
	int numA = 0;

	int answer = 0; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d %d", &numA, &numB); 

		printf("%d\n", henry(numA, numB));
	}

	fclose(f);
	return 0;
}

int henry(int a, int b) {
	int n = 1;
	while ((a * n - b) != 0){
		if (a * n > b){
			a = a * n - b;
			b = b * n;
		}
		n++;
	}
	return n;
}
