

#include <stdio.h>
#include <stdlib.h>
void main(void)
{
	FILE *inFile;
	int circle, H,W,N,i; 

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d %d %d", &H, &W, &N);
		
			i = (N%H == 0) ? (N / H) : (N / H + 1);
			printf("%d", (N%H == 0) ? (H) : (N%H));
			if (i<10) printf("0");
			printf("%d\n",i);
	}
	fclose(inFile);
}
