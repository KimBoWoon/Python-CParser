#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int main() {

	ifstream instream("input.txt");
	int numtestcases, num, count = 0;

	instream>>numtestcases;

	for(int i=0; i<numtestcases;i++){
		instream>>num;

		int *arr = new int[num];
		for(int i=0;i<num;i++)
			instream >> arr[i];

		while(1) {
			int flag=0;
			int temp = 0;
			for(int i = 0 ; i < num ; i++){
				if(i!=num-1){
					if(i==0)
						temp = abs(arr[i] - arr[i+1]);
					else
						arr[i] = abs(arr[i] - arr[i+1]);
				}
				else
					arr[i] = abs(arr[0] - arr[i]);
			}
			arr[0] = temp;

			count++;
			if(count == 1000){
				cout << "LOOP" << endl;
				break;
			}
			for(int i = 0 ; i < num ; i++){
				if(arr[i]!=0){
					flag++;
					break;
				}
			}
			if(flag == 0){
				cout << "ZERO" << endl;
				break;
			}
			else
				continue;
		}
		count =0;
	}
	instream.close();
}
