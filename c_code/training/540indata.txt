

#include <stdio.h>
#include <stdlib.h>

int getFactorial(int num);
void permute(char *str, int length, int num, int count);
void print(char *str, int count);
void freeChar(char *str);

void main() {
	char temp[13] = { 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', '\0' };
	int numTestCase;
	int i, j;
	int length, num;
	FILE *f;
	char *str = NULL;

	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCase);

	for (i = 0; i < numTestCase; i++) {
		fscanf(f, "%d %d", &length, &num);
		str = (char *)malloc(length + 1);
		for (j = 0; j < length; j++)	
			*(str + j) = temp[j];
		*(str + j) = '\0';
		permute(str, --length, num, 0);
		freeChar(str);	
	}
	
}

int getFactorial(int num) {	
	if (num == 0)
		return 1;
	else {
		return num * getFactorial(num - 1);
	}
}

void permute(char *str, int length, int num, int count) {
	int check = getFactorial(length);

	if (num - check > 0) {
		num -= check;
		count++;
	}
	else {	
		print(str, count);
		if (length<= 0) {
			printf("\n");
			return;
		}
		count = 0;
		length--;
	}
	permute(str, length, num, count);
}

void print(char *str, int count) {	
	int i = 0;
	while (1) {
		if (*(str + i) == '0') {
			i++;
			count++;
			continue;
		}
			
		if (i == count) {
			printf("%c", *(str + i));
			*(str + i) = '0';
			break;
		}
		i++;
	}
}

void freeChar(char *str) {
	if (str != NULL)
		free(str);
}
