#include <iostream>
#include <fstream>
#include <math.h>

bool is_prime(int num);
void goldbach(int left_side, int right_side);

int main(void){

	std::ifstream inFile("input.txt");

	int testCases = 0;
	inFile >> testCases;

	int number = 0;
	for (int i = 0; i < testCases; i++){

		inFile >> number;
		goldbach(number / 2, number / 2);

	}

	return 0;
}

void goldbach(int left_side, int right_side){

	if (is_prime(left_side) && is_prime(right_side)){
		std::cout << left_side << " " << right_side << std::endl;
	}
	else{
		goldbach(left_side - 1, right_side + 1);
	}

}

bool is_prime(int num){

	for (int i = 2; i <= int(sqrt(num)); i++){
		if (num % i == 0){
			return false;
		}
	}
	return true;
}
