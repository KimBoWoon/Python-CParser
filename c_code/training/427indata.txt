

#include<stdio.h>
#include<stdlib.h>

int gcd(int a, int b)
{
	if(b==0)	
		return a;
	else
		gcd(b,a%b);	
}
void main(void){
	FILE *in_file;
	int no_case;
	int a,b;
	int i;

	in_file = fopen("input.txt","r");	
	if(in_file == NULL) exit(1);

	fscanf(in_file , "%d" , &no_case);

	for(i=0; i<no_case; i++)
	{
		fscanf(in_file, "%d %d",&a,&b);
		
		printf("%d\n",gcd(a,b));	
	}

	fclose(in_file);
}
