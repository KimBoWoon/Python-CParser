#include <iostream>
#include <fstream>

using namespace std;

int fn(int n, int m, int cnt, int max){
	if (cnt >= max)
		return n;
	else
		return fn(m, n+m, cnt+1, max);
}

int main(){
	ifstream fin("input.txt");

	int nCase;
	fin >> nCase;

	for (int ci = 0; ci < nCase; ++ci)
	{
		int n;
		fin >> n;
		cout << fn(0, 1, 0, n) << endl;
	}
}
