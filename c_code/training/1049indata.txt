
#include<iostream>
#include<fstream>
#define MAX 1000

using namespace std;

int abs(int n)
{
	return  n < 0 ? -n : n;
}

int isZero(int arr[],int n){
	for (int i = 0; i < n; i++) if (arr[i] != 0) return 1;
	return 0;
}

int isLoop(int arr[][16], int n, int cnt){
	for (int i = 0; i < cnt; i++){
		int check = 0;
		for (int j = 0; j < n; j++){
			if (arr[i][j] != arr[cnt][j]) check = 1;
		}
		if (!check) return 0;
	}
	return 1;
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 0;

	int test_case;
	in >> test_case;
	for (int i = 0; i < test_case; i++){
		int n, arr[MAX][16], cnt = 0, zero;
		in >> n;
		for (int j = 0; j < n; j++) in >> arr[0][j];
		while ((zero = isZero(arr[cnt], n)) && isLoop(arr, n, cnt++)){
			for (int j = 0; j < n; j++){
				if (j != n - 1) arr[cnt][j] = abs(arr[cnt - 1][j] - arr[cnt - 1][j + 1]);
				else arr[cnt][j] = abs(arr[cnt - 1][j] - arr[cnt - 1][0]);
			}
		}
		if (!zero) cout << "ZERO" << endl;
		else cout << "LOOP" << endl;
	}

	in.close();
	return 0;
}
