
#include <stdio.h>
#include <stdlib.h>

int factorial(int n){
	return (n > 1) ?  n* factorial (n-1): 1;
}

void  permutation(char *str,int n, int k){
	int i;
	if(n<=0) return;
	int fct = factorial(n-1);
	printf("%c",str[k/fct]);
	for ( i=k/fct;i<n;i++) str[i]=str[i+1];
	permutation(str, n-1, k%fct); 
}

void main(){
	
	FILE *in_file;
	int no_cases;

	in_file = fopen("input.txt", "r");

	if(in_file == NULL) exit(1);

	fscanf(in_file,"%d", &no_cases);

	while(no_cases--){
		int n, k,i;
		fscanf(in_file,"%d %d", &n, &k);
		char *str = (char*)malloc(sizeof(char)*n);
		for(i=0;i<n;i++) str[i]=97+i;
		permutation(str,n,k-1);
		printf("\n");
	}
	fclose(in_file);
}
