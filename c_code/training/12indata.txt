
#include<iostream>
#include<cmath>
#include<fstream>

using namespace std;

int main(){
	ifstream f; 
	f.open("input.txt"); 
	int num = 0; 
	int j = 0; 

	f >> num;
	for (int i = 0;i<num;i++){
		int primeNum = 0;
		f >> primeNum;
		for (j = 2;j<=(int)sqrt(primeNum);j++){ 
			if (primeNum%j == 0){
				cout << 0 << endl;
				break;
			}
		}
		if (j == (int)sqrt(primeNum)+1){
			cout << 1 << endl;
		}
	}
	return 0;
}
