#include <stdio.h>
#include <stdlib.h>
#define MAXSIZE 9
#define MARK 1
#define UNMARK 0

typedef struct Point {int x,y;} point;
point direction[8] = {{1,-2},{2,-1},{2,1},{1,2},{-1,2},{-2,1},{-2,-1},{-1,-2}};

int board[MAXSIZE][MAXSIZE], path[MAXSIZE][MAXSIZE]; 
int KNightTour(int Width, int Height, point pos, int count){
	int i;
	point next;
	if(count == Width * Height)
		return 1;
	for(i=0; i<MAXSIZE; i++){
		next.x = pos.x +direction[i].x;
		next.y = pos.y +direction[i].y;

		if(next.x > 0 && next.x <=Height && next.y > 0 && next.y <= Width && board[next.y][next.x] !=MARK){
			board[next.y][next.x] = MARK;
			path[next.y][next.x] = count+1;
			
			if(KNightTour(Width, Height, next, count+1))
				return 1;
			board[next.y][next.x] = UNMARK;
		}
	}
	return 0;
}
void PrintTour(int Width, int Height){ 
	int i,j;
	for(i=1; i<=Width; i++){
		for(j=1; j<=Height; j++){
			printf("%d ",path[i][j]);
		}
		printf("\n");
	}
}
	
int SetBoard(FILE *file){ 
	int i,j, Width, Height;
	point start;
	fscanf(file,"%d %d", &Width ,&Height);
	fscanf(file,"%d %d", &start.y ,&start.x);
	for(i=1; i<=Width; i++){
		for(j=1; j<=Height; j++){
			board[i][j] = UNMARK;
		}
	}
	board[start.y][start.x] = MARK;
	path[start.y][start.x] = 1;
	if(KNightTour(Width,Height,start,1)){
		printf("1\n");
		PrintTour(Width, Height);
	}
	else{
		printf("0\n");
	}
}
void main(void)
{
	FILE *file;
	int i,j,testcase;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &testcase);
	for(i=0; i<testcase; i++)
	{
		SetBoard(file); 
	}
	fclose(file);
}
