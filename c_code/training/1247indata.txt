
#include<iostream>
#include<fstream>
#define MAX 1000

using namespace std;

int abs(int n)
{
	return  n < 0 ? -n : n;
}

int isPrime(int n){
	int flag = 1;
	for (int i = 2; i <= n / 2; i++){
		if (n%i == 0){
			flag = 0;
			break;
		}
	}
	return flag;
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 0;

	int test_case, arr[MAX], cnt = 0;
	for (int i = 2; i < MAX; i++) if (isPrime(i)) arr[cnt++] = i; 

	in >> test_case;
	while (test_case--){
		int n, min1, min2, sum = MAX;
		in >> n;
		for (int i = 0; i < cnt; i++){
			for (int j = i; j < cnt; j++){
				if (arr[i]>n || arr[j]>n) break;
				else if (arr[i] + arr[j] == n && abs(arr[i]-arr[j]) < sum){
					sum = abs(arr[i] - arr[j]);
					min1 = arr[i];
					min2 = arr[j];
				}
			}
		}
		cout << min1 << " " << min2 << endl;
	}

	in.close();
	return 0;
}
