
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a,int b)
{
	if (b == 0) return a;
	if (b > 0) return gcd(b, a % b);
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while(t--)
	{
		int a,b;
		in >> a>>b;
		cout<<gcd(a,b)<<endl;
	}
	in.close();
}
