

#include <iostream>
#include <fstream>
using namespace std;
int main()
{
	ifstream instream("input.txt");
	int numtestcases, numdata, check = 0;
	instream >> numtestcases;
	for (int i = 0; i < numtestcases; i++){
		instream >> numdata;
		for (int j = 2; j < numdata; j++){
			if (numdata % j == 0){
				check++;
			}
			if (check >= 1) break;
		}
		if (check == 0) cout << "1" << endl;
		else cout << "0" << endl;
		check = 0;
	}
	instream.close();
}
