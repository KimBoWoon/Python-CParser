


#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

void makeCheckArray(int* numArray, int* check);
int main()
{
	ifstream inStream;
	int numTest, num;
	inStream.open("input.txt");
	if (inStream.fail()) exit(1);

	inStream >> numTest;

	for (int i = 0; i < numTest; i++)
	{
		int numArray[50];
		int check[1001];
		makeCheckArray(numArray, check);
		inStream >> num;
		if (check[num] == 1) num = 1;
		else num = 0;
		cout << num << endl;

	}
	inStream.close();
}

void makeCheckArray(int* numArray, int* check){
	
	for (int i = 1; i < 50; i++)
	{
		numArray[i] = (i * (i + 1)) / 2;
	} 

	for (int i = 1; i < 50; i++)
	{
		for (int j = i; j < 50; j++)
		{
			for (int k = j; k < 50; k++)
			{
				if (numArray[i] + numArray[j] + numArray[k] <= 1000)
				{
					check[numArray[i] + numArray[j] + numArray[k]] = 1;
				}
			}
		}
	}

}
