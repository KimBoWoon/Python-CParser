

#include<stdio.h>
#include<stdlib.h>

void main(void)
{
	FILE *in_file;
	int i,j,k,l;
	int no_case;
	int data;
	int t1,t2,t3;
	int num;

	in_file = fopen("input.txt" , "r");		
	if(in_file==NULL)
		exit(1);

	fscanf(in_file, "%d" , &no_case);

	for(i=0; i<no_case; i++)
	{
		num=0;
		fscanf(in_file, "%d", &data);

		for(j=1; j<=44; j++){		
			t1=(j*(j+1))/2;			
			if(t1>=data || num==1)	
				break;
			for(k=1; k<=44; k++){
				t2=(k*(k+1))/2;		
				if(t2>=data || num==1)	
					break;
				for(l=1; l<=44; l++){
					t3=(l*(l+1))/2;
					if(t1+t2+t3==data)	
					{
						num=1;
						break;
					}
					if(t3>=data)	
						break;
				}
			}
		}
		printf("%d\n",num);
	}
	fclose(in_file);
}
