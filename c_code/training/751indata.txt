#include<cmath>
#include<fstream>
#include<cstring>
using namespace std;

int Palindrome(char text[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (text[first] != text[last])
		return 0;
	else 
		return Palindrome(text, first + 1, last - 1);

}


int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	char text[1024];
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> text;
		cout <<Palindrome(text, 0, strlen(text) - 1)<< endl;
	}
	return 0;
}
