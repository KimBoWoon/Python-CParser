#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <functional>

using namespace std;

int c;
int minNum=200000000;
int arr[1000000];

int main(){
    	FILE* file;
       int cases;

    	file = fopen("input.txt", "r");
 
        fscanf(file, "%d", &cases);

        while(cases--){
				c=0;
				minNum = 200000000;
              int num, k, cp;
              fscanf(file, "%d%d", &num, &k);
		
        		for(int i=0; i<num; i++){
          			fscanf(file, "%d", &arr[i]);
				}

				sort(arr, arr + num, less<int>());

				int i,j;
				i=0,j=num-1;

				while( i<j ){
					cp = k - (arr[i] + arr[j]);
					if( cp >= 0 ) i++;
					else j--;

					cp=abs((double)cp);
					if(cp == minNum){ c++; }
					else if(cp < minNum){ minNum = cp; c = 1; }
				}

        		printf("%d\n", c);
        	}
       return 0;
}
