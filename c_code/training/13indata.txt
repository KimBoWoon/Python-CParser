

#include <stdio.h>
#include <stdlib.h>
#include <math.h>
void main(void)
{

	FILE *in_file;
	int no_cases;
	int i, j, data, isPrime;
	in_file = fopen("input.txt","r");

	fscanf(in_file,"%d",&no_cases);
	if(in_file == NULL) exit(1);
	
	for(i=0; i<no_cases; i++){
		isPrime = 1;
		fscanf (in_file,"%d",&data);
		for(j=2; j<=(int)sqrt(data); j++) { 
			if (data%j ==0){
				isPrime=0;
				break;
			}
		}
		printf("%d\n",isPrime);
	}
}
