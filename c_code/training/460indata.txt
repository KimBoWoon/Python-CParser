
#include <cstdio>
#include <cstdlib>

int GCD(int number1, int number2){
	return (!number2) ? (number1) : GCD(number2, number1%number2);
}
int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}

	int testCases = 0,
		number1 = 0,
		number2 = 0;
	
	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		fscanf(readStream, "%d %d", &number1, &number2);
		printf("%d\n", GCD(number1, number2));
	}
	fclose(readStream);
}
