
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

char str[1000];
int checkPalindrome(char str[], int first,int last)
{
	if(last<=first) return 1;
	else if(str[first]!=str[last]) return 0;
	else return checkPalindrome(str,first+1,last-1);
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in>>t;
	while(t--)
	{
		in>>str;
		cout<<checkPalindrome(str,0,strlen(str)-1)<<"\n";
	}
	
}
