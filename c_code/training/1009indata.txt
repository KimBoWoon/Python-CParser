int check(int length,int c[])
{
	int i,j;
	int newc[15];
	for(i=0; i<length; i++)
	{
		if(i!=length-1) {
			newc[i]=abs(c[i]-c[i+1]);}
		else {newc[i]=abs(c[i]-c[0]);}
	}
	for(i=0; i<length; i++)
	{
		c[i] = newc[i];
	}
	for(i=0; i<length; i++)
	{
		if(c[i] != 0) return 0;
	}
	return 1;
}
