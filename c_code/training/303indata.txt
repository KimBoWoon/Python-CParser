#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b){
    if(b == 0)  return a;
    return gcd(b, a % b);
}

int main(){
    ifstream file;
    file.open("input.txt");
    int cases;
    file >> cases;
    while(cases--){
        int first_c, first_m, second_c, second_m;
        file >> first_c >> first_m >> second_c >> second_m;
        int last_c, last_m;
        last_c = first_c * second_m + second_c * first_m;
        last_m = first_m * second_m;
        int Gcd = gcd(last_c, last_m);
        cout << last_c / Gcd << " " << last_m / Gcd << endl;
    }
}
