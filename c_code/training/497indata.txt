#include <iostream>
#include <fstream>
using namespace std;

int uclead(int a, int b){
    if(b == 0)  return a;
    
    return uclead(b, a % b);
}

int main(){
    ifstream file;
    file.open("input.txt");
    
    int cases;
    file >> cases;
    
    while(cases--){
        int a, b;
        file >> a >> b;
        cout << uclead(a, b) << endl;
    }
    return 0;
}
