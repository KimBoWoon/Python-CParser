#include<iostream>
#include<fstream>
using namespace std;
int a, b, c, d;
int calGcd(int a, int b) {
	if (b == 0) return a;
	else calGcd(b, a%b);
}
void addFoun(int gcd) {
	int val1 = d / gcd;	
	int val2 = b / gcd;
	int numerator = val1*a + val2*c;
	int denominator = val1*b;
	int gcd1 = calGcd(denominator, numerator);
	cout << numerator / gcd1 << " " << denominator / gcd1 << endl;
}
int main() {
	ifstream fin;
	fin.open("input.txt");
	int num,gcd;
	fin >> num;
	for (int cnt = 0; cnt < num; cnt++) {
		fin >> a >> b >> c >> d;
		gcd = calGcd(b, d);
		addFoun(gcd);
	}
}
