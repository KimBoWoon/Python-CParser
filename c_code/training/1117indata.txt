#include<iostream>
#include<fstream>
using namespace std;
int Theorem(int num);
int main(){
	ifstream fin;
	fin.open("input.txt");
	if (!fin) {
		cout << "file error" << endl;
		return 0;
	}
	int num,val,test[45],result[1001];
	for (int cnt = 1; cnt < 45; cnt++){		
		test[cnt] = cnt*(cnt + 1) / 2;
	}
	for (int cnt = 0; cnt < 1001; cnt++){
		result[cnt]=0;
	}
	for (int i = 1; i < 45; i++){
		for (int j = i; j < 45; j++){
			for (int k = j; k < 45; k++){
				if (test[i] + test[j] + test[k] < 1001) {
					result[test[i] + test[j] + test[k]] = 1;	
				}
			}
		}
	}
	fin >> num;
	for (int cnt = 0; cnt < num; cnt++){
		fin >> val;
		cout << result[val] << endl;
	}
}
