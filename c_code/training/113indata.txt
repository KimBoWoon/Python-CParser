


#include <stdio.h>
#include <stdlib.h>

int mac(int n)
{
	if( n>100 )
		return n-10;
	else
		return mac(mac(n+11));
}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int N = 0;														

		fscanf(inFile, "%d", &N);										

		printf("%d\n", mac(N));
	}

	fclose(inFile);														
}
