

#include <iostream>
#include <fstream>
#include <cstdlib>

int proveMinority(int testnum);

int main()
{
	std::fstream inputfile; 
	int testcase; 
	int testnum; 

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> testnum;
		std::cout << proveMinority(testnum) << std::endl;
	}

	return 0;
}
int proveMinority(int testnum)
{
	for (int i = 2; i < testnum; i++)
	{
		if (testnum % i == 0)
			return 0;
	}
	
	return 1;
}
