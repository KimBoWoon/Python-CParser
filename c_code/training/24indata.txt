

#include <stdio.h>
#include <stdlib.h>
void main(void)
{
	FILE *inFile;
	int circle, num, i; 

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d", &num);
		if (num == 2 || num == 3) { 
			printf("1\n"); continue;
		}
		for (i = 2; i <= num / 2; i++){ 
			if (num%i == 0) { 
				printf("0\n"); break; 
			}
			if (i == num / 2)
				printf("1\n"); 
		}
	}
	fclose(inFile);
}
