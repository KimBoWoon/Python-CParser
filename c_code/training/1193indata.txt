

#include <stdio.h>
#define criterionSquare(a,b) (b*b <= a ? 1:0) 
int checkPrime(int n)
{
	int b = 2;
	while (criterionSquare(n, b)){
		if (!(n - (n / b)*b)){ 
			return 0;
		}b++;
	}
	if (!criterionSquare(n, b)) return 1;
}

int main()
{
	unsigned int a, t, p, q;
	FILE *fp;
	fp = fopen("input.txt", "r"); 
	fscanf(fp, "%d", &t); 
	while (t--){
		fscanf(fp, "%d", &a);

		if (a & 1)
		{
			p = a / 2; 
			q = (a / 2) + 1; 
			while (1)
			{
				if (2 == checkPrime(p) + checkPrime(q)){ 
					if (p < q)
					{
						printf("%d %d\n", p, q); 
						break;
					}
					else
					{
						printf("%d %d \n", q, p);
						break;
					}
				}
				else
				{
					p++; q--;
					continue;
				}
			}
		}
		else 
		{
			p = a / 2; 
			q = a / 2; 
			while (1)
			{
				if (2 == checkPrime(p) + checkPrime(q)){ 
					if (p < q)
					{
						printf("%d %d\n", p, q); 
						break;
					}
					else
					{
						printf("%d %d \n", q, p);
						break;
					}
				}
				else
				{
					p++; q--;
					continue;
				}
			}
		}
	}
	fclose(fp);
	return 1;
}
