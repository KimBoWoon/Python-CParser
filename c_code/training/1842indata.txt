
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;
int parent(int i,char *str, int *result)
{
	int close = 0;
	for(int j=i+1; i<strlen(str); j++)
	{
		if(str[j] == '(')
		{
			j = parent(j,str,result);
		}
		else if(str[j] ==')')
		{
			close = 1;
			return j;
		}
	}
	*result = close;
}
int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일 열수 없음" << endl;
		return 0;
	}
	int testcase;
	instream >> testcase;
	for(int z=0; z<testcase; z++)
	{
		char str[51];
		instream >> str;
		int result = 1;
		if(strlen(str) % 2 != 0 || str[0] == ')' || str[strlen(str)-1] == '(')
			result = 0;
		else
		{
			for(int i=0; i<strlen(str); i++)
			{
				if(str[i] == '(')
					i=parent(i, str, &result);
				else if(str[i] == ')')
				{
					result = 0;
					break;
				}

			}
		}
		if(result == 0)
			cout << "NO" << endl;
		else if(result == 1)
			cout << "YES" << endl;

	}
}
