#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){
    if( b== 0 ) return a;
    else{
        return (a>b)? gcd(b, a%b) : gcd(a, b%a);
    }
}

int main(void){
    FILE* file;
    file = fopen("input.txt","r");
    
    int cases;
    fscanf(file, "%d", &cases);
    while(cases--){
        int a,b;
        fscanf(file, "%d %d", &a, &b);
        printf("%d\n", (a>b)? gcd(a, b) : gcd(b, a));
    }
    return 0;
}
