
#include <stdio.h>
#include <stdlib.h>

void value(FILE *infile);
int prime(int a);

int main()
{
	FILE *infile;
	int numcases;
	int t;

	infile =fopen("input.txt","r");
	if(infile ==NULL) exit(1);

	fscanf(infile, "%d", &numcases);

	for (t = 0; t < numcases; t++)
	{
		value(infile);
	}
    return 0;
}

void value(FILE *infile)
{
    int a;
    int n1,n2;
    int s1,s2;

	fscanf(infile, "%d", &a);
   
    n1 = a/2;
    n2 = a-n1;
    
    while(1)
    {
       s1 = prime(n1);
       s2 = prime(n2);

       if(s1 == 1 && s2 == 1)
       {
		   printf("%d %d\n", n1, n2);
           break;
       }

       else
       {
          n1 -= 1;
          n2 += 1;
       }
    } 
}

int prime(int a)
{
	int i;
    for(i =2; i*i <= a; i++)
    {
       if(a%i == 0)
       {
          return 0;
       }
    }
    return 1;
}
