

#include <stdio.h>
int GCD(unsigned int a,unsigned int b) 
{
	unsigned int temp;
	if (a < b) 
	{
		temp = a;
		a = b;
		b = temp;
	}
	if (!b) return a; 
	return GCD(b, a%b); 
}

int main()
{
	unsigned int t, a, b, c, d, lcm;
	FILE *fp;
	fp = fopen("input.txt", "r");
	if (fp  == NULL) return 0; 
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%d %d %d %d", &a, &b, &c, &d); 
		lcm = GCD(b, d) *(b / GCD(b, d) * (d / GCD(b, d))); 
		printf("%d %d\n", (a*(lcm / b) + c*(lcm / d)) / GCD((a*(lcm / b) + c*(lcm / d)), lcm), lcm / GCD((a*(lcm / b) + c*(lcm / d)), lcm)); 
	}
	fclose(fp); 
	return 1;
}
