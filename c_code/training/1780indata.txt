

#include <stdio.h>
#define MAX_SIZE 50
int checkParenthesis(char* str, int size) 
{
	int i, count = 0;
	for (i = 0; i < size; i++)
	{
		if (str[i] == '(') count++; 
		else if (str[i] == ')')
		{
			if (count < 0) count = 0;
			count--; 
		}
	}
	return count == 0 ? 1 : 0; 
}
int main()
{
	int t, i;
	char parenthesis[MAX_SIZE];
	FILE *fp;
	fp = fopen("input.txt", "r"); 
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%s", parenthesis); 
		i = 0;
		while ((parenthesis[i] == '(') || (parenthesis[i] == ')')) i++;
		if (checkParenthesis(parenthesis, i)) printf("YES\n"); 
		else printf("NO\n");
	}
	fclose(fp); 
	return 1;
}
