


#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>

using namespace std;
int parenthesisCheck(char arr[], int cnt, int i);		

int main()
{
	ifstream inStream;
	int numTestCases;		
	char pArr[51] = { NULL };

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> pArr;
		if (parenthesisCheck(pArr, 0, 0) == 0)
		{
			cout << "YES" << endl;
		}
		else
		{
			cout << "NO" << endl;
		}
	}

	inStream.close();
	return 0;
}

int parenthesisCheck(char arr[], int cnt, int i)
{
	if (arr[0] == ')')
	{
		return 0;
	}

	if (arr[i] == '(')
	{
		cnt++;
	}
	else
	{
		cnt--;
		if (cnt < 0)
		{
			return 1;
		}
	}
	i++;

	if (i == strlen(arr))
	{
		return cnt;
	}
	else
	{
		parenthesisCheck(arr, cnt, i);
	}
}
