




#include <iostream>
#include <fstream>

int factorial(int testCase);

int main() {
	std::ifstream inFile;
	inFile.open("input.txt");

	int testCaseNum;

	inFile >> testCaseNum;

	for (int i = 0; i < testCaseNum; i++) {
		int testCase = 0;

		inFile >> testCase;

		std::cout<<factorial(testCase)<<std::endl;
	}
}

int factorial(int testCase) {
	if (testCase == 0)
		return 1;
	else
		return testCase * factorial(testCase - 1);
}
