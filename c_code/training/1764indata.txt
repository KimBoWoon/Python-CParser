

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream infile;
	int testcases;
	int num;
	int arr[9999];

	infile.open("input.txt");

	infile >> testcases;
	for (int i = 0; i < testcases; i++)
	{
		infile >> num;
		int flag = 0;

		for (int div = 2; div <= 64; div++)
		{
			int tmpNum = num;
			int index = 0;
			int k = 0, l = 0;

			while (tmpNum != 0)
			{
				arr[index++] = tmpNum % div;
				tmpNum /= div;
			}

			for (k, l = index - 1; k <= l; k++, l--)
			{
				if (arr[k] == arr[l])
					continue;
				else
					break;
			}

			if (k > l)
			{
				flag = 0;
				break;
			}
			else
				flag = 1;

		}

		if (flag == 0)
			cout << 1 << endl;
		else
			cout << 0 << endl;
	}

	return 0;
}
