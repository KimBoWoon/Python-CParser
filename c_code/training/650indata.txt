

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int factorial(int a);		

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error"<<endl;
		exit(1);
	}

	int no_case;
	instream>>no_case;

	for(int i=0; i<no_case; i++)		
	{
		int fac;
		instream>>fac;

		int res = factorial(fac);

		cout<<res<<endl;
	}
}
int factorial(int a)			
{
	if(a <= 1)
		return 1;
	else
		return a * factorial(a-1);
}
