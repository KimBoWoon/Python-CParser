	int numOfStr;

void swap(char str[],int n1,int n2){
	char temp = str[n2];
	str[n2] = str[n1];
	str[n1] = temp;
}
int fact(int n){
	int fac = 1;
	for (int i = 1; i <= n; i++){
		fac *= i;
	}
	return fac;
}
void ord(char str[], int begin, int end){
	for (int i = begin; i < end; i++){
		for (int j = begin; j < i; j++){
			if (str[j]>str[i]){
				swap(str, j, i);
			}
		}
	}
}
