




#include <iostream>
#include <string>
#include <fstream>

int factorial(int n);

int main() {
	std::ifstream inFile;
	int testCaseNum;

	inFile.open("input.txt");

	inFile >> testCaseNum;

	for (int i = 0; i < testCaseNum; i++) {
		int size = 0, count = 0, f = 0;
		std::string str = "abcdefghijkl";
		std::string result = "";
		
		inFile >> size >> count;
		
		for (int j = factorial(--size), k = count - 1; size >= 1; size--) {
			int select = k / j;
			k %= j;

			result += str[select];
			str.erase(select, 1);
			j /= size;
		}
		std::cout << result << str[0] << std::endl;
	}
}

int factorial(int n) {
	if (n == 0)
		return 1;
	else
		return n * factorial(n - 1);
}
