

#include <iostream>
#include <fstream>
using namespace std;
void ACMHotel(int H, int W, int N);

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases = 0;
	int H = 0, W=0, N=0;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++){
		inFile >> H >> W >> N;
		ACMHotel(H, W, N);
	}
	return 0;
}
void ACMHotel(int H, int W, int N){
	int HCount, WCount;
	HCount = N %  H; 
	if (N > H && HCount == 0){
		HCount = H;
		WCount = N / H;
	}
	else if (N > H){
		WCount = N / H;
		WCount++;
	}
	else if (N < H){
		HCount = N % H;
		WCount = 1;
	}
	else if (H == N){
		HCount = H;
		WCount = 1;
	}
	if (WCount < 10){
		cout << HCount << 0 << WCount << endl;;
	}
	else{
		cout << HCount << WCount << endl;
	}
}
