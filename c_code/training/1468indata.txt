

#include <stdio.h>

int main() {
	int i, mul;
	int no_cases, H, W, N;
	FILE *fp;

	if( (fp = fopen("input.txt", "rt")) == NULL) {
		return -1;
	}

	fscanf(fp ,"%d", &no_cases);
	for(i=0; i<no_cases; i++) {
		fscanf(fp, "%d %d %d", &H, &W, &N);
		printf("%d\n", ((N-1)%H+1)*100+(N-1)/H+1); 
	}
	fclose(fp);
	return 0;
}
