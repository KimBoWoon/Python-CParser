

#include <iostream>
#include <fstream>

using namespace std;

class Henry    
{   
    long long up,low;
public:
    Henry(int up,int low){ this->up=up; this->low=low;}
    long last_deno();   
};


long Henry::last_deno()
{

    while(up>1)  
     {           
        if(low%up==0)
        return low/up;

        long long i= low/up +1;
        up=up*i-low; 
        low=low*i; 
     }

    return low;

}

int main()
{
    
    ifstream file;
    
    file.open("input.txt");
    
    if(file.is_open()==false) 
        cout<<"입력파일이 없습니다."<<endl;
    
    else
    {
        int case_num;
        file>>case_num;
        
        for(int i=0; i<case_num; i++)
        {
            int up,low;
            file>>up>>low;
            Henry henry(up,low);
            cout<<henry.last_deno()<<endl;
            
        }
        
    }
    
    file.close();
    
}
