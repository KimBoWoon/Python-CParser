
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int recursive(char str[], int first, int last)
{
	if (last <= first)
	{
		return 1;
	}
	else if (str[first] != str[last])
	{
		return 0;
	}
	else
		return recursive(str, first + 1, last - 1);
}

int main()
{
	FILE *in_file;
	int i;
	int numTestCases;
	char word[2000];

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++)
	{
		fscanf(in_file, "%s", word);
		printf("%d\n", recursive(word, 0, strlen(word)-1));
	}
	fclose(in_file);
}
