#include <stdio.h> 
#include <stdlib.h>
#include <math.h>

void ducciTest(int max, int arr[], int arr2[]);

int main()
{

	FILE* inFile; 

	int numCases;
	int noData;
	int ducci[16] = { 0 };
	int ducciTmp[16] = { 0 };

	inFile = fopen("input.txt", "r");

	if (inFile == NULL) 
	{
		exit(1);
	}


	fscanf(inFile, "%d", &numCases); 
	while (numCases--) 
	{
		fscanf(inFile, "%d", &noData);  


		for (int i = 0; i<noData; i++)
			fscanf(inFile, "%d", &ducci[i]);

		ducciTest(noData, ducci, ducciTmp);
	}

	fclose(inFile); 
	return 0;
}


void ducciTest(int max, int arr[], int arr2[])
{
	int count = 0;
	int tmp = 0;

	while (1) 
	{
		for (int j = 0; j<max; j++)
		{
			if (j != max - 1) 
			{
				arr2[j] = abs(arr[j] - arr[j + 1]);
			}

			else if (j == max - 1)
			{
				arr2[j] = abs(arr[j] - arr[0]);
			}
		}

		count++;
		if (count>1000)
		{
			printf("LOOP\n");
			break;
		}

		else
		{
			for (int k = 0; k<max; k++)
			{
				arr[k] = arr2[k];
				tmp += arr2[k];

			}

			if (tmp == 0)
			{
				printf("ZERO\n");
				break;
			}

			else
			{
				tmp = 0;
				continue;

			}

		}

	}

}
