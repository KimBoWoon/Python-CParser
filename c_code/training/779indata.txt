#include<stdio.h>
#include<string.h>

int palindrome(char* list, int pre, int post)
{
	if(pre >= post)	return 1;
	else if(list[pre] != list[post]) return 0;
	else return palindrome(list, pre+1, post-1);
}

void main()
{
	FILE*inFile = fopen("input.txt", "r");
	int testNum;
	int i;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		char list[1000];
		fscanf(inFile, "%s", &list);
		printf("%d\n", palindrome(list, 0, strlen(list)-1));
	}

	fclose(inFile);
}
