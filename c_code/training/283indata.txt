#include <stdio.h> 
#include <stdlib.h> 

 
 unsigned int gcd(unsigned int x, unsigned int y); 
 
 
 void main(void) 
{ 
 	FILE *file; 
 	int i, nocase; 
 	 
 	file = fopen("input.txt", "r"); 
 	if(file == NULL)  
		exit(1); 

 
 	fscanf(file, "%d", &nocase); 
 
 
 	for(i=0; i<nocase; i++) 
 	{ 
 		unsigned int n1, n2, d1, d2;	 
 		unsigned int num, den;	
 		unsigned int gcdTwoDen;	
 		unsigned int gcdDenNum; 

 
 		fscanf(file, "%u %u %u %u", &n1, &d1, &n2, &d2);	
 		 
 		gcdTwoDen = gcd(d1, d2);	
 		 
 		den = d1 * d2 / gcdTwoDen;	
 		num = (n1 * d2 / gcdTwoDen) + (n2 * d1 / gcdTwoDen);	  
 		 
 		gcdDenNum = gcd(num, den);	
 
 
 		printf("%u %u\n", num/gcdDenNum, den/gcdDenNum); 
 	} 
	 
 	fclose(file); 
 } 
 
 
 unsigned int gcd(unsigned int x, unsigned int y) { 
 	if(x == 0)	
 		return y;	
 	 
 	if(y == 0)	 
 		return x;	
 
 
 	return ((x >= y) ? (gcd(y, x%y)) : (gcd(x, y%x)));	
 }
