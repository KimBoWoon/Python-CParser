#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>
#include<stdlib.h>

void main(){
	FILE *in_file;

	int no_cases;
	int h, w, n;
	int i,a,b;

	in_file=fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d %d", &h, &w, &n);

		a = (n - 1) % h + 1;    
		b = (n - 1) / h + 1;
	
		printf("%d%02d\n",a, b);
	
	}

	fclose(in_file);
}
