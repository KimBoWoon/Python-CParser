
#include <iostream>
#include <fstream>

class TriangleNum
{
private:
	int TriangleNumList[45];
public:
	TriangleNum()
	{
		for ( int i = 1 ; i*(i+1)/2 < 1000 ; i++)
			TriangleNumList[i] = i*(i+1)/2;
	}
	bool isItTriangleNum(const int& num)
	{
		for(int n1 = 1 ; n1 <= 44 ; n1 ++)
			for (int n2 = 1; n2 <= 44 ; n2 ++)
				for (int n3 = 1; n3 <= 44 ; n3 ++)
				{
					int sum = TriangleNumList[n1] + TriangleNumList[n2] + TriangleNumList[n3];
					if ( sum == num)
						return true;
					else if ( sum < num)
						continue;
					else
						break;
				}
		return false;
	}

};

int main ()
{
	std::fstream input;
	input.open("input.txt");
	if ( input.fail() )
	{
		std::cout << "file open err" << std::endl;
		return 0;
	}
	int caseNum;
	input >> caseNum;
	TriangleNum tm;
	for ( int i = 0 ; i < caseNum ; i ++)
	{
		int testNum;
		input >> testNum;
		if ( tm.isItTriangleNum(testNum) )
			std::cout << 1 << std::endl;
		else
			std::cout << 0 << std::endl;
	}
	return 0;
}
