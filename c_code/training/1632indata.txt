#include <cstdio>
#include <cstring>
#include <climits>
using namespace std;

int arr[502];
int sum[502];
int d[502][502];

int main(void) {
	freopen("input.txt", "r", stdin);
	int t;
	scanf("%d", &t);

	while (t--) {
		int n;
		scanf("%d", &n);

		memset(d, 127, 501 * 501 * sizeof(unsigned int));

		for (int i = 1; i <= n; i++) {
			scanf("%d", &arr[i]);

			d[i][i] = 0;
		}

		sum[0] = 0;
		for (int i = 1; i <= n; i++) {
			sum[i] = sum[i - 1] + arr[i];
		}

		for (int i = 2; i <= n; i++) {
			for (int j = 1; j <= n - i + 1; j++) {
				int t = j, p = j + i - 1;

				int min = INT_MAX;
				for (int k = t; k < p; k++) {
					if (min > d[t][k] + d[k + 1][p]) min = d[t][k] + d[k + 1][p];
				}
				d[t][p] = (sum[p] - sum[t - 1]) + min;
			}
		}

		printf("%d\n", d[1][n]);
	}

	return 0;
}
