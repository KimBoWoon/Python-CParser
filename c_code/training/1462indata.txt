#include <stdio.h>
#include <stdlib.h>

void main()
{
	FILE *in_file;

	int i;
	int numcases;

	in_file = fopen("input.txt","r"); 

	fscanf(in_file, "%d", &numcases); 

		for(i=0; i < numcases; i++)
		{	
			int h, w, n;
			int res1 = 0;
			int res2 = 0;
			int a;

			fscanf(in_file,"%d %d %d", &h, &w, &n);
			if((n % h) == 0)
				res1 = h *100;
			else	
				res1 = (n % h) *100;
	         
            for(a=1; a<=w; a++){
	             if( n <= (h*a))
		          break;
            	}	  

		    res2 = res1 + a;
			printf("%d\n", res2);
		}

	fclose(in_file); 
}
