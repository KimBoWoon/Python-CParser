


#include <iostream>
#include <fstream>
#include<cstdlib>
using namespace std;

int ucl(int a, int b); 

int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int a, b, gcd;
		inStream >> a >> b;
		if (a >= b) 
		{
			gcd = ucl(a, b);
		}
		else
		{
			gcd = ucl(b, a); 
		}

		cout << gcd << endl; 
		

	}
	inStream.close();
	return 0;
}

int ucl(int a, int b) 
{ 
	if (b == 0) 
		return a;
	else 
		return ucl(b, a%b);
}
