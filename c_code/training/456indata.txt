#include <stdio.h>

int gcd(int a, int b){
    if(b==0) return a;
    else return gcd(b, a%b);
}

void getGcd(FILE* fin){
    int a, b=0;
    fscanf(fin, "%d %d", &a, &b);    
    printf("%d\n", gcd(a,b));       
}

int main(){
    int testCases, cases, n=0;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        getGcd(fin);        
    
    fclose(fin);            
    
    return 0;           
}
