
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>

int main(){
	FILE *fp;
	int i;
	int numcases;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);

	for(i=0;i<numcases;i++){
		int h,w,n;
		int n1=0,n2=0;

		fscanf(fp,"%d %d %d",&h,&w,&n);

		if(n%h==0){
			n1=h;
			n2=(n-1)/h+1;
		}
		else{
			n1=n%h;
			n2=(n-1)/h+1;
		}

		printf("%d%02d\n",n1,n2);
	}
	fclose(fp);
}
