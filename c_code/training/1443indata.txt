




#include <iostream>
#include <fstream>

void searchHotelRoom(int height, int width, int bookedRooms);

int main(){
	std::ifstream inFile;
	inFile.open("input.txt");

	int testCaseNum;
	int height;
	int width;
	int bookedRooms;

	inFile>>testCaseNum;

	for(int i = 0; i < testCaseNum; i++){
		inFile>>height>>width>>bookedRooms;

		searchHotelRoom(height, width, bookedRooms);
	}
}


void searchHotelRoom(int height, int width, int bookedRooms){

	int temp = bookedRooms%height;	
	int temp1 = bookedRooms/height;	

	if(temp == 0 && temp1 < 10){
		std::cout<<height<<"0"<<temp1<<std::endl;
	}
	else if(temp == 0 && temp1 > 10){
		std::cout<<height<<temp1<<std::endl;
	}
	else if(temp != 0 && temp1 < 10){
		std::cout<<temp<<"0"<<(temp1 + 1)<<std::endl;
	}
	else{
		std::cout<<temp<<(temp1 + 1)<<std::endl;
	}

}
