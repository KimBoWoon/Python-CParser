
#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
	FILE *inFile=NULL;
	inFile = fopen("input.txt","r");
	int t=0;
	fscanf(inFile, "%d", &t);
	for (int i = 0; i < t; i++)
	{
		vector<int> v;
		int n=0, K=0, value=0, cmp,k, max_K=-100000001, ans=0;
		fscanf(inFile, "%d %d", &n, &K);
		v.resize(n);
		for (int j = 0; j < n; j++)
		{
			fscanf(inFile,"%d", &value);
			v[j] = value;
		}
		sort(v.begin(), v.end());
		cmp = v.size() - 1;
		for (int j = 0; j < n; j++)
		{
			int max_k_K = -100000001, count_k=0;
			for (k = cmp; k > j; k--)
			{
				if (j >= k)
					break;
				int abs_k = abs(v[j] + v[k] - K), abs_K=abs(max_k_K - K);
				if (abs_k > abs_K)
				{
					cmp = k + 1;
					break;
				}
				if (abs_k < abs_K)
				{
					max_k_K = v[j] + v[k];
					count_k = 1;
				}
				else 
					count_k++;
			}
			if (abs(max_k_K - K) < abs(max_K - K))
			{
				max_K = max_k_K;
				ans = count_k;
			}
			else if(abs(max_k_K - K) == abs(max_K - K))
			{
				ans += count_k;
			}
			if (j >= k)
				break;
		}
		printf("%d\n", ans);
	}
	fclose(inFile);
}
