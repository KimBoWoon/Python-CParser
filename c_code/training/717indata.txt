

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <string>
#include <fstream>
#include <iostream>

using namespace std;

int PalinDrome(char c[], int first, int last);

int main(void) {

	ifstream dataFile("input.txt");

	int caseNum;
	dataFile >> caseNum;
	
	for (int i = 0; i < caseNum; i++) {
		char str[1000];
		dataFile >> str;

		int si = 0;
		int strLength = 0;		

		while (str[si] != NULL) {			
				strLength++;
				si++;		
		}
		cout << PalinDrome(str, 0, strLength-1) << endl;		
	}
	

	dataFile.close();

	return 0;
}

int PalinDrome(char c[], int first, int last) {

	if (last <= first) {
		return 1;
	}
	else if (c[first] != c[last]) {
		return 0;
	}
	else {
		return PalinDrome(c, first + 1, last - 1);
	}
}
