
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int macFunc(int n);

int main()
{
	int count=0;
	ifstream file;
	file.open("input.txt");
	file >> count;

	int num = 0;
	for (int i = 0; i < count; i++)
	{
		file >> num;
		cout << macFunc(num)<<endl;
	}
	return 0;
}

int macFunc(int n) 
{
	if (n > 100)
		return n - 10;
	else
		return macFunc(macFunc(n + 11));
}
