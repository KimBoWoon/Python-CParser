#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>

using namespace std;

void paren(fstream &infile);

int main()
{   
	fstream infile;
	int numcases;
	int t;
	char vps[50];

	infile.open("input.txt");

    if(infile.fail())
    {
       cerr << "파일열기오류" <<endl;
       exit(1);
    }

   infile >> numcases;
   infile.getline(vps,50);

	for (t = 0; t < numcases; t++)
	{
		paren(infile);
	}
	return 0;
}

void paren(fstream &infile)
{
   char a[50];
   int count = 0;
   int leng;
   int i;

   infile.getline(a,50);
   leng = strlen(a);

   if(a[0] == '(' && a[leng-1] == ')')
   {
      for(i = 1; i<leng-1; i++)
      {
         if(a[i] == '(')
         {
            count ++;
         }
         else if(a[i] == ')')
         {
            count --;
         }
      }
	  if(count == 0)
      {
		 cout << "YES" << endl;
	  }
      else
        cout << "NO" << endl;
   }

   else
        cout << "NO" << endl;
}
