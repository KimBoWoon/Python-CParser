#include <stdio.h>
#include <stdlib.h>

void main()
{

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){

		int num;
		int divisor;
		int value=1;

		fscanf(in_file,"%d",&num);
		for(divisor=2;divisor*divisor<=num;divisor++){

			if(num%divisor==0){
				
				value=0;
				break;
			}
		}

		printf("%d\n", value);
	}
	fclose(in_file);
}
