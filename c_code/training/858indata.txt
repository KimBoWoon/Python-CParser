#include<stdio.h>
#include<stdlib.h>
int Fibonacci(int n);
int main(void)
{
	FILE *in_file;
	int no_cases, no_data;
	int i;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   


	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d ", &no_data);
		{
			printf("%d\n", Fibonacci(no_data));  
		}
	}
	fclose(in_file);  
}

int Fibonacci(int n)
{
	if (n == 0)
		return 0;
	else if (n == 1)
		return 1;
	else if (n > 1 && n <= 30)
		return Fibonacci(n-1)+Fibonacci(n - 2);
}
