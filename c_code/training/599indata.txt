

#include <iostream>
#include <fstream>
using namespace std;
int factorialRecursive(int num);

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases = 0;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++)
	{
		int num = 0;
		inFile >> num;
		cout << factorialRecursive(num) << endl;
	}
}
int factorialRecursive(int num){

	if (num == 0){
		return 1;
	}
	else{
		return num*factorialRecursive(num - 1);
	}
}
