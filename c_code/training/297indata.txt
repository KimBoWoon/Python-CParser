#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

void cal(fstream & infile); 
int max(int a, int b);
int min(int a, int b, int max);

int main()
{
	int nocases;

	fstream infile;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i = 0; i < nocases; i++)
	{
		cal(infile);
	}
}

void cal(fstream & infile)
{
	int a,b,c,d;
	int num,num1;
	int calnum;
	int k;

	infile >> a;
	infile >> b;
	infile >> c;
	infile >> d;

	k = b*d;

	a = a*(k/b);
	c = c*(k/d);

	num = a+c;

	calnum = max(k,num);

	cout << num/calnum << " " << k/calnum << endl;
}

int max(int a, int b)
{
	if(b == 0)
		return a;
	else
		max(b, a%b);
}
