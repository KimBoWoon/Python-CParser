

#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;
void printGoldbach(int test);
bool checkPrime(int num, int i);
int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	if (inStream.fail()) exit(1);
	int numTest;
	inStream >> numTest;
	for (int i = 0; i < numTest; i++)
	{
		int test;
		inStream >> test;
		printGoldbach(test);
	}
}
void printGoldbach(int test)
{
	int plusNum;
	int minusNum;

	if (test % 2 == 0){
		plusNum = test / 2;
		minusNum = plusNum;
	}
	else{
		plusNum = test / 2 + 1;
		minusNum = test / 2;
	}

	for (int k = 0; k < test / 2; k++)
	{
		if (checkPrime(plusNum, plusNum - 1) && checkPrime(minusNum, minusNum - 1))
		{
			cout << minusNum << " "<< plusNum;
			break;
		}
		plusNum++;
		minusNum--;
	}
	cout << endl;
}
bool checkPrime(int num, int i)
{
	for (int k = 2; k < num; k++)
	{
		if (num%k == 0) return false;
	
	}
	return true;
}
