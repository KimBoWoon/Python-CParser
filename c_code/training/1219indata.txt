


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int sosu(int num);		

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value;
	int primeArr[168] = { 0, };		
	int z = 0;

	for (int i = 2; i < 1000; i++)
	{
		if (sosu(i) == 1)
		{
			primeArr[z] = i;
			z++;
		}
	}

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value;
		int rArr[2] = { 0, 1000 };	
		int r = 0;
		int absol = 0;
		int rabsol = 1000;

		for (int l = 0; l < 168; l++)			
		{
			for (int k = 0; k < 168; k++)
			{
				if (primeArr[l] + primeArr[k] == value)
				{				
					absol = abs(primeArr[l] - primeArr[k]);		
					
					if (rabsol > absol)
					{
						rArr[0] = primeArr[l];
						rArr[1] = primeArr[k];
					}
					rabsol = absol;
				}
			}
		}
		cout << rArr[0] << " " << rArr[1] << endl;
	}

	inStream.close();
	return 0;
}

int sosu(int num)
{
	int isPrime = 1;	
	int nam;				

	for (int j = 2; j < num; j++)
	{
		nam = num % j;

		if (nam == 0)
		{
			isPrime = 0;
			break;
		}

		else
			isPrime = 1;
	}
	return isPrime;
}
