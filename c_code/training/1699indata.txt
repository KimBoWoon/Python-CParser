#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>
#include <cstdlib>

using namespace std;

int fn(int n, int m)
{
	vector<char> v;

	while (1)
	{
		int num = n % m;
		v.push_back(num + '0');
		n /= m;
		if (n == 0)
			break;
	}

	int result = 1;
	for (int i = 0; i < v.size() / 2; ++i) {
		if (v[i] != v[v.size() - 1 - i]) {
			result = 0;
			break;
		}
	}

	return result;
}

int main() {

	ifstream inStream;
	inStream.open("input.txt");

	int case_num;

	inStream >> case_num;
	for (int case_cnt = 0; case_cnt < case_num; ++case_cnt) {
		int num;
		inStream >> num;
		int result = 0;
		for (int i = 2; i < 64; ++i) {
			if (fn(num, i) == 1) {
				result = 1;
				break;
			}
		}
		cout << result << endl;
	}
	return 0;
}
