


#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)
int palindrome(char* str, int first, int last); 

void main(void)
{
	FILE* inFile;
	int testCase = 0;
	char str[1001];
	int i = 0;
	inFile = fopen("input.txt", "r"); 
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase); 

	for (i = 0; i < testCase; i++)
	{
		fscanf(inFile, "%s", &str);
		printf("%d\n", palindrome(str, 0, strlen(str) - 1));
	}
	fclose(inFile);
}

int palindrome(char* str, int first, int last)
{
	if (last<=first) 
	{
		return 1;
	}
	else if (str[first] != str[last])
	{
		return 0;
	}
	else
		return palindrome(str, first + 1, last - 1);
}
