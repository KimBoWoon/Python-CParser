
#include <stdio.h>
#include <stdlib.h>

int f(int a,int b,int c,int d);
int g(int a,int b);

int main()
{
	FILE *infile;
	int numcases;
	int t,a,b,c,d,s,w;

	infile = fopen("input.txt","r");
	if(infile==NULL) exit(1);

	fscanf(infile, "%d", &numcases);

	for (t = 0; t < numcases; t++)
	{
		fscanf(infile, "%d %d %d %d", &a, &b, &c, &d);
		f(a,b,c,d);
	}
	return 0;
}

int f(int a,int b,int c,int d)
{
	int s, w;
	int n;
	s = a*d + c*b;
	w = b*d;

	n=g(s,w);

	s=s/n;
	w=w/n;

	printf("%d %d\n",s,w);
}

int g(int a, int b)
{
	if(b==0)
		return a;
	else
		return g(b, a%b);		
}
