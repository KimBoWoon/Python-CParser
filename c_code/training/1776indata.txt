#include <cstdio>

int number[30];

bool ispalindrome(int head, int tail) {
	if (head >= tail) return true;
	else if (number[head] != number[tail]) return false;
	else return ispalindrome(head + 1, tail - 1);
}
bool changeBase(int n, int base, int cnt) {
	if (n == 0) {
		return ispalindrome(0, cnt-1);
	}
	else {
		number[cnt] = n % base;
		return changeBase(n / base, base, cnt + 1);
	}
}

int main(void) {
	freopen("input.txt", "r", stdin);
	int t;
	scanf("%d", &t);

	while (t--) {
		int n;
		scanf("%d", &n);

		bool result = false;
		for (int i = 2; i <= 64; i++) {
			result = result || changeBase(n, i, 0);
		}

		printf("%d\n", result);
	}
	return 0;
}
