#include <iostream>
#include <fstream>
#include <cstdlib>

bool checkTupleisZero(const int tuple[15], int tupleLen)
{
	for (int i = 0; i < tupleLen; i++)
		if (tuple[i] != 0)
			return false;
	return true;
}

void nextStep(int tuple[15], int tupleLen)
{
	int first = tuple[0];
	for (int i = 0; i < tupleLen - 1; i++)
		tuple[i] = abs(tuple[i] - tuple[i + 1]);
	tuple[tupleLen - 1] = abs(tuple[tupleLen - 1] - first);
}

bool doesItHaveSequence(int tuple[15], int tupleLen)
{
	int count = 0;
	while (!checkTupleisZero(tuple, tupleLen))
	{
		if (count >= 1000)
			return false;
		count++;
		nextStep(tuple, tupleLen);
	}
	return true;
}


int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail())
	{
		std::cerr << "file open err" ;
		return 1;
	}

	int caseNum;
	input >> caseNum;
	for (int i = 0; i < caseNum; i++)
	{
		int tuple[15] = { 0 };
		int tupleLen;
		input >> tupleLen;
		for (int i = 0; i < tupleLen; i++)
			input >> tuple[i];

		std::cout << (doesItHaveSequence(tuple, tupleLen) ? "ZERO" : "LOOP") << std::endl;
	}

	return 0;
}
