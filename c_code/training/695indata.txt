#include <iostream>
#include <fstream>
#include <string>

using namespace std;

bool fn(string & s, int n)
{
	if (n == s.length() / 2)
		return true;
	else
		return s[n] == s[s.length() - n - 1] && fn(s, n + 1);
}

int main(){
	ifstream fin("input.txt");

	int nCase;
	fin >> nCase;

	for (int ci = 0; ci < nCase; ++ci)
	{
		string s;
		fin >> s;
		cout << (fn(s, 0) ? 1 : 0) << endl;
	}
}
