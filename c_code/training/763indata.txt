
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int pal(char str[], int first, int last){
	if(last<=first) 
		return 1;
	else if(str[first]!=str[last])
		return 0;
	else 
		return pal(str,first+1,last-1);
}

int main(){
	FILE *fp;
	int i;
	int numcases;
	char input[1000];
	int result=0;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);
	
	for(i=0;i<numcases;i++){
	
		fscanf(fp,"%s",&input);
		
		result=pal(input,0,strlen(input)-1);

		printf("%d\n",result);
		
	}
	fclose(fp);
}
