#include <iostream>
#include <fstream>

using namespace std;

int fn(int a, int b){
	if (b == 0)
		return a;
	else
		return fn(b, a % b);
}

int main(){
	ifstream fin("input.txt");

	int nCase;
	fin >> nCase;

	for (int ci = 0; ci < nCase; ++ci)
	{
		int a, b;
		fin >> a >> b;
		cout << fn(a, b) << endl;
	}
}
