#include<stdio.h>

int Euclid(int a, int b)		
{
	if (b == 0) return a;
	else return Euclid(b, a%b);
}

int Henry(int a, int b)
{
	int t, gcd;
	if(a==1) return b;
	else{
		t = b/a +1;
		a = a*t-b; b = b*t;		
		gcd = Euclid(a, b);
		a /= gcd; b /= gcd;
		Henry(a, b);
	}
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testNum, i;
	int a, b;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i)
	{
		fscanf(inFile, "%d %d", &a, &b);
		printf("%d\n", Henry(a, b));
	}
	fclose(inFile);
}
