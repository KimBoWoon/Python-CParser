
#include <iostream>
#include <fstream>
#include <cmath>

int a[15];
int n;

void DucciSequence(int num) {
	bool zero = true;
	if (n > 1000) {
		std::cout << "LOOP" << std::endl;
		return;
	}

	for (int i = 0; i < num; i++)
		if (a[i] != 0) zero = false;
	if (zero) {
		std::cout << "ZERO" << std::endl;
		return;
	}
	int a_z = a[0];
	for (int i = 0; i < num; i++){
		if (i + 1 < num) a[i] = std::abs(double(a[i]) - double(a[i + 1]));
		else a[i] = std::abs(double(a[i]) - double(a_z));
	}
	n++;
	DucciSequence(num);
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase,num;
	std::string str;
	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	while (numcase--) {
		fin >> num;
		n = 0;
		for (int i = 0; i < num; i++)
			fin >> a[i];
		DucciSequence(num);
	}
	fin.close();
}
