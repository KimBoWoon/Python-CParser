
#include<iostream>
#include<fstream>
using namespace std;

int abs(int a, int b){
	return a > b ? a - b : b - a;
}
void check(int* num){
	if (num[1]++ == 1000){
		cout << "LOOP" << endl;
		return;
	}
	int sum = 0;
	int temp = num[2];
	for (int i = 2; i < num[0]; i++){
		num[i] = abs(num[i], num[i + 1]);
		sum += num[i];
	}
	num[num[0]] = abs(temp, num[num[0]]);
	sum += num[num[0]];
	if (sum == 0){
		cout << "ZERO" << endl;
		return;
	}
	check(num);
}
int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;
	while (cases--){
		int size;
		inFile >> size;
		int* num = new int[size+2];
		for (int i = 2; i < size+2; i++){
			inFile >> num[i];
		}
		num[0] = size+1;
		num[1] = 0;
		check(num);
	}
}
