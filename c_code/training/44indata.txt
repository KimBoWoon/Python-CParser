#include <iostream>
#include <fstream>

using namespace std;

int main(void)
{
	fstream a;
	int numcases;
	int num;
	bool primer=false;
	a.open("input.txt");
	a>>numcases;
	for(int i=0; i<numcases; i++)
	{
		a>>num;
		for(int j=2; j<num; j++)
		{
			if(num%j==0)
				primer=true;

			if(primer)
			{	cout<<"0"<<endl;
				break;
			}
			
		}
		if(!primer)
		{
			cout<<"1"<<endl;
		}
		primer=false;
	}
	return 0;
}
