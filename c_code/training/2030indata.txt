#include <iostream>
#include <algorithm>
#include <cstdlib>
#include <fstream>
using namespace std;

int arr[1000000];

int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while (t--)
	{
		int n, k;
		int min;
		int cnt = 0;
		in >> n >> k;
		for (int i = 0; i < n; i++) in >> arr[i];
		sort(arr, arr + n);
		min = abs(arr[0] + arr[n-1] - k);
		for (int i = 0,j=n-1; i<j;)
		{
			int val = arr[i] + arr[j] - k;
			int tmp = abs(val);
			if (tmp < min) min = tmp;
			if (val < 0)i++;
			else j--;
		}
		for (int i = 0, j = n - 1; i < j;)
		{
			int val = arr[i] + arr[j];
			if (k + min == val) cnt++;
			if (val > k) j--;
			else i++;
		}
		for (int i = 0, j = n - 1; min!=0 && i < j;)
		{
			int val = arr[i] + arr[j];
			if (k - min == val) cnt++;
			if (val > k) j--;
			else i++;
		}
		cout << cnt << endl;
	}
}
