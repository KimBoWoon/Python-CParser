



#include<iostream>
#include<fstream>

using namespace std;

int main(void) {
	ifstream inStream;
	inStream.open("input.txt");
	int numTestCases;
	inStream >> numTestCases;
	int count=0;
	int arr[50];

	for (int j = 1; j < 50; j++)
	{
		if ((j*(j+1))/2 > 1000) break;
		arr[j] = (j*(j+1))/2;
		count++;
	}

	for (int i = 0; i < numTestCases; i++) {
		
		int number;
		bool ok = false;
		inStream >> number;

		for (int a = 0; a < count; a++){
			for (int b = 0; b < count; b++) {
				for (int c = 0; c < count; c++) {
					if (arr[a] + arr[b] + arr[c] == number) {
						cout << "1" << endl;
						ok = true;
					}
					if (ok == true) break;
				}
				if (ok == true) break;
			}
			if (ok == true) break;
		}
		if (ok == false) cout << "0" << endl;

	}
}
