#include <stdio.h>
 
int sum[501][501];
 
int main(void) {
    FILE* file;
    int cases;
    int a, i, j, k, h, s[501][501], m = 99999999;
    file = fopen("input.txt", "r");
    fscanf(file, "%d", &cases);
 
    while (cases--) {
        fscanf(file, "%d", &k);
 
        for (i = 0; i < k; i++) {
            fscanf(file, "%d", &s[i][i]);
        }
 
        for (i = 1; i < k; i++) {
            for (j = i - 1; j != -1; j--) {
                s[j][i] = s[i][i] + s[j][i - 1];
            }
        }
 
        for (i = 1; i < k; i++) {
            for (j = i - 1; j != -1; j--) {
                a = i - 1;
                for (h = i; h != j; h--) {
                    if (i == j)
                        m = 0;
                    else
                        if (m > sum[h][i] + sum[j][a])
                            m = sum[h][i] + sum[j][a];
                    a--;
                }
                sum[j][i] = s[j][i] + m;
                m = 99999999;
            }
        }
 
        printf("%d\n", sum[0][k - 1]);
    }
    return 0;
}
