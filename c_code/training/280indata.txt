

#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b) {
	if (b == 0) return a;
	else return gcd(b, a%b);
}
void fountain(int a, int b, int c, int d) {
	int res1, res2, lcm;
	lcm = (b/gcd(b,d))*(d/gcd(b,d))*gcd(b,d);
	res1=a*(lcm/b)+c*(lcm/d);
	res2=lcm;
	printf("%d %d\n", res1/gcd(res1,res2), res2/gcd(res1,res2));
}

void main(void) 
{
	int test=0, i, a, b, c, d;
	FILE *file = fopen("input.txt", "r"); 
	if(NULL == file) exit(1);

	fscanf(file, "%d", &test);

	for(i=0; i<test; i++) 
	{
		fscanf(file, "%d %d %d %d", &a, &b, &c, &d);
		fountain(a, b, c, d);
	}
	fclose(file);
}
