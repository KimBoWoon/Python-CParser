#include <iostream>
#include <fstream>

int gcd(int inputNum1, int inputNum2)
{
	if (inputNum2 == 0)
		return inputNum1;
	else
		gcd(inputNum2, inputNum1%inputNum2);
}

int main()
{
	std::ifstream file("input.txt");
	int testCase = 0, a, b, c, d, e, f,g;
	a = b = c = d = e = f = g=0;
	file >> testCase;
	while (testCase--)
	{
		file>>a >> b >> c >> d;
		if (b > d)
			f = b*d/gcd(b, d);
		else
			f = b*d / gcd(d, b);
		e = a*(f / b) + c*(f / d);
		if (e > f)
			g = gcd(e, f);
		else
			g = gcd(f, e);
		std::cout << e/g << " " << f/g << std::endl;
	}
}
