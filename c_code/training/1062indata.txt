

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main(void) 
{
	int test, n;
	ifstream file;
	file.open("input.txt"); 
	file >> test;

	for(int i=0; i<test; i++) 
	{
		int D[15]={0};
		int count=0, first=0;
		file >> n;
		for(int j=0; j<n; j++) {
			file >> D[j]; 
		}
		while(1) {
			int zero=0;
			first = D[0];
			for(int k=0; k<n; k++) {
				if(D[k]==0) zero++;
			}
			if(zero==n) {
				cout << "ZERO" << endl;
				break;
			}
			if(count >= 1000) {
				cout << "LOOP" <<endl;
				break;
			}
			for(int j=0; j<n; j++) {
				if(j == n-1)
					D[j] = abs(D[j]-first);
				else D[j] = abs(D[j]-D[j+1]);
			}
			count++;
		}
	}
	return 0;
}
