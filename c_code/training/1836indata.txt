
#include <cstdio>
#include <cstdlib>
#include <cstring>

bool parenthesis(char* str, int flag, int count, int length){

	if(flag == length && count == 0) return true;
	else if(flag == length) return false;
	
	if(str[flag] == '(') count += 1;
	else count -= 1;

	if(count < 0) return false;
	
	if(parenthesis(str, flag + 1, count, length)) return true;

	return false;
}
int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}

	int testCases = 0;
	char stringArray[50];
	
	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		fscanf(readStream, "%s", stringArray);
		printf("%s\n", (parenthesis(stringArray, 0, 0, strlen(stringArray))) ? "YES" : "NO");
	}
	fclose(readStream);
}
