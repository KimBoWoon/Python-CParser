

#include <iostream>
#include <fstream>

using namespace std;

int factorial(int num); 

int main()
{
	ifstream infile;
	int testCase;
	int num; 

	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num;

		cout << factorial(num)<< endl; 
	}

	infile.close();

	return 0;
}

int factorial(int n)
{
	if (n <= 1) 
		return 1;
	else
		return n*factorial(n - 1);
}
