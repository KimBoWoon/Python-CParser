

#include <stdio.h>
#include <math.h>

int criterionTriangularNumber(int x)
{
	int i = 1;
	while ((i*i + i) <= 2 * x) 
	{
		if ((i*i + i) == 2 * x) return 1; 
		i++;
	}
	return 0; 
}

int main()
{
	FILE *fp;
	int t, k, flag = 0, i, j, n;
	fp = fopen("input.txt", "r"); 
	if (fp == NULL)
	{
		return 0;
	}
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%d", &k); 
		for (i = 1; i < sqrt(k)*sqrt(k); i++)
		{
			if (criterionTriangularNumber(i)) 
			{
				for (j = 1; j < sqrt(k)*sqrt(k); j++)
				{
					if (criterionTriangularNumber(j)) 
					{
						for (n = 1; n < sqrt(k)*sqrt(k); n++)
						{
							if (criterionTriangularNumber(n)) 
							{
								if ((i + j + n) == k) 
								{
									printf("1\n"); 
									flag = 3;
									break; 
								}
							}
						}
					}
					if (flag == 3)
						break; 
				}
			}
			if (flag == 3)
				break; 
		}
		if (flag != 3)
			printf("0\n"); 
		flag = 0; 
	}

	fclose(fp); 
	return 1;
}
