#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;


int Ducci(int DucciNum[], int numcase)
{
	int storeDucci[1024][15] = {0}; 
	int sum = 0;
	int duci;

	for (int i = 0; i < numcase; i++)
	{
		if (i == numcase - 1)
			storeDucci[0][i] = abs(DucciNum[i] - DucciNum[0]);
		else
			storeDucci[0][i] = abs(DucciNum[i] - DucciNum[i + 1]);
	}

	for (int j = 1; j < 1001; j++) { 
		duci = storeDucci[j-1][0];
		for (int i = 0; i < numcase; i++)
		{
			if (i == numcase - 1)
				storeDucci[j][i] = abs(storeDucci[j-1][i] - duci);
			else
				storeDucci[j][i] = abs(storeDucci[j - 1][i] - storeDucci[j-1][i + 1]);
		}
		for (int k = 0; k < numcase; k++) {
			sum += storeDucci[j][k];
		}
		if (sum == 0) {
			cout << "ZERO" << endl;
			return 0;
		}
		sum = 0;
	}


	cout << "LOOP" << endl;
	return 0;
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int numcase;
	int DucciNum[15] = {0};
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> numcase;
		for (int j = 0; j < numcase; j++)
			instream >> DucciNum[j];

		Ducci(DucciNum, numcase);
	}
	return 0;
}
