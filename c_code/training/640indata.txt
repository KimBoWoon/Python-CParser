
#include<iostream>
#include<fstream>

using namespace std;

int factorial(int t){
	if (t <=1) return 1;
	return factorial(t - 1) * t;
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;

	int n;
	in >> n;
	for (int i = 0; i < n; i++){
		int t;
		in >> t;
		cout << factorial(t) << endl;
	}
	in.close();
	return 0;
}
