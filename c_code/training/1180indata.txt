#include <stdio.h>

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int inputK = 0; 
	int Tn[46] = { 0 }; 
	int isEureka[1001] = { 0 }; 

	int n;
	for (n = 1; n <= 45; n++) { 
		Tn[n] = n*(n + 1) / 2;
	}
	int i, j, k;
	for (i = 1; i <= 45; i++){
		for (j = 1; j <= 45; j++){
			for (k = 1; k <= 45; k++) {
				if (Tn[i] + Tn[j] + Tn[k] <= 1000)
					isEureka[Tn[i] + Tn[j] + Tn[k]] = 1;
			}
		}
	}

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d", &inputK); 

		printf("%d\n", isEureka[inputK]);

	}

	fclose(f);
	return 0;
}
