#include <iostream>
#include <fstream>
#define MAX 20
using namespace std;
int ncase;
int testcase;
int i, j, k, n;
int num[MAX];

int main()
{
	ifstream instream("input.txt");
	instream >> ncase;
	for (i = 0; i < ncase; i++)
	{
		instream >> testcase;
		for (int j = 0; j < testcase; j++)
		{
			instream >> num[j];
		}
		num[testcase] = num[0];
		for (k = 0; k < 1000; k++)
		{
			for (n = 0; n < testcase; n++)
			{
				num[n] = num[n] - num[n + 1];
				if (num[n] < 0) num[n] = -num[n];
			}
			num[testcase] = num[0];
		}
		for (j = 0; j < testcase; j++)
		{
			num[testcase+1] += num[j];
		}
		if (num[testcase + 1] == 0)
		{
			cout << "ZERO" << endl;
		}
		else
		{
			cout << "LOOP" << endl;
		}
	}
}
