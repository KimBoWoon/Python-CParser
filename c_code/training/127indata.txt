#include <stdio.h>
#include <stdlib.h>

int mcCarthy(int );

int main()
{
   FILE *fp;
   int i;
   int no_cases,H,W,N;

   fp=fopen("input.txt","r");
   if(fp==NULL) exit(1);

   fscanf(fp, "%d",&no_cases);

   for(i=0; i<no_cases; i++){
	   int num;
	   fscanf(fp,"%d",&num);

	   printf("%d\n",mcCarthy(num));
   }
   return 0;
   fclose(fp);
}

int mcCarthy(int a)
{
	if(a>100)
		return a-10;
	else
		return mcCarthy(mcCarthy(a+11));
}
