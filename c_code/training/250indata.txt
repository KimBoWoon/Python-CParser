#include <iostream>
#include <fstream>

int gcd(int num1, int num2);
int add_mol(int lcm_number, int mol1, int deno1, int mol2, int deno2);

int main(void){

	std::ifstream inFile("input.txt");
	if (inFile.fail()){
		std::cout << "not file" << std::endl;
		return 0;
	}

	int testCases = 0;
	inFile >> testCases;
	
	int mol1 = 0;
	int mol2 = 0;
	int deno1 = 0;
	int deno2 = 0;

	int lcm_number = 0;
	int reduction_number = 0;

	for (int i = 0; i < testCases; i++){
		inFile >> mol1 >> deno1 >> mol2 >> deno2;

		lcm_number = deno1 * deno2 / gcd(deno1, deno2);

		reduction_number = gcd(add_mol(lcm_number, mol1, deno1, mol2, deno2), lcm_number);

		std::cout << add_mol(lcm_number, mol1, deno1, mol2, deno2) / reduction_number << " " << lcm_number / reduction_number << std::endl;
	}
	return 0;
}

int gcd(int num1, int num2){

	if (num1 < 0 || num2 < 0){
		return -1;
	}

	if (num2 == 0){
		int temp = num1;
		num1 = num2;
		num2 = temp;
	}

	int r = 0;
	r = num1 % num2;
	num1 = num2;
	num2 = r;

	if (r == 0){
		return num1;
	}
	else{
		return gcd(num1, num2);
	}
}

int add_mol(int lcm_number, int mol1, int deno1, int mol2, int deno2){

	return (lcm_number / deno1) * mol1 + (lcm_number / deno2) * mol2;
}
