#include <fstream>
#include <iostream>
using namespace std;
int gcd(int a, int b)
{
   if (b == 0) 
   	return a;
   else 
   	return gcd(b, a % b);
}
bool chk(int deno1, int nume1, int deno2, int nume2)
{
   if (nume1 * deno2 > nume2 * deno1)
   	return true;
   else
   	return false;
}
void chk2(int& deno, int& nume, int deno2, int nume2)
{
   int nume1 = nume;
   int deno1 = deno;
   int G=0;
   nume1 *= deno2;
   nume2 *= deno1;
   G=gcd((nume1 - nume2), (deno1 * deno2));
   nume = (nume1 - nume2) / G;
   deno = (deno1 * deno2) / G;
}
int henry(int deno, int nume)
{
	   if (nume == 1)
	   	return deno;
	   else
	   {
	   	for (int i = 2; i < deno; i++){
	    		if (chk(deno, nume, i, 1))
	    		{
	            		chk2(deno, nume, i, 1);
	            		break;
	         		}
	    	}
	   	return henry(deno, nume);
	   }
}
int main(){
   int Num = 0;
   ifstream inputFile("input.txt");
   inputFile >> Num;
   for (int i = 0; i < Num; i++)
   {
      int deno = 0, nume = 0;
      inputFile >> nume >> deno;
      cout << henry(deno, nume) << endl;
   }
   return 0;
}
