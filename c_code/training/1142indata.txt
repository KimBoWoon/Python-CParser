





#include <iostream>
#include <fstream>
#include <vector>

bool isTriangleNumber(int testCase, std::vector<int> tempVec);

int main(){
	std::ifstream inFile;
	inFile.open("input.txt");

	int numTestCases;

	inFile>>numTestCases;

	for(int i = 0; i < numTestCases; i++){
		std::vector<int> tempVec;
		int testCase;

		inFile>>testCase;

		for(int i = 1; i < testCase; i++){
			if((i * (i + 1)) / 2 >= testCase)
				break;
			else{
				tempVec.push_back((i * (i + 1)) / 2);
			}
		}
	std::cout<<(isTriangleNumber(testCase, tempVec))<<std::endl;

	}
}

bool isTriangleNumber(int testCase, std::vector<int> tempVec){
	for(int i = 0; i < tempVec.size(); i++){
		for(int j = 0; j < tempVec.size(); j++){
			for(int k = 0; k < tempVec.size(); k++){
				if(tempVec[i] + tempVec[j] + tempVec[k] == testCase)
					return true;
			}
		}
	}
	return false;
}
