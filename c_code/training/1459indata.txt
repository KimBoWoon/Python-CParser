#include<stdio.h>
#include<stdlib.h>

void main(){

	FILE *file;

	int numCases;
	int i,j;
	int H,W,N;

	file = fopen("input.txt","r");
	if(file == NULL) exit(1);

	fscanf(file, "%d", &numCases);
	

	for(i=0; i<numCases; i++){
		fscanf(file, "%d%d%d", &H, &W, &N);

		for(j=1; j<(W+1); j++){
			
			if(N>H){
				N-=H;
			}
			else
				break;

			
		}
			if(j>=10)
		printf("%d%d\n",N,j);

			else
				printf("%d0%d\n",N,j);
			
	}


}
