
#include <iostream> 
#include <fstream>
#include <string>
using namespace std;

int checkPalindrome(string s, int first, int last)
{
	if(last<=first) return 1;
	else if(s[first] != s[last]) return 0;
	else return checkPalindrome(s,first+1, last-1);
}
int main()
{
	int t;
	string st;
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>t;
	for(int i=0; i<t; i++)
	{
		inStream>>st;
		cout<<checkPalindrome(st,0,st.size()-1)<<endl;
	}
}
