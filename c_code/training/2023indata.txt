
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int cnt=0;
int input[1000010]={0};

int partition(int a[], int low, int high)
{
   int i, j;
   int pivot, pivotPos, tmp;
   pivot = a[low];
   j = low;
   for (i = low + 1; i <= high; i++)
   {
      if (a[i] < pivot)
      {
         j++;
         
         tmp = a[i]; a[i] = a[j]; a[j] = tmp;
      }
   }
   pivotPos = j;
   tmp = a[low]; a[low] = a[pivotPos]; a[pivotPos] = tmp;
   return pivotPos;
}
void quickSort(int v[], int low, int high)
{
   int pivotPos;
   if (high > low)
   {
      pivotPos = partition(v, low, high);
      quickSort(v, low, pivotPos - 1);
      quickSort(v, pivotPos + 1, high);
   }
}

void twonum(int n,int k){

	int s=0,e=n-1;
	int min=1000000000;
	while(s<e){
		if(abs(k-input[s]-input[e])<min){
			min=abs(k-input[s]-input[e]);
			cnt=1;
		}
		else if(abs(k-input[s]-input[e])==min)
			cnt++;

		if(input[s]+input[e]<k) 
			s++;
		else
			e--;
	}
}

int main(){
	FILE *fp;
	int numcases;
	int i,j;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);
	fscanf(fp,"%d",&numcases);

	for(i=0;i<numcases;i++){
		int n,k;

		fscanf(fp,"%d %d",&n,&k);

		for(j=0;j<n;j++)
			fscanf(fp,"%d",&input[j]);

		quickSort(input,0,n-1);
		cnt=0;
		twonum(n,k);

		printf("%d\n",cnt);

	}
	fclose(fp);
}
