
#include<stdio.h>

int recursion(int a){
	if (a==0) return 0;
	else if (a == 1) return 1;
	else return recursion(a-1)+recursion(a-2);
}

int main(){

	FILE *inFile;
	int circle, num;
	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d", &num);
		printf("%d\n", recursion(num));
	}
	return 0;
}
