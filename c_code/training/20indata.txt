




#include<stdio.h>
#include<stdlib.h>

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		int value;
		int isPrime = 1;
		int divisor;

		fscanf(inFile, "%d", &value);

		for(divisor=2; divisor < value; divisor++)
		{
			if(value % divisor == 0)
			{
				isPrime=0;
				break;
			}
		}
		printf("%d\n", isPrime);
	}
	fclose(inFile);
}
