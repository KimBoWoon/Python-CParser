#include <iostream>
#include <string>
#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>

using namespace std;

bool checkPrime(int num){
	if (num < 4) return true;
	for (int k = 3; k <= num; k = k + 2){
		if (k == num){
			return true;
		}
		if (num%k == 0) {
			return false;
		}
	}
}

int main() {
	int _cases;
	fstream inStream;

	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1;
	}

	inStream >> _cases;

	for (int i = 0; i < _cases; i++) {
		int number;
		inStream >> number;
		int a, b;
		if (number == 4) cout << 2 << " " << 2 << endl;
		else{
			for (int j = 3; j <= number - j; j = j + 2)
			{
				if (checkPrime(j)){
					if (checkPrime(number - j)){
						a = j;
						b = number - j;
					}
				}
			}
			cout << a << " " << b << endl;
		}
	}
	inStream.close();		
	return 0;
}
