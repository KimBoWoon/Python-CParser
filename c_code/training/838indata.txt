


#include <iostream>
#include <fstream>
using namespace std;

int fibo(int n)
{
	if(n<=1)
	{
		return n;
	}
	else
	{
		return fibo(n-1)+fibo(n-2);
	}
}

int main()
{
	fstream in;
	in.open("input.txt");
	int numcases;
	in>>numcases;
	for(int i=0; i<numcases; i++){
		int n,m;
		in>>n;
		m=fibo(n);
		cout<<m<<endl;
	}
}
