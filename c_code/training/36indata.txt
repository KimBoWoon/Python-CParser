
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
	ifstream inStream;
	int cases;
	bool flag = true;

	inStream.open("input.txt");
	inStream>>cases;

	while( cases-- ){
		int num;
		inStream>>num;

		if( num != 2 && num % 2 == 0 ) cout<<"0"<<endl;
		else{
			flag = true;
			for(int i=3; i<=(int)sqrt(num); i+=2){
				if( num % i == 0 ){
					flag = false;
					break;
				}
			}
			if( flag )	cout<<"1"<<endl;
			else		cout<<"0"<<endl;
		}
	}

	return 0;
}
