#include <stdio.h>

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int inputNum = 0; 
	int divNum = 2; 
	int isPrime = 1; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d", &inputNum); 

		for (divNum=2; divNum <= inputNum / 2; divNum++) {
			isPrime = 1;
			if (inputNum % divNum == 0) {
				isPrime = 0; 
				break;
			}
			else
				continue;
		}

		printf("%d\n", isPrime);
	}

	fclose(f);
	return 0;
}
