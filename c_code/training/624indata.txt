#include<iostream>
#include<fstream>

using namespace std;

int fac(int n){
	if (n == 0)
		return 1;
	return n*fac(n - 1);
}
int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;
	int fact[13];
	for (int i = 0; i < 13; i++){
		fact[i] = fac(i);
	}
	while (cases--){
		int num;
		inFile >> num;
		cout << fact[num] << endl;
	}
	return 1;
}
