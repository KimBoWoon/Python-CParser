#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int DucciList[1000][15];

int check_loop(int T, int index)			
{
	int i, j;
	int isItLoop;
	int x = 0;

	for(i=0; i<T; ++i) x += DucciList[index][i];
	if(x == 0) return 0;

	for(i=0; i<index; ++i) {
		isItLoop = 1;
		for(j=0; j<T; ++j){
			if(DucciList[i][j] != DucciList[index][j]) isItLoop = 0;
		}
		if(isItLoop) break;
	}

	return isItLoop;
}

void make_ducci_sequence(int T, int index)	
{
	int i;

	for(i=0; i<T; ++i) {
		DucciList[index+1][i] = abs(DucciList[index][i] - DucciList[index][(i+1)%T]);
	}
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testNum, T;
	int i, j;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		fscanf(inFile, "%d ", &T);
		for(j=0; j<T; ++j) fscanf(inFile, "%d", &DucciList[0][j]);
		
		j = 0;
		while(1) {
			if(j==1000) { printf("%s", "ZERO\n"); break; }
			make_ducci_sequence(T, j);
			if(check_loop(T, j+1)){ printf("%s", "LOOP\n"); break; }
			++j;
		}
	}
	fclose(inFile);
}
