#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

int sosu[10000]={0,};							 
int output1=0, output2=0;						 

void sosu_save()
{
	int i, j, k=0;
	int count = 0;

	for(i=1; i<=10000; i++)
	{
		for (j = 2; j <= sqrt(i) + 1; j++)		  
		{
			if (i == 2)							  
				continue;
			else
			{
				if((i == 1) || (i % j == 0))	  
				{
					count++;
					break;
				}
			}
		}
		if (count == 0)							  
		{
			sosu[k]= i;
			k++;
		}
		if(i == 10000)							  
		{
			sosu[k] = 10001;
		}
		count = 0;
	}
}

int partition(int start, int end, int even_number)
{

	if(start > end)										
	{
		return 0;
	}
	if((sosu[start] + sosu[end]) > even_number)			
	{
		end--;
	}
	else if((sosu[start] + sosu[end]) < even_number)	
	{
		start++;
	}
	else if((sosu[start] + sosu[end]) == even_number)	
	{
		output1 = sosu[start];
		output2 = sosu[end];

		start++;
		end--;
	}
	partition(start, end, even_number);					

	return 0;
}
int main()
{
	int numcase;
	int even_number;
	int start, end;

	FILE * in_file;
	in_file = fopen("input.txt", "r");

	sosu_save();

	fscanf(in_file,"%d", &numcase);

	while(numcase--)
	{
		fscanf(in_file, "%d", &even_number);
		end=0;

		while(1)
		{
			if(even_number <= sosu[end])				
				break;
			end++;      
		}   
		end--;  
		start = 0;
		partition(start, end, even_number);
		printf("%d %d\n", output1, output2); 
	}
}
