#include <iostream>
#include <fstream>
using namespace std;

int prime[10001] = { 0,0,1,1 };

int main()
{
	ifstream in;
	in.open("input.txt");

	int t;
	in >> t;
	for (int i = 4; i <= 10000; i++)
	{
		for (int div = 2; div*div <= i; div++)
		{
			if (i%div == 0)
			{
				prime[i] = 0;
				break;
			}
			else prime[i] = 1;
		}
	}
	while (t--)
	{
		int n;
		in >> n;
		for (int i = n / 2; i>0; i--)
		{
			if (prime[i] == 1 && prime[n - i] == 1)
			{
				cout << i << " " << n - i << endl;
				break;
			}
		}
	}
	in.close();
}
