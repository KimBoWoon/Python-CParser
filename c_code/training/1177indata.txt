
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

int tsum(int n){      
	return n*(n+1)/2;
}
int tmax(int i){       
	int max=1;
	if(i!=1){
	while(i>tsum(max)){
		max++;
	}
	max-=1;
	}
	else
		max=1;
	return max;
}
int ureka(int i){
	int num; 
	num=i;
	int count=0;
	int max = tmax(num);
	int all[100] = { 0 };
	for(int j=1;j<=max;j++)
		all[j]=tsum(j);

		for (int a = 1; a <= max; a++)
			for (int b = 1; b <= max; b++)
				for (int c = 1; c <= max; c++)
					if (all[a] + all[b] + all[c] == num)
							count=1;
	
	return count;
}

int main(void){
	ifstream infile;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error"<<endl;
		exit(1);
	}
	int num;
	int x;
	infile >> num;
	for(int i=0;i<num;i++){
		infile >> x;
		cout << ureka(x) << endl;
	}

	return 0;
}
