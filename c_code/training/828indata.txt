
#include<stdio.h>
#include <stdlib.h>
int recursive(int a)
{
	if (a <= 1) 
	{
		return a;
	}
	else
	{
		return recursive(a - 2) + recursive(a - 1);
	}
}
int main()
{
	FILE *in_file;
	int i;
	int numTestCases;
	int num;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++)
	{
		fscanf(in_file, "%d", &num);

		printf("%d\n", recursive(num)); 
	}
	fclose(in_file);
}
