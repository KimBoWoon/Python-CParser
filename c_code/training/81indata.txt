#include <stdio.h>
#include <stdlib.h>


void main (void)
{
 FILE *in_file;
 int i,no_cases;

 in_file = fopen("input.txt","r");
 if (in_file == NULL) exit(1);

 fscanf(in_file,"%d",&no_cases);
 
 for(i=0; i < no_cases; i++)
 {
  int data;
  int prime = 1;
  int i;

  fscanf(in_file,"%d",&data);

  for(i=2; i*i<=data; i++)
  {
   if(data%i==0)
   {
    prime = 0;
    break;
   }
  }
  printf("%d\n",prime);
 }
 fclose(in_file);
}
