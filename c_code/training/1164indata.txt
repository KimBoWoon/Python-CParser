

#include <stdio.h>
#include <stdlib.h>

void main()
{
	FILE *file;
	int i, j, k, l;
	int test;
	int num;
	int a=0, b=0, c=0;
	int sum=0;

	file = fopen("input.txt", "r"); 
	if(file==NULL) exit(1);

	fscanf(file, "%d", &test);
	for(i=0; i<test; i++)
	{
		fscanf(file, "%d", &num);

		for(j=1; j<45; j++)
		{
			a = (j*(j+1))/2;

			for(k=1; k<45; k++)
			{
				b = (k*(k+1))/2;

				for(l=1; l<45; l++)
				{
					c = (l*(l+1))/2;

					sum = a + b + c;
					
					if(num == sum)
					{
						printf("1\n");
						break;
					}
					
				}
				if(num == sum)
					break;
			}
			if(num == sum)
					break;		
		}
		if(num != sum)
			printf("0\n");			
	}
	fclose(file);
}
