
#include <stdio.h>
#include <stdlib.h>

void swap(int *a, int *b)
{
   int temp = *a;
   *a = *b;
   *b = temp;
}
int GCD(int a, int b)
{
   if (a <= b)
      swap(&a, &b);
   if (b == 0)			
      return a;
   GCD(b, a%b);
}

int main()
{
   int gcd = 0, result_up = 0, result_down = 0;
   int testcase, i, GLC;
   int up1, down1, up2, down2;

   FILE * in_file;
   in_file = fopen("input.txt", "r");
   fscanf(in_file, "%d", &testcase);

   for (i = 0; i < testcase; i++)
   {
      fscanf(in_file, "%d %d %d %d", &up1, &down1, &up2, &down2);
      gcd = GCD(down1, down2);											
      GLC = down1 * down2 / gcd;										
      result_up = (up1 * (GLC / down1)) + (up2 * (GLC / down2));		
      result_down = GLC / GCD(result_up, GLC);							
      result_up /= GCD(result_up, GLC);									
      printf("%d %d\n", result_up, result_down);
   }
   fclose(in_file);
}
