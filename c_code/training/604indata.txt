#include<iostream>
#include<fstream>
using namespace std;

int calculate(int n)
{
	int i, fact = 1;
	for (i = 1; i <= n; i++)
		fact *= i;

	cout << fact << endl;
	return 0;
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testCase;
	instream >> testCase;

	for (int i = 0; i<testCase; i++)
	{
		int  num;
		instream >> num;
		calculate(num);
	}
}
