

#include<iostream>
#include<fstream>

using namespace std;
int gcd(int fst_input, int sec_input)
{
	if (sec_input == 0)
		return fst_input;

	else
		return gcd(sec_input, fst_input%sec_input);
}

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if (instream.fail())
	{
		cerr << "input file open failed";
		return 0;
	}

	int testcase;
	instream >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		int den;
		int nume;
		instream >> nume >> den;

		while (nume != 1)
		{
			int x = den / nume;
			if (nume%den != 0)
				x += 1;

			nume = nume*x - den;
			den = den*x;
			int g = gcd(nume, den);
			nume /= g;
			den /= g;
		}
		cout << den << endl;

	}


	return 0;
}
