
#include <fstream>
#include <iostream>
using namespace std;
int McCarthy(int n){
	if (n > 100) return n - 10;
	else return McCarthy(McCarthy(n + 11));
}


int main(){

	int caseNum = 0;
	ifstream inputFile("input.txt");

	inputFile >> caseNum;
	for (int i = 0; i < caseNum; i++){
		int testNum = 0;
		inputFile >> testNum;

		cout << McCarthy(testNum) << endl;
	}

	return 0;
}
