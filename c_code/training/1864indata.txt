



#include <stdio.h>
#include <stdlib.h>

char string[50] ={0};

void main(void)
{
	FILE *inFile;
	int i,j;
	int x,y,m;
	int numCases;
	int count ;
	int num_A,num_B;
	
	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0;i<numCases;i++)
	{
		fscanf(inFile, "%s ",string );

		num_A = 0;
		num_B = 0;
		count = 0;

		for(j=0;j<50;j++)
		{
			if(string[j] == '(')
			{
				num_A++;
				count++;
			}
			if(string[j]==')')
			{
				num_B++;
				count++;
			}
		}

		if(string[0] ==')' || string[count-1] =='(')
			printf("NO \n");
		else if(num_A == num_B)
			printf("YES \n");
		else
			printf("NO \n");



		count =0;
		for(j=0;j<50;j++)
		{
			string[j] = 0;
		}


	}
	
	
	
	fclose(inFile); 

}
