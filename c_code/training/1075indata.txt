#include <stdio.h>
#include <stdlib.h>
#include <math.h>
void copyArray(int* newArray,int* array,int size)
{
	int i = 0;
	for(i=0; i<size; i++)
		array[i]=newArray[i];
}
void checkDucci(int* array,int size)
{
	int i = 0;
	int j = 0;
	int newArray[size];
	for(i=0; i<1000; i++)
	{
		int m = 0;
		
		for(j=0; j<size; j++)
		{
			
			if(j!=size-1) newArray[j] = abs(array[j]-array[j+1]); 
			else newArray[j] = abs(array[j]-array[0]);
			
		}
		copyArray(newArray,array,size);
		for(m=0; m<size; m++) if(newArray[m]!=0) break;
		if(m==size)
		{
				printf("ZERO\n");
				return;
		}
	}
	printf("LOOP\n");

	
}

void main()
{
	int datacase = 0;
	int i = 0;
	FILE* infile = fopen("input.txt","r");

	fscanf(infile,"%d",&datacase);
	
	for(i=0; i<datacase; i++)
	{
		int* array = NULL;
		int size = 0;

		for(j=0; j<size; j++) fscanf(infile,"%d",array+j);
