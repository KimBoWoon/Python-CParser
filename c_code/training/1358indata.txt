

#include <iostream>
#include <fstream>

using namespace std;

int gcd(int a, int b) {
	if (b == 0) return a;
	else return gcd(b, a%b);
}
int fountain(int a, int b) {
	int res1, res2, g, d=2;
	if(a==1) return b;
	while(1) {
		if(a*d >= b) {
			res1 = a*d-b;
			res2 = b*d;
			g = gcd(res1, res2);
			a = res1/g;
			b = res2/g;
			if(a==1) break;
		}
		else d++;
	}
	return b;
}
int main(void) 
{
	int test, a, b;
	ifstream file;
	file.open("input.txt"); 
	file >> test;

	for(int i=0; i<test; i++) 
	{
		file>>a>>b;
		cout<<fountain(a,b)<<endl;
	}
	file.close();
	return 0;
}
