

#include <iostream>
#include <fstream>

using namespace std;


void IsPrimeNumber() {

int numTestCases; 
int testCase; 
int cnt = 0;


	ifstream inFile;

	inFile.open("input.txt");
	
	inFile >> numTestCases; 

	for(int i = 0; i < numTestCases; i++)
	{
		int cnt = 0;
		inFile >> testCase;
		
		for(int j = 2; j < testCase ; j++)
		{
			if((testCase % j) == 0)
			{
				cnt++;
			}
		}

		if(cnt == 0) cout << 1 << endl;
		else { cout <<  0 << endl;}

	}

}

int main()
{

	IsPrimeNumber();
	
}
