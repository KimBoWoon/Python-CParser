#include <stdio.h>
#include <string.h> 
#include <stdlib.h>

int fib(int n);

void main(void)
 {
	 FILE *in_file;
	int no_cases; 
	int i,n;
	
	
	in_file = fopen("input.txt","r");
	if(in_file == NULL) exit(1);
	fscanf(in_file,"%d",&no_cases);
	for(i=0; i<no_cases; i++){
		fscanf(in_file,"%d",&n);
		fib(n);
		printf("%d\n", fib(n));
		
	}
}

int fib(int n)
{
if (n <= 1) 
return n;
else 
return fib(n-1)+fib(n-2);
}
