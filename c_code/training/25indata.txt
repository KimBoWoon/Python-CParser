

#include<stdio.h>
#include<stdlib.h>

void main(void)
{
FILE *in_file;
int nocase=0,data=0;
int i=0,j=0;
int num=0;
in_file = fopen("input.txt","r");
if(in_file == NULL) exit(1);

fscanf(in_file, "%d", &nocase);

for(i=0; i<nocase; i++)
{
 fscanf(in_file, "%d", &data);
 for(j=2; j<data; j++)
 { 
	if(data%j==0)	
	{
		num=0;
		printf("%d\n",num);
		break;
	}
 }
 if(j==data){
	num=1;
	printf("%d\n",num);
 }
}

fclose(in_file);
}
