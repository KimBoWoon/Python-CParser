
#include <iostream>
#include <fstream>

using namespace std;

int factorial(int num);	
						

int main()
{
	ifstream inFile;
	inFile.open("input.txt");

	int numTestCase;	
	inFile >> numTestCase;
	
	while(numTestCase--)	
	{
		int data;	
		inFile >> data;

		cout << factorial(data) << endl;	
	}	

	inFile.close();
	return 0;
}

int factorial(int num)
{
	if (num == 0)	
		return 1;

	else
		return num * factorial(num - 1);	
}
