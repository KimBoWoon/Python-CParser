
#include <iostream>
#include <fstream>
using namespace std;
int number[45];
int main(){
	ifstream in;
	in.open("input.txt");
	int n,checknum,check;
	in >> n;
	number[1] = 1;
	for (int i = 2; i <= 44; i++){
		number[i] = number[i - 1]+i;
	}
	for (int i = 0; i < n; i++){
		in >> checknum;
		check = 0;
		for (int j = 1; j <= 44; j++){
			for (int k = 1; k <= 44; k++){
				for (int p = 1; p <= 44; p++){
					if ((number[j] + number[k] + number[p]) == checknum){
						check = 1;
						break;
					}
				}
				if (check == 1)
					break;
			}
			if (check == 1)
				break;
		}
		cout << check << endl;
	}
	return 0;
}
