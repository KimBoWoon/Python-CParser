#include <iostream>
#include <fstream>
using namespace std;

int main() {
	std::ifstream inStream; 
	int h, w, n, cases;
	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;

	for (int i = 0; i < cases; i++) {
		inStream >> h >> w >> n; 
		if (h == 1) cout << 100 + n << endl; 
		else
			cout << (n % h == 0 ? h % n * 100 + n / h : n%h * 100 + n / h + 1) << endl;
	}
	inStream.close();		
	return 0;
}
