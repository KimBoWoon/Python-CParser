

#include<iostream>
#include<fstream>
using namespace std;

int fac(int num)
{
	if(num==0)
		return 1;
	else
		return num*fac(num-1);
}
void set(char *arr,int size)
{
	for(int i=0;i<size;i++)
		arr[i]=97+i;	
}
void modify(char *arr, int size, int delNum)
{
	for(int i=0;i<size;i++)
		if(delNum==i)
		{
			cout<<arr[delNum];
			for(int j=delNum;j<size-1;j++)
				arr[j]=arr[j+1]; 
			break;
		}
}
void permutation(int charNum,int num)
{
	int temp = charNum-1;
	int *arr = new int[charNum]; 
	char *alphabet = new char[charNum]; 
	set(alphabet,charNum); 


	num-=1;
	for(int i = temp;i>=1;i--) 
	{
		arr[temp-i] = num/fac(i);
		num%=fac(i);
	}
	arr[temp]=0; 
	
	int tt = charNum;
	for(int i=0;i<charNum;i++)
	{
		modify(alphabet,tt--,arr[i]);		
	}
	cout<<endl;
	
	delete[] alphabet;
	delete[] arr;
}

int main()
{
	fstream instream("input.txt",ios::in);
	int testcase;
	instream>>testcase;

	while(testcase--)
	{
		int charNum,num; 
		instream>>charNum>>num;
		permutation(charNum,num);
	}

	instream.close();
	return 0;
}
