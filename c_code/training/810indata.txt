

#include<iostream>
#include<fstream>

using namespace std;

class Fibonacci
{
public:
	int Fibonacci_method(int num)     
	{
		if (num <= 1)         
			return num;
		else                  
			return Fibonacci_method(num - 1) + Fibonacci_method(num - 2);
	}
};



int main()
{
	ifstream inStream;
	int testcase;
	Fibonacci Fib;


	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << " input file open failed";
		return 0;
	}

	inStream >> testcase; 

	for (int i = 0; i < testcase; i++)
	{
		int testnum;

		inStream >> testnum;

		cout << Fib.Fibonacci_method(testnum) << endl;
	}


	return 0;

}
