

#include <iostream>
#include <fstream>
using namespace std;
#define sum(n) (n)*(n+1)/2 

bool Eureka(int value){
	int result;
	for (int i = 1; i < 45; i++){
		for (int j = 1; j < 45; j++){
			for (int k = 1; k < 45; k++){
				if ((sum(i) + sum(j) + sum(k)) == value){
					return true;
				}
				else
					result = 0;
			}
		}
	}
	if (result == 0)
		return false;
}

int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;

	for (int z = 0; z < testcases; z++){
		int value;
		in >> value;
		
		cout << Eureka(value) << endl;
	}

}
