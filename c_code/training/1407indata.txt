



#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)

void main(void){
	FILE* inFile;
	int testCase = 0, height = 0, width = 0, numberOfPeople=0, roomNumber=0, widthCount=1; 
	int i = 0;
	inFile = fopen("input.txt", "r");      
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase);   

	for (i = 0; i < testCase; i++)   
	{
		fscanf(inFile, "%d %d %d", &height, &width, &numberOfPeople); 
		for (; numberOfPeople > height; numberOfPeople -= height)   
		{
			widthCount++;
		}
		roomNumber += (numberOfPeople * 100 + widthCount); 
		printf("%d\n", roomNumber);  
		roomNumber = 0;  
		widthCount = 1;  
	}
	fclose(inFile);
}
