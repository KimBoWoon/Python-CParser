

#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int fibo(int n);

int main()
{
	int numOfCase = 0;	
	int n; 

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> n;

		cout << fibo(n) << endl;

	}
	input.close();
	return 0;
}

int fibo(int n){
	if (n == 0){
		return 0;
	}
	else if (n == 1){
		return 1;
	}
	else{
		return (fibo(n - 1) + fibo(n - 2));
	}
}
