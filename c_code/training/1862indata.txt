#include <iostream>
#include <fstream>
#include <stack>
#include <string>
using namespace std;
string PS;
int cnt = 0;
int vps(int len, int i, stack<int> s)
{
	if (i<len)
	{
		if (PS[i] == '(')
		{
			s.push(PS[i]);
			cnt++;
		}
		else
		{
			if (s.empty()) return cnt = -1;
			else
			{
				s.pop();
				cnt--;
			}
		}
		i++;
		return vps(len, i, s);
	}
	else return cnt;
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while (t--)
	{
		stack<int> s; 
		in >> PS; 
		int len = PS.size();
		cnt = 0;
		vps(len, 0, s);
		if (cnt != 0) cout << "NO" << "\n";
		else cout << "YES" << "\n";
	}
}
