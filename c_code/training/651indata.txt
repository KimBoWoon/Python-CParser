




#include<iostream>
#include<fstream>

int f(int n);

int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){

		int n;
		is >> n;
		
		std::cout<<f(n)<<std::endl;

	}

}


int f(int n){

	if ( n <=1 ) 
		return 1;

	else 
		return n*f(n-1);
}
