  



#include <iostream>
#include <fstream>
using namespace std;
int main(){
	ifstream inFile("input.txt");  
	int no_cases=0;
	inFile>> no_cases;
	for(int l =0; l < no_cases; l++){
		int num=0;
		inFile>>num;
		int arr[100]={0},sum=0,cnt=1;
		for(int i=1; i<100; i++){ 
		
			arr[cnt]=(i*(i+1))/2;
			if(arr[cnt]>=num)
				break;
			cnt++;
		}
		bool flag=0;
		for(int i=1; i<cnt; i++){
			for(int j=1; j<cnt; j++){
				for(int k=1; k<cnt; k++){
					if(arr[i]+arr[j]+arr[k]==num) {
						flag=1;
						break;
					}
				}
				if(flag==1)
					break;
			}
			if(flag==1)
				break;
		}
		if(flag==1)
		cout<<"1"<<endl;
		else
			cout<<"0"<<endl;
	}
	inFile.close();
	return 0;
}
