
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int isPrime(int a)
{
	int i;
	int divisor = 0;

	for (i = 1; i <= a; i++) 
	{
		if (a%i == 0){
			divisor++;

			if (divisor == 2) 
				break;
		}
	}

	if (i != a)
		return 0;
	else
		return 1;
}

void main(){
	FILE *in_file;
	int no_cases, no_data;
	int i;

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d", &no_data);
		
		printf("%d\n", isPrime(no_data));
	}

	fclose(in_file);
}
