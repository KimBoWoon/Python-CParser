

#include <iostream>
#include <fstream>

using namespace std;

int main(void){
	ifstream in;
	in.open("input.txt");
	
	int testcases;
	int H, W, N;
	int floor=0, room = 0;
	in >> testcases;
	for (int i = 0; i < testcases; i++){
		in >> H >> W >> N;

		if (N % H == 0)
			room = N / H;
		else
			room = N / H + 1;

		if (N % H == 0)
			floor = H * 100;
		else
			floor = N%H*100;

		cout << room + floor << endl;
	}
}
