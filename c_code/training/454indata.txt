

#include<iostream>
#include<fstream>
using namespace std;

int gcd(int num1, int num2)
{	
	if(num2==0) 
		return num1;
	else
		return gcd(num2,num1%num2); 
}

int main()
{
	fstream instream("input.txt",ios::in);
	int test;
	instream>>test;

	while(test--)
	{
		int num1,num2;
		instream>>num1>>num2;
		cout<<gcd(num1,num2)<<endl;
	}
}
