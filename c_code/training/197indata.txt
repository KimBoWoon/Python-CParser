#include<iostream>
#include<fstream>
using namespace std;

int McCarthy(int n){
	if (n > 100)	return n - 10;
	else{ 
		int m = McCarthy(n + 11);
		return McCarthy(m);
	}
}

int main(){
	ifstream inStream;
	int numCases;
	int number;

	inStream.open("input.txt");

	inStream >> numCases;

	for (int i = 0; i<numCases; i++){
		inStream >> number;

		cout << McCarthy(number) << endl;
	}

	inStream.close();
}
