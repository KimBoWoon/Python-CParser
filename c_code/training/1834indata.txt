


#include <stdio.h>
#include <string.h>

#define MAX 50

int count(char *str,int i,int cnt){
	if(i==strlen(str)||cnt<0) return cnt; 
	if(str[i]=='(') return count(str,++i,++cnt); 
	else return count(str,++i,--cnt); 
}

void Parenthesis(char *str){
	int i=0,cnt=0;

	if(count(str,i,cnt)==0) printf("YES\n"); 
	else if(count(str,i,cnt)!=0) printf("NO\n"); 
}

int main(){
	int cases,i;
	char str[MAX];
	FILE *infile=fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0;i<cases;i++){
		fscanf(infile,"%s", str);
		Parenthesis(str);
	}
	fclose(infile);
	return 0;
}
