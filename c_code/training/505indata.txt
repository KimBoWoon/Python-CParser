#include <iostream>
#include <fstream>

using namespace std;

int gcd(int a, int b) {
	if (b == 0)
		return a;
	else
		return gcd(b, a % b);
}

int main(void) {
	ifstream fin("input.txt");

	int nCase; fin >> nCase;
	
	while (nCase--) {
		int num1, num2; fin >> num1 >> num2;
		cout << gcd(num1, num2) << endl;
	}

	fin.close();
	return 0;
}
