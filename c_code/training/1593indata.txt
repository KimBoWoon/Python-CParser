#include <iostream>
#include <fstream>
using namespace std;
int inputArr[503];
int result[503][503];
int cal[503];
int minSum(int start, int end);
int main() {
    fstream in;
    in.open("input.txt");
    int numcases;
    in>>numcases;
    for(int i=0; i<numcases; i++)
    {
        int num;
        in>>num;
        for(int j=1; j<=num; j++)
        {
            in>>inputArr[j];
            cal[j]=cal[j-1]+inputArr[j];
        }
        for(int j=1; j<=num; j++)
        {
            result[j][j]=inputArr[j];
        }
        cout<<minSum(1, num)<<endl;
    }
    return 0;
}
int minSum(int start, int end)
{
    for(int i=1; i<=end-start; i++)
    {
        for(int j=1; j<end; j++)
        {
            int min=0;
            int temp=0;
            int k=i+j;
            if(k<=end){
                for(int p=j; p<k; p++)
                {
                    if(p==j && p+1==k)
                    {
                        temp=result[j][p]+result[p+1][k];
                    }
                    else if(p!=j && p+1 ==k)
                    {
                        temp=cal[k-1]-cal[j-1]+result[j][p]+result[p+1][k];
                    }
                    else if(p==j && p+1 != k)
                    {
                        temp=cal[k]-cal[j]+result[j][p]+result[p+1][k];
                    }
                    else{
                        temp=cal[k]-cal[j-1]+result[j][p]+result[p+1][k];
                    }
                    if(min==0)
                    {
                        min=temp;
                    }
                    else if(temp<min)
                    {
                        min=temp;
                    }
                }
                result[j][k]=min;
            }
        }
    }
    return result[start][end];
}
