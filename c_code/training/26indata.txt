
#include <iostream>
#include <cstdlib>
#include <fstream>
using namespace std;

int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int	number = 0;
	int cnt = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0;i < numTestCases;i++)
	{
		inStream >> number;
		for (int j = 2;j < number;j++)
		{
			if (number % j == 0)
			{
				cnt++;
				break;
			}
		}
		if (cnt >= 1)
			cout << "0" << endl;
		else
			cout << "1" << endl;
		cnt = 0;
	}
	

	inStream.close();
	return 0;
}
