

#pragma warning(disable : 4996) 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int RecursivePalindrome(char str[], int start, int end); 

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;
	char strData[1001];

	inFile = fopen("input.txt", "r"); 

	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &numCases); 


	for (i = 0; i<numCases; i++)
	{
		fscanf(inFile, "%s", &strData); 

		printf("%d\n", RecursivePalindrome(strData, 0, strlen(strData) - 1)); 
	}

	fclose(inFile); 
}

int RecursivePalindrome(char str[], int start, int end)
{
	if (start < end)
	{
		if (str[start] != str[end]) return 0;					
		RecursivePalindrome(str, start + 1, end - 1);
	}
	else return 1;
}
