


#include <stdio.h>
#include <stdlib.h>

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int j;
		int k;
		int H = 0;														
		int W = 0;														
		int N = 0;														
		int l = 0;														
		int R = 0;														


		fscanf(inFile, "%d", &H);										
		fscanf(inFile, "%d", &W);										
		fscanf(inFile, "%d", &N);										

		for(j=1; j<=W; j++)												
		{
			for(k=1; k<=H; k++)											
			{
				l++;
				if(N == l)												
				{
					R = k*100 + j;										
					break;
				}
			}
		}

		printf("%d\n", R);
	}

	fclose(inFile);														
}
