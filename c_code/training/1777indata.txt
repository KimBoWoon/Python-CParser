#include <cstdio>
#include <cstring>

int arr[20];

int getNumByRadix(int num, int radix) {
	int cnt = 0;

	while (num) {
		arr[cnt++] = num % radix;
		num /= radix;
	}

	return cnt;
}

bool isPalindrome(int cnt) {
	for (int i = 0; i < cnt; ++i, --cnt) {
		if (arr[i] != arr[cnt - 1]) {
			return false;
		}
	}
	return true;
}

void palindrome(int num) {
	for (int i = 2; i <= 64; ++i) {
		if (isPalindrome(getNumByRadix(num, i))) {
			printf("1\n");
			return;
		}
	}

	printf("0\n");
}

int main() {
	freopen("input.txt", "r", stdin);
	int nCase;
	scanf("%d", &nCase);

	while (nCase--) {
		memset(arr, 0, sizeof(arr));

		int num;
		scanf("%d", &num);

		palindrome(num);
	}
	
	return 0;
}
