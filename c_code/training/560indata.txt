
#include <iostream>
#include <fstream>

char str[12] = { 'a','b','c','d','e','f','g','h','i','j','k','l' };

char* make;

int Factorial(int num) {

	if (num <= 0) {
		return 1;
	}
	return num * Factorial(num - 1);
}

void seachStr(int n , int max) {

	if (max <= 0) {
		std::cout << std::endl;
		return;
	}
	int division = Factorial(max - 1);

	std::cout << make[n / division];
	for (int i = n / division; i < max - 1; i++) {
		make[i] = make[i+1];
	}
	make[max - 1] = '.';
	
	seachStr(n % division, max - 1);
}


int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num1, num2;

	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	while(numcase--) {

		fin >> num1 >> num2 ;

		make = new char[num1];
		for (int i = 0; i < num1; i++) make[i] = str[i];		
		seachStr(num2 - 1, num1);
	}
	fin.close();
}
