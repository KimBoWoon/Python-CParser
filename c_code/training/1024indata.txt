

#include<iostream>
#include<fstream>
#include<string>

using namespace std;

void ducci(int arr[], int num, int test);
int main()
{
	int numOfCase = 0; 
	int num = 0;	
	
	int arr[15];

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> num;
		for (int j = 0; j < num; j++){
			input >> arr[j];
		}
		ducci(arr, num, 1);
	}
	input.close();
	return 0;
}
void ducci(int arr[], int num, int test){
	int temp[15];
	for (int j = 0; j < num; j++){ 
		if (j < num - 1){
			temp[j] = arr[j] - arr[j + 1];
			if (temp[j] < 0){
				temp[j] = -temp[j];
			}
		}
		else{
			temp[j] = arr[j] - arr[0];
			if (temp[j] < 0){
				temp[j] = -temp[j];
			}
		}
	}
	arr = temp;
	int sum = 0;
	for (int j = 0; j < num; j++){
		sum += arr[j];
	}
	if (sum == 0){
		cout << "ZERO" << endl;
		return;
	}
	else{
		if (test == 1000){
			cout << "LOOP" << endl;
			return;
		}
		ducci(arr, num, test + 1);
	}
}
