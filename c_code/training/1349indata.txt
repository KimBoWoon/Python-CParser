#include <iostream>
#include <fstream>
#include <cstdlib>

int SearchHenly(int a, int b);
int gcd(int a, int b);
int main()
{
	std::fstream inStream;
	inStream.open("input.txt");
	if (inStream.fail())
	{
		std::cout << "input file error!!";
		exit(1);
	}
	int numcases;
	inStream >> numcases;
	for (int i = 0; i < numcases; i++)
	{
		int i_bunmo, i_bunja;
		inStream >> i_bunja >> i_bunmo;
		std::cout << SearchHenly(i_bunja, i_bunmo) << std::endl;
	}
	return 0;
}

int SearchHenly(const int a, const int b)
{
	if (a == 1)
		return b;

	int henlybunmo, henlybunja;
	int cmpnum = (b / a) + (b%a ==0 ? 0 : 1);

	henlybunja = a*cmpnum - b;
	henlybunmo = b*cmpnum;

	int dividenum = gcd(henlybunja, henlybunmo);
	henlybunja /= dividenum;
	henlybunmo /= dividenum;
	
	return SearchHenly(henlybunja, henlybunmo);
}
int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
