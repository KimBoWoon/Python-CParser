
#include <iostream>
#include <fstream>
#include <list>

using namespace std;

int fact(int n) {
	return (n > 0) ? fact(n - 1) * n : 1;
}

int main(void) {
	ifstream in("input.txt");

	int t;

	in >> t;

	for (int testcase = 0; testcase < t; testcase++) {
		int n, cnt;


		in >> n >> cnt;

		list<char> alphabets;
		for (int i = 0; i < n; i++) alphabets.push_back('a' + i);	
		
		list<char> result; 
		int flagnum = fact(n);
		int sum = 0;
		for (int i = n; i >= 1; i--) {
			int divide = flagnum / i;

			for (list<char>::iterator it = alphabets.begin(); it != alphabets.end(); it++) {
				if (sum <= cnt && cnt <= sum + divide) {
					result.push_back(*it);
					alphabets.erase(it);
					break;
				}
				else {
					sum += divide;
				}
			}

			flagnum /= i;
		}


		for (list<char>::iterator it = result.begin(); it != result.end(); it++) {
			cout << *it;
		}
		cout << endl;
	}

	return 0;
}
