#include <iostream>
#include <fstream>

int gcd(int inputNum1, int inputNum2)
{
	if (inputNum2 == 0)
		return inputNum1;
	else
		gcd(inputNum2, inputNum1%inputNum2);
}

int main()
{
	std::ifstream file("input.txt");
	int inputNum1 = 0, inputNum2 = 0, testCase = 0;
	file >> testCase;
	while (testCase--)
	{
		file >> inputNum1>>inputNum2;
		std::cout << gcd(inputNum1,inputNum2) << std::endl;
	}
}
