#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int Ducci[1000][15];

int steps(int T, int index)
{
	int i, j;
	int check;
	int x = 0;

	for(i=0; i<T; ++i) x += Ducci[index][i];
	if(x == 0) return 0;

	for(i=0; i<index; ++i) {
		check = 1;
		for(j=0; j<T; ++j){
			if(Ducci[i][j] != Ducci[index][j]) check = 0;
		}
		if(check) break;
	}

	return check;
}

void sequence(int T, int index)
{
	int i;

	for(i=0; i<T; ++i) {
		Ducci[index+1][i] = abs(Ducci[index][i] - Ducci[index][(i+1)%T]);
	}
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testcase, T;
	int i, j;

	fscanf(inFile, "%d", &testcase);
	for(i=0; i<testcase; ++i){
		fscanf(inFile, "%d ", &T);
		for(j=0; j<T; ++j) fscanf(inFile, "%d", &Ducci[0][j]);
		
		j = 0;
		while(1) {
			if(j==1000) {
			    printf("%s", "ZERO\n"); break; 
			}
			sequence(T, j);
			if(steps(T, j+1)) {
			    printf("%s", "LOOP\n"); break;
			}
			++j;
		}
	}
	fclose(inFile);
}
