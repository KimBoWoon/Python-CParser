#include <stdio.h>
#include <stdlib.h>
int recursiveFactorial(int num);

void main()
{
    FILE *f;
    int count;        
    int i;            
    int number;    

    f = fopen("input.txt", "r");    
    if(f==NULL)    exit(1);    

    fscanf(f, "%d", &count);    

    for(i=0; i<count; i++)
    {
        int result;    
        fscanf(f, "%d", &number);    
        result = recursiveFactorial(number);    
        printf("%d\n", result);    
    }
}
int recursiveFactorial(int num)
{
    if(num <= 1)    
        return 1;
    else    
        return num*recursiveFactorial(num-1);
}
