
#include<iostream>
#include<fstream>
using namespace std;

int gcd(int a, int b);

int main()
{
	int count = 0;

	ifstream file;
	file.open("input.txt");
	file >> count;

	int a=0, b=0;
	for (int i = 0; i < count; i++)
	{
		file >> a >> b;
		cout<<gcd(a, b)<<endl;
	}

}

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
