


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int Tri(int n); 
int main()
{
	ifstream inStream;
	int numTestCases;
	int firstArray[45];
	int secondArray[45];
	int thirdArray[45];
	int testNum = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases; 
	
	for (int i = 0; i < 45; i++)
	{
		firstArray[i] = secondArray[i] = thirdArray[i] = Tri(i); 
	}

	for (int i = 0; i < numTestCases; i++) 
	{
		
		int check = 0;
		
		inStream >> testNum;
		
		for (int a = 0; a < 45; a++)
		{
			for (int b = 0; b < 45; b++)
			{
				for (int c = 0; c < 45; c++)
				{
					int x = firstArray[a] + secondArray[b] + thirdArray[c]; 
					if (x == testNum) 
					{
						check = 1; 
						a = b = c = 45; 
					}
				}
			}
		}

		cout << check << endl;

	}
	return 0;
	inStream.close();

}

int Tri(int n) 
{
	if (n <= 1)
		return 1;
	else
		return Tri(n - 1) + n;
}
