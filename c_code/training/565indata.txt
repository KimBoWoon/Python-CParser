

#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int factorial(int inputNum)
{
	if (inputNum <= 1)
		return 1;
	else
		return inputNum*factorial(inputNum - 1);
}

void perm(char arr[], int num1, int num2)
{
	int fact, res, first;
	for (int i = 0; i < num1; i++) {
		fact = factorial(num1 - i - 1);
		res = num2 / fact;

		if (num2%fact == 0)
			res = res - 1;

		for (int j = i + res; j > i; j--) {
			first = arr[j - 1];
			arr[j - 1] = arr[j];
			arr[j] = first;
		}
		num2 = num2 - (fact*res);
	}

	if (num2 == 2) {
		first = arr[num1 - 2];
		arr[num1 - 2] = arr[num1 - 1];
		arr[num1 - 1] = first-1;
	}
}
int main()
{
	ifstream instream("input.txt");
	int numtestcases, num1, num2;
	char alp[12] = { 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l' };
	char str[12];
	instream >> numtestcases;
	for (int i = 0; i < numtestcases; i++){
		instream >> num1 >> num2;
		for (int i = 0; i < num1; i++)
			str[i] = alp[i];

		perm(str, num1, num2);
	
		for (int i = 0; i<num1; i++) {
			cout << str[i];
		}
		cout << endl;
	}
	instream.close();
}
