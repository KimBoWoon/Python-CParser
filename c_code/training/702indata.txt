
#include<iostream>
#include<string>
#include<fstream>

using namespace std;

int palindrome(int startNum, int endNum, string strBuff){
	if (startNum >= endNum){
		return 1;
	}
	else if (strBuff[startNum] != strBuff[endNum]){
		return 0;
	}
	else{
		return palindrome(startNum+1, endNum-1, strBuff);
	}
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, startNum = 0, endNum = 0;
	string strBuff;

	f >> numCount;
	for (int i = 0;i<numCount;i++){
		f >> strBuff;
		endNum = strBuff.length()-1;
		cout << palindrome(startNum, endNum, strBuff) << endl;
	}
	return 0;
}
