

#include <stdio.h>
#include <stdlib.h>

int mccarthy(int n);

int main(void) {
	FILE *input;
	int no_cases;
	int i;
	int data;

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d", &data);
		printf("%d\n", mccarthy(data));
	}

	fclose(input);
	return 0;
}

int mccarthy(int n) {
	if (n > 100)
		return n - 10;
	else
		return mccarthy(mccarthy(n + 11));
}
