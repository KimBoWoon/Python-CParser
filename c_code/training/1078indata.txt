
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

using namespace std;

void ducci(fstream &infile, int num)
{
	int snum[1000][15];
	int count = 0;
	int count1 = 0;
	int count2 = 0;
	int count3 = 0;
	int num2 = 0;

	for(int i = 0; i<num; i++)
		infile >> snum[0][i];

	while(1)
	{
		if(count1 < num-1)
		{
			snum[count+1][count1] = abs(snum[count][count1]-snum[count][count1+1]);
			count1 ++;
		}

		else if(count1 == num-1)
		{
			snum[count+1][count1] = abs(snum[count][0] - snum[count][count1]);

			for(int i = 0; i<num; i++)
			{
				if(snum[count+1][i] == 0)
				{
					count2 ++;
				}
			}

			if(count2 == num)
			{
				cout << "ZERO" << endl;
				break;
			}

			else
				count2 = 0;

			for(int i = 0; i < count; i++)
			{
				for(int j = 0; j < num; j++)
				{
					if(snum[i][j] == snum[count+1][j])
					{
						count3 ++;
					}
				}

				if(count3 == num)
				{
					cout << "LOOP" << endl;
					num = 1;
					break;
				}

				else
					count3 = 0;
			}

			if(num == 1)
				break;
			
			count1 = 0;
			count++;
		}
	}
}
int main()
{
	fstream infile;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "input file error" << endl;
		exit(1);
	}
	int cases;
	int num;
	infile >> cases;

	for(int i = 0; i < cases; i++)
	{
		infile >> num;
		ducci(infile,num);
	}

	return 0;
}
