#include <stdio.h>

int gcd(int a, int b){
	if (b == 0) 
		return a;
	else		
		return gcd(b, a%b);
}

void Henry(int a,int b){
	int n,g;
	if(a<=1) printf("%d\n",b);
	else{
		n=(b-1)/a+1;
		a=a*n-b;
		b=b*n;
		g=gcd(a,b);
		Henry(a/g,b/g);
	}
}

int main()
{
	FILE *file=fopen("input.txt","r");
    int cases,a,b;

	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);

    while(cases--){
        fscanf(file,"%d %d",&a,&b);
		Henry(a,b);
    }
}
