#include <stdio.h>
#include <stdlib.h>	
#define MAX 1000
#define TRUE 1
#define FALSE 0

int is_zeros_tuple(int *a, int n);

int main() {
	FILE *fp = fopen("input.txt", "r");
	int testCaseNumber = 0, testCases = 0, idx = 0;
	int isZero = 0;
	int *pAry = 0;

	fscanf(fp, "%d", &testCaseNumber);
	while (testCaseNumber--) {
		isZero = 0; 
		fscanf(fp, "%d", &testCases);
		pAry = (int*)malloc(sizeof(int) * testCases);
		
		for (idx = 0; idx < testCases; idx++) {
			fscanf(fp, "%d", &pAry[idx]);
			isZero += pAry[idx];
		}
		isZero == 0 ? printf("ZERO\n") : 
			is_zeros_tuple(pAry, testCases) == TRUE ? printf("ZERO\n") : printf("LOOP\n");
	}
	free(pAry);
	fclose(fp);
}

int is_zeros_tuple(int *a, int n) {
	int i = 0, j = 0, k = 0, a0 = 0;
	
	for (i = 0; i < MAX; i++) {
		k = 0; 
		a0 = a[0]; 

		for (j = 0; j < n - 1; j++) {
			a[j] = (a[j] - a[j + 1] > 0 ? a[j] - a[j + 1] : a[j + 1] - a[j]);
		}
		a[n - 1] = (a[n - 1] - a0 > 0 ? a[n - 1] - a0 : a0 - a[n - 1]); 
		for (j = 0; j < n; j++) {
			k += a[j];
		}

		if (k == 0) return TRUE;
	}
	
	return FALSE;
}
