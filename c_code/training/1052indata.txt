

#include<iostream>
#include<fstream>
#include<cstdlib>

using namespace std;

bool Ducci(int *arr, int number);

int main(void)
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {

		int number;
		bool end = false;
		inStream >> number;
		int *arr = new int[number];
		
		for (int j = 0; j < number; j++)
			inStream >> arr[j];

			for (int j = 0; j < 1000; j++){
				if (Ducci(arr, number) == true){
					cout << "ZERO" << endl;
					end = true;
					break;
				}
			}
			if (end == false)  cout << "LOOP" << endl;

	}
}

bool Ducci(int *arr, int number){

	int temp = arr[0];
	for (int i = 0; i < number-1; i++){
		arr[i] = abs(arr[i] - arr[i + 1]);
	}
	arr[number - 1] = abs(arr[number - 1] - temp);
	
	for (int i = 0; i < number; i++){
		if (arr[i] != 0) return false;
	}
	return true;

}
