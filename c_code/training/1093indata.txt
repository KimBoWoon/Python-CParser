
#include <fstream>
#include <iostream>
using namespace std;
int eureka[50];

void setIndex(int* index1, int* index2, int* index3, int testNum){
	int eurekaNum = 1, index = 2;
	while (testNum != eurekaNum)
	{
		eureka[(*index1)++] = eurekaNum;
		eurekaNum += index++;

		if (eurekaNum <= (testNum / 2))	(*index2)++;
		if (eurekaNum <= (testNum / 3)) (*index3)++;
		if (eurekaNum > testNum) return;
	}
}

int isEureka(int index1, int index2, int index3, int testNum){
	for (int i = 0; i <= index1; i++){
		for (int j = 0; j <= index2; j++){
			for (int k = 0; k <= index3; k++){
				if (eureka[i] + eureka[j] + eureka[k] == testNum){
					return 1;
				}
				else if (eureka[i] + eureka[j] + eureka[k] > testNum){
					break;
				}
			}
		}
	}
	return 0;
}


int main(){

	int caseNum = 0;

	ifstream inputFile("input.txt");
	inputFile >> caseNum;

	for (int i = 0; i < caseNum; i++){
		int testNum = 0;
		int index1 = 0, index2 = 0, index3 = 0;
		inputFile >> testNum;

		setIndex(&index1, &index2, &index3, testNum);
		cout << isEureka(index1, index2, index3, testNum) << endl;
		
		
	}

	return 0;
}
