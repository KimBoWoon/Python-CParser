#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");
	int testcase;
	instream >> testcase;
	for(int z=0; z<testcase; z++)
	{
		int number;
		int rest[20];
		
		instream >> number;
		int flag = 0;
		
		for(int divisor=2; divisor<=64; divisor++)
		{
			int a=0;
			int tempnumber = number;
			while(tempnumber != 0) 
			{
				rest[a] = tempnumber % divisor;
				tempnumber /= divisor;
				a++;
			}
			int count = 0;
			for(int i=0; i<a/2; i++) 
			{
				if(rest[i] == rest[a-1-i])
				{
					flag=1;
				}
				else
				{
					flag=0;
					break;
				}

				
			}
			if(flag ==1)
				break;
			
		}
		cout << flag <<  endl;
	}
}
