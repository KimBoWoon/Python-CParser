#include <stdio.h>
#include <stdlib.h>
#include <string.h>

static char str[1001] = { 0, };

int palin(int start, int end)
{
	if (start >= end)						
		return 1;							
	else if (str[start] == str[end])
		return palin(start + 1, end - 1);
	else
		return 0;
}

int main(void)
{
	FILE * in_file;
	int numTestCase;
	int i = 0;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &numTestCase);
	for (i = 0; i < numTestCase; i++)
	{
		int len = 0;

		fscanf(in_file, "%s", str);
		len = strlen(str);

		printf("%d\n", palin(0, len - 1));
		memset(str, 0, 1001);
	}
	fclose(in_file);
}
