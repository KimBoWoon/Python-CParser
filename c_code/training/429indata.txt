


#include <iostream>
#include <cstdlib>
#include <fstream>
using namespace std;

int gcd(int n1, int n2)
{
	if (n2 == 0)
		return n1;
	else
		return gcd(n2,n1 % n2);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num1 = 0, num2 = 0, result = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> num1 >> num2; 
		if (num1 >= num2)
			result = gcd(num1, num2);
		else
			result = gcd(num2, num1);
		cout << result << endl;
	}
	inStream.close();
	return 0;
}
