#include <iostream>
#include <fstream>

using namespace std;

int facto(int num) {
	if (num == 0)
		return 1;
	else
		return num * facto(num - 1);
}

int main(void) {
	ifstream fin("input.txt");

	int nCase; fin >> nCase;
	while (nCase--) {
		int num; fin >> num;
		cout << facto(num) << endl;
	}

	fin.close();
	return 0;
}
