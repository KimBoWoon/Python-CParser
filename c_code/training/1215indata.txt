

#include<stdio.h>

int count = 0;

void makePrime(int num){
	if ((checkPrime(num - count) == 1) && (checkPrime(num + count) == 1)){
		printf("%d %d\n", num - count, num + count);
	}
	else {
		count++;
		makePrime(num);
	}
 }

int checkPrime(int num){
	int i;
	if (num < 4) { 
		return 1;
	}
	for (i = 2; i <= num / 2; i++){ 
		if (num%i == 0) { 
			return 0;
		}
		if (i == num / 2)
			return 1;
	}

}

int main(){

	FILE *inFile;
	int circle, num;

	inFile = fopen("input.txt", "r"); 
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d", &num);
		count = 0;
		makePrime(num / 2);
	}
	return 0;
}
