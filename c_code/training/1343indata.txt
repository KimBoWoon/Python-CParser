


#include <stdio.h>

int Euclid(int num1, int num2){
	if(num2==0) return num1;	
	else return Euclid(num2, num1%num2);	
}

void Sub(int *n1,int *n2,int m1,int m2){
	int k1,k2,divisor;

	k1=(*n1)*m2-m1*(*n2); 
	k2=(*n2)*m2; 
	divisor = Euclid(k2,k1); 
	*n1=k1/divisor;
	*n2=k2/divisor;
}

void Henry(int num,int den){
	int i;
	for(i=2;;i++){
		if(num==1){
			printf("%d\n",den); 
			return;
		}
		if(num*i > den) 
			Sub(&num,&den,1,i); 
	}
}

int main(){
	int cases,numerator,denominator,i;
	FILE *infile=fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0;i<cases;i++){
		fscanf(infile,"%d %d",&numerator,&denominator);
		Henry(numerator,denominator);
	}
	fclose(infile);
	return 0;
}
