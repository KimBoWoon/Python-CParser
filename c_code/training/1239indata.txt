#include <stdio.h>
#include <stdlib.h>

int IsPrime(int n);					
void GoldbachsConjecture(int n);	

void main(void)
{
	FILE *inFile;
	int i;
	int numCases, numData;

	inFile = fopen("input.txt", "r"); 

	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &numCases); 
	for (i = 0; i<numCases; i++)
	{
		fscanf(inFile, "%d", &numData); 

		GoldbachsConjecture(numData);
	}

	fclose(inFile); 
}

int IsPrime(int n)
{
	int i;
	int flag = 1;

	for (i = 2; i*i <= n; i++)
		if (n % i == 0) { flag = 0; break; }	

	if (flag) return 1;
	else return 0;
}

void GoldbachsConjecture(int n)
{
	int i, j;
	int minI, minJ;

	for (i = 2; i <= n / 2; i++)
		for (j = 2; j < n; j++)
			if (IsPrime(i) && IsPrime(j) && (n == i + j) && i <= j)
			{
				minI = i; minJ = j;
			}
	printf("%d %d\n", minI, minJ);
}
