#include <stdio.h>
#include <stdlib.h>

int prime[200];

int seekPrime ()
{
	int i = 0;
	int j = 0;
	int n = 1;
	prime[0]=2;
	for(i=3; i<1000; i++)
	{
		for(j=2; j<i; j++)
		{
			if(i%j==0) break;
		}
		if(i==j) prime[n++]=i;
	}
	return n;
}
void gold(int a,int primeSize)
{

	int i = 0;
	int j = 0;
	for(i=0; i<primeSize; i++)
	{
		for(j=i; j>=0; j--)
		{
			if(a-prime[i]-prime[j]==0)  
			{
				printf("%d %d\n",prime[j],prime[i]);
				return;
			}
		}
	}

}	

void main()
{
	int datacase = 0;
	int i = 0;
	int primeSize = seekPrime();
	FILE* infile = fopen("input.txt","r");	
	fscanf(infile,"%d",&datacase);
	
	for(i=0; i<datacase; i++)
	{
	
		int even = 0;
		fscanf(infile,"%d",&even);
		gold(even,primeSize);
	
		
	}
}
