
#include <iostream>
#include <fstream>

using namespace std;

int euclid(int a, int b){

	if(b==0)
		return a;
	else
		euclid(b, a%b);
}
int main(){

	ifstream inStream;
	inStream.open("input.txt");		

	int num_case;	

	inStream >> num_case;

	for(int i=0; i< num_case ; i++){

		int a,b;		

		inStream >> a >> b;		
		cout<<euclid(a,b)<<endl;
	}

}
