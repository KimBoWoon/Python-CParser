#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void pop(char str[],int index,int count);
int fact(int a);

int main()
{
	int i,j,k,numTestCases;
	FILE *fp;
	fp=fopen("input.txt","r");
	if(fp==NULL) exit(1);
	fscanf(fp,"%d",&numTestCases);

	for(i=0; i<numTestCases; i++){
		int number,count,newCount,newNumber;
		int letter; 
		int rest;
		char str[25]={'a','b','c','d','e','f','g','h','i','j','k','l'};
		char answerStr[25]={' '};
		fscanf(fp,"%d %d",&number,&count);
		
		newNumber=number;
		newCount=count-1;
		for(j=0; j<number; j++){
			letter=(newCount)/fact(newNumber-1);
			answerStr[j]=str[letter];
			pop(str,letter,newNumber);
			rest=(newCount)%fact(newNumber-1);
			newNumber--;
			newCount=rest;
			printf("%c",answerStr[j]);
		}
		printf("\n");
	}

	return 0;
	fclose(fp);
}

int fact(int a) 
{
	if(a<=1)
		return 1;
	else
		return a*fact(a-1);
}
	
void pop(char str[], int index,int count)
{
	int i;
	for(i=index; i<count; i++){
		str[i]=str[i+1];
	}
}
