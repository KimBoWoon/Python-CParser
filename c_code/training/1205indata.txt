#include <iostream>
#include <fstream>
using namespace std;
bool isPrime(int n)
{
    for(int i=2;i*i<=n;i++)
        if(n%i==0)return false;
    return true;
}
int main()
{
	ifstream fs;
	fs.open("input.txt");
    int n,k;
    fs>>n;
    while(n--)
    {
        fs>>k;
        for(int i=k/2;i>1;i--)
        {
            if(isPrime(i)&&isPrime(k-i))
            {
                cout<<i<<" "<<k-i<<"\n";
                break;
            }
        }
    }
}
