



#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int gcd(int a, int b);

int main() {
    
    FILE* fp;
    
    int i, num_cases, a1, b1, a2, b2, c1, c2, d;
    
    fp = fopen("input.txt", "r");
    if(fp == NULL)  exit(1);
    
    fscanf(fp, "%d", &num_cases);
    
    for(i=0; i<num_cases; i++) {
        fscanf(fp, "%d %d %d %d", &a1, &b1, &a2, &b2);
        
        c1 = a1*b2 + a2*b1;
        c2 = b1*b2;
        
        d = gcd(c1, c2);
        
        printf("%d %d\n", c1/d, c2/d);
    }
    
    fclose(fp);
    
}

int gcd(int a, int b) {
    
    if(b == 0)
        return a;
    else
        return gcd(b, a%b);
}
