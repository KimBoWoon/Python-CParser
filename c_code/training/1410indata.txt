




#include<stdio.h>
#include<stdlib.h>

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		int H,W,B;
		int a,b;

		fscanf(inFile, "%d%d%d", &H, &W, &B);

		a = (B-1) % H+1;
		b = (B-1) / H+1;

		printf("%d%02d\n", a,b);

	}

	fclose(inFile);
}
