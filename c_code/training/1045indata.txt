

#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;
int flag=0;

int check(int **arr,int * compare,int num, int row) 
{
	int count=0;
	int zeroCount=0;
	
	for(int i=0;i<row;i++)
	{	for(int j=0;j<num;j++)
		{
			if(arr[i][j]==compare[j]) 
				count++;
			if(arr[i][j]==0)
				zeroCount++;
			
			if(zeroCount==num)
			{	
				cout<<"ZERO"<<endl;
				flag=1;
				return 0;
			}
			if(count==num)
			{	
				cout<<"LOOP"<<endl;
				flag=1;
				return 0;
			}
		}
		count=0;
		zeroCount=0;
	}
}

void ducci(int **arr, int num)
{
	int *compare = new int[num];
	for(int i=1;i<1000;i++)
	{
		if(flag==1)
			break;
		for(int j=0;j<num;j++)
		{
			if(j==(num-1)) 
			{
				int temp = abs(arr[i-1][j]-arr[i-1][0]);
				arr[i][j] = temp;
			}
			else if(j<num-1) 
			{
				int temp =  abs(arr[i-1][j]-arr[i-1][j+1]);
				arr[i][j] = temp;
			}
			compare[j]=arr[i][j]; 
		}
		check(arr,compare,num,i);
	}

	delete[] compare;
}

int main()
{
	ifstream instream("input.txt",ios::in);
	int testCase;
	instream>>testCase;
	while(testCase--)
	{
		int num; 
		instream>>num;
		
		int **ducciArr = new int*[1000];
		for(int i=0;i<1000;i++)
			ducciArr[i] = new int[num];

		flag=0;
		for(int i=0;i<num;i++)
			instream>>ducciArr[0][i];
		ducci(ducciArr, num);
	
		for(int i=0;i<1000;i++)
			delete[] ducciArr[i];
		delete[] ducciArr;
	}
	
	
	instream.close();
	return 0;
}
