
#include <iostream>
#include <fstream>
#include <cstring> 

using namespace std;

bool palindrome(char str[], short first, short last) 
{
	if(last <= first)
		return true;
	else if(str[first] != str[last])
		return false;
	else
		return palindrome(str, first+1, last-1); 
}

int main()
{
	ifstream in;
	in.open("input.txt");

	short t;
	in>>t;

	char string[1001];

	while(t--){
		in>>string;
		cout<<palindrome(string, 0, strlen(string)-1)<<endl;
	}
	in.close();
}
