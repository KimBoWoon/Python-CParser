

#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int gcd(int num1, int num2);
void reduc(int* num1, int* num2);

int main()
{
	int numOfCase = 0;	
	int a1, a2;	
	int b1, b2;	
	int sum1, sum2;

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> a1;
		input >> a2;
		input >> b1;
		input >> b2;

		reduc(&a1, &a2);
		reduc(&b1, &b2);

		sum2 = a2*b2; 
		sum1 = a1*b2 + b1*a2;

		reduc(&sum1, &sum2);

		cout << sum1 << " " << sum2 << endl;

	}
	input.close();
	return 0;
}
int gcd(int num1, int num2)
{
	int a, b;
	if (num1 >= num2){
		a = num1;
		b = num2;
	}
	else{
		a = num2;
		b = num1;
	}
	int r;
	if (b == 0){
		return a;
	}
	do
	{
		r = a%b;
		a = b;
		b = r;
	} while (r != 0);

	return a;
}
void reduc(int *num1, int *num2){
	int a = gcd(*num1, *num2); 
	*num1 /= a;
	*num2 /= a;
}
