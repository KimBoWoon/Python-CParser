
#include <iostream>
#include <fstream>

using namespace std;

int m(int n){

	if(n > 100)
		return n-10;
	else
		return m(m(n+11));
}

int main(){

	ifstream inStream;
	inStream.open("input.txt");		

	int num_case;	

	inStream >> num_case;

	for(int i=0; i< num_case ; i++){
		int test;
		inStream >> test;

		cout<< m(test) <<endl;
	}
}
