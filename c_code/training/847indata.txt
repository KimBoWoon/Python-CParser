#include <fstream>

int fibonacci(int n);

int main() {
	std::ifstream inFile;
	inFile.open("input.txt");

	int testCaseNum;

	inFile >> testCaseNum;

	for(int i = 0; i < testCaseNum; i++) {
		int testCase;
		inFile >> testCase;

		std::cout << fibonacci(testCase) << std::endl;
	}
}

int fibonacci(int testCase) {
	if (testCase == 0)
		return 0;
	else if (testCase == 1)
		return 1;
	else
		return fibonacci(testCase - 1) + fibonacci(testCase - 2);
}
