
#include<iostream>
#include<fstream>

using namespace std;

int fibo(int n);

int main(void){
    
    ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {
	    
	    int n;
	    inStream >> n;
	    cout<<fibo(n)<<endl;
	}
    
    
    
    
}
int fibo(int n){
    
    if( n <1)
        return 0;
    else if(n < 2)
        return 1;
    else 
        return fibo(n-1) + fibo(n-2);
}
