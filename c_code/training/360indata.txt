

#include<iostream>
#include<fstream>
using namespace std;

int roomCount=1; 

void sort(int *arr,int size)
{
	int temp;
	
	for(int i=1;i<=size;i++)
		for(int j=i+1;j<=size;j++)
		{
			if(arr[i]<arr[j])
			{
				temp = arr[i];
				arr[i] = arr[j];
				arr[j] = temp;
			}
		}
}

void sizePrint(int *arr,int size)
{
	sort(arr,size);
	for(int i=1;i<=size;i++)
			cout<<arr[i]<<" ";
		cout<<endl;
}

void roomCheck(char **room,int *roomSize,int x, int y, char cross, char dot)
{
	if(room[x][y]!='.')
		return;
	else
	{ 
		room[x][y]=(char)roomCount;
		roomSize[roomCount]++;
		roomCheck(room,roomSize,x-1,y,cross,dot);
		roomCheck(room,roomSize,x+1,y,cross,dot);
		roomCheck(room,roomSize,x,y+1,cross,dot);
		roomCheck(room,roomSize,x,y-1,cross,dot);
	}
}

int main()
{
	fstream instream("input.txt",ios::in);
	int testcase;
	instream>>testcase;
	int row,col;
	char** room;
	char lastNum;

	while(testcase--)
	{
		int roomSize[100]={0}; 
		instream>>col>>row;
		room = new char*[row]; 
 		for(int i=0;i<row;i++)	
			room[i] = new char[col];

		for(int i=0;i<row;i++)	
			for(int j=0;j<col;j++)
				instream>>room[i][j];

		for(int i=0;i<row;i++)
			for(int j=0;j<col;j++)
			{
				if(room[i][j]=='.')
				{
					roomCheck(room,roomSize,i,j,'+','.');
					roomCount++;
				}
			}
		lastNum=roomCount-1; 
		roomCount=1; 
			
		cout<<(int)lastNum<<endl;
		sizePrint(roomSize,(int)lastNum);
	}
	
	instream.close();

	for(int i=0;i<row;i++)
		delete[] room[i];
	delete[] room;

	return 0;
}
