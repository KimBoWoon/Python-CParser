

#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b) {
	if (b == 0) return a;
	else return gcd(b, a%b);
}
void main()
{
	int test_case, a, b, i;
	FILE *file = fopen("input.txt", "r");

	fscanf(file, "%d", &test_case);

	for (i = 0; i < test_case; i++) {
		fscanf(file, "%d %d", &a, &b);

		printf("%d\n", gcd(a, b));
	}
	fclose(file);
}
