


#include <iostream>
#include <cstdlib>
#include <fstream>
using namespace std;
int judgeNum(int * arr, int n, int c)
{
	int sum = 0;
	for (int i = 0;i < c;i++)
	{
		for (int j = 0;j < c;j++)
		{
			for (int k = 0;k < c;k++)
			{
				sum = arr[i] + arr[j] + arr[k];
				if (sum == n) return 1;
			}
		}
	}
	return 0;
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int arr[998] = {0,};
	int recurNum = 0;
	int temp = 0;
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0;i < 998;i++) arr[i] = (i + 1) * (i + 2) / 2;
	for (int i = 0;i < numTestCases;i++)
	{
		int cnt = 0;
		inStream >> recurNum;
		while (arr[cnt] < recurNum) cnt++;
		cout << judgeNum(&arr[0],recurNum, cnt) << endl;
	}
	inStream.close();
	return 0;
}
