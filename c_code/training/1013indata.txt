


#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#pragma warning(disable : 4996)

void main(void)
{
	FILE* inFile;
	int testCase = 0, i, size, j, k, sum = 0;
	int ducciArr[15] = { 0 }, accountArr[15] = { 0 };
	inFile = fopen("input.txt", "r"); 
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase); 
	for (i = 0; i < testCase; i++)
	{
		fscanf(inFile, "%d",&size); 
		for (j = 0; j < size; j++)
		{
			fscanf(inFile, "%d", &ducciArr[j]);
		}
		for (k = 0; k < 1000; k++)
		{
			for (j = 0; j < size; j++)
			{
				if (j == (size - 1))
				{
					accountArr[j] = abs(ducciArr[j] - ducciArr[0]);
				}
				else
				{
					accountArr[j] = abs(ducciArr[j] - ducciArr[j + 1]);
				}
			}
			for (j = 0; j < size; j++)
			{
				ducciArr[j] = accountArr[j];
			}
		}
		for (j = 0; j < size; j++)
		{
			sum += ducciArr[j];
		}
		if (sum == 0)
		{
			printf("ZERO\n");
		}
		else
			printf("LOOP\n");
		sum = 0;
		for (j = 0; j < size; j++)
		{
			ducciArr[j] = 0;
			accountArr[j] = 0;
		}
	}
	fclose(inFile);
}
