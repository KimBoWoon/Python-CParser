



#include <stdio.h>
#include <stdlib.h>


void main(void)
{
	FILE *inFile;
	int i,j;
	int x,y,z;
	int numCases;
	int N;
	int count =0;
	int tri_num[44]= { 0 };


	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0;i<numCases;i++)
	{
		count =0;
		fscanf(inFile, "%d ",&N );

		

		for(j=0;j<44;j++)
		{
			tri_num[j] = (j+1)*(j+2)/2 ;
		}


		
		

		for(x=0;x<44;x++)
		{
			for(y=0;y<44;y++)
			{
				for(z=0;z<44;z++)
				{
					if(N == tri_num[x] + tri_num[y] + tri_num[z])
						count = 1;
				}
			}
		}
		
		if(count==1)
			printf("1 \n");
		else
			printf("0 \n");
		
	}
	
	
	fclose(inFile); 

}
