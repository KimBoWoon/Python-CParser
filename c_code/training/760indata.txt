

#include <stdio.h>
#include <stdlib.h>

int cpal(char str[], int f, int l) {
	if (l <= f) return 1;
	else if (str[f] != str[l]) return 0;
	else return cpal(str, f + 1, l - 1);
}
void main()
{
	int test_case, i;
	char str[1000];
	FILE *file = fopen("input.txt", "r");

	fscanf(file, "%d", &test_case);

	for (i = 0; i < test_case; i++) {
		fscanf(file, "%s", str);

		printf("%d\n", cpal(str, 0, strlen(str)-1));
	}
	fclose(file);
}
