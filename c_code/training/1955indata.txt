#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int pyth(int s, int a);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int team[1001][2];
		int n, m;
		inStream >> n >> m;

		for (int j = 1; j <= n; j++){
			team[j][0] = 0;
			team[j][1] = 0;
		}
		int team1, team2;
		int win, lose;

		for (int j = 0; j < m; j++){
			inStream >> team1 >> team2 >> win >> lose;
			team[team1][0] += win;
			team[team2][0] += lose;
			team[team1][1] += lose;
			team[team2][1] += win;
		}

		int min = 10001;
		int max = 0;
		int s;

		for (int j = 1; j <= n; j++){
			if (team[j][0] == 0 && team[j][1] == 0){
				s = 0;
			}
			else
			s = pyth(team[j][0], team[j][1]);

			if (s < min)
				min = s;

			if (s>max)
				max = s;
		}

		cout << max << endl;
		cout << min << endl;
	}

	inStream.close();
	return 0;
}

int pyth(int s, int a){
	double num1 = 1000 * s * s;
	double num2 = s * s + a * a;

	return num1 / num2;
}
