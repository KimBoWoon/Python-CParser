

#include<iostream>
#include<fstream>

using namespace std;

class Factorial
{
private:
	int num;
public:
	int factorial_method(int num)
	{
		if (num <= 1)
			return 1;
		else
			return num*factorial_method(num - 1);

	}

};

int main()
{
	ifstream inStream;
	int testcase;
	Factorial fac;

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << " input file open failed";
		return 0;
	}

	inStream >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int fac_num;
		inStream >> fac_num;
		cout<< fac.factorial_method(fac_num)<<endl;
		
	}

	return 0;
}
