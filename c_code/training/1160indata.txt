
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int T,A[45],D[1001];
int main()
{
	ifstream inStream;
	int no_cases,i,j,k;
	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"input file openning failed. \n ";
		exit(1);
	}
	for (i=1;i<45;i++)
		A[i] = i*(i+1)/2;
	for (i=1;i<45;i++) 
		for (j=i;j<45;j++)
			for (k=j;k<45;k++) 
				if (A[i]+A[j]+A[k] < 1001) 
					D[A[i]+A[j]+A[k]] = 1;
	inStream>>no_cases;
	for(int i=0; i<no_cases; i++){
	    inStream>>T;
		for (int i=0; i< 1001; i++)
			if(D[i]==1&&i==T)
				cout<<"1"<<endl;
			else if( D[i]!=1&&i==T)
				cout <<"0"<<endl;
	}
	return 0;
}
