
#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b)
{
   if(b==0)
      return a;
   else
      return gcd(b,a%b);
}

int main()
{
   fstream in;
   in.open("input.txt");
   int numcases;
   in>> numcases;
   for(int i =0; i<numcases; i++){
      int a,b,c,d;
      in>>a>>b>>c>>d;
	        
      int A,B;
	  A=a*d+b*c;
	  B=b*d;
	  int maxDivison=gcd(A,B);
      cout<<A/maxDivison<<" "<<B/maxDivison<<endl;
   }

   return 0;
}
