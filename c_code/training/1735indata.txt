

#pragma warning(disable : 4996) 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int isPalindrom(char*, int, int);
char* convertNumber(int, int);

void main(void)
{
	FILE *inFile;
	int i, j;
	int flag;
	int numCases, numData;
	char* strNum = NULL;

	inFile = fopen("input.txt", "r"); 	

	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &numCases); 	
	for (i = 0; i < numCases; i++)
	{
		fscanf(inFile, "%d", &numData); 

		flag = 0;
		for (j = 2; j < 65; j++)
		{
			strNum = convertNumber(numData, j);
			if (isPalindrom(strNum, 0, strlen(strNum) - 1)) { flag = 1; break; }	
		}
		free(strNum);
		printf("%d\n", flag);
	}

	fclose(inFile); 			
}

int isPalindrom(char* str, int left, int right)
{
	int i, j;
	for (i = left, j = right; i <= j; i++, j--)
		if (str[i] != str[j]) return 0;
	return 1;
}
char* convertNumber(int num, int n)
{
	int i, idx = 0;
	char base[64];
	char* convertStr = (char*)malloc(sizeof(char) * 22);

	for (i = 0; i < 64; i++)
		base[i] = (char)((int)'0' + i);		

	while (num)
	{
		convertStr[idx++] = base[num%n];	
		num /= n;							
	}
	convertStr[idx] = '\0';					

	return convertStr;
}
