
#include <iostream>
#include <fstream>
using namespace std;

int fact(int n)
{
	if(n==0)
		return 1;
	else
		return n*fact(n-1);
}

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase;i++)
	{
		int num;
		instream>> num;
		cout<< fact(num)<<endl;
	}
}
