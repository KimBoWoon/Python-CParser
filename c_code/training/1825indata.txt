

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

void parenthesis(string str);

int main()
{
	int numTestCases;
	string str;
	
	ifstream inFile;
	inFile.open("input.txt");
	inFile >> numTestCases; 

	if(inFile.is_open() == false) {cout << "파일을 열 수 없습니다" <<endl;}

		for(int i = 0 ; i < numTestCases ; i++) {
			inFile >> str;
			parenthesis(str);
		}
}

void parenthesis(string str)
{	
	int left = 0;
	int right = 0;
	int i = 0;

	while(1){
		if(str[i] != '(' && str[i] != ')') break; 
		if(str[i] == '(') {left++; i++;} 
		else if(str[i] == ')') {right++; i++;}
	}
	
	if(left == right && str[i-1] != '(') cout << "YES" << endl;
	else cout <<"NO"<<endl;
}
