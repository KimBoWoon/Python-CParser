 

 
#include <stdio.h>

int main()
{
    FILE *fp;
    int value,n,w,h;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&value);
    while(value--) 
    {
        fscanf(fp,"%d %d %d",&h,&w,&n); 
        printf("%d%02d\n",(n-1)%h+1,(n-1)/h+1);
    }  
    fclose(fp); 
    return 0;
}
