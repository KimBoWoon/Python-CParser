#include <stdio.h>
#include <stdlib.h>
#include<string.h>
#define MAXNUM 1000

int Palindrome(char str[],int start, int last);

void main()
{
	FILE *file;
	int i=0;
	int t;
	char cstr[MAXNUM];
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%s", &cstr);
		printf("%d\n", Palindrome(cstr,0, strlen(cstr)-1));
	}
	fclose(file);
}
int Palindrome(char str[],int start, int last)
{
	if (last <= start) 
		return 1; 
	else if(str[start]!= str[last]) 
		return 0; 
	else 
		return Palindrome(str, start+1, last-1); 
}
