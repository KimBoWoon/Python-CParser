
#include<iostream>
#include<fstream>
using namespace std;

bool isPrime(int num);

int main(void)
{
	int count = 0;
	ifstream file;
	file.open("input.txt");
	file >> count;

	int num;
	for (int i = 0; i<count; i++)
	{
		file >> num;

		if (isPrime(num))
			cout << "1" << endl;
		else
			cout << "0" << endl;
	}
	return 0;

}

bool isPrime(int num)
{
	for (int i = 2; i <= num / 2; i++)
	{
		if (num%i == 0) 
			return false;
	}
		return true;
}
