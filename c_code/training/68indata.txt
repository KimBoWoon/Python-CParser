#include<stdio.h>
#include<stdlib.h>

void main(void){
	FILE *in_file;
	int i;
	int numCases;

	in_file = fopen("input.txt","r");
	if(in_file == NULL) exit(1);

	fscanf(in_file, "%d",&numCases);

	for(i=0; i<numCases; i++){
		int value;
		int isPrime = 1;
		int divisor;

		fscanf(in_file, "%d", &value);

		for(divisor = 2; divisor*divisor<=value; divisor++)
		{
			if(value % divisor == 0){
				isPrime = 0;
				break;
			}
		}

		printf("%d\n", isPrime);
	}
	fclose(in_file);
}
