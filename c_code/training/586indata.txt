 

 
#include <stdio.h>

long factorial(int value);
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test); 
    while(test--) 
    {
        int value;
        fscanf(fp,"%d",&value); 
        printf("%ld\n",factorial(value)); 
    }
    fclose(fp); 
    return 0;
}
long factorial(int value)
{
    if(value == 1 || value == 0) 
    {
        return 1;
    }
    else 
    {
        return value*factorial(value-1);
    }
}
