
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

void FloodFill(char fig[][100], int x, int y, char noRoom, char room, int &size);

int main()
{
	int count = 0;
	ifstream file;
	file.open("input.txt");
	file >> count;
	
	char room[100][100];

	for (int i = 0; i < count; i++)
	{
		int m = 0, n = 0;
		file >> m >> n;

		for (int j = 0; j < n; j++) 
		{
			for (int k = 0; k < m; k++)
				file >> room[j][k];
		}

		int numOfRoom[100] = { 0 }; 
		int num = 0, size = 0; 

		for (int j = 1; j < n - 1; j++)
		{
			for (int k = 1; k < m - 1; k++)				
			{
				size = 0;
				if (room[j][k] == '.')
				{
					num++;
					FloodFill(room, j, k, '+', '.', size);
					numOfRoom[num] = size;
				}
			}
		}

		cout << num << endl; 

		int temp = 0;
		for (int j = 1; j <= num; j++) 
		{
			for (int k = j; k <=num; k++)
			{
				if (numOfRoom[j] < numOfRoom[k])
				{
					temp = numOfRoom[k];
					numOfRoom[k]= numOfRoom[j];
					numOfRoom[j] = temp;
				}
			}
		}
		for (int j = 1; j<= num; j++)
			cout << numOfRoom[j]<<" "; 
		cout << endl;
	}
	return 0;
}

void FloodFill(char fig[][100], int x, int y, char noRoom, char room, int &size)
{
	if (fig[x][y] == room)
	{
		size++;
		fig[x][y] = noRoom;
	}
	if (fig[x - 1][y] == room)
		FloodFill(fig, x - 1, y, noRoom, room, size);
	if (fig[x+1][y] == room)
		FloodFill(fig, x + 1, y, noRoom, room, size);
	if(fig[x][y-1] == room)
		FloodFill(fig, x, y - 1, noRoom, room, size);
	if (fig[x][y + 1] == room)
		FloodFill(fig, x, y + 1, noRoom, room, size);
	else
		return;
}
