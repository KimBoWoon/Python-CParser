

#include <stdio.h>
#include <stdlib.h>


int up(int n)
{
	int a=1;
	int i;

	for (i = 2; i < n+1; ++i)
		a = a * i;
	return a;
}

void change(char s[55], int n, int m)
{
	int i;
	int j;
	int f;

	if (n <= 0)
	{
		printf("\n");
		return;
	}

	else
	{
		j = 0;
		f = up(n - 1);

		if (f <= m)
		{
			j = m / f;
			m %= f;
		}

		printf("%c",s[j]);

		for (i = j; i < 27; ++i)
			s[i] = s[i + 1];
		change(s, n - 1, m);

	}

}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int n, m;
		char s[55] = "abcdefghijklmnopqrstuvwxyz";

		fscanf(inFile, "%d", &n);									
		fscanf(inFile, "%d", &m);									

		change(s, n, m-1);
	}
	fclose(inFile);
}
