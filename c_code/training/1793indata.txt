

#include<iostream>
#include<fstream>
#include<vector>
#include<string>
using namespace std;

vector<int> callStack;

void strCheck(string str, int idx, int len){
	if (len == idx)
		return;
	else{
		if (str[idx] == '(')
			callStack.push_back(1);
		else if (str[idx] == ')')
			if (!callStack.empty())
				callStack.pop_back();
		idx++;		
		strCheck(str, idx, len);
	}
	return;
}

int main(){
	ifstream File;
	int numCase = 0;
	string str;

	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> str;
		strCheck(str, 0, str.length());
		callStack.empty() ? cout << "YES" << endl : cout << "NO" << endl;
		callStack.clear();
	}
	File.close();
	return 0;
}
