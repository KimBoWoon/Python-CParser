



#include <iostream>
#include <fstream>

bool isPrimeNumber(int num);

int main()
{
	int num;
	int numtestCase;

	std::ifstream infile;
	infile.open("input.txt");

	infile >> numtestCase;
	for(int i= 0 ; i<numtestCase; i++)
	{
		infile >> num;
		if( isPrimeNumber(num))
			std::cout<<"1"<<std::endl;
		else
			std::cout<<'0'<<std::endl;	
	}


	return 0;
}


bool isPrimeNumber(int num)
{
	if( num == 1 )
		return true;

	for(int i=2 ; i*i<=num; i++)
	{
		if( num % i == 0 )
			return false;
	}


	return true;



}
