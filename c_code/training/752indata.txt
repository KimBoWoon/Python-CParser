#include <iostream>
#include <cstdlib>
#include <fstream>
#include <cstring>

int checkPalindrome(char str[], int first, int last);
int main()
{
	std::fstream infile;
	int testcase;
	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << " input file error";
		exit(1);
	}
	infile >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		char line[1001];
		infile >> line;
		std::cout << checkPalindrome(line, 0, strlen(line) - 1) << std::endl;
	}
	return 0;
}

int checkPalindrome(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first + 1, last - 1);
}
