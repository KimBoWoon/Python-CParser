


#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main() {
	FILE* in_file;
	int i, j, cnt = 1;
	int no_cases, no_data;

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d\n", &no_cases); 

	for (i = 0; i<no_cases; i++) {
		fscanf(in_file, "%d\n", &no_data);

		for (j = 2; j <= ((int)sqrt(no_data)); j++)
		{
			if (no_data%j == 0) {	
				cnt = 0;		
				break;		
			}
		}
		printf("%d\n", cnt);
		cnt = 1;
	}
	fclose(in_file);
	return 0;
}
