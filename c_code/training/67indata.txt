


#include <stdio.h>
#include <stdlib.h>

void main(void)
{
	FILE *file;
	int i, j, k;
	int test, data;
	int primenum;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		fscanf(file, "%d", &data);
		primenum=data-1;
		
		while(1) 
		{
			if (primenum==1) 
			{
				printf("1\n");
				break;
			}
			else if (data%primenum==0) 
			{
				printf("0\n");
				break;
			}
			primenum--; 
		}
	}
	fclose(file); 
}
