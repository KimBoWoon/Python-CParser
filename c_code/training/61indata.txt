
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream){
		cout << "파일을 열 수 없습니다." << endl;
		return 0;
	}
	int case_number;
	instream >> case_number;
	for(int i=0; i<case_number; i++) 
	{
		int number;
		instream >> number; 
		for(int j=2; j<number+2; j++)
		{
			if(number == j){ 
				cout << "1" << endl;
				break;
			}
			else if(number%j == 0){ 
				cout << "0" << endl;
				break;
			}
		}
			
	}
		
	
	return 0;
}
