

#include <stdio.h>
#include <stdlib.h>

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file, "%d", &num_cases);

	for(i=0; i<num_cases; i++){

		int value;			
		int a,b,c,k;
		int count=0;		
		int total[1000];	
		int tri[50];		
		
		fscanf(in_file, "%d", &value);		

		for(k=1; k<45; k++){

			tri[k]=k*(k+1)/2;	
		}

		for(a=1; a<45; a++){
			for(b=a; b<45; b++){
				for(c=b; c<45; c++){

					if(tri[c]+tri[b]+tri[a]==value){

						count=1;	
					}				
	
				}
			}
		}
		printf("%d\n",count);		

	}
	fclose(in_file);
}
