#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int Chae(int num1, int num2)
{
	if (num2 == 0)
		return num1;
	else
		return Chae(num2, num1%num2);
}

int main()
{
	int gatsu = 0;	
	int num1 = 0;
	int num2 = 0;
	ifstream inStream;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> gatsu;
	for (int i = 0; i < gatsu; i++)
	{
		inStream >> num1;
		inStream >> num2;
		cout<<Chae(num1, num2)<<endl;
	}
	inStream.close();
	return 0;
}
