#include <iostream>
#include <fstream>
#include <iomanip>

using std::cout;
using std::endl;
using std::ifstream;

int main() {
	ifstream fin("input.txt");
	int nCase;
	int height, width, n;

	fin >> nCase;
	cout << std::setfill('0');

	for (int i = 0; i < nCase; i++) {
		fin >> height >> width >> n;

		cout << (n % height ? n % height : height);
		cout << std::setw(2) << (n / height + (n % height ? 1 : 0)) << endl;
	}

	fin.close();
	return 0;
}
