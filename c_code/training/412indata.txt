#include <iostream>
#include <cstdlib>
#include <fstream>

using namespace std;
std::ifstream inStream;

int fibo(int num, int tar){
	return tar == 0 ? num : fibo(tar, num%tar);
}

int main() {
	
	int cases, num, tar;
	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;
	
	for (int i = 0; i < cases; i++) {
		inStream >> num >> tar;
		cout << fibo(num, tar) << endl;

	}
	inStream.close();		
	return 0;
}
