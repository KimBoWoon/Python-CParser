#include <stdio.h>
#include <stdlib.h>

int cnt = 0;	
int index = 0;	
static char number[20];	


void baseConversion(int n, int base);		
int checkPalindrome(char n[], int first, int last);	


void main()	
{
	FILE *f;
	int i;
	int count;

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);
	for(i=0; i<count; i++)
	{
		int j;
		int num;
		int result;
		cnt = 0;
		index = 0;

		fscanf(f, "%d", &num);
		
		for(j=2; j<=64; j++)	
		{
			baseConversion(num, j);	
			result = checkPalindrome(number, 0, cnt);	
			if(result)	
				break;	
			cnt = 0;	
			index = 0;
		}
		printf("%d\n", result);	
	}
	fclose(f);
}
void baseConversion(int n, int base)	
{
	static char baseTable[] = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ+*";

	if(n>=base)
	{
		cnt++;
		baseConversion(n/base, base);
	}
	number[index++] = baseTable[n%base];		
}

int checkPalindrome(char *str, int first, int last)	
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first+1, last-1);
}
