
#include <iostream>
#include <fstream>
using namespace std;
bool checkPalindrone(int *str,int first,int last)		
{
    if(first>=last)
    {
        return 1;
    }
    else if(str[first] != str[last])
    {
        return 0;
    }
    else
    {
        return checkPalindrone(str,first+1,last-1);
    }
}

int main() {
    fstream in;
    in.open("input.txt");
    int numcases;
    in>>numcases;
    for(int i=0; i<numcases; i++)
    {
        int num;
        in>>num;
       
        int str[100000];
        for(int j=2; j<=64; j++){
            int temp=num;
            int k=0;
            while(temp!=0)
            {
                str[k]=temp%j;
                temp/=j;
                k++;
            }
            if(checkPalindrone(str, 0, k-1))
            {
                cout<<"1"<<endl;
                break;
            }
            else if(j==64)
            {
                cout<<"0"<<endl;
            }
        }
    }
    return 0;
}
