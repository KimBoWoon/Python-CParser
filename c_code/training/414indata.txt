
#include<iostream>
#include<string>
#include<fstream>

using namespace std;

int euclid(int a, int b){
	if (b == 0){
		return a;
	}
	else{
		return euclid(b, a%b);
	}
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, aBuff = 0, bBuff = 0;
	f >> numCount;
	for (int i = 0;i<numCount;i++){
		f >> aBuff >> bBuff;
		cout << euclid(aBuff, bBuff) << endl;
	}

	return 0;
}
