#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int recursivePalindrome(char arr[], int first, int last);

void main()
{
	FILE *f;
	int count;	
	int i;		
	char str[1001];		

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);	

	for(i=0; i<count; i++)
	{
		int result;	
		int l;		
		fscanf(f, "%s", &str);	
		l = strlen(str)-1;	
		result = recursivePalindrome(str, 0, l);	
		printf("%d\n", result);	
	}
}
int recursivePalindrome(char arr[], int first, int last)
{
	if(first >= last)	
		return 1;		
	else if(arr[first] != arr[last])	
		return 0;		
	else
		return recursivePalindrome(arr, first+1, last-1);	
}
