#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int palindrome(char str[], int first, int last){
	if(last <= first)
		return 1;
	else if(str[first] != str[last])
		return 0;
	else 
		return palindrome(str, first+1, last-1);
}
void main()
{
	FILE *in_file;
	int i,n;
	int numcases;
	char str[1000];              
	int first, last;
	int l;
	in_file = fopen("input.txt","r"); 

	fscanf(in_file, "%d", &numcases); 

		for(i=0; i < numcases; i++)
		{


			for(n = 1; n <= 1000; n++){
				fscanf(in_file, "%s", str);
				l = strlen(str);
				break;
			
			}
			printf("%d\n", palindrome(str, 0, strlen(str)-1));   
		}
	fclose(in_file);
}
