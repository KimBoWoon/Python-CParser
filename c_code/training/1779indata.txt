#include<stdio.h>
#include<string.h>

char *count[2] = {"NO", "YES"};
char VPS[50];
int l;

char* judge(int n, int left, int right)
{
	if(n==l){
		if(left == right)	return count[1];
		else				return count[0];
	}
	else{
		if(right > left) return count[0];
		else{
			if(VPS[n] == '(')	left += 1;
			else				    right += 1;

			judge(n+1, left, right);
		}
	}
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testcase;
	int i;

	fscanf(inFile, "%d", &testcase);
	for(i=0; i<testcase; ++i)
	{
		fscanf(inFile, "%s", &VPS);
		l = strlen(VPS);
		printf("%s\n", judge(0, 0, 0));
	}
	fclose(inFile);
}
