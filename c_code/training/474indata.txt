
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;
int gcd(int a, int b);
int main()
{
	ifstream inStream;
	int no_cases,value1,value2;
	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"Input file openning failed.\n";
		exit(1);
	}
	inStream>>no_cases;
	for(int i=0; i<no_cases; i++)
	{
		inStream>>value1;
		inStream>>value2;
		cout<<gcd(value1,value2)<<endl;
	}
	return 0;
}
int gcd(int a, int b)
{
	if (b == 0) 
		return a;
	else 
		return gcd(b, a%b);
}
