


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

void prime(int value);         

int main(void)
{
   ifstream inf;
   int cases, value;

   inf.open("input.txt");
   if(inf.fail())
   {
      cerr<<"Input file opening failed./n";
      exit(1);
   }

   inf>>cases;
   for(int i=0; i<cases; i++)
   {
      inf>>value;
      prime(value);
   }
}

void prime(int value)
{
   int count(0);               

   for(int i=2; i<value; i++)
   {
      if(value%i==0)                  
      {
         cout<<"0"<<endl;
         break;
      }
      else
         count++;
   }
   
   if(count==value-2)            
      cout<<"1"<<endl;
}
