#include <stdio.h>
#include <stdlib.h>

int judgePrime (int num)
{
	if(num!=2&&num%2==0) return 0;
	else
	{
		int i =0;
		int limit =num/3;
		for(i=2; i<=limit; i++)
		{
			if(num%i==0) return 0;
		}
	}
	return 1;
}
	
int main ()
{
	int dataCase=0;
	int i=0;
		
	FILE* file;
	file  = fopen("input.txt","r");
       
	fscanf(file,"%d",&dataCase);
	
        
	for(i=0; i<dataCase; i++)
	{
		int num=0;
		fscanf(file,"%d",&num);
		int result = 0;
		result =judgePrime(num);
		printf("%d\n",result);
	}
	fclose(file);
	return 0;
}
