

#include<iostream>
#include<fstream>
#include<list>
#include<cstdlib>
using namespace std;

list<int*> li;
int size = 0;

bool isZero(int* arr){
	bool zero = true;
	int temp = arr[0];
	for (int i = 0; i < size; i++){
		if (arr[i] != temp){
			zero = false;
			break;
		}
	}
	return zero;
}

bool isDupl(int* arr){
	bool dupl = false;
	int cnt = 0;
	for (list<int*>::iterator it = li.begin(); it != li.end(); ++it){
		cnt = 0;
		for (int i = 0; i < size; i++){
			if (arr[i] != (*it)[i])
				break;
			cnt++;
		}
		if (cnt == size){
			dupl = true;
			break;
		}
	}
	return dupl;
}

void calc(int* arr){
	while (true){
		int* temp = new int[size];
		for (int i = 0; i < size; i++){
			temp[i] = arr[i];
		}
		if (isZero(arr)){
			cout << "ZERO" << endl;
			return;
		}
		else{
			if (isDupl(arr)){
				cout << "LOOP" << endl;
				return;
			}
			else{
				li.push_back(temp);
				int n = arr[0];
				for (int i = 0; i < size-1; i++){
					arr[i] = abs(arr[i] - arr[i + 1]);
				}
				arr[size - 1] = abs(arr[size - 1] - n);
			}
		}
	}
	return;
}

int main(){
	ifstream File;
	int numCase = 0;
	int num = 0;
	int* arr = NULL;
	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> size;
		arr = new int[size];
		for (int i = 0; i < size; i++){
			File >> num;
			arr[i] = num;
		}
		calc(arr);
		li.clear();
		delete[] arr;
	}
	File.close();
	return 0;
}
