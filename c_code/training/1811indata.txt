
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int i=0, left=0, right=0;

void parenthesis(char *p){

	if(i==strlen(p)){  
		if(left == right) 
			printf("YES\n");
		else              
			printf("NO\n");
	}else{
		if(p[i] == '(')
			left++;
		else if(p[i] == ')'){
			right++;
			if(left<right){  
				printf("NO\n");
				return;
			}
		}
		i++;
		parenthesis(p);  
	}
}

int main(){

	FILE *inFile;
	int t;
	char p[51];

	if((inFile=fopen("input.txt","r")) == NULL)
		exit(1);

	fscanf(inFile, "%d", &t);
	while(t--){
		fscanf(inFile, "%s",p);
		parenthesis(p);
		i=left=right=0;
	}
	fclose(inFile);
	return 0;
}
