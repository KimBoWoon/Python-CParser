#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int tuple[20] = { 0, };

void Ducci(int tuple_n)
{
	int max = 1001;											
	int count = 0;											
	int tuple_zero;											
	int i;

	while (max--)
	{
		count = 0;
		tuple_zero = tuple[0];

		for (i = 0; i < tuple_n; i++)
		{
			if (i == (tuple_n - 1))
				tuple[i] = abs(tuple[i] - tuple_zero);
			else
				tuple[i] = abs(tuple[i] - tuple[i + 1]);

			if (tuple[i] == 0)
				count++;
		}
		if (count == tuple_n)
		{
			printf("ZERO\n");
			break;
		}
		else
		{
			if (max == 0)
				printf("LOOP\n");
		}
	}
}
int main()
{
	int numcase;
	int tuple_n;
	int i;

	FILE * in_file;
	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d", &numcase);

	while (numcase--)
	{
		fscanf(in_file, "%d", &tuple_n);
		for (i = 0; i < tuple_n; i++)
		{
			fscanf(in_file, "%d", &tuple[i]);
		}
		Ducci(tuple_n);
	}
	fclose(in_file);
}
