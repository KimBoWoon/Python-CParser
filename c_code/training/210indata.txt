


#include<stdio.h>

int gcd(int _a, int _b)     
{
	if(_b == 0)
	{
		return _a;
	}
	else
	{
		int remainder = _a % _b;
		return gcd(_b, remainder);
	}
}

void addFraction(int *_resultA, int *_resultB, int _a, int _b, int _c, int _d)  
{
	int divisor;

	*_resultA = _a * _d + _c * _b;
	*_resultB = _b * _d;
	divisor = gcd(*_resultA, *_resultB);

	*_resultA /= divisor;
	*_resultB /= divisor;
}

int main(void)
{
	FILE *fp = NULL;
	int testCase;

	fp = fopen("input.txt", "r");                   
	fscanf(fp, "%d", &testCase);                    
	while(testCase--)
	{
		int a, b, c, d, resultA, resultB;
		fscanf(fp, "%d %d %d %d", &a, &b, &c, &d);  
		addFraction(&resultA, &resultB, a, b, c, d);
		printf("%d %d\n", resultA, resultB);        
	}

	fclose(fp);
	return 0;
}
