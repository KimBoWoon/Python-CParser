
#include <iostream>
#include <fstream>
using namespace std;

bool isEurekaT(int num);

int main()
{
	int count=0;
	ifstream file;
	file.open("input.txt");
	file >> count;
	
	int num;
	for (int i = 0; i < count; i++)
	{
		file >> num;
		if (isEurekaT(num))
			cout << "1" << endl; 
		else
			cout << "0" << endl; 
	}
}

bool isEurekaT(int num)
{
	int T = 1, n = 0, total = 0;
	for (n = 1; T < num; n++)
		T = (n*(n + 1)) / 2; 

	for (int m = 1; m < n; m++) 
	{
		for (int j = 1; j < n; j++)
		{
			for (int k = 1; k < n; k++)
			{
			total = ((m*(m + 1)) + (k*(k + 1)) + (j*(j + 1))) / 2;

			if (total == num)
				return true; 
			else
				total = 0;
			}
		}
	}

	return false;
}
