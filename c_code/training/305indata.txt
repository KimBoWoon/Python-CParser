

#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){

	if(b==0)	return a;
	else		return gcd(b,a%b);
}

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){


		int s1, m1, s2, m2;
		int s_fin, m_fin;
		fscanf(in_file,"%d %d %d %d",&s1,&m1,&s2,&m2);

		m_fin = m1*m2;
		s_fin = (s1*m2)+(s2*m1);

		printf("%d %d\n", s_fin/gcd(s_fin,m_fin), m_fin/gcd(s_fin,m_fin));

	}
	fclose(in_file);
}
