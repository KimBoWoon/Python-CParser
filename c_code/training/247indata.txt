#include<iostream>
#include<fstream>

using namespace std;

int foundGcd(int a, int b){
	if (b == 0){
		return a;
	}
	else{
		return foundGcd(b, a%b);
	}
}

void calcFountain(int numerA,int numerB,int denomA, int denomB){
	int numer, denom, gcd;

	denom = denomA * denomB;
	numer = numerA * denomB + numerB * denomA;

	gcd = foundGcd(denom, numer);

	denom /= gcd;
	numer /= gcd;

	cout << numer << " " << denom << endl;
}

int main()
{
	int testCase;
	int numerA, denomA;
	int numerB, denomB;

	ifstream file;
	file.open("input.txt");

	file >> testCase;
	for (int tCase = 0; tCase < testCase; tCase++){
		file >> numerA >> denomA;
		file >> numerB >> denomB;
		calcFountain(numerA, numerB, denomA, denomB);
	}

	return 0;
}
