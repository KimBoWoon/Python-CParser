#include <iostream>
#include <fstream>
#include <vector>
using namespace std;

bool isUsed[12];
int first_size;

int fac(int num) {
	int temp = 1;

	for (int i = 1; i <= num; i++)	temp *= i;
	return temp;
}

void permutation(int size, int num) {
	if (size == 0)	return;
	if (size == 1) {
		for (int i = 0; i < 12; i++)
			if (!isUsed[i]) {
				char print = 'a' + i;
				cout << print;
				break;
			}
		return;
	}

	int cases = fac(size);
	int index = (num - 1) / (cases / size);

	int count = index;
	for (int i = 0; i < first_size; i++) {
		
		if (isUsed[i] == false) {
			if (count != 0) {
				count--;
				continue;
			}
			char print = 'a' + i;
				cout << print;
				isUsed[i] = true;
				break;
		}
	}
	
	permutation(size - 1, num - ((cases / size) * index));
}

int main() {
	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;
	while (cases--) {
		for (int i = 0; i < 12; i++)	isUsed[i] = false;
		int num;
		file >> first_size >> num;

		permutation(first_size, num);

		cout << endl;
	}

	return 1;
}
