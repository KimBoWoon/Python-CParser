

#include<iostream>
#include<fstream>
#include<list>
#include<string>
using namespace std;

int fac[12];
list<char> cList;
string str;

void findPermul(int size, int seq){
	if (size == 0){
		char ch = cList.back();
		str += ch;
	}
	else{
		int listIter = seq / fac[size - 1];
		list<char>::iterator iter = cList.begin();
		for (int i = 0; i < listIter; i++){
			++iter;
		}
		char ch = *iter;
		str += ch;
		cList.remove(ch);
		seq %= fac[size - 1];
		findPermul(--size, seq);
	}
}

int main(){	
	ifstream File;
	int numCase = 0;
	int size = 0, seq = 0;

	File.open("input.txt");
	File >> numCase;
	fac[0] = 1;
	for (int i = 1; i < 12; i++){
		fac[i] = fac[i - 1] * (i + 1);
	}
	for (int iter = 0; iter < numCase; iter++){
		File >> size >> seq;
		for (int i = 0; i < size; i++){
			cList.push_back('a' + i);
		}
		findPermul(--size, --seq);
		cout << str << endl;
		cList.clear();
		str.clear();
	}
	File.close();
	return 0;
}
