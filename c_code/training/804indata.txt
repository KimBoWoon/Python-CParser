
#include<iostream>
#include<string>
#include<fstream>

using namespace std;

int fibonacci(int numBuff){
	if (numBuff <= 1){
		return numBuff;
	}
	else{
		return fibonacci(numBuff-1) + fibonacci(numBuff-2);
	}
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, numBuff;

	f >> numCount;
	for (int i = 0;i<numCount;i++){
		f >> numBuff;
		cout << fibonacci(numBuff) << endl;
	}
	return 0;
}
