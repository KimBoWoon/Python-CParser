

#include <stdio.h>
unsigned int Fibonacci(n) { 
	if (!n) return 0; 
	else if (!(n - 1)) return 1; 
	else
		return Fibonacci(n - 1) + Fibonacci(n - 2); 
}
int main()
{
	unsigned int t, n;
	FILE *fp;
	if ((fp = fopen("input.txt", "r")) == NULL) return 0; 
	fscanf(fp, "%d", &t); 
	while (t--){
		fscanf(fp, "%d", &n); 
		printf("%u\n", Fibonacci(n)); 
	}fclose(fp);
}
