#include<iostream>
#include<fstream>
using namespace std;

int fac(int num)
{
	if(num==0)
		return 1;
	else
	{
		return num*fac(num-1);
	}
}

int main()
{
	fstream instream("input.txt",ios::in);
	int testCase;
	instream>>testCase;
	
	for(int i=0;i<testCase;i++)
	{
		int num;
		instream>>num;
		cout<<fac(num)<<endl;
	}
	return 0;
}
