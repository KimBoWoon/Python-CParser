#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char bracket[55] = { '0', };

int Parenthesis(int start, int end, int position)
{
	if (end > start)														
	{
		printf("NO\n");
	}
	else if ((bracket[position] == '(') || (bracket[position] == ')'))		
	{
		if (bracket[position] == '(')
			start++;
		else
			end++;

		position++;
		Parenthesis(start, end, position);
	}
	else if (start == end)													
	{
		printf("YES\n");
	}
	else																	
	{
		printf("NO\n");
	}

	return 0;
}
int main()
{
	int numcase;
	int start, end, position;

	FILE * in_file;
	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d\n", &numcase);

	while (numcase--)
	{
		start = 0;
		end = 0;
		position = 0;

		memset(bracket, '0', sizeof(bracket));
		fgets(bracket, sizeof(bracket), in_file);
		Parenthesis(start, end, position);
	}
	fclose(in_file);
}
