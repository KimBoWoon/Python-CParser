


#include <iostream>
#include <fstream>
#include<cstdlib>
using namespace std;

int fibo(int n); 

int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int x;
		inStream >> x;
		cout << fibo(x) << endl; 
		
	}
	inStream.close();
	return 0;
}

int fibo(int n)
{
	if (n <= 1) 
		return n;
	else 
		return fibo(n - 1) + fibo(n - 2);
}
