
#include <cstdio>
#include <cstdlib>

int factorial(int number){
	static int memoization[13];

	if(memoization[number] != 0) return memoization[number];

	if(number == 0) return 1;
	else return memoization[number] = number*factorial(number-1);
}
int compare(const void* str1, const void* str2){
	
	if(*(char*)str1 > *(char*)str2) return 1;
	else if(*(char*)str1 == *(char*)str2) return 0;
	else return -1;
}
int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}

	int testCases = 0,
		length = 0,
		number = 0,
		factorialNum = 0;
	
	fscanf(readStream, "%d", &testCases);

	while(testCases--){
		
		fscanf(readStream, "%d %d", &length, &number);
		char alphabet[12] = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l'};

		number -= 1;
		factorialNum = length - 1;

		for(int i = 0; i < (length - 1); ++i){
			printf("%c", (alphabet[number / factorial(factorialNum - i)]));
			alphabet[number / factorial(factorialNum - i)] = alphabet[factorialNum - i];
			qsort(alphabet, (length - 1 - i), sizeof(char), compare);
			number %= factorial(factorialNum - i);
		}
		printf("%c\n", alphabet[0]);
	}
	fclose(readStream);
}
