



#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>


void main(void)
{
	FILE *inFile;
	int i,j,k;
	int numCases;
	int len ;
	int check;
	int a,b;
	char word[1000][1000]={'#'};

	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0;i<numCases;i++)
	{
		len =0;
		fscanf(inFile, "%s",word[i]);	


		for(j=0;j<1000;j++)
		{
			if(isalpha(word[i][j]))
			{
				len++;
			}	
		}

		check=0;
		
		for(k=0;k<len/2;k++)
		{	
			if(word[i][k] != word[i][len-k-1])
			{
				check++;
			}
		}


		if(check == 0)
		{
			printf("1\n");
		}
		else
			printf("0\n");
		
	
	}
	
	fclose(inFile); 

}
