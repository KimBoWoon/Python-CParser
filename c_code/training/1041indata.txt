
#include <iostream>
#include <fstream>

#define MAX 1000

using namespace std;

int main(){

	ifstream inStream;

	inStream.open("input.txt");

	int num_case;

	inStream >> num_case;

	for(int i = 0; i <num_case; i++){

		int size;

		inStream >> size;

		int *arr = new int[size];		
		int *comp = new int[size];		

		for(int j = 0; j<size; j++)
		{
			inStream >> arr[j];
		}
		comp = arr;		

		int a = 0;
		int cnt = 0;

		int first = arr[0];

		while(cnt<MAX)
		{	
			if(a < size-1){
				if(comp[a+1] > comp[a]){
					arr[a] = comp[a+1]-comp[a];
					a++;
				}
				else{
					arr[a] = comp[a]-comp[a+1];
					a++;
				}
			}
			else{
				if(first > comp[a])	{
					arr[a] = first-comp[a];
					a=0;
					first = arr[0];
				}
				else{
					arr[a] = comp[a] - first;
					a=0;
					first=arr[0];
				}
			}
			cnt++;
		}
		int flag =0;	
		for(int i=0; i<size ; i++)
		{
			if(arr[i] != 0)
				flag =1 ;
		}

		if(flag == 1)
			cout<<"LOOP"<<endl;
		else
			cout<<"ZERO"<<endl;
	}

}
