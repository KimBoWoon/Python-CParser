#include <iostream>
#include <cstdlib>
#include <fstream>

using namespace std;
std::ifstream inStream;

int Maxdiv(int a,int b){
	return b == 0 ? a : Maxdiv(b, a%b);
}

int main() {
	
	int cases, a, b, c ,d;
	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;
	
	for (int i = 0; i < cases; i++) {
		inStream >> a >> b >> c >> d;
		cout << (a * d + b * c) / Maxdiv(a * d + b * c, b * d) << " " << 
			b * d / Maxdiv(a * d + b * c, b * d) << endl;

	}
	inStream.close();		
	return 0;
}
