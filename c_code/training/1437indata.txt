

#include <stdio.h>
#include <stdlib.h>

void main(void)
{
	FILE *in_file;
	int i;
	int no_cases;
	int hotelHeight, hotelWidth, hotelGuest;
	int hotelRoomNumber;
	
	in_file = fopen("input.txt", "r"); 
	
	if (in_file == NULL) exit(1);
	fscanf(in_file, "%d", &no_cases); 

	for (i = 0; i<no_cases; i++)
	{
		
		fscanf(in_file, "%d %d %d", &hotelHeight, &hotelWidth, &hotelGuest); 
		
		
		hotelRoomNumber = ((hotelGuest-1) % hotelHeight +1) * 100 + ((hotelGuest-1) / hotelHeight + 1);

		printf("%d\n", hotelRoomNumber);
	}

	fclose(in_file); 
}
