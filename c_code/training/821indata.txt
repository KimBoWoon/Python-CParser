#include<iostream>
#include<fstream>
using namespace std;
int Finbo(int val);
int main(){
	ifstream fin;
	fin.open("input.txt");
	int num, val;
	fin >> num;
	for (int cnt = 0; cnt < num; cnt++){
		fin >> val;
		cout << Finbo(val) << endl;
	}
}
int Finbo(int val){
	if (val <= 1) return val;		
	else return Finbo(val - 1) + Finbo(val - 2);		
}
