


#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b)
{
	if( b==0 )
		return a;
	else
		return gcd(b, a%b);
}


void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int x1 = 0;														
		int x2 = 0;
		int y1 = 0;
		int y2 = 0;
		int z1 = 0;
		int z2 = 0;


		fscanf(inFile, "%d", &x1);
		fscanf(inFile, "%d", &x2);
		fscanf(inFile, "%d", &y1);
		fscanf(inFile, "%d", &y2);

		x1 = x1 * y2;
		y1 = y1 * x2;
		x2 = x2 * y2;

		z1 = (x1 + y1) / gcd((x1+y1),x2); 
		z2 = x2 / gcd((x1+y1),x2);

		printf("%d %d\n",z1,z2);

	}

	fclose(inFile);														
}
