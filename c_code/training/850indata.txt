
#pragma warning(disable:4996)

#include <cstdio>
#include <cstdlib>

unsigned int fibonacci(int num){

	static unsigned int memoization[31];
	
	if(memoization[num]) return memoization[num];

	if(num == 1 || num == 2){
		memoization[num] = 1;
		return memoization[num];
	}
	return memoization[num] = fibonacci(num-1) + fibonacci(num-2);
	
}
int main(int argc, char* argv[]){

	FILE* readStream = 0x00L;
	readStream = fopen("input.txt", "r");

	if(readStream == 0x00L){
		fprintf(stderr, "Don`t open the file.\n");
		return 2;
	}
	
	int testCases = 0, num = 0;
	fscanf(readStream, "%d", &testCases);
	
	while(testCases--){
		fscanf(readStream, "%d", &num);
		printf("%d\n", (num==0) ? 0 : fibonacci(num));
	}
	fclose(readStream);
}
