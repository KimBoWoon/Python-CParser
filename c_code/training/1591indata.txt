

#include <iostream>
#include <fstream>

using namespace std;
int merge[503][503];

int merging(int arr1[], int arr2[], int n){
	int sum;
	for(int i=1;i<n;i++){
		for ( int j=1;j<=n-i;j++){
			sum =i+j;
			int min =merge[j][j] + merge[j + 1][sum];
			for( int k=j+1;k<sum;k++){
				int res =  merge[j][k] + merge[k+1][sum];
				min = (min <res) ? min : res;
			}
			merge[j][sum] =min + arr2[sum] - arr2[j-1];
		}
	}
	return merge[1][n];
}

int main(){
	ifstream instream;
	instream.open("input.txt");

	int no_cases;
	instream>>no_cases;

	while(no_cases--){
		int n;
		instream>> n;
		int *arr=new int [n+1];
		int *arr1 = new int [n+1]; 
		for ( int i=1;i<=n;i++){
			instream>>arr[i];
			arr1[i] = arr1[i-1] + arr[i];

		}
		cout<<merging(arr,arr1,n)<<endl;
	}
	instream.close();
}
