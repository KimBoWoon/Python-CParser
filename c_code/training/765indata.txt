

#include <iostream>
#include <fstream>
#include <cstring>

int palin(char str[], int first, int last)
{
	if(last <= first)
		return 1;
	else if(str[first] != str[last])
		return 0;
	else
		return palin(str, first+1, last-1);

}

int main(int argc, char* argv[])
{
	std::ifstream file;
	file.open("input.txt");

	int test;
	file >> test;

	for(int i=0; i<test; i++)
	{
		char str[1000]; 
		file >> str;
		std::cout << palin(str, 0, strlen(str)-1) << std::endl;
	}
return 0;
}
