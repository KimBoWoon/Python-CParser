#include<iostream>
#include<fstream>
using namespace std;


int ok(int* a, int num)
{
	int right = 0;
	int check = 1;
	int left = num;
	while (right < left)
	{
		if (a[right] != a[left])
		{
			check = 0;
			break;
		}
		right++;
		left--;
	}
	return check;
}


int main()
{
	ifstream instream;
	instream.open("input.txt");

	if (instream.fail())
	{
		cerr << "input file open failed";
	}

	int B_num;
	int num;
	int check;
	int testcase;

	int B_div;
	int B_rem[31];
	int k = 0;

	instream >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		instream >> B_num;

		for (int j = 2; j < 65; j++)
		{
			num = B_num;
			while (B_num)
			{
				B_rem[k++] = B_num % j;
				B_num /= j;
			}
			check = ok(B_rem, k - 1);
			if (check)
				break;
			B_num = num;
			k = 0;
		}
		cout << check << endl;
		k = 0;
	}
	return 0;
}
