


#include <stdio.h>

#define MAX 1000001

int Palindromic(int num){
	int i,j,p,k=0,tmp,flag;
	int RR[MAX]; 

	for(i=2;i<=64;i++){
		tmp=num;
		k=0;
		while(tmp>0){
			RR[k++]=tmp%i; 
			tmp/=i;
		}
		for(j=0,p=k-1;j<k/2;j++){ 
			if(RR[j]!=RR[p--]){
				flag=0;
				break;
			}
			else
				flag=1;
		}
		if(flag)
			return 1;
	}
	return 0;
}

int main(){
	int num,cases;
	FILE* fp=fopen("input.txt","r");
	fscanf(fp,"%d",&cases);
	while(cases--){
		fscanf(fp,"%d",&num);
		printf("%d\n",Palindromic(num));
	}
	fclose(fp);
	return 0;
}
