
#include<iostream>
#include<fstream>

using namespace std;

int isPalind(int arr[], int start, int end){
	if (start >= end) return 1;
	if (arr[start] != arr[end]) return 0;
	return isPalind(arr, start + 1, end-1);
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;

	int test_case;
	in >> test_case;
	int *arr = new int[30];
	while (test_case--){
		int n;
		in >> n;
		int flag = 0;
		for (int i = 2; i <= 64; i++){
			int t = n, cnt = 0;
			while (t > 0){
				arr[cnt++] = t % i;
				t /= i;
			}
			if (isPalind(arr, 0, cnt - 1)){
				flag = 1;
				break;
			}
		}
		cout << flag << endl;
	}
	delete arr;

	in.close();
	return 0;
}
