#include <iostream>
#include <fstream>
using namespace std;

int getGcd(int n1, int n2){
    if(n2==0)
        return n1;
    else
        getGcd(n2, n1 % n2);
}

int main(){
    ifstream inStream;
    inStream.open("input.txt");
    int test_case;
    inStream >> test_case;

    for(int i=0; i<test_case; i++){
        int n1, n2;
        inStream >> n1 >> n2;
	if(n1==0)
            cout << n2 << endl;
        else
            cout << getGcd(n1, n2) << endl;
    }
}
