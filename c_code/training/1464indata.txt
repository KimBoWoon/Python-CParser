

#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int main()
{
	ifstream instream("input.txt");
	int numtestcases, height, width, num, yy, xx;
	int yyxx;
	instream >> numtestcases;
	for (int i = 0; i < numtestcases; i++){
		instream >> height >> width >> num;
		yy = num % height;
		if (num%height == 0){
			xx = (num / height);
			yyxx = (height * 100) + xx;
		}
		else{
			xx = (num / height) + 1;
			yyxx = (yy * 100) + xx;
		}
		cout << yyxx << endl;
	}
	instream.close();
}
