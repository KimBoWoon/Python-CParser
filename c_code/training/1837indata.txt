
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;
void checkString(char check[], int count, int score){
	if (check[count] == '(')
		checkString(check, count + 1, score + 1);
	else if (check[count] == ')'){
		if (score < 0)
			cout << "NO" << endl;
		else
			checkString(check, count + 1, score - 1);
	}
	else if (count == strlen(check)){
		if (score == 0)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n;
	char check[50];
	in >> n;
	in.getline(check, 256);
	for (int i = 0; i < n; i++){
		in.getline(check, 256);
		checkString(check, 0, 0);
	}
	return 0;
}
