


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main(void)
{
   ifstream inStream;
   int cases;
   int length=0;
   int width=0; 
   int rankOfCustomer=0;
   int roomTotalNum = width * length;
   int customer[10000]={0};

   inStream.open("input.txt");
   if(inStream.fail())
   {
      cerr<<"Input file opening failed./n";
      exit(1);
   }

   inStream>>cases;

   for(int i=0; i<cases; i++)
   {
      inStream>> width >> length >> rankOfCustomer;
 
   int rank = 0;  
   for(int i=1; i<=length; i++)   
   {
	   for(int j=1; j<=width; j++)
	   {
		   customer[rank]= i + 100 * j;
		   rank++;
	   }
   }
	cout << customer[rankOfCustomer-1]<< endl;
   }
}
