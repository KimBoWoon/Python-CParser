


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int MC(int val);

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value;				

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value;
		cout << MC(value) << endl;
	}

	inStream.close();
	return 0;
}

int MC(int val)
{
	if (val > 100)
	{
		return val - 10;
	}
	else
	{
		return MC(MC(val + 11));
	}
}
