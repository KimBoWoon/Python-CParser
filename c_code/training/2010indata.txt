
#include<iostream>
#include<fstream>
#include<cstdlib>

using namespace std;

void quickSort(int arr[], int left, int right) {
	int i = left, j = right;
	
	int pivot = arr[(left + right) / 2];
	while (i <= j) {
		while (arr[i] < pivot) i++;
		while (arr[j] > pivot) j--;
		if (i <= j) {
			int t;
			t = arr[i];
			arr[i] = arr[j];
			arr[j] = t;
			i++;
			j--;
		}
	};
	if (left < j) quickSort(arr, left, j);
	if (i < right) quickSort(arr, i, right);
}

int main(){
	ifstream in;
	in.open("input.txt");

	if (!in) return 1;
	
	int test_case;
	in >> test_case;
	while(test_case--){
		int m, k, min = 999999999, cnt = 0;
		in >> m >> k;
		int *arr = new int[m];
		for (int i = 0; i < m; i++) {
			in >> arr[i];
		}
		quickSort(arr, 0, m-1);
		int left = 0, right = m - 1;
		while (left < right){
			int t = arr[left] + arr[right];
			int t2 = abs(k - t);
			if (t2 < min){
				min = t2;
				cnt = 1;
			}
			else if (t2 == min) cnt++;
			t < k ? left++ : right--;
		}
		delete arr;
		cout << cnt << endl;
	}

	in.close();
	return 0;
}
