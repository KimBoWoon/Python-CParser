


#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>
#include <string.h>

using namespace std;

int palindrome(char string[], int first, int last);

int main()
{
	ifstream inStream;
	int numTestCases;		
	char str[1000] = {};		

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> str;
		cout << palindrome(str, 0, strlen(str)-1) << endl;
	}

	inStream.close();
	return 0;
}

int palindrome(char string[], int first, int last)
{
	if (last <= first)
	{
		return 1;
	}

	else if (string[first] != string[last])
	{
		return 0;
	}

	else
	{
		return palindrome(string, first + 1, last - 1);
	}
}
