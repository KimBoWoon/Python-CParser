

#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>

using namespace std;

int Palindrome(char str[], int first, int last);	

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error"<<endl;
		exit(1);
	}

	int no_case;
	instream>>no_case;

	for(int i=0; i<no_case; i++)		
	{
		char arr[1001];			
		instream>>arr;

		cout<<Palindrome(arr, 0, strlen(arr)-1)<<endl;
		
	}
}

int Palindrome(char str[], int first, int last)		
{
	if(last <=first)
		return 1;
	else if(str[first] != str[last])
		return 0;
	else
		return Palindrome(str, first+1, last-1);
}
