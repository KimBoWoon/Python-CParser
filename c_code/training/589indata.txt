#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int Fact(int num)
{
	if (num <= 1)
		return 1;
	else
		return num*Fact(num - 1);
}

int main()
{
	int gatsu = 0;
	int num = 0;

	ifstream inStream;
	inStream.open("input.txt");

	inStream >> gatsu;

	for (int i = 0; i < gatsu; i++)
	{
		inStream >> num;
	
		cout << Fact(num) << endl;
	}
	inStream.close();
	return 0;
}
