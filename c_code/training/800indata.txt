
#include <iostream>
#include <fstream>

using namespace std;

int fibonacci(int num);	


int main()
{
	ifstream inFile;
	inFile.open("input.txt");

	int numTestCase;	
	inFile >> numTestCase;

	while (numTestCase--)	
	{
		int num;	
		inFile >> num;

		cout << fibonacci(num) << endl;	
	}

	inFile.close();
	return 0;
}

int fibonacci(int num)
{
	if (num == 0)	
		return 0;
	if (num == 1)	
		return 1;
	else
		return fibonacci(num - 1) + fibonacci(num - 2);	
}
