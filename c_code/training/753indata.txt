void Palindrome(int f , int l) {	

	if (f >= l) {	
		std::cout << "1" << std::endl;
	}
	else {
		if (str.at(f) == str.at(l)) Palindrome(f + 1, l - 1);
		else std::cout << "0" << std::endl;
	}
}
