#include <stdio.h>
#include <stdlib.h>



void main(void) {
FILE *in_file;
int i, no_cases;

in_file = fopen("input.txt","r");
if(in_file == NULL) exit(1);

fscanf(in_file, "%d", &no_cases);

for (i=0; i<no_cases; i++) {
int h, w, n, h_room, w_room;
fscanf (in_file, "%d %d %d", &h, &w, &n);

if (h < 1 || w > 99 || n < 1 || n > h*w)
printf ("data is not correct");

if (n % h == 0) {
h_room = h;
w_room = n / h;
}

else {
h_room = n % h;
w_room = (n / h) +1;
}

printf("%d%02d\n", h_room, w_room);
}
fclose(in_file);
}
