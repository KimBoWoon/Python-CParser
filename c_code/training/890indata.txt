
#include <stdio.h>
#include <stdlib.h>

int fibonacci(int n){

	if(n==0)		return 0;
	else if(n==1)	return 1;
	else if(n>1)	return fibonacci(n-1)+fibonacci(n-2);
}

void main(){

	FILE *in_file;
	int i;
	int num_cases;
	
	in_file=fopen("input.txt","r");
	if(in_file == NULL) exit(1);

	
	fscanf(in_file,"%d", &num_cases);

	for(i=0; i<num_cases; i++){

		int value;

		fscanf(in_file, "%d", &value);

		printf("%d\n", fibonacci(value));

	}
	fclose(in_file);
}
