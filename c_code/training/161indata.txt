
#include<iostream>
#include<fstream>

using namespace std;

int McCarthy(int n){
	if (n > 100) return n-10;
	else McCarthy(McCarthy(n + 11));
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;

	int t;
	in >> t;
	for (int i = 0; i < t; i++){
		int n;
		in >> n;
		cout << McCarthy(n) << endl;
	}

	in.close();
	return 0;
}
