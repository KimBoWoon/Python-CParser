int Palindrome(int* text, int first, int last)
{
	if (last <= first)
		return 1;
	else if (text[first] != text[last])
		return 0;
	else 
		return Palindrome(text, first + 1, last - 1);

}

int baseConversion(int testNum,int base, int* ansArr)
{
	int i=0;
	while (testNum != 0) { 
		ansArr[i++] = testNum % base;
		testNum /= base;		
	}			

	return Palindrome(ansArr, 0, i-1);
}
