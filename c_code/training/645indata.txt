#include<iostream>
#include<fstream>
#include <cstdlib>

int factorial(int n);

int main()
{
	std::fstream infile;
	int testcase;
	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << " file open error";
		exit(1);
	}
	infile >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		int num;
		infile >> num;
		std::cout << factorial(num) << std::endl;
	}
	return 0;
}

int factorial(int n)
{
	if (n <= 1) return 1;

	return n*factorial(n - 1);
}
