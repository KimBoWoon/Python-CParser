
#include <iostream>
#include <fstream>

int uclid(int a, int b);

int main()
{
	std::ifstream infile("input.txt");
	int numCase;
	infile >> numCase;

	for(int i=0; i<numCase; i++)
	{
		int a, b;
		infile >> a >> b;
		std::cout<<uclid(a,b)<<std::endl;
	}
}

int uclid(int a, int b)
{
	if( b==0)
		return a;
	else
		return uclid(b, a%b);
}
