#include<stdio.h>
#include<stdlib.h>

int main(){
    FILE *in_file;
    int i;
    int no_cases;
    
    in_file = fopen("input.txt","r");
    if(in_file == NULL) exit(1);
    
    fscanf(in_file,"%d",&no_cases);
    
    for(i=0; i<no_cases; i++)
    {
        int N, H, W;
        int a, b;
        
        fscanf(in_file, "%d %d %d",&H, &W,&N);
        a = (N - 1) % H + 1;    
        b = (N - 1) / H + 1;    
         printf("%d%02d\n", a, b);
    }
    fclose(in_file);
}
