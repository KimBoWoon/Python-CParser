#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int num,temp;
		instream >> num;

		int div = 2;
		int res = 0;

		while (div<65)
		{
			int neck[20];
			int n = 0; 
			int flag = 1;  
			temp = num;
			while (temp !=0)
			{
				neck[n++] =temp% div;    
				temp /= div;
			}

			for (int j = 0; j < n / 2 ; j++)
			{
				if (neck[j] == neck[n - 1 - j])    
					continue;
				else		
				{
					flag = 0;break;
				}
			}

			if (flag == 1)
			{
				res = 1; break;
			}
			div++;
		}
		cout << res << endl;
	}
}
