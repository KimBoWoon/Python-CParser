
#include<iostream>
#include<fstream>
#include<math.h>

using namespace std;

int trinum(int num);

int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;
	while (cases--){
		int num;
		bool flag = false;
		inFile >> num;
		for (int i = (sqrt(1+8*num)-1)/2; i > 0; i--){
			if (num - trinum(i) < 0){
				continue;
			}
			if (flag)
				break;
			for (int j = 1; j < num; j++){
				if (num - trinum(i) - trinum(j) < 0){
					break;
				}
				if (flag)
					break;
				for (int k = 1; k < num; k++){
					if (num - trinum(i) - trinum(j) - trinum(k) == 0){
						flag = true;
						break;
					}
					if (num - trinum(i) - trinum(j) - trinum(k) < 0){
						break;
					}
				}
			}
		}
		if (flag)
			cout << 1 << endl;
		else
			cout << 0 << endl;

		flag = false;

	}
}

int trinum(int num){
	return num*(num + 1) / 2;
}
