#include<iostream>
#include<fstream>
using namespace std;

class Init{
public:
	int calc(int num);
};

int Init::calc(int num){
	for(int i=2; i<100000; i++){
		if(num % i == 0 && i != num){
			return 0;
			break;
		}
	}
	return 1;
}

int main(){
	Init init;

	ifstream inStream;
	int numCases;
	int number;

	inStream.open("input.txt");

	inStream >> numCases;

	for(int i=0; i<numCases; i++){
		inStream >> number;

		cout << init.calc(number) << endl;
	}

	inStream.close();
}
