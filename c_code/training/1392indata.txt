
#include<stdio.h>

int main(void)
{
	FILE *fp = NULL;
	int testLine;

	fp = fopen("input.txt", "r");

	fscanf(fp, "%d", &testLine);
	while(testLine--)
	{
		int w, h, n, floorNumberTmp, floorNumber, roomNumber;
		fscanf(fp, "%d %d %d", &h, &w, &n);

		floorNumberTmp = n % h;
		if(floorNumberTmp == 0)
		{
			floorNumber = h;
			roomNumber = n / h;
		}
		else
		{
			floorNumber = floorNumberTmp;
			roomNumber = n / h + 1;
		}

		printf("%d", floorNumber);
		if((int)(roomNumber * 0.1f) == 0)
		{
			printf("0");
		}
		printf("%d\n", roomNumber);
	}
	
	fclose(fp);
	return 0;
}
