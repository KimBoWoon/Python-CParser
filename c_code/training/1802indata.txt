#include<stdio.h>
#include<stdlib.h>

int pa(int,int,int ,int , char []);
int pb(int,int,int ,int , char []);
void result(int,int,int, char a[]);

int main()
{
	FILE *fp;
	int i,j,k,numTestCases;
	char parent[50]={' ',};
	fp=fopen("input.txt","r");
	if(fp==NULL) return ;
	fscanf(fp,"%d",&numTestCases);
	for(i=0; i<numTestCases; i++){
		char par;
		int cntOpen=0,cntClose=0;
		int cnt=0;
		int index=0;
		fscanf(fp,"%s",parent);
		for(j=0; j<50; j++){
			cnt++;
			if(parent[cnt]==NULL)
				break;
		}
		result(pa(cntOpen,cntClose,index,cnt,parent),pb(cntOpen,cntClose,index,cnt,parent),cnt,parent);
	}
	return 0;
	fclose(fp);
}

int pa(int cO,int cC,int idx,int cnt,char a[])
{
	
	if(idx==cnt){
		return cO;
	}
	else if(a[idx]=='('){
		cO++;
		pa(cO,cC,idx+1,cnt,a);
	}
	else if(a[idx]==')'){
		cC++;
		pa(cO,cC,idx+1,cnt,a);
	}
}

int pb(int cO,int cC,int idx,int cnt,char a[])
{
	
	if(idx==cnt){
		return cC;
	}
	else if(a[idx]=='('){
		cO++;
		pb(cO,cC,idx+1,cnt,a);
	}
	else if(a[idx]==')'){
		cC++;
		pb(cO,cC,idx+1,cnt,a);
	}
}

void result(int pre,int after,int cnt,char a[])
{
	if(pre<after || pre>after)
		printf("NO\n");
	else if(pre==after){
		if(a[0]==')')
			printf("NO\n");
		else if(a[cnt-1]=='(')
			printf("NO\n");
		else
			printf("YES\n");
	}
}
