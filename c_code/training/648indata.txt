
#include<iostream>
#include<fstream>
using namespace std;

int factorial(int num);

int main()
{
	int count = 0;

	ifstream file;
	file.open("input.txt");
	file >> count;

	int num = 0;
	for (int i = 0; i < count; i++)
	{
		file >> num;
		cout << factorial(num) << endl;
	}

	return 0;
}

int factorial(int num)
{
	if (num == 0)
		return 1;
	else
		return num*factorial(num - 1);
}
