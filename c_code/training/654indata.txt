

#include <stdio.h>
#include <stdlib.h>

int factorial(int num) {
	if (num==0) return 1;
	else return num*factorial(num-1);
}
void main(void) 
{
	int test=0, i, num;
	FILE *file = fopen("input.txt", "r"); 
	if(NULL == file) exit(1);

	fscanf(file, "%d", &test);

	for(i=0; i<test; i++) 
	{
		fscanf(file, "%d", &num);
		printf("%d\n", factorial(num));
	}
	fclose(file);
}
