#include <iostream>
#include <fstream>

void palindrome(int number);

int main(){

	int testCases = 0;
	std::ifstream inFile("input.txt");
	inFile >> testCases;

	for (int i = 0; i < testCases; i++){
		int number = 0;
		inFile >> number;
		palindrome(number);
	}

	return 0;
}

void palindrome(int number){
	int temp = number;
	
	for (int i = 2; i <= 64; i++) {
		int numbers[30] = { 0 };
		int count = 0;
		number = temp;
		while (1) {
			if (number == 0 || count == 29) {
				break;
			}
			numbers[count++] = number % i;
			number /= i;
		}
		for (int j = 0; j < count; j++){
			if (numbers[j] != numbers[count - j - 1]){
				break;
			}
			if (j == count - 1){
				std::cout << 1 << std::endl;
				return;
			}
		}

	}
	std::cout << 0 << std::endl;
}
