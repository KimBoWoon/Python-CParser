#include <stdio.h>
#include <stdlib.h>
int t,n;
int arr[45];
int arr2[1001];
FILE *file;
void cal();
void main()
{	
	int a;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	cal();
	for(a=0; a<t; a++)
	{
		fscanf(file, "%d", &n); 
		printf("%d\n",arr2[n]);
	}	
	fclose(file);	
}
void cal()
{
	int i,j,k;
	for (i=1;i<45;i++) {
		arr[i] = i*(i+1)/2;
	}
	for (i=1;i<45;i++) 
	{
		for (j=i;j<45;j++) 
		{
			for (k=j;k<45;k++) 
			{
				if (arr[i]+arr[j]+arr[k] < 1001) 
					arr2[arr[i]+arr[j]+arr[k]]=1;
			}
		}
	}
}
