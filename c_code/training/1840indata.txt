
#include <iostream>
#include <string>
#include <fstream>
#include <cstdlib>

bool check_vps(int &checknum,std::string teststring);

int main()
{
	std::fstream infile;
	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << "input file error";
		exit(1);
	}
	int testcase;
	std::string teststring;
	infile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int checknum = 0;
		infile >> teststring;
		if (check_vps(checknum, teststring))
		{
			std::cout << "YES" << std::endl;
		}
		else std::cout << "NO" << std::endl;

	}
	return 0;
}

bool check_vps(int &checknum, std::string teststring)
{
	if (teststring.at(teststring.length() - 1) == ')')
	{
		checknum--;
	}
	else
	{
		checknum++;
	}
	teststring.erase(teststring.length() - 1);

	if (teststring.length() == 0)
	{
		if (checknum == 0) return true;
		else return false;
	}
	if (checknum > 0) return false;
	return check_vps(checknum, teststring);
}
