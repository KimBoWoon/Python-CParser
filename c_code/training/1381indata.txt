#include <iostream>
#include <fstream>
#define MAX 2147483647
using namespace std;


int gcd(int a, int b) {
	if (b == 0)  return a;
	return gcd(b, a % b);
}

int main() {
	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;
	while (cases--) {
		int num_c, num_m;
		file >> num_c >> num_m;
		int multi_m = num_m;
		for (int i = 2;; i++) {
			if (num_c == 1) {
				cout << num_m << endl;
				break;
			}

			if (num_c * i - num_m < 0) {
				if (multi_m * (i + 1) <= MAX)	continue;
				else {
					cout << num_m << endl;
					break;
				}
			}

			num_c = num_c * i - num_m;
			num_m *= i;
			int g = gcd(num_c, num_m);
			num_c /= g;		num_m /= g;

			multi_m *= i;

		}
	}


	return -1;
}
