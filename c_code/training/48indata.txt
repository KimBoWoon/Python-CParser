#include <iostream>
#include <fstream>
using namespace std;

int prime(int num){
	int i;
	for (i = 2; i < num; i++)
		if (num%i == 0) break;
	
	if (i == num)
		return 1;
	else
		return 0;
}

int main(void){
	ifstream in("input.txt");
	int testcases;
	in >> testcases;

	int value = 0;
	for (int i = 0; i < testcases; i++){
		in >> value;
		cout << prime(value) << endl;
	}
	return 0;
}
