
#include <iostream>
#include <fstream>
using namespace std;

int gcd(int n1,int n2)
{
	if(n2==0)
		return n1;
	else
		gcd(n2,n1%n2);
}

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >>testcase;
	for(int i=0;i<testcase;i++)
	{
		int a,b,c,d ;
		instream >> a>>b>>c>>d;

		int num1, num2;
		num1 = a*d+c*b;
		num2 = b*d;

		cout<< num1/gcd(num1,num2)<<" "<<num2/gcd(num1, num2)<<endl;
	}

}
