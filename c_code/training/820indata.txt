


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int Fibo(int val);

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value;				

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value;
		cout << Fibo(value) << endl;
	}

	inStream.close();
	return 0;
}

int Fibo(int val)
{
	if (val == 0)
	{
		return 0;
	}
	else if (val == 1)
	{
		return 1;
	}
	else
	{
		return Fibo(val - 1) + Fibo(val - 2);
	}
}
