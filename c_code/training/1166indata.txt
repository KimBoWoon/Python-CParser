
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream>> testcase;

	for(int i=0;i<testcase;i++)
	{
		int num,res=0;
		instream >> num;
		
		for(int a=1;a*(a+1)/2<num;a++)    
		{
			for(int b=1;b*(b+1)/2<num;b++)
			{
				for(int c=1 ;c*(c+1)/2<num;c++)
				{
					if(num==(((a*(a+1))/2)+((b*(b+1))/2)+((c*(c+1))/2)))     
						res=1;
				}
			}
		}
		cout<<res<<endl;
	}

}
