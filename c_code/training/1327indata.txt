#include <iostream>
#include <fstream>


using namespace std;

typedef long long lld;

lld gcd(lld a, lld b){
	return b ? gcd(b, a%b) : a;
}

struct henry{
	henry(lld p, lld q){
		lld g = gcd(p, q);
		a = p / g;
		b = q / g;
	}
	lld a, b;
};

int main()
{
	int T;
	int a, b;
	ifstream file;
	file.open("input.txt");

	file >> T;
	while(T--){
		file >> a >> b;
		henry z(a, b);
		while (z.a > 1){
			lld n = (z.b - 1) / z.a + 1;
			z = henry(z.a*n - z.b, z.b*n);
		}
		printf("%lld\n", z.b);
	}

	return 0;
}
