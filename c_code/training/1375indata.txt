#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

void cal(int a, int b);

int main()
{
	int nocases;
	int a,b;

	fstream infile;
	
	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i = 0; i<nocases; i++)
	{
		infile >> a;
		infile >> b;
		cal(a,b);
	}

	return 0;
}
