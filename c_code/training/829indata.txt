#include <stdio.h>
#include <stdlib.h>

int fibo(int n){
    if( n==0 )       return 0;
    else if(n==1) return 1;
    else                 return fibo(n-1) + fibo(n-2);
}

int main(void){
    FILE* file;
    file = fopen("input.txt","r");
    
    int cases;
    fscanf(file, "%d", &cases);
    while(cases--){
        int n;
        fscanf(file, "%d", &n);
        printf("%d\n",  fibo(n));
    }
    return 0;
}
