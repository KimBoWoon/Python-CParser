

#include<stdio.h>

int DetermindEureka(int n, int* arr, int arrSize)
{
        int i, j, k;
        int eureka = 0;

        for(i=0; i<44; ++i)
                for(j=0; j<44; ++j)
                        for(k=0; k<44; ++k)
                                if(arr[i]+arr[j]+arr[k]==n)     eureka=1;

        return eureka;
}

int main()
{
        FILE*inFile;
        int testNum = 0;
        int num = 0;
        int i, j;
        int triangleNum[45];

        inFile = fopen("input.txt", "r");

        fscanf(inFile, "%d", &testNum);
        for(j=1; j<45; ++j)
                triangleNum[j-1] = j*(j+1)/2;

        for(i=0; i<testNum; ++i)
        {
                fscanf(inFile, "%d", &num);
                printf("%d\n", DetermindEureka(num, &triangleNum[0], 45));
        }
        fclose(inFile);
        return 0;
}
