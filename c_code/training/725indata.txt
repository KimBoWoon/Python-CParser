  

#include <iostream>
#include <fstream>
using namespace std;
int checkPalindrome(char str[], int first, int last); 

int main()
{
	ifstream inFile("input.txt");  
	int no_cases=0,num=0;
	inFile>> no_cases;
	for(int l =0; l < no_cases; l++){
		char str[1001]={0,};
		inFile>>str;
		int len=0;
		for(len=0; str[len]!='\0'; len++);
		cout<<checkPalindrome(str,0,len-1)<<endl;

	}
	return 0;
}
int checkPalindrome(char str[], int first, int last)
{ if(last <= first) 
	return 1; 
	else if(str[first] != str[last]) 
		return 0; 
	else 
		return checkPalindrome(str, first+1, last-1); 
}
