
#include <stdio.h>
#include <stdlib.h>

int factorial(int n){

	if(n<=1)	return 1;
	else		return n*factorial(n-1);
}


void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){

		int value;

		fscanf(in_file,"%d",&value);

		printf("%d\n", factorial(value));

	}
	fclose(in_file);
}
