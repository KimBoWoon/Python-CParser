

#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int gcd(int num1, int num2);

int main()
{
	int numOfCase = 0;	
	int num1, num2;

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> num1;
		input >> num2;

		cout << gcd(num1, num2) << endl;
	}
	input.close();
	return 0;
}
int gcd(int num1, int num2)
{
	int a, b;
	if (num1 >= num2){
		a = num1;
		b = num2;
	}
	else{
		a = num2;
		b = num1;
	}
	int r;
	if (b == 0){
		return a;
	}
	do
	{
		r = a%b;
		a = b;
		b = r;
	} while (r != 0);

	return a;
}
