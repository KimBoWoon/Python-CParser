

#pragma warning(disable:4996)

#include<stdio.h>
#include<stdlib.h>
#include<string.h>


int Palindrome_check(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return Palindrome_check(str, first + 1, last - 1);
}



int main()
{
	FILE *in_file;
	int i;
	int testcase;
	char test_line[1001];

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
	{
		printf("file open failed");
		return 0;
	}

	fscanf(in_file, "%d", &testcase);

	for (i = 0; i < testcase; i++)
	{
		
		fscanf(in_file, "%s", &test_line);

		printf("%d \n",Palindrome_check(test_line, 0, strlen(test_line) - 1));


	}

	fclose(in_file);

	return 0;
}
