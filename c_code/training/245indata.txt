#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){
    if( b== 0 ) return a;
    else{
        return (a>b)? gcd(b, a%b) : gcd(a, b%a);
    }
}
int main(void){
    FILE* file;
    file = fopen("input.txt","r");
    
    int cases;
    fscanf(file, "%d", &cases);
    while(cases--){
        int a,b,c,d;
        fscanf(file, "%d%d%d%d",  &a,&b,&c,&d);
		 int upNum,doNum;
		 upNum = a*d + b*c;
		 doNum = b*d;
		 int gcdNum = (upNum>doNum)? gcd(upNum, doNum) : gcd(doNum, upNum);
        printf("%d %d\n", upNum/gcdNum, doNum/gcdNum);
         }
    return 0;
}
