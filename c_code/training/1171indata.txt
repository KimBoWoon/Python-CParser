

#include <iostream>
#include <fstream>
#include <string.h>

using namespace std;

int checkEureka(int k);

int main()
{
	ifstream infile;
	int testCase;
	int k; 
	
	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> k;
		cout << checkEureka(k) << endl;


	}
	infile.close();

	return 0;
}

int checkEureka(int num)
{
	int T[45]; 
	int sum;
	int result=0; 
	for (int i = 1; i < 45; i++)
		T[i] = i*(i + 1) / 2;

	for (int i = 1; i < 45; i++)
	{
		for (int j = 1; j < 45; j++)
		{
			for (int k = 1; k < 45; k++)
			{	
				sum = T[i] + T[j] + T[k];
				if (num == sum)
				{
					result = 1;
					break;
				}
				else
					continue;

			}
		}
	}

	return result;

}
