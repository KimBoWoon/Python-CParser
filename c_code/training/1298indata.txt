


#include<stdio.h>

void henry(int _a, int _b)
{
	int i = 2;
	int a;
	do
	{
		a = _a * i - _b;        
		
		if(a < 0)                   
		{
			continue;
		}

		if(a > 0)                   
		{
			_a = a;
			_b *= i;
			continue;
		}
		
		if(a == 0)                  
		{
			printf("%d\n", i);      
			return;
		}
	} while(i++);
}

int main(void)
{
	FILE *fp = NULL;
	int testCase;

	fp = fopen("input.txt", "r");       
	fscanf(fp, "%d", &testCase);        
	while(testCase--)
	{
		int a, b;
		fscanf(fp, "%d %d", &a, &b);    
		henry(a, b);                    
	}

	fclose(fp);                         
	return 0;
}
