#include<stdio.h>
#include<stdlib.h>
#include<math.h>

int main()
{
	FILE * in_file;
	int i, j;
	int no_cases, no_data;
	int count=0;

	in_file = fopen("input.txt", "r");
	if(in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &no_cases);

	for(i=0; i<no_cases; i++)
	{
		fscanf(in_file, "%d", &no_data);

		for (j = 2; j <= sqrt(no_data) + 1; j++)	    
		{
			if (no_data == 2)                           
				continue;
			else
			{
				if((no_data == 1) || (no_data % j == 0)) 
				{
					count++;
					break;
				}
			}
		}
		if (count == 0)									 
			printf("1\n");
		else
			printf("0\n");
		count = 0;
	}
	fclose(in_file);

}
