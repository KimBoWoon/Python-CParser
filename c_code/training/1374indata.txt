#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main(){
	ifstream inStream;
	int numTestCases;

	inStream.open("input.txt");
	
	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++){
		int s1, m1;
		int ans = 1;
		inStream >> s1 >> m1;

		while ((s1 * ans - m1) != 0){
			if (s1 * ans > m1){
				s1 = s1 * ans - m1;
				m1 = m1 * ans;
			}
			ans++;
		}
		cout << ans << endl;
	}
	inStream.close();
	return 0;
}
