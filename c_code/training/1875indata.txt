#include <stdio.h>
#include <string.h>

int Parenthesis(char s[]){

	int i;
	int count_a=0; int count_b=0;
	int length = strlen(s);
	
	for(i=0; i<length; i++){

 		if(s[i]=='(')		count_a++;
		else if(s[i]==')')	count_b++;
	}

	if(s[0]=='('&&s[length-1]==')'){

		if(count_a==count_b)			return 1;
		else if(count_a!=count_b)		return 0;
	}

	else return 0;
}

void main(){

	FILE *in_file;
	int i, num_cases;
	in_file=fopen("input.txt","r");

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){

		char s[51];

		fscanf(in_file,"%s", s);

		
		if(Parenthesis(s)==1)		printf("YES\n");
		else if(Parenthesis(s)==0)	printf("NO\n");
	}
}
