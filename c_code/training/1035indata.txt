#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

void init_numbers(std::ifstream& inFile, int numbers[15], int count);
void ducci_sequence(int numbers[15], int count, int index);

int main(void){

	std::ifstream inFile("input.txt");
	if (inFile.fail()){
		std::cout << "not file" << std::endl;
	}

	int testCases = 0;
	inFile >> testCases;

	int count = 0;
	

	for (int i = 0; i < testCases; i++){

		int numbers[15] = { 0 };
		inFile >> count;
		
		init_numbers(inFile, numbers, count);
		ducci_sequence(numbers, count, 0);
	}

	return 0;
}

void init_numbers(std::ifstream& inFile, int numbers[15], int count){
	
	for (int i = 0; i < count; i++){
		inFile >> numbers[i];
	}
}

void ducci_sequence(int numbers[15], int count, int index){

	int first_number = numbers[0];
	for (int i = 0; i < count; i++){

		if (i == count - 1){
			numbers[i] = abs(numbers[i] - first_number);
		}
		else{
			numbers[i] = abs(numbers[i] - numbers[i + 1]);
		}
	}

	int zero_count = 0;
	for (int i = 0; i < count; i++){

		if (numbers[i] == 0){
			zero_count++;
		}
		else{
			break;
		}
	}
	if (zero_count == count){
		std::cout << "ZERO" << std::endl;
		return;
	}
	else{
		if (index == 1000){
			std::cout << "LOOP" << std::endl;
			return;
		}
		else{
			ducci_sequence(numbers, count, index + 1);
		}
	}
}
