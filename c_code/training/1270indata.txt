
#include <stdio.h>
#define false 0
#define true 1
#define MAXINPUT 1000


int is_prime(int n){
	int i;
	if(n==2)
		return true;
	if ( n%2 == 0)
		return false;
	for(i=3; i*i<n+1; i+=2)
		if(n%i==0)
			return false;
	return true;
}



void getGoldbach(int arr[MAXINPUT], int num, int idx1, int idx2){
	if( arr[idx1] && arr[idx2] ){
		printf("%d %d\n", idx1, idx2);
		return ;
	}
	else return getGoldbach(arr, num, idx1-1, idx2+1);
}

int main(void){
	FILE *fp = fopen("input.txt", "r");
	int arr[MAXINPUT+1] = {0};
	int i;
	int numcase;

	for(i=0; i<=MAXINPUT; i++){
		arr[i]=is_prime(i);
	}

	fscanf(fp, "%d", &numcase);
	for(i=0; i<numcase; i++){
		int num;
		fscanf(fp, "%d", &num);
		getGoldbach(arr, num, num/2, num-num/2); 
	}
}
