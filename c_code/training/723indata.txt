

#include <iostream>
#include <fstream>
#include <string>

using namespace std;


int palindrome(string str,int first,int last)  
{
    if(last<=first)  
        return 1;
    else if(str.at(first)!=str.at(last)) 
        return 0;
    else 
        return palindrome(str,first+1,last-1);  
}

int main()
{

    ifstream file;
    
    file.open("input.txt");
    
    if(file.is_open()==false) 
        cout<<"입력파일이 없습니다."<<endl;
    
    else
    {	
        int case_num;
        file>>case_num;
        
        for(int i=0; i<case_num; i++)
        {   
            string str;
            file>>str;
            cout<<palindrome(str,0,str.length()-1)<<endl;
        }

    }

        file.close();

}
