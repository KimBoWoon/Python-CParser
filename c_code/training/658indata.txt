
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>

int fac(int n){
	if(n==0) 
		return 1;
	else
		return n*fac(n-1);
}

int main(){
	FILE *fp;
	int i;
	int numcases;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);
	
	for(i=0;i<numcases;i++){
		int n;
		int result=0;

		fscanf(fp,"%d",&n);
		result=fac(n);

		printf("%d\n",result);
		
	}
	fclose(fp);
}
