#include <stdio.h>
#include <stdlib.h>
#include <math.h>
int main()
{
	FILE *in;
	int t;
	in = fopen("input.txt","r");
	if(in==NULL) exit(1);
	fscanf(in,"%d",&t);
	while(t--)
	{
		int h,w,n;
		int room;
		fscanf(in,"%d %d %d",&h,&w,&n);
		if(n%h==0) room = h*100 + n/h;
		else room = (n%h)*100+floor((double)n/h)+1;
		printf("%d\n",room);
	}
}
