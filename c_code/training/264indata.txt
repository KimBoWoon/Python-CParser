


#include <stdio.h>

#define COMPARE_BIG(A,B) (A>B)?A:B
#define COMPARE_SMALL(A,B) (A>B)?B:A

int Euclid(int num1, int num2){
	if(num2==0) return num1;	
	else if(num2>0) return Euclid(num2, num1%num2);	
}

void Sum(int n1,int n2,int m1,int m2){
	int k1,k2,divisor;

	k1= n1*m2+m1*n2; 
	k2= n2*m2; 
	divisor = Euclid(COMPARE_BIG(k1,k2), COMPARE_SMALL(k1,k2)); 

	printf("%d %d\n", k1/divisor,k2/divisor);
}

int main(){
	int cases,i;
	int n1,n2,m1,m2;

	FILE *infile;
	infile = fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0;i<cases;i++){
		fscanf(infile,"%d %d %d %d",&n1,&n2,&m1,&m2);
		Sum(n1,n2,m1,m2);
	}
	fclose(infile);
	return 0;
}
