



#include <stdio.h>
#include <stdlib.h>

int Bach(int n)
{
	int i;
	
	for(i = n/2; i > 1; i--)	
		if(!(n%i))	
			return 0;	
	
	return 1;	
}

void main(void) 
{
	FILE *inFile;
	int i;
	int numCases;
	int num;
	
	inFile = fopen("input.txt", "r");

	if(inFile == NULL) 
			exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i = 0; i < numCases; i++) 
	{
		int j;

		fscanf(inFile, "%d", &num);	

		num/=2;						

		for(j = 0; j <= num-2; j++) 
		{	
			if(Bach(num-j) && Bach(num+j))
			{
				printf("%d %d\n", num-j, num+j);
				break;
			}
		}
	}
	
	fclose(inFile);
}
