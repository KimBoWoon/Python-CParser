
#include<stdio.h>
#include <stdlib.h>
int recursive(int a, int b)
{
	if (a >= b)
	{
		if (b == 0)
		{
			return a;
		}
		else
		{
			return recursive(b, a%b);
		}
	}
	else
	{
		if (a == 0)
		{
			return b;
		}
		else
		{
			return recursive(a, b%a);
		}
	}
}

int main()
{
	FILE *in_file;
	int i;
	int numTestCases;
	int num1, num2;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++)
	{
		fscanf(in_file, "%d %d", &num1, &num2);
		printf("%d\n", recursive(num1, num2));
	}
	fclose(in_file);
}
