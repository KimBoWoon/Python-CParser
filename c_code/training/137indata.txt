

#include <iostream>
#include <fstream>


using namespace std;


int McCarthy(int n) 
{
    if(n>100)
        return n-10;
    else
        return McCarthy(McCarthy(n+11)); 
}

int main()
{

    ifstream file;
    
    file.open("input.txt");
    
    if(file.is_open()==false) 
        cout<<"입력파일이 없습니다."<<endl;
    
    else
    {	
        int case_num;
        file>>case_num;
        
        for(int i=0; i<case_num; i++)
        {   
            int num;
            file>>num;
            cout<<McCarthy(num)<<endl;
        }

    }

        file.close();

}
