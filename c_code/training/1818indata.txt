#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define max 51		

void PS(char *str);	

int lCount=0, rCount=0, len=0;	

void main()
{
	FILE *f;
	int i;
	int count;
	
	char ps[max];

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);
	for(i=0; i<count; i++)
	{
		lCount=0, rCount=0, len=0;	
		fscanf(f, "%s", &ps);
		PS(ps);	
	}
	fclose(f);
}
void PS(char *str)	
{
	if(strlen(str) == lCount + rCount)	
	{
		if(lCount == rCount)		
			printf("YES\n");		
		else						
			printf("NO\n");			
	}
	else							
	{
		if(str[len] == '(')			
			lCount++;				
		else if(str[len] == ')')	
			rCount++;				
		if(rCount>lCount)			
		{
			printf("NO\n");			
			return;					
		}
		else						
		{
			len++;					
			PS(str);				
		}
	}
}
