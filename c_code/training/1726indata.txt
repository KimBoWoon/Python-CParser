#include <iostream>
#include <fstream>

using namespace std;
int main(){
	
	ifstream file;
	file.open("input.txt");

	int t, n;
	int flag;
	
	file >> t;

	while (t--){
		file >> n;
		flag = 1;

		for (int i = 2; i <= 64; i++){
			int reverse = 0, rem, temp;
			temp = n;

			while (temp != 0){
				rem = temp % i;
				reverse = reverse * i + rem;
				temp /= i;
			}

			if (reverse == n){
				cout << 1 << endl;
				flag = 0;
				break;
			}
		}
		if (flag){
			cout << 0 << endl;
		}
	}
	return 0;
}
