#include <iostream>
#include <fstream>
using namespace std;

bool primeNumber(int n);

int main() {
	ifstream in;
	int testCase = 0;

	in.open("input.txt");

	in >> testCase;

	for (int i = 0; i < testCase; i++) {
		int n = 0;

		in >> n;

		if (primeNumber(n))
			cout << 1 << endl;
		else
			cout << 0 << endl;
	}
}

bool primeNumber(int n) {
	for (int i = 2; i * i <= n; i++) {
		if (n % i == 0)
			return false;
	}
	return true;
}
