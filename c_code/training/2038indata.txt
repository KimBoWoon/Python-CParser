#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;

int abs(int num) {
	return (num < 0) ? num * -1 : num;
}
int main() {
	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;
	while (cases--) {
		int set, k;
		file >> set >> k;
		int *data = new int[set];
		for (int i = 0; i < set; i++)
			file >> data[i];

		sort(data, data + set);
		int index = 0, std = k;


		int gap = 2147483647, count = 0;
		for (int i = 0; i < set - 1; i++) {
			const int num = k - data[i];
			int start = 0, end = set - 1, mid = (start + end) / 2;
			while (start <= end) {
				mid = (start + end) / 2;

				if (data[mid] == num)	break;
				if (end - start <= 1)	break;

				if (num > data[mid])
					start = mid;
				else if (num < data[mid])
					end = mid;
			}

			for (int j = -1; j < 2; j++) {
				if (mid + j == i)	continue;
				if (mid + j < i)	continue;

				int _gap = abs(k - (data[mid + j] + data[i]));
				if (gap > _gap) {
					gap = _gap;
					count = 1;
				}
				else if (gap == _gap)
					count++;
			}
		}

		cout << count << endl;
	}

	return 0;
}
