

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int factorial(int n);
void popList(int index, int n);
void makeResult(int n, int k, int index);

char alpha[12] = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l'};
char rearray[12];

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int cases, size, rank;
	int i, j;

	fscanf(inFile, "%d", &cases);
	for(i=0; i<cases; ++i){
		fscanf(inFile, "%d %d", &size, &rank);
		for(j=0; j<size; ++j) 
			rearray[j] = alpha[j];

		makeResult(size, rank, 1);
		printf("\n");
	}
	
	fclose(inFile);
}

int factorial(int n)
{
	if(n<=1) 
		return 1;
	return n*factorial(n-1);
}

void popList(int index, int n)
{
	int i;

	if(index == strlen(rearray)-1) 
		rearray[index] = NULL;
	else 
		for(i=index; i<n; ++i)	
			rearray[i] = rearray[i+1];
}

void makeResult(int n, int k, int index)
{
	int fac = factorial(n-index);
	int life = k / fac;
	int etc = k % fac;
	int i = 0;

	if(n==index) {
		printf("%c", rearray[0]);
		return;
	}

	if(etc==0 && life != 0) { 
		life -= 1; etc = fac; } 

	printf("%c", rearray[life]);

	popList(life, n);
	makeResult(n, etc, index+1);
}
