

#include <iostream>
#include <fstream>
#include <cmath>
using namespace std;

void checking(int *arr, int arrNum){
	int cnt = 0; 
	for (int i = 0; i < arrNum; i++){
		if (arr[i] == 0)
			cnt++;
	}
	if (cnt == arrNum)
		cout << "ZERO" << endl;
	else
		cout << "LOOP" << endl;
}
double abs(double &a){
	if (a <= 0)
		return (-a);
	else
		return a;
}
int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;
	int arr[15];
	int arrNum;
	int newArr[15];
	int temp;
	for (int i = 0; i < testcases; i++){
		in >> arrNum;
		for (int j = 0; j < arrNum; j++){in >> arr[j];}

		for (int k = 0; k < 1000; k++){
			if (k%arrNum == 0)
				temp = arr[k%arrNum];
			if (k%arrNum ==arrNum-1)
				arr[k%arrNum] = abs(double(temp) - double(arr[k%arrNum]));
			else
				arr[k%arrNum] = abs(double(arr[(k + 1) % arrNum]) - double( arr[k%arrNum] ));
		}
		checking(arr, arrNum);
	}
}
