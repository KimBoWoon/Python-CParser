#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int palind(char str[],int n, int len) 
{
	if(n>=len)
		return 1;

	else if(str[n]==str[len])
		return palind(str,n+1,len-1);
	else
		return 0;

}

int main ()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase;i++)
	{
		char str[1000];
		instream >>str;
		cout<<palind(str,0,strlen(str)-1)<<endl;;
	}
}
