#include <iostream>
#include <fstream>
#include <cstdlib>

int gcd(int a, int b);

int main()
{
	std::fstream infile;
	int testcase;
	int num1, num2;

	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << "error";
		exit(1);
	}

	infile >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		infile >> num1 >> num2;
		std::cout<<gcd(num1, num2)<<std::endl;
	}
	return 0;
}

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
