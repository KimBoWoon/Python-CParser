#include <stdio.h>

int fib(int n){
    if(n < 2) return n;
    else return (fib(n-1) + fib(n-2));
}

int main(){
    int testCases, cases, n=0;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++){
        fscanf(fin, "%d", &n);
        printf("%d\n", fib(n));        
    }
    fclose(fin);            
    
    return 0;           
}
