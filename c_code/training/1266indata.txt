
#include <iostream>
#include <fstream>
using namespace std;

int isPrime(int num){ 
	int divi=num/2;

	for(divi=2;divi*divi<=num;divi++){
		if(num%divi==0){
			return 0;
		}
	}
	return 1;
}

void goldbach(int n1,int n2,int n){ 
	
	if(isPrime(n1)==1&&isPrime(n2)==1){
			if(n1+n2==n){
				cout<<n1<<" "<<n2<<endl;
				return;
			}
		}
		else
			goldbach(n1-1,n2+1,n);
		
	
}
int main(){
	ifstream fp;
	int numcases;
	int n;

	fp.open("input.txt"); 

	fp>>numcases;

	for(int i=0;i<numcases;i++){

		fp>>n;
		int n1=n/2;
		int n2=n/2;
		goldbach(n1,n2,n);

	}
	return 0;
}
