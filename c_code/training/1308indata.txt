


#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#pragma warning(disable : 4996)
int uclid(int num1, int num2)
{
	if (num2 == 0) 
	{
		return num1;
	}
	else 
	{
		return uclid(num2, num1%num2);
	}
}
void main(void)
{
	FILE* inFile;
	int testCase = 0, i, up, down, henryNum=0, gcd;
	int prime[1000] = { 0 };
	int realPrime[170] = { 0 };
	inFile = fopen("input.txt", "r"); 
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase); 
	for (i = 0; i < testCase; i++)
	{
		fscanf(inFile, "%d %d", &up, &down);
		gcd = uclid(up, down);
		up /= gcd;
		down /= gcd;
		henryNum = (down-1) / (up + 1);
		while ((up*henryNum - down)!=0)
		{
			if (up*henryNum - down > 0)
			{
				up = up*henryNum - down;
				down = down*henryNum;
				henryNum = (down - 1) / (up + 1);
				gcd = uclid(up, down);
				up /= gcd;
				down /= gcd;
			}
			else
				henryNum++;
		}
		printf("%d\n", henryNum);
	}
}
