

#include <stdio.h>

int euclid(int m, int n){
	return (n==0) ? m : euclid(n, m%n);
}
void fraction(int a,int b){
	int gcd = euclid(a,b);
	printf("%d %d\n",a/gcd,b/gcd);
}

void main(){

	FILE *in_file;
	int no_cases;

	in_file = fopen("input.txt", "r");

	if(in_file == NULL) exit(1);

	fscanf(in_file,"%d", &no_cases);

	while(no_cases--){
		int a,b,c,d;
		fscanf(in_file,"%d %d %d %d", &a,&b,&c,&d);
		fraction(a*d+b*c,b*d);
	}
	fclose(in_file);
}
