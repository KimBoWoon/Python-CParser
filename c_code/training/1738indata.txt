#include <iostream>
#include<fstream>
#include <cstring>

using namespace std;

void check(int num)
{
		int flag;
		int value;
		char pallen[1000000];

		for (int j = 2; j <= 64; j++) {
			int k = 0;
			value = num;
			while (value > 0) {
				pallen[k++] = value%j;
				value /= j;
			}
			int pe = k;
			for (int k = 0; k < pe / 2; k++) {
				if (pallen[k] != pallen[pe - k - 1]) {
					flag = 0;
					break;
				}
				else flag = 1;
			}

			if (flag == 1) {
				cout << "1" << endl;
				break;
			}
		}
		if(flag !=1) cout << "0" << endl;

}

int main() {
	int cases;
	ifstream instream("input.txt",ios::in);
	instream >> cases;

	for (int i = 0; i < cases; i++) 
	{
		int num;
		instream>>num;
		check(num);
	}

	return 0;
}
