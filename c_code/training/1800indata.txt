#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int p_open, p_close; 
int p_num; 
char parenthesis[50]; 

int check_VPS(char parenthesis[], int index); 

void main(void)
{
	FILE *inFile;
	int TestCases, i;
	int result;

	inFile = fopen("input.txt", "r");
	if (inFile == NULL) exit(1);

	fscanf(inFile, "%d", &TestCases); 

	for (i = 0; i<TestCases; i++)
	{
		fscanf(inFile, "%s", &parenthesis); 

		p_open = 50, p_close = 0;
		p_num = strlen(parenthesis); 

		result = check_VPS(parenthesis, 0); 

		if (result == 1) printf("YES\n");   
		else printf("NO\n");                
	}

	fclose(inFile);
}


int check_VPS(char parenthesis[], int index)    
{
	if (p_num % 2 == 1) return 0;             

	if (parenthesis[index] == '.')             
		check_VPS(parenthesis, index + 1);
	else if (parenthesis[index] == '(')
	{
		p_open = index;                           
		check_VPS(parenthesis, index + 1);        
	}
	else if (parenthesis[index] == ')')            
	{
		p_close = index;                           
		if (p_open < p_close) 
		{
			parenthesis[p_open] = '.';             
			parenthesis[p_close] = '.';
			p_num = p_num - 2;                      
			p_open = 50, p_close = 0;               

			if (p_num == 0) 
				return 1;
			else
				check_VPS(parenthesis, 0); 
		}                                   
		else
			return 0;       
	}
	else return 0;
}
