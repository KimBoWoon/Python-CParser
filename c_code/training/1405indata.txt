#include<iostream>
#include<fstream>
using namespace std;
int main()
{
	ifstream fs;
	fs.open("input.txt");
    int k;
    fs>>k;
    while(k--)
    {
        int h,w,n,s=0;
         
        fs>>h>>w>>n;
        if(n%h==0)s--;
        cout<<(n%h==0?h:n%h);
        if(n/h+s+1<10)cout<<"0";
        cout<<n/h+s+1<<"\n";
    }
}
