

#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)

int gcd(int a, int b);
int gcd(int a, int b)
{
	if (b == 0) 
		return a;
	else 
		return gcd(b, a%b);
}

int main()
{
	FILE *in_file;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	int n1,n2, TestCases, i = 0;

	fscanf(in_file, "%d", &TestCases);

	for (i = 0; i < TestCases; i++){
		fscanf(in_file, "%d", &n1);
		fscanf(in_file, "%d", &n2);
		printf("%d\n",gcd(n1, n2));
	}
}
