

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int fib(int n);

void main(void) {
	FILE *inFile;
	int caseNum;

	int i;
	int n;

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &caseNum);

	for (i = 0; i < caseNum; i++) {
		fscanf(inFile, "%d", &n);

		printf("%d\n", fib(n));
	}
}

int fib(int n) {
	if (n == 0) {
		return 0;
	}

	if (n == 1) {
		return 1;
	}

	if (n > 1) {
		return fib(n - 1) + fib(n - 2);
	}
}
