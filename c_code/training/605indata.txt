


#include <iostream>
#include <fstream>
#include<cstdlib>
using namespace std;

int factorial(int n); 

int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int n;
		inStream >> n;
		cout << factorial(n) << endl; 
	}
	return 0;
	inStream.close();
}

int factorial(int n)
{
	if (n <= 1) 
	{
		return 1;
	}
	else
	{
		return n*factorial(n - 1); 
	}
}
