
#include<iostream>
#include<string>
#include<cmath>
#include<fstream>

using namespace std;

void ducci(int *compArray, int compCount, int loopCount){ 
	int sumResult = 0;
	for (int i = 0;i<compCount;i++){ 
		sumResult += compArray[i];
	}
	if (sumResult == 0){ 
		cout << "ZERO" << endl;
	}
	else if (loopCount > 1000){ 
		cout << "LOOP" << endl;
	}
	else{
		for (int i = 0;i<compCount;i++){ 
			compArray[i] = compArray[i] - compArray[i+1];
			if (compArray[i] < 0){ 
				compArray[i] = -compArray[i];
			}
		}
		compArray[compCount] = compArray[0];
		loopCount++;
		ducci(compArray, compCount, loopCount); 
	}
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, compCount = 0, loopCount = 0;
	int *compArray;
	f >> numCount;

	for (int i = 0;i<numCount;i++){
		f >> compCount;
		compArray = new int[compCount];
		for (int j = 0;j<compCount;j++){
			f >> compArray[j];
		}
		compArray[compCount] = compArray[0];
		ducci(compArray, compCount, loopCount);
	}
	return 0;
}
