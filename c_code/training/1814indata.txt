  

#include <iostream>
#include <fstream>
using namespace std;
int Parenthesis(char str[],int begin,int left,int len);
int main(){

	ifstream inFile;
	inFile.open("input.txt");
	int no_cases=0; 
	inFile >> no_cases;
	for (int test = 0; test < no_cases; test++){
		char str[50]={0,};
		inFile>>str;
		int len=0;
		for(len=0; str[len]!=0; len++);

		int result=Parenthesis(str,0,0,len);
		if(result==1)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
	return 0;
}
int Parenthesis(char str[],int begin,int left,int len) 
{
	if(str[begin]=='(')
		left++;
	if(str[begin]==')')
		left--;
	if(left<0) 
		return 1;
	if(begin==len-1){ 
		if(left==0)
		  return 2;
		else 
		  return 1;
	}
	Parenthesis(str,begin+1,left,len); 
	
}
