
#include<iostream>
#include<fstream>

using namespace std;

int fibonacci(int n) {
	if (n == 0)
		return 0;
	else if (n==1)
		return 1;
	else
		return fibonacci(n-1)+ fibonacci(n - 2);
}


int main()
{
	int testCase, num = 0;
	fstream instream("input.txt", ios::in);
	instream >> testCase;
	for (int q = 0; q < testCase; q++)
	{
		instream >> num;
		cout << fibonacci(num) << endl;
	}
}
