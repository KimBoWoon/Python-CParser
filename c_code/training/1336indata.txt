
#include <iostream>
#include <fstream>

using namespace std;

int henry(int up, int down, int start){
	int sol;
		while(true)
		{
			int share = down * start;
			int sec_up1, sec_up2;			

			sec_up1 = start*up;
			sec_up2 = down;
			if(sec_up1> sec_up2)
			{
				up = sec_up1-sec_up2;
				if(up%share == 0)
					share/=up;
				down = share;
				start=down/up;
			}
			else if(sec_up1 == sec_up2)
			{
				sol = start;
				share = 0;
				break;
			}
			else
			{
				start++;
			}

		}
		return sol;
}
int main(){

	ifstream inStream;

	inStream.open("input.txt");

	int num_case;

	inStream >> num_case;

	for(int i = 0; i <num_case; i++){

		int up,down;
	
		inStream >> up >> down;

		cout<<henry(up,down, 2)<<endl;
	}

}
