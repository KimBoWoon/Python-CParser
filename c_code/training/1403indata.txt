
#include <stdio.h>
#include <stdlib.h>
void main(void)
{

	FILE *in_file;
	int no_cases;
	int i, j, data1,data2, data3;
	in_file = fopen("input.txt","r");

	fscanf(in_file,"%d",&no_cases);
	if(in_file == NULL) exit(1);
	
	for(i=0; i<no_cases; i++){
		fscanf (in_file,"%d %d %d",&data1,&data2,&data3);
		if(data3%data1==0)
		printf("%d%02d\n",data1,data3/data1);
		else
		printf("%d%02d\n",data3%data1,data3/data1+1);
		}
		
}
