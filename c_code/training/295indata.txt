
#include <stdio.h>
#include <stdlib.h>


void simpleFraction (int numerator, int denominator);
int findDivisor (int a , int b); 


int main(int argc, const char * argv[]) {
    FILE* infile  = fopen("input.txt","r");
    int datacase;
    int i;
    
    fscanf(infile,"%d",&datacase);
    for(i=0; i<datacase; i++)
    {
        int numerator1, numerator2;
        int denominator1, denominator2;
        int parameter[2];
        
        fscanf(infile, "%d %d %d %d", &numerator1,&denominator1,&numerator2,&denominator2);
        
        parameter[0] =  numerator1*denominator2 + numerator2*denominator1;
        parameter[1] = denominator1 * denominator2;
        
        
        simpleFraction(parameter[0], parameter[1]);
        
        
        
    }
    fclose(infile);
    return 0;
    
}

void simpleFraction(int numerator, int denominator)
{
    int maxdivisor = 1;
    
    if(numerator>denominator)
        maxdivisor = findDivisor(numerator, denominator);
    else
        maxdivisor = findDivisor(denominator,numerator);
    
    
    printf("%d %d\n", numerator/maxdivisor, denominator/maxdivisor);
}


int findDivisor(int a, int b)
{
   
    
    if(a%b==0)
        return b;
    else return findDivisor(b, a%b);

}
