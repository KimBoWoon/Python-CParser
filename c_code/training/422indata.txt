
#include<stdio.h>
#include<stdlib.h>

int euculidean(int a,int b)
{
	if(b == 0)
		return a;

	else
		return euculidean(b, a%b);

}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;
	int p,q;

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		fscanf(inFile, "%d%d", &p, &q);

		printf("%d\n", euculidean(p,q));

	}

	fclose(inFile);
}
