

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int checkPalindrome(char data[], int first, int last) {
	if (last <= first)
		return 1;
	else if (data[first] != data[last])
		return 0;
	else
		return checkPalindrome(data, first + 1, last - 1);
}

int change(int a, int b) {
	int i = 0, result;
	char c[101] = { "" };
	while (a >= b) {
		c[i] = a % b + 48; 
		a /= b;
		i++;
	}
	c[i] = a + 48;
	result = checkPalindrome(c, 0, strlen(c) - 1);
	return result;
}

int main(void) {
	FILE *input;
	int no_cases;
	int i, j, value;

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);
	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d", &value);
		for (j = 2; j < 65; j++) {
			if (change(value, j)) {
				printf("%d\n", 1);
				break;
			} else if (j == 64) {
				printf("%d\n", 0);
			}
		}
	}

	fclose(input);
	return 0;
}
