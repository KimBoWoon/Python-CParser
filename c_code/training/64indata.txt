


#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>

int prime(int a);

int main(void)
{
	FILE *in_file;
	int i, j;
	int no_cases, no_data;

	in_file = fopen("input.txt", "r");  
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   

	for (i = 0; i<no_cases; i++)
	{
		fscanf(in_file, "%d", &no_data);

		if (no_data == 0)      return 0;

		if (prime(no_data) == 1) printf("1\n");
		if (prime(no_data) == 0) printf("0\n");

		

	}
	fclose(in_file); 

	return 0;
}

int prime(int a)
{
	int j;

	for (j = 2; j <= a; j++)
	{
		if (a%j == 0)
		{
			if (a == j)    return 1;
			if (a != j)    return 0;
		}
	}

	return 0;
}
