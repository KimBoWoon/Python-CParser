#include <iostream> 
#include <fstream>
using namespace std;
int gcd(int i, int j)
{
	if(j==0)
		return i;
	return gcd(j, i%j);
}
int main()
{
	int t,a,b,tmp;
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>t;
	for(int i=0; i<t; i++)
	{
		inStream>>a>>b;
		if(a>b)
		{
			tmp=a;
			a=b;
			b=tmp;
		}
		cout<<gcd(a,b)<<endl;
	}
}
