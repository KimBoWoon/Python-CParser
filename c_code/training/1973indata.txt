#include<cstdio>
#include<cmath>
#include<algorithm>
int arr[1000000];
using namespace std;
int main()
{
	int n;
	FILE* f = fopen("input.txt","r");
	fscanf(f,"%d",&n);
	while(n--)
	{
		int s,k,v=200000001,ans=0;
		fscanf(f,"%d %d",&s,&k);
		for(int i=0;i<s;i++)
		{
			fscanf(f,"%d",&arr[i]);
		}
		sort(arr,arr+s);
		for(int i=0,j=s-1;i<j;)
		{
			int sum=arr[i]+arr[j],t;
			t=sum-k;
			if(abs(t)<v)v=abs(t);
			if(t==0)break;
			else if(t>0)j--;
			else i++;
		}
		for(int i=0,j=s-1;i<j;)
		{
			int sum=arr[i]+arr[j];
			if(sum==k-v)ans++;
			if(sum<=k-v)i++;
			else j--;
		}
		if(v)
		{
			for(int i=0,j=s-1;i<j;)
			{
				int sum=arr[i]+arr[j];
				if(sum==k+v)ans++;
				if(sum<=k+v)i++;
				else j--;
			}
		}
		printf("%d\n",ans);
	}
}
