

#include<iostream>
#include<fstream>
#include<vector>
#include<string>
using namespace std;

int size = 0;
string str;

bool pelin(string str, int idx){
	if (idx == size / 2)
		return true;
	if (str[idx] == str[size - 1 - idx])
		return pelin(str, ++idx);
	else
		return false;
}

void changeStr(int num, int base){
	if (num == 0)
		return;
	char c = '0' + num % base;
	num /= base;
	changeStr(num, base);
	str += c;
	return;
}

int main(){
	ifstream File;
	int numCase = 0;
	int num = 0;


	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		bool res = false;
		File >> num;
		for (int i = 2; i <= 64; i++){
			str = "";
			changeStr(num, i);
			size = str.length();
			if (res = pelin(str, 0))
				break;
		}
		cout << res << endl;
	}
	File.close();
	return 0;
}
