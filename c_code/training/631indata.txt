
#include <iostream>
#include <fstream>

using namespace std;

int factorial(int n){

	if(n <= 1)
		return 1;
	else
		return n*factorial(n-1);		

}
int main(){

	ifstream inStream;
	inStream.open("input.txt");		

	int num_case;	

	inStream >> num_case;

	for(int i=0; i< num_case ; i++){

		int test;
		inStream >> test;
		cout<<factorial(test)<<endl;
	}
}
