  

#include <iostream>
#include <fstream>
using namespace std;
int abs(int a);
int main(){

	ifstream inFile;
	inFile.open("input.txt");
	int no_cases=0; 
	inFile >> no_cases;
	for (int test = 0; test < no_cases; test++){
		int arr[15]={0,},brr[1000][15]={0,};
		int num=0;
		inFile>>num;
		for(int i=0; i<num; i++){
			inFile>>arr[i];
			brr[0][i]=arr[i];
		}
		int flag=1;
		for(int i=0; i<1000; i++){
			flag=1;
			for(int j=0; j<num; j++){
				brr[i][j]=arr[j];
				if(brr[i][j]!=0){
					flag=0;
				}
			}
			if(flag==1)
				break;
			int tmp=arr[0];
			for(int j=0; j<num; j++)
			{
				if(j!=num-1)
				   arr[j]=abs(arr[j]-arr[j+1]);
				else
					arr[j]=abs(arr[j]-tmp);
			}
			
			for(int j=0; j<=i; j++)
			{
				flag=2;
				for(int k=0; k<num; k++)
				{
					if(arr[k]!=brr[j][k]){
						flag=3;
						break;
					}
				}
			 if(flag==2)
				break;
			}
			if(flag==2)
				break;

		}
		if(flag==1)
			cout<<"ZERO"<<endl;
		else if(flag==2)
			cout<<"LOOP"<<endl;
		
	}
	return 0;
}
int abs(int a)
{
	if(a<0)
		return -a;
	else
		return a;
}
