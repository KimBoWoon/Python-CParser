#include <stdio.h>
#include <stdlib.h>

void main()
{
	FILE *file;
	int i=0, a=0;
	int t, n;

	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d", &n);
		a=n-1;
		while(1)
		{
			if(a==1)
			{
				printf("1\n");
				break;
			}
			else if(n%a==0)
			{
				printf("0\n");
				break;
			}
			a--;
		}
	}
	fclose(file);
}
