#include <cstdio>
using namespace std;

#define MIN(x,y) (x<y)?x:y

int T, N, Z;
int M[502][502], F[502], C[502];

int main() {
    FILE *inFile;
    inFile = fopen("input.txt", "r");

    for(fscanf(inFile, "%d", &T); T--; )
    {
        fscanf(inFile, "%d", &N);

        for (int i=1; i<=N; ++i) {
            fscanf(inFile, "%d", &F[i]);
            C[i] = C[i-1] + F[i];
        }

        for (int i=1; i<=N; ++i)
            for (int j=i+1; j<=N; ++j)
                M[i][j] = 199999999;

        for (int i=1; i<N; ++i)
        {
            for (int j=1; j<=N-i; ++j)
            {
                Z = i+j;
                for (int k=j; k<Z; ++k)
                    M[j][Z] = MIN(M[j][Z], M[j][k]+M[k+1][Z]+C[Z]-C[j-1]);
            }
        }

        printf("%d\n", M[1][N]);
    }

    fclose(inFile);
    return 0;
}
