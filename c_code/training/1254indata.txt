

#include <iostream>

int Prime(int prinum[],int number)
{
	prinum[0] = 2;
	int count = 1;
	for(int i=3; i<number; i++)
	{
		int check = 1;
		for(int j=2; j<i; j++)
		{
			if(i%j == 0)
			{	
				check = 0;
			}
		}
		if(check==1)
		{
			prinum[count] = i; 
			count ++;
		}
	}
	return count;
}
int main()
{

		int *prinum = new int [number]; 
		Prime(prinum,number); 
		count = Prime(prinum,number); 
		int a=0,b=1000000,c=0,d=0; 
		for(int i=0; i<count; i++)
		{
			for(int j=0; j<count; j++)
			{
				if(prinum[i]+prinum[j] == number)
				{
					c = prinum[i]; 
					d = prinum[j];
					if(abs(a-b) >= abs(c-d))
