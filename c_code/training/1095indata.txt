
#include <fstream>
#include <iostream>

#define TRIANGLE_NUM 1	
#define NON_TRIANGLE_NUM 0
#define SIZE 30	

using namespace std;

void perform(ifstream &inFile);	
int checkTriangleNumber(int testNum);	

int main()
{
	int numTestCase;	
	ifstream inFile;
	
	inFile.open("input.txt");
	inFile >> numTestCase;

	for (int i = 0; i < numTestCase; i++)	
	{
		perform(inFile);	
	}

	inFile.close();

	return 0;
}

void perform(ifstream &inFile)
{
	int testNum;	
	
	inFile >> testNum;

	int check = checkTriangleNumber(testNum);	

	cout << check << endl;	
}

int checkTriangleNumber(int testNum)
{
	

	
	int triNum_1[SIZE],
		triNum_2[SIZE],
		triNum_3[SIZE];

	

	for (int i = 0, n = 1; i < SIZE; i++, n++)
		triNum_1[i] = triNum_2[i] = triNum_3[i] = n * (n + 1) / 2;
	
	
	
	for (int i = 0; i < SIZE; i++)
		for (int j = 0; j < SIZE; j++)
			for (int k = 0; k < SIZE; k++)
				if (triNum_1[i] + triNum_2[j] + triNum_3[k] == testNum) return TRIANGLE_NUM;	

	return NON_TRIANGLE_NUM;	
}
