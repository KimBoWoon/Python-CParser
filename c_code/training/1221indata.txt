bool isPrime(int n){
	for (int i = 2; i < n; i++){
		if (n%i == 0){	
			return 0;
		}
	}
	return 1;
}

void goldba(int n){
	for (int i = (n / 2); i >= 2; i--){
		if (isPrime(i) && isPrime(n - i)){
			cout << i << " " << n - i << endl;
			return;
		}
	}
}
