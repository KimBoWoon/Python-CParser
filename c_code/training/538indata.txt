#include<iostream>
#include<fstream>

using namespace std;
ifstream inFile("input.txt");
int fac(int n){
	return n ==0? 1:fac(n-1)*n;
}

class Permu{
private :
	int len;
	int th;
	bool* charBox;
public : 
	Permu(){
		inFile >> len >> th;
		charBox = new bool[len];
		for (int i = 0; i < len; i++){
			charBox[i] = false;
		}
		trip(len);
		cout << endl;
	}
	void trip(int k){
		if (k == 0){
			return;
		}
		int num = (th - 1) / fac(k - 1);
		for (int i = 0; i <= num; i++){
			if (charBox[i])
				num++;
		}
		cout << (char)('a' + num);
		charBox[num] = true;
		th = (th - 1) % fac(k - 1)+1;
		trip(k - 1);
	}
};

int main(){
	int cases;
	inFile >> cases;
	while (cases--){
		Permu p;
	}
	return 1;
}
