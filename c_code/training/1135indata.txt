
#include <iostream>
#include <fstream>

using namespace std;

int main(){

	ifstream inStream;
	int num_case;

	inStream.open("input.txt");

	inStream >> num_case;

	for(int i = 0; i < num_case; i++)
	{
		int result;		
		inStream >> result;		

		int comp = 1;		
		int a= 2;
		int cnt = 0;

		int output =0;

		while(true)
		{
			if(comp >= result)
			{
				comp=1;
				a=2;
				break;
			}
			else
			{
				comp = comp + a;
				a++;
				cnt++;
			}
		}			

		int *arr = new int[cnt];

		for(int index =0; index <cnt; index++){
			arr[index] =comp;
			comp = comp + a;
			a++;
		}

		int test;

		for (int i=0; i<cnt; i++)
		{
			for (int j =0; j<cnt; j++)
			{
				for (int k = 0; k <cnt ; k++)
				{
					test = arr[i]+arr[j]+arr[k];

					if(test == result)
						output = 1;
				}
			}
		}
		delete arr;
		cout<<output<<endl;
	}
}
