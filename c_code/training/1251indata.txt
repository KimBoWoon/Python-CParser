#include<iostream>
#include<fstream>
#include<cstdlib>
#include<cmath>
using namespace std;


int prime(double num) 
{
	double root = sqrt(num); 
	int cnt = 0;
	int dest = (int)num;
	for (int i = 2; i <= root; i++)
		if (dest%i == 0) cnt++;

	if (cnt>0) return 0; 
	else return 1; 
}

void Goldbach(int num)
{
	int GoldNum[1001] = {0};
	int ans[1000][2] = {0};
	int cnt = 0; int cnt2 = 0;
	int ansNum=0;
	
	for (int i = 2; i < num; i++)
	{
		if (prime(i)) 
			GoldNum[cnt++]=i;
		
	}


	for (int i = 0; i < cnt; i++)
		for (int j = 0; j < cnt; j++ )
			if (num == GoldNum[i] + GoldNum[j]) {
				ans[cnt2][0] = GoldNum[i];
				ans[cnt2][1] = GoldNum[j];
				cnt2++;
			}

	int number=abs( ans[0][0]-ans[0][1]);
	for (int i = 1; i < cnt2; i++)
	{
		if (number > abs(ans[i][0] - ans[i][1]))
		{
			ansNum = i;
			number = abs(ans[i][0] - ans[i][1]);
		}
	}


	cout << ans[ansNum][0] << " " << ans[ansNum][1] << endl;
				


}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int testNum;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> testNum;
		Goldbach(testNum);
	}
	return 0;
}
