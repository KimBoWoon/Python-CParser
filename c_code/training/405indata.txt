 

 
#include <stdio.h>

long gcd(long m, long n);
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    while(test--) 
    {
        long m,n; 
        fscanf(fp,"%ld %ld",&m,&n); 
        if(m >= n) printf("%ld\n",gcd(m,n)); 
        else printf("%ld\n",gcd(n,m)); 
    }
    fclose(fp); 
    return 0;
}
long gcd(long m, long n) 
{
    if(n == 0) 
    {
        return m;
    }
    else 
    {
        return gcd(n,m%n);
    }
}
