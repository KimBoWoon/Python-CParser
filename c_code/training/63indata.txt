


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

bool isPrime(int num)
{
	int count = 0;

	for (int i = 2; i*i <= num; i++)
	{
		if (num % i == 0)
		{
			return false;
		}
	}

	return true;

}

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if (instream.fail())
	{
		cerr << "Error";
		exit(1);
	}

	int no_cases;
	instream >> no_cases;

	for (int i = 0; i < no_cases; i++)
	{
		int num;
		instream >> num;

		cout << isPrime(num) << endl;
	}

	return 0;
}
