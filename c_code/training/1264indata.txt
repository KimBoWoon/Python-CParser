 #include <stdio.h> 
 #include <stdlib.h> 
 
 
 int isNatNum(int n); 
 
 
 void main(void) { 
 	FILE *file; 
 	int i, nocase, num; 
 	 
 	file = fopen("input.txt", "r"); 
 	if(file == NULL)  
 		exit(1); 
 
 
 	fscanf(file, "%d", &nocase); 
 
 
 	for(i = 0; i < nocase; i++) { 
 		int j; 
 		fscanf(file, "%d", &num);	
 		num/=2;						
 
 
 		for(j = 0; j <= num-2; j++) {	 
 			if(isNatNum(num-j) && isNatNum(num+j)) {	
				printf("%d %d\n", num-j, num+j); 
 				break; 
 			} 
 		} 
 	} 
 	 
 	fclose(file); 
 } 
 
 
 int isNatNum(int n) { 
 	int i; 
 	 
 	for(i = n/2; i > 1; i--)	
 		if(!(n%i))	
 			return 0;	
 	 
 	return 1;	
 }
