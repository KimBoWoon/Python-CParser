#include <iostream>
#include <fstream>

int Factorial(int n)
{
	if (n <= 1)
		return 1;
	else
		return n*Factorial(n - 1);
}

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail())
	{
		std::cerr << "file open err" ;
		return 1;
	}

	int caseNum;
	input >> caseNum;
	for (int i = 0; i < caseNum; i++)
	{
		int n;
		input >> n;
		std::cout << Factorial(n) << std::endl;
	}
	return 0;
}
