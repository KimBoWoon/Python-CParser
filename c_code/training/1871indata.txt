

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void parenthesis(char str[], int left, int right, int n);

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	char str[50]; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%s", &str); 

		if (strlen(str) % 2 != 0)
			printf("NO\n");
		else if (str[0] == ')' || str[strlen(str) - 1] == '(')
			printf("NO\n");
		else
			parenthesis(str, 1, 0, 1);
	}

	fclose(f);
	return 0;
}

void parenthesis(char str[], int left, int right, int n) {
	if (strlen(str) == n){
		if (left - right == 0)
			printf("YES\n");
		else
			printf("NO\n");
		return;
	}
	else {
		if (str[n] == '(') {
			if (right == 0)
				parenthesis(str, left + 1, right, n + 1);
			else
				parenthesis(str, left, right - 1, n + 1);
		}
		else {
			if (left == 0)
				parenthesis(str, left, right + 1, n + 1);
			else
				parenthesis(str, left - 1, right, n + 1);
		}
	}

}
