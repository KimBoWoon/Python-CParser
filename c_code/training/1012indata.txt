

#include <iostream>

int a[15];										

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++)
	{
		int n;
		inFile >> n;
		for (int j = 0; j < n; j++)			
			inFile >> a[j];

		ducciSeuquence(a,n,1);
	}
	return 0;
}
void ducciSeuquence(int a[], int n, int cnt){
	int aOne = a[0];		
	if (cnt == 1000)			
	{
		cout << "LOOP" << endl;
		return;
	}
	else if (checkZero(a, n, 0) == 0){	
		cout << "ZERO" << endl;
		return;
	}
	else{
	for (int i = 0; i < n; i++){
		if (i == n - 1){
			if (a[i]>aOne){ a[i] = (a[i] - aOne); }
			else { a[i]=aOne - a[i]; }
		}
		else{
			if (a[i] > a[i + 1]){ a[i] = (a[i] - a[i + 1]); }
			else{a[i]= a[i + 1] - a[i]; }
		}
	}
	return ducciSeuquence(a, n, cnt+1);
	}
}

int checkZero(int a[], int end, int start){	
	if (start == end){	
		return 0;
	}
	else{
		if (a[start] == 0){
			return checkZero(a, end, start + 1);
		}
		else return 1;
	}
}
