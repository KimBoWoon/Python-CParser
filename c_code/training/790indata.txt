#include <cstdio>
#include <cstring>

using namespace std;

int isPalindrome(char * str, int left, int right) {
	if (left > right)
		return 1;
	else if (str[left] == str[right])
		return isPalindrome(str, left + 1, right - 1);
	else
		return 0;
}

int main(void) {
	FILE * fp = fopen("input.txt", "r");

	int nCase; 
	fscanf(fp, "%d", &nCase);

	while (nCase--) {
		char str[1001];
		fscanf(fp, "%s", str);
		printf("%d\n", isPalindrome(str, 0, strlen(str) - 1));
	}

	fclose(fp);
	return 0;
}
