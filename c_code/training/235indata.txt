


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int son1 = 0, mom1 = 0, son2 = 0, mom2 = 0, son3 = 0, mom3 = 0;
	int st1 = 0, st2 = 0, st3 = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}
	
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> son1 >> mom1 >> son2 >> mom2;	
		st1 = gcd(son1, mom1);
		st2 = gcd(son2, mom2);
		son1 /= st1;
		mom1 /= st1;
		son2 /= st2;
		mom2 /= st2;
		son3 = son1 * mom2 + son2 * mom1;
		mom3 = mom1 * mom2;
		st3 = gcd(son3, mom3);
		son3 /= st3;
		mom3 /= st3;

		cout << son3 << " " << mom3 << endl;
	}
	return 0;
}
