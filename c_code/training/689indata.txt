#include <iostream>
#include <fstream>
#include <cstring>

using namespace std;

int M(char *str, int a , int n);

int main()
{
	ifstream infile;

	infile.open("input.txt");
	
	int numTestCases = 0;
	char result[1000];
	
	
	
	infile >> numTestCases;
	
	
	while(numTestCases--)
	{
		infile >> result;
		
		cout<<M(result,0, strlen(result)-1)<<endl;
		
	}

	
	return 0;
}
int M(char str[], int a , int n)
{
	if( n <= a)
		return 1;
	else if( str[a] != str[n])
		return 0;
	else
		return M(str, a+1 , n-1);

	
}
