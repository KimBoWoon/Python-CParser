
#include <iostream>
#include <fstream>

using namespace std;


int main(){

	int *arr = NULL;
    int nNum;
    int i, j;
 	int num_case;
	int maximumA, maximumB;

	ifstream inStream;
    inStream.open("input.txt");
    inStream >> num_case;

    for(int t = 0 ; t<num_case; t++){
	    inStream >> nNum;

        arr = new int[nNum+1];			

		for ( i = 0; i <= nNum; i++ )		
               arr[i] = 0;
 
        for ( i = 2; i <= nNum; i++ )		
        {
			if ( arr[i] == 1 )
				continue;
			j = i;
			while ( (j += i) <= nNum )
			{
				arr[j] = 1;	   
			}
		}
 
		for ( i = 2; i < nNum; i++ )
        {
                for ( j = i; j < nNum; j++ )
               {
				   if ( (arr[i] == 1) || (arr[j] == 1) )
						continue;
             
				   if ( i + j == nNum )				
                   {
					   maximumA=i;
					   maximumB=j;
				   }
				}
		}
		
		        
		delete [] arr;
			cout<<maximumA<<" " <<maximumB<<endl;

   }
}
