

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int gcd(int a, int b);			

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error"<<endl;
		exit(1);
	}

	int no_case;
	instream>>no_case;

	for(int i=0; i<no_case; i++)
	{
		int numer1, denom1, numer2, denom2;
		instream>>numer1>>denom1>>numer2>>denom2;

		int res_numer, res_denom;

		res_numer = (numer1 *denom2) + (numer2 * denom1);
		res_denom = (denom1 * denom2);						

		int agcd;
		agcd = gcd(res_numer, res_denom);					

		res_numer/=agcd;									
		res_denom/=agcd;

		cout<<res_numer<<" "<<res_denom<<endl;
	}
}

int gcd(int a, int b)			
{
	if(b == 0)
		return a;
	else
		return gcd(b, a%b);
}
