






#include<iostream>
#include<cmath>
#include<fstream>
using namespace std;


int prime(double num) 
{
	double root = sqrt(num); 
	int cnt = 0;
	int dest = (int)num;
	for (int i = 2; i <= root; i++)
		if (dest%i == 0) cnt++;

	if (cnt>0) return 0; 
	else return 1; 
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		double num;
		instream >> num;
		cout << prime(num) << endl;
	}
	return 0;
}
