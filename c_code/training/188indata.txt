

#include <iostream>
#include <fstream>
#include <string.h>

using namespace std;

int McCarthy(int n);

int main()
{
	ifstream infile;
	int testCase;
	int num; 
	
	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num;

		cout << McCarthy(num) << endl;
	}

	infile.close();

	return 0;
}

int McCarthy(int n)
{
	if (n > 100)
		return n - 10;
	else
		return McCarthy(McCarthy(n + 11));
}
