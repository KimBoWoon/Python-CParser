#include <stdio.h>

int check_prime(int value){

	int divisor;
	int check=1;

	for(divisor=2;divisor*divisor<=value;divisor++){

		if(value%divisor==0)		check=0;	
	}

	return check;
}

void G_Conj(int value_a, int value_b){

	int a,b;
	
	a=check_prime(value_a);	b=check_prime(value_b);

	if((a==1)&&(b==1))	printf("%d %d\n", value_a, value_b);
	else	G_Conj(value_a-1, value_b+1);
}
	
void main(){

	FILE *in_file;
	int i,num_cases;
	in_file=fopen("input.txt","r");
	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){
		
		int value;
		int value_a;
		int value_b;

		fscanf(in_file,"%d",&value);

		value_a=value/2; value_b=value/2;

		G_Conj(value_a, value_b);
	
	}
}
