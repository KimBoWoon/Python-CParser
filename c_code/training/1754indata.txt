
#include <iostream>
#include <fstream>
using namespace std;
int arr[10000000];
int n;
int out;
void baseConversion(int num, int base)
{
	if (num >= base)
		baseConversion(num / base, base);
	arr[n++] = num%base;
}
int checkPalindrome(int first, int last)
{
	if (last <= first)
		return 1;
	else if (arr[first] != arr[last])
		return 0;
	else
		return checkPalindrome(first + 1, last - 1);
}
void pal(int num, int base)
{
	n = 0;
	baseConversion(num, base);
	int res = checkPalindrome(0, n-1);
	if (base < 65) {
		if (res == 1) out = 1;
		else pal(num, base + 1);
	}
	else return;
}
int main(void)
{
	int test, num;
	ifstream file;
	file.open("input.txt"); 

	file >> test;
	for (int i = 0; i < test; i++) {
		file >> num;
		out = 0;
		pal(num, 2);
		cout << out << endl;
	}
	file.close();
	return 0;
}
