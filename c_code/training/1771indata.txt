#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

string notation(int data, int type){
	if (data < type){
		string temp = "";
		temp += data + 1;
		return temp;
	}


	string temp = "";

	while (data){
		temp += data % type + 1;
		data /= type;
	}

	return temp;
}

bool palin(string data){
	int size = data.length();
	if (size < 2)	return true;

	for (int i = 0; i < size / 2; i++){
		if (data[i] != data[size - 1 - i])	return false;
	}
	return true;
}

int main(){
	ifstream file;
	file.open("input.txt");
	int cases;
	file >> cases;
	while (cases--){
		int num;
		file >> num;
		bool isPalin = false;
		for (int i = 2; i <= 64; i++){
			if (num < i){
				isPalin = true;
				break;
			}
			string temp = notation(num, i);
			isPalin = palin(temp);
			if (isPalin)	break;
		}

		if (isPalin)	cout << 1 << endl;
		else			cout << 0 << endl;
	}


	return 0;
}
