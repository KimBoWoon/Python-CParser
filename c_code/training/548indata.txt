
#include <iostream>
#include <string>
#include <cstring>
#include <fstream>

char popAlphabet(char alphabet[], int index)
{
	char popChar = alphabet[index];
	int i;
	for ( i = index ; i < strlen(alphabet)-1 ; i ++)
		alphabet[i] = alphabet[i+1];
	alphabet[i] = '\0';
	return popChar;
}

std::string permutation(int numOfString, int n)
{
	char alphabet[] = "abcdefghijkl";
	alphabet[numOfString] = '\0';

	int factorial[12] = {1};
	for ( int i = 2 ; i <= 12 ; i ++)
		factorial[i-1] = i*factorial[i-2];

	std::string permString = "";
	n -= 1;
	for ( int i = numOfString - 1; i >= 0 ; i --)
	{
		permString += popAlphabet(alphabet, n/factorial[i-1]);
		n %= factorial[i-1];
	}

	return permString;
}


int main()
{
	std::ifstream input;
	input.open("input.txt");
	if ( input.fail())
	{
		std::cerr << "file open err";
		return 0;
	}
	int caseNum;
	input >> caseNum;
	for ( int i = 0 ; i < caseNum ; i ++)
	{
		int numOfString, n;
		input >> numOfString >> n;
		std::cout << permutation(numOfString, n) << std::endl;
	}

	return 0;
}
