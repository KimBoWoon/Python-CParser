#include <iostream>
#include <fstream>
#include <cstring>
#include <stdlib.h>
#include <string.h>
#include <algorithm>
 
using namespace std;
 
int S[1000005] = {0,};
int main()
{
 int numcase, n, k;
 int i, j;
 int count = 0;
 int min = 300000001;

 ifstream fin;
 fin.open("input.txt");
 fin>>numcase;
 while(numcase--)
 {
  	fin>>n>>k;
  	for(i=0; i<n; i++)
 	{
   		fin>>S[i];
  	}
  	sort(S, S+n);      
  	i = 0;             
  	j = n-1;
  	min = 300000001;
  	count = 0;
  
	while(i<j)         
 	{
   		if(abs(k - (S[i] + S[j])) < min)   
   		{     
    			min = abs(k - (S[i] + S[j]));
    			count = 1;
   		}
   		else if(abs(k - (S[i] + S[j])) == min)  
   		{
    			count++;
   		}
   		if((S[i] + S[j]) < k)     
    			i++;
   		else         		  
    			j--;
  	}
  	cout<<count<<endl;
 }
 fin.close();
}
