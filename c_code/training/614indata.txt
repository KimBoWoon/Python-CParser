

#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

int factorial(int n);
int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	if (inStream.fail()) exit(1);
	int numTest,num;
	inStream >> numTest;

	for (int i = 0; i < numTest; i++){
		inStream >> num;
		cout << factorial(num) << endl;
	}

}
int factorial(int n)
{
	if (n <= 1) return 1;
	return n*factorial(n-1);

}
