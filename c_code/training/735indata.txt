
#include <stdio.h>
#include <stdlib.h>
#include <string.h>										

int checkP(char s[], int f, int l){						
	if(l<=f)													
		return 1;
	else if(s[f]!=s[l])										
		return 0;
	else														
		return checkP(s, f+1, l-1);
}

int main(){
	FILE *inFile;
	int i;
	int testCase;
	inFile=fopen("input.txt", "r");						
	fscanf(inFile, "%d", &testCase);

	for(i=0; i<testCase; i++){							
		char s[1000];
		fscanf(inFile, "%s", s);

		printf("%d\n", checkP(s, 0, strlen(s)-1));
	}
	return 0;
}
