int parenthesis(char str[], int n)
{
	int length = strlen(str);
	if (length % 2 == 1)
		return 0;

	for (int i = n; i < length; i++)
	{
		if (str[0] == ')' || str[length - 1] == '(')
			return 0;
		else if (str[i] == ')')
			return 1;
		else if (str[i] == '(')
			return parenthesis(str, i+1);
	}	
}
