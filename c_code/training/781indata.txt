

#include <stdio.h>
#include <stdlib.h>
#include <string.h>


int checkPalindrome(char str[], int first, int last);

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	char str[1000]; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%s", str);

		printf("%d\n", checkPalindrome(str, 0, strlen(str)-1));

	}

	fclose(f);
	return 0;
}

int checkPalindrome(char str[], int first, int last) {
	if (last <= first) 
		return 1;
	else if (str[first] != str[last])
	{
		return 0;
	}
	else
		return checkPalindrome(str, first+1, last-1);
}
