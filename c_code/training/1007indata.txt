

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int check(int data[], int no_datas);

int main(void) {
	FILE *input;
	int no_cases, no_datas;
	int i, j;
	int try, print;
	int data[15];

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);
	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		try = 1000;
		print = 0;
		fscanf(input, "%d", &no_datas);
		for (j = 0; j < no_datas; j++) {
			fscanf(input, "%d", &data[j]);
		}

		while (try--) {
			if (check(data, no_datas)) {
				printf("ZERO\n");
				print = 1;
				break;
			}

		}

		if (!print) {
			printf("LOOP\n");
		}

	}

	fclose(input);
	return 0;

}

int check(int data[], int no_datas) {
	int i;
	int newData[no_datas];

	for (i = 0; i < no_datas; i++) {
		newData[i] =
				i == (no_datas - 1) ?
						abs(data[i] - data[0]) : abs(data[i] - data[i + 1]);
	}

	for (i = 0; i < no_datas; i++) {
		data[i] = newData[i];
	}

	for (i = 0; i < no_datas; i++) {
		if (newData[i] != 0)
			return 0;
	}
	return 1;
}
