


#include <stdio.h>
#include <stdlib.h>

#define MAX 15

int Ducci(int size, int *data, int cnt){
	int temp[size],flag=0,i;
	if(cnt==1000) return 0; 
	for(i=0;i<size;i++){
		temp[i]=abs(data[i]-data[(i+1)%size]); 
		if(temp[i]==0) flag++; 
	}
	if(flag==size) return 1; 
	else return Ducci(size,temp,cnt+1); 
}

int main(){
	int i,j,cases,size,data[MAX],cnt=0;

	FILE *infile=fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0;i<cases;i++){
		fscanf(infile,"%d",&size);
		for(j=0;j<size;j++)
			fscanf(infile,"%d",&data[j]);

		if(Ducci(size,data,cnt+1)==1) printf("ZERO\n"); 
		else printf("LOOP\n"); 
	}
	fclose(infile);
	return 0;
}
