




#include<iostream>
#include<fstream>

int m(int n);

int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){
		
		int n;
		is >> n;

		std::cout<<m(n)<<std::endl;
	}

}

int m(int n){

	if( n > 100) 
		return n - 10;

	else
	    return m(m(n+11));


	
	
}
