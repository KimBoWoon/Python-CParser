
#include <iostream>
#include <fstream>
using namespace std;
int mac(int n){
	if (n > 100)
		return (n - 10);
	else
		return mac(mac(n + 11));
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n,number;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> number;
		cout << mac(number) << endl;
	}
	return 0;
}
