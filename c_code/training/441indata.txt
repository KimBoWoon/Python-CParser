#include<iostream>
#include<fstream>

using namespace std;

int gcd(int a, int b){
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int main(){

	ifstream file;
	file.open("input.txt");

	int testCase;
	int data1, data2;

	file >> testCase;

	for (int i = 0; i < testCase; i++){
		file >> data1 >> data2;
		cout << gcd(data1, data2) << endl;
	}

	return 0;
}
