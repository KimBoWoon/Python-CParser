
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;
int mc91(int n);
int main()
{
	ifstream inStream;
	int no_cases,value1;

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"Input file openning failed.\n";
		exit(1);
	}
	inStream>>no_cases;
	for(int i=0; i<no_cases; i++)
	{
		inStream>>value1;
		cout<<mc91(value1)<<endl;
	}
	return 0;
}
int mc91(int n)
{
	if (n > 100) {
		return n - 10;
	} else {
		return mc91(mc91(n+11));
	}
}
