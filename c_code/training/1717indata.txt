


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value, rv;
	int arr[65][20];		

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		int check = 0;
		inStream >> value;

		for (int j = 2; j <= 64; j++)		
		{
			rv = value;
			int k = 0;
			while(rv > 0)		
			{
				arr[j][k] = rv%j;	
				rv = rv / j;
				k++;
			}

			int half = k / 2;
			int rc = 0;
			for (int l = 0; l < half; l++)
			{
				if (arr[j][l] != arr[j][k - 1])		
					break;
				else if (arr[j][l] == arr[j][k - 1])	
					rc++;
				k--;
			}
			if (rc == half)		
			{
				check = 1;
				break;
			}
		}
		cout << check << endl;
	}
	inStream.close();
	return 0;
}
