#include<iostream>
#include<fstream>
#include<cstring>
using namespace std;
int movement[8][2]={{-2,-1},{-2,1},{-1,-2},{-1,2},{1,-2},{1,2},{2,-1},{2,1}},board[8][8],w=0,h=0;
bool visit[8][8];
bool Tour(int x,int y,int cnt)
{
	
	if(x<0||x>=w||y<0||y>=h||visit[x][y])return false;
	if(cnt==w*h)
	{
		visit[x][y]=true;
		board[x][y]=cnt;
		return true;
	}
	visit[x][y]=true;
	bool ret=false;
	for(int i=0;i<8;i++)
		if(Tour(x+movement[i][0],y+movement[i][1],cnt+1))
			ret=true;
	if(ret)
	{
		board[x][y]=cnt;
		return true;
	}
	visit[x][y]=false;
	return false;
}
int main()
{
	ifstream fs;
	fs.open("input.txt");
	int n;
	fs>>n;
	while(n--)
	{
		memset(visit,0,sizeof(visit));
		int x,y;
		fs>>w>>h>>x>>y;
		if(!Tour(x-1,y-1,1)) cout<<"0\n";
		else
		{
			cout<<"1\n";
			for(int i=0;i<w;i++)
			{
				for(int j=0;j<h;j++)
					cout<<board[i][j]<<" ";
				cout<<"\n";
			}
		}
	}
}
