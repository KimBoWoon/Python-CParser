

#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>

using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if (instream.fail())
	{
		cerr << "Error" << endl;
		exit(1);
	}

	int no_case;
	instream >> no_case;
	char field[50] = {0};

	while(no_case--)	
	{
		while(instream){
			instream>>field;
			break;
		}
		int Paren = 0;

		for(int i=0 ; i<strlen(field); i++)
		{
			if(Paren <0)
				break;
			if(field[i] == '(')
				Paren++;
			else
				Paren--;
		}
		
		if(Paren ==0)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}


}
