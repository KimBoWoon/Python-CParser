

#include<stdio.h>

int main(){

	FILE *inFile;
	int circle;
	int a, b, xn; 
	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d %d", &a, &b);
		for (xn = 2; (a*xn - b) != 0; xn++){
			if ((a*xn - b) > 0){
				a = a*xn - b;
				b *= xn;
			}
		}
		printf("%d\n", xn);
	}
	return 0;
}
