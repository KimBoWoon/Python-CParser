

#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){						
	if(b==0)
		return a;								
	else
		gcd(b, a%b);						
}

int main(){
	FILE *inFile;
	int i;
	int testCase;
	inFile=fopen("input.txt", "r");						
	fscanf(inFile, "%d", &testCase);

	for(i=0; i<testCase; i++){							
		int n1, n2;
		fscanf(inFile, "%d %d", &n1, &n2);
		printf("%d\n", gcd(n1, n2));
	}
	return 0;
}
