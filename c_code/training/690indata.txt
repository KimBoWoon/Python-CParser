#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;
using std::strlen;

int Palin(char str[], int in, int out){
	if (out <= in)
		return 1;
	else if (str[in] != str[out])
		return 0;
	else
		return Palin(str, in + 1, out - 1);
}

int main(){

	int testCase;
	char line[1001];

	ifstream file;
	file.open("input.txt");

	file >> testCase;
	for (int tCase = 0; tCase < testCase; tCase++){
		file >> line;
		cout << Palin(line, 0, std::strlen(line) - 1) << endl;
	}
	return 0;
}
