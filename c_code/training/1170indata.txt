#include <stdio.h>
#include <stdlib.h>
int eureka[1001]={0};
int check[1001]={0};
int main()
{
	FILE *in;
    int t=0;
    int n=0;
    int res=0;
    int i,j,p;
     
    for(i=1;i*(i+1)/2<1000;i++)
    {
        eureka[i]=i*(i+1)/2;
        n=i;
    }
    for(i=1; i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            for(p=1;p<=n;p++)
            {
                res = eureka[i]+eureka[j]+eureka[p];
                if(res>1000) break;
                else check[res]=1;
            }
        }
    }
	in = fopen("input.txt","r");
    fscanf(in,"%d",&t);
    while(t--)
    {
        int val;
        fscanf(in,"%d",&val);
        printf("%d\n",check[val]);
    }
}
