


#include <stdio.h>
#include <stdlib.h>

#define MAX 255

int makeTri(int *arr){		
	int num=0,i=0;
	while(num<1000){
		num+=i+1;
		arr[i++]=num;
	}

	return i;
}

int detectEureka(int data){		
	int i=0,j,k,max;
	int *triangle;

	triangle = (int*)malloc(sizeof(int)*MAX);

	max = makeTri(triangle);

	for(i=0; i<max; i++){
		for(j=0; j<max; j++){
			for(k=0; k<max; k++){
				if(data == triangle[i]+triangle[j]+triangle[k]){
					free(triangle);
					return 1;
				}
			}
		}
	}
	free(triangle);

	return 0;
}

void result(int flag){		
	if(flag == 1)
		printf("1\n");
	else
		printf("0\n");
}

int main(){
	int i,cases,data;
	int flag;

	FILE *infile;
	infile = fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0; i<cases; i++){
		fscanf(infile,"%d",&data);

		flag = detectEureka(data);
		result(flag);
	}

	fclose(infile);
	return 0;
}
