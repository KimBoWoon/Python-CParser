
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <vector>

using namespace std;
 
#define pb push_back
#define sz(v) ((int)(v).size())
 
int t, n;
 
int main()
{
   fstream infile;
   infile.open("input.txt");

   if(infile.fail())
   {
      cerr << "¿À·ù" << endl;
      exit(1);
   }

    for (infile >> t;t--;){
       infile >> n;
        bool ans = 0;

        for (int b=2;b<65;b++){
            vector <int> arr;

            for (int v=n;v;v/=b) arr.pb(v%b);
            int n = sz(arr);
            bool sw = 0;

            for (int j=0;j<n;j++) if (arr[j] != arr[n-j-1]) sw = 1;
            if (!sw){ ans = 1; break; }
        }
        printf("%d\n", (int)ans);
    }

   return 0;
}
