#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int k_temp = -1;
char data_string[100] = { 0, };

char *itoafunc(int val, int base)			
{
	static char buf[35] = { 0, };
	int k;
	k = 30;

	for (; val && k; --k, val /= base)
	{
		buf[k] = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ!@"[val % base];
	}
	k_temp = k + 1;
	return &buf[k + 1];
}
int palin(int start, int end)				 
{
	if (start >= end)                        
		return 1;                            
	else if (data_string[start] == data_string[end])
		return palin(start + 1, end - 1);
	else
		return 0;
}
int main()
{
	int numcase;
	int data;
	int radix = 2;
	int result;
	FILE * in_file;
	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d", &numcase);

	while (numcase--)
	{
		fscanf(in_file, "%d", &data);
		radix = 2;							
		while (radix < 65)					
		{
			strcpy(data_string, itoafunc(data, radix));
			result = palin(0, 30 - k_temp);

			if (result == 1)				
			{
				printf("1\n");
				break;
			}
			memset(data_string, 0, 100);	
			radix++;
			k_temp = -1;
		}
		if (radix == 65)					
			printf("0\n");
	}
}
