#include <stdio.h>
#include <string.h> 
#include <stdlib.h>

int mac(int a);

void main(void)
 {
	 FILE *in_file;
	int no_cases; 
	int i,m;
	
	
	in_file = fopen("input.txt","r");
	if(in_file == NULL) exit(1);
	fscanf(in_file,"%d",&no_cases);
	for(i=0; i<no_cases; i++){
		fscanf(in_file,"%d",&m);
		printf("%d\n", mac(m));
		
	}
}

int mac(int a)
{
	if(a>100)
		return a-10;
	else
		return mac(mac(a+11));
}
