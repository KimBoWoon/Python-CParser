



#include<stdio.h>

long factorial(int n)   
{
	if(n == 0)
	{
		return 1;
	}
	else
	{
		return n * factorial(n-1);
	}
}

int main()
{
	FILE *fp = NULL;
	int testCase;
	long result;

	fp = fopen("input.txt", "r");   
	fscanf(fp, "%d", &testCase);    
	while(testCase--)
	{
		int n;
		fscanf(fp, "%d", &n);       
		result = factorial(n);      
		printf("%ld\n", result);    
	}

    fclose(fp);
	return 0;
}
