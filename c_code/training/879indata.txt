



#include <stdio.h>
#include <stdlib.h>



void main(void)
{
	FILE *inFile;
	int i,j;
	int numCases;
	int count =0;
	int num;
	int sum[1000] ={0};

	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0;i<numCases;i++)
	{
		
		fscanf(inFile, "%d ",&num );
		sum[0]=0;
		sum[1]=1;
		for(j=0;j<1000;j++)
		{
			if(j==0)
				sum[j]=0;
			else if(j==1)
				sum[j]=1;
			else
			{
				sum[j] = sum[j-1] + sum[j-2];
			}

		}

		printf("%d\n",sum[num]);
	}

		

	
	
	fclose(inFile); 

}
