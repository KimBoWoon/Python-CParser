
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int arr[15] = { 0, };
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while (t--)
	{
		int n,cnt;
		in >> n;
		for (int i = 0; i < n; i++) in >> arr[i];
		for (int i = 0; i < 1000;i++)
		{
			cnt = 0;
			int tmp = arr[0];
			for (int j = 0; j < n; j++)
			{
				if (j == n - 1) arr[j] = abs(arr[n - 1] - tmp);
				else arr[j] = abs(arr[j] - arr[j + 1]);
				if (arr[j] == 0) cnt++;
			}
			if (cnt == n) break;
		}
		if (cnt == n) cout << "ZERO" << "\n";
		else cout << "LOOP" << "\n";
	}
	in.close();
}
