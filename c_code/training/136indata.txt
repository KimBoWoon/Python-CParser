
#include <iostream>
#include <fstream>
using namespace std;

unsigned short mcCathy(unsigned short n){
	if(n > 100) return n-10; 
	else return mcCathy(mcCathy(n+11)); 
}

int main(){
	ifstream in;
	in.open("input.txt");

	unsigned short t,n;
	in>>t;

	while(t--){
		in>>n; 
		cout<<mcCathy(n)<<endl;
	}
	in.close();
}
