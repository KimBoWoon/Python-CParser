

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int gcd(int a, int b);

void main(void) {
	FILE *inFile;
	int caseNum;

	int i;
	int n1,n2;

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &caseNum);

	for (i = 0; i < caseNum; i++) {
		fscanf(inFile, "%d %d", &n1 , &n2);
		printf("%d\n", gcd(n1, n2));		
	}
}

int gcd(int a, int b) {
	if (b == 0) {
		return a;
	}
	return gcd(b, a%b);
}
