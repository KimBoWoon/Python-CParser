#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

void repeatPallen(int num);
int checkPallen(int num, int k);
int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	if (inStream.fail()) exit(1);

	int numTest;

	inStream >> numTest;

	for (int k = 0; k < numTest; k++)
	{
		int num;
		inStream >> num;
		repeatPallen(num);
	}
}
void repeatPallen(int num)
{
	int result = 0;
	for (int k = 2; k <= 64; k++)
	{
		if (checkPallen(num, k) == 1){
			result = 1;
			break;
		}

	}
	if (result == 0) cout << 0 << endl;
	else cout << 1 << endl;

}
int checkPallen(int num, int k){
	int * numArray = new int[200];
	int count = 0;
	while (true)
	{
		if (num == 0) break;
		numArray[count] = num%k;
		num = num / k;
		count++;
	}

	for (int k = 0; k < count; k++)
	{
		if (numArray[k] != numArray[count - k-1]){
			return 0;
		}
	}
	return 1;
}
