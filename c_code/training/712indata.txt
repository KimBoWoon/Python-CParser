
#include <iostream>
#include <cstdlib>
#include <fstream>
#include <cstring>
using namespace std;

int calculate(char str[], int first, int last);



int main()
{
	std::fstream inputfile;
	int testcase;
	char str[1001];

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> str;
		
		cout << calculate(str, 0, strlen(str) - 1) << endl;
	
	}

	return 0;
}

int calculate(char str[], int first, int last)
{

	if (last <= first)
		return 1;

	else if (str[first] != str[last])
		return 0;

	else
		return calculate(str, first + 1, last - 1);

}
