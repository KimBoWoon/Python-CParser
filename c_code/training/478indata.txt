
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);

}
void main(){
	FILE *in_file;
	int no_cases, no_data1,no_data2;
	int i;
	

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d", &no_data1,&no_data2);
	
		printf("%d\n", gcd(no_data1, no_data2));
		
	}

	fclose(in_file);
}
