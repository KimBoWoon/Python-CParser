

#include <fstream>
#include <iostream>
#include <cstring>
using namespace std;

int factorial(int n){
	if (n == 0) return 1;
	else return n * factorial(n - 1);
}

void swap(char* a, char* b){
	char temp = *a;
	*a = *b;
	*b = temp;
}

void sort(int Num, char *str)
{
	int max = Num;
	int i, j;
	for (i = 0; i < max; i++){
		for (j = i + 1; j < max; j++){
			if (str[i] > str[j]) swap(&str[i], &str[j]);
		}
	}
}

void swap(char **a, char **b)
{
	char *c;
	c = *a;
	*a = *b;
	*b = c;
}


int main(){

	int caseNum = 0;

	ifstream inputFile("input.txt");
	inputFile >> caseNum;
	for (int i = 0; i < caseNum; i++){
		char str[] = "abcdefghijkl";
		int testLen, idx;
		inputFile >> testLen >> idx;
		int index = 0;
		idx = idx - 1;
		if (idx != 0){
			for (int i = 0; i < testLen; i++){
				index = idx / factorial(testLen - i - 1);
				idx = idx % factorial(testLen - i - 1);
				swap(str[i], (str + i)[index]);
				sort(testLen - i - 1, (str + i + 1));
			}
		}

		for (int i = 0; i < testLen; i++){
			cout << str[i];
		}
		cout << endl;
	}

	return 0;
}
