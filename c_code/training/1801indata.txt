

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;
int main()
{
	int numTestCases;
	char test[50];
	ifstream inStream("input.txt");	
	inStream >> numTestCases;
	inStream.getline(test, 50); 
	for (int i = 0; i < numTestCases; i++)
	{
		int count = 0;
		int testCount = 0;
		int check = 0;
		inStream.getline(test, 50); 
		for (int i = 0; i < 50; i++)
		{
			count++; 
			if (test[i] != '(' && test[i] != ')')
				break; 
		}
		count = count - 1;
		if (test[0] == ')') 
			check = 0;
		else
		{
			if (test[count - 1] == '(') 
				check = 0;
			else
			{
				for (int i = 0; i < count; i++) 
				{
					if (test[i] == '(')
						testCount++;
					if (test[i] == ')')
						testCount--;
				}
				if (testCount == 0) 
					check = 1;
				else
					check = 0;
			}
		}

		if (check == 0) 
			cout << "NO" << endl;
		else if (check == 1)
			cout << "YES" << endl;
	}

	inStream.close();
	return 0;
}
