


#include <stdio.h>

#define MAXSIZE 500+1
#define MIN(A,B) A>B?B:A

int MergingFiles(int num,int *file){
	int i,j,k,min;
	int merge[MAXSIZE][MAXSIZE]={0};
	int sum[MAXSIZE][MAXSIZE]={0};

	if(num==1)
		return file[1];

	for(i=1;i<=num;i++){
		for(j=i;j>0;j--){
			if(i==j)
				sum[i][j]=file[i];
			else
				sum[j][i]=sum[j][i-1]+sum[i][i];  
		}
	}

	for(i=2;i<=num;i++){
		for(j=i-1;j>0;j--){
			for(k=j;k<i;k++){
				if(k==j) min=merge[j][k]+merge[k+1][i]+sum[j][i];
				min=MIN(min,(merge[j][k]+merge[k+1][i]+sum[j][i])); 
			}
			merge[j][i]=min; 
		}
	}
	return merge[1][num]; 
}

int main(){
	int cases,num,i;
	int file[MAXSIZE];
	FILE *fp=fopen("input.txt","r");
	fscanf(fp,"%d",&cases);

	while(cases--){
		fscanf(fp,"%d",&num);
		for(i=1;i<=num;i++)
			fscanf(fp,"%d",&file[i]);
		printf("%d\n",MergingFiles(num,file));
	}
	fclose(fp);
	return 0;
}
