
#include <iostream>
#include <fstream>
#include <string>
#include <vector>

bool pel(std::string s, int first, int last);

std::string changenum(int n, int base)
{
	std::string temp;

	int num = n;
	while( num != 0 )
	{
		char t = num%base+33;
		temp.push_back(t);
		num /= base;
	}

	return temp;
}

int main()
{
	std::vector<char> v;
	std::ifstream infile("input.txt");

	int numCases;
	
	infile >> numCases;

	for(int i =0 ; i<numCases; i++)
	{
		std::string s;
		int n;
		infile >> n;
		int chk=0;
		for( int j=2; j< 65; j++)
		{
			s = changenum(n,j);
			if(pel(s,0,s.length()-1))
			{
				chk =1;
				std::cout<<1<<std::endl;
				break;
			}
		}

		if ( chk == 0 )
			std::cout<<0<<std::endl;
	}
	return 0;
}

bool pel(std::string s, int first, int last)
{
	if( last <= first)
		return true;
	else if ( s.at(first) != s.at(last) )
		return false;
	else
		return pel(s,first+1,last-1);
}
