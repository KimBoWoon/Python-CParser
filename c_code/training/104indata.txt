#include <iostream>
#include <fstream>
using namespace std;

int M(int n);

int main()
{
	ifstream infile;

	infile.open("input.txt");
	
	int numTestCases;
	int result = 0;
	

	infile >> numTestCases;

	while(numTestCases--)
	{
		infile >> result;
		
		cout<< M(result) << endl;
	}

	
	return 0;
}
int M(int n)
{
	if(n > 100)
		return n - 10;
	else if(n <= 100)
		return M(M(n+11));
}
