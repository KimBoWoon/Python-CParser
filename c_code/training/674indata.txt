#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int cal(int a); 

long num = 1;

int main()
{
	fstream infile;

	int nocases;
	int a;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i = 0; i < nocases; i++)
	{
		infile >> a;

		cout << cal(a)<< endl;

		num = 1;
	}
	return 0;
}

int cal(int a)
{
	if(a > 1)
	{
		return a*cal(a-1);
	}

	return num;
}
