
#include <stdio.h>
#include <math.h>

int isPrime(int testNum, int div){
    if(sqrt(testNum) < div) return 1;
    if(testNum % div == 0) return 0;
    return isPrime(testNum, ++div);
}

void perform(int n, int index){
    if(isPrime(n-index, 2) && isPrime(index, 2))
        printf("%d %d\n", n-index, index);
    else return perform(n, ++index);
}

void goldbach(FILE *fin){
    int n = 0;
    
    fscanf(fin, "%d", &n);
 
    perform(n, n/2);
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        goldbach(fin);      
    
    fclose(fin);            
    
    return 0;           
}
