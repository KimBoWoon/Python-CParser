#include <iostream>
#include <fstream>
#include <string>

int main()
{
	std::ifstream file("input.txt");
	int testCase = 0;
	file >> testCase;

	while (testCase--)
	{
		std::string g;
		int countR = 0, countL = 0;
		file >> g;
		for (int i = 0; i < g.size(); i++)
		{
			if (g[i] == '(')
				countL++;
			if (g[i] == ')')
				countR++;
		}
		if (g[g.size() - 1] == ')'&&countR == countL)
			std::cout << "YES";
		else
			std::cout << "NO";
		std::cout << std::endl;
	}

}
