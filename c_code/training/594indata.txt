

#include <stdio.h>
#include <stdlib.h>

static factresult[13] = { 0 };
int fact(int n);

void main(void) {
	FILE *input;
	int no_cases;
	int data;
	int i;

	factresult[0] = 1;
	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d", &data);
		factresult[data] =fact(data);
		printf("%d\n", factresult[data]);

	}

	fclose(input);

}

int fact(int n) {
	if (n == 1) {
		return 1;
	} else {
		if (factresult[n] != 0) {
			return factresult[n] * fact(1);
		} else {
			return n * fact(n - 1);
		}
	}
}
