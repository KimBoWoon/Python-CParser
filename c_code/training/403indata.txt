

#include <stdio.h>
unsigned int Euclidean(unsigned int a,unsigned int b) { 
	if (!b) return a; 
	return Euclidean(b, a%b); 
}
int main()
{
	unsigned int t, a,b, temp;
	FILE *fp;
	if ((fp = fopen("input.txt", "r")) == NULL) return 0; 
	fscanf(fp, "%d", &t); 
	while (t--){
		fscanf(fp, "%d %d", &a,&b); 
		if (a < b) 
		{
			temp = a;
			a = b;
			b = temp;
		}
		printf("%u\n", Euclidean(a,b)); 
	}fclose(fp);
}
