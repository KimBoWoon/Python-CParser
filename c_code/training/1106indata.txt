

#include<iostream>
#include<fstream>
using namespace std;

int main()
{
	ifstream inStream;
	int testcase;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "input file open failed\n";
		return 0;
	}

	inStream >> testcase;

	int a[45];

	for (int j = 1; j < 46; j++)
	{
		a[0] = 0;
		a[j] = j*(j + 1) / 2;
	}

	for (int i = 0; i < testcase; i++)
	{
		int checknum;

		inStream >> checknum;

		bool ck=false;
		

		for (int j = 1; j < 46; j++)
		{
			for (int k = 1; k < 46; k++)
			{
				for (int l = 1; l < 46; l++)
				{

					if ((a[j] + a[k] + a[l]) == checknum)
					{
						ck = true;
						break;
					}
					
				}
				if (ck == true)
					break;
			}
			if (ck == true)
				break;
		}
		
		cout << ck << endl;
	}

	return 0;
}
