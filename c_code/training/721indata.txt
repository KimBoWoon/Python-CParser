{

	if (inStream.fail()) exit(1);
	int numTest,k=0;
	inStream >> numTest;
	string test;
	getline(inStream, test);
	for (int k = 0; k < numTest; k++){
		inStream >> test; 
		cout << palindrome(test, 0, test.size()-1)<<endl;
	}
}
int palindrome(string str, int first,int last){
	if (last <= first) return 1;
	else if (str[first] != str[last]) return 0;
	else return palindrome(str, first + 1, last - 1);

}
