
#include <iostream>
#include <fstream>
using namespace std;

void ducci(int num ,int *input);
int main()
{
	ifstream fp;
	int numcases;
	int n;
	int input[20];
	int sum=0;
	fp.open("input.txt");

	fp>>numcases;

	for(int i=0;i<numcases;i++)
	{
		fp>>n;

		for(int i=0;i<n;i++)
			fp>>input[i];
		
		ducci(n,input);
	}
}

void ducci(int num ,int *input)
{
	int sum=0;

	for(int i=0;i<1000;i++)
	{
		input[num]=input[0];
		 sum=0;
		for(int j=0;j<num;j++)
		{
			input[j]=input[j]-input[j+1];

			if(input[j]<0)
				input[j]=-input[j];
			sum+=input[j];
		}
			
		if(sum==0)
		{
			cout<<"ZERO"<<endl;
			return;
		}
	}
	cout<<"LOOP"<<endl;
}
