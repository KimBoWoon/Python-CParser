
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int Palindrome(char array[], int first_word, int last_word){

	if(first_word >= last_word)							return 1;
	else if(array[first_word] != array[last_word])		return 0;
	else	Palindrome(array, first_word+1, last_word-1);	
}

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL)	exit(1);

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){

		char list[1000];
		int a;

		fscanf(in_file, "%s", &list);

		a = strlen(list);

		printf("%d\n", Palindrome(list, 0, a-1));
	}
	fclose(in_file);
}
