

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int num=0;
int goldbach(int n)
{
	int i, j;
	int result = 0;
	int a = n/2, b=0, c=0;
	int min=0, k=0;
	int arr2[1000][2]={0};
	int arr[1000]={0};

	if(n = a+a) 
	{
		c=a-1;
		while(1) 
		{
			if (c==1)
			{
				result = 1;
				return printf("%d %d\n", a, a);
			}
			else if (a%c==0) 
			{
				result = 0;
				break;
			}
			c--;
		}
	}	
	
	if(result == 0) 
	{
		for(i=2; i<n; i++)
		{
			b=i-1;
			arr[0]=1;
			while(1) 
			{
				if (b==1) 
				{
					arr[num+1] = i;
					num++;
					break;
				}
				else if (i%b==0) 
				{
					break;
				}	
				b--;
			}
		}

		for(i=0; i<num; i++) 
		{
			for(j=0; j<num; j++)
			{
				if(n == arr[i] + arr[j])
				{
					if(min==0 || (min > abs(arr[i]-arr[j])))
					{
						min = abs(arr[i]-arr[j]);
						arr2[k][0] = arr[i];
						arr2[k][1] = arr[j];
						k++;
					}
				}
			}
		}
		
		for(i=0; i<k; i++) 
		{
			if(abs(arr2[i][0]-arr2[i][1]) == min)
			{
				if(arr2[i][0]>arr2[i][1])
					return printf("%d %d\n", arr2[i][1], arr2[i][0]);
				else
					return printf("%d %d\n", arr2[i][0], arr2[i][1]);
			}
		}
	}
}
	

void main()
{
	FILE *file;
	int i;
	int test;
	int n;

	file = fopen("input.txt", "r");
	fscanf(file, "%d", &test);
	for(i=0; i<test; i++)
	{
		fscanf(file, "%d", &n);

		goldbach(n);
		num=0;
	}
	fclose(file);
}
