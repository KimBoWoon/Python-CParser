
#include<stdio.h>

char isEurekaTheorem[1001] = {0, };

void init()
{
	int triangleNumber[45];
	int i, j, k;

	for(i=0; i<45; i++)
	{
		triangleNumber[i] = (int)(i * (i + 1) * 0.5f);
	}

	for(i=1; i<45; i++)
	{
		for(j=1; j<45; j++)
		{
			for(k=1; k<45; k++)
			{
				int sum = triangleNumber[i] + triangleNumber[j] + triangleNumber[k];
				if(sum <= 1000)
				{
					isEurekaTheorem[sum] = 1;
				}
			}
		}
	}
}

int main(void)
{
	FILE *fp = NULL;
	int testLine;

	init();	
	fp = fopen("input.txt", "r");

	fscanf(fp, "%d", &testLine);
	while(testLine--)
	{
		int testNumber;
		fscanf(fp, "%d", &testNumber);
		printf("%d\n", isEurekaTheorem[testNumber]);
	}

	fclose(fp);
	return 0;
}
