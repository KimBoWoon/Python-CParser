

#include <stdio.h>
#include <stdlib.h>
int main(void)
{
	FILE *in_file;
	int i, j;
	int no_cases, num;
	int result, data;
	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);
	fscanf(in_file, "%d", &no_cases); 
	for (i = 0; i<no_cases; i++)
	{
		fscanf(in_file, "%d", &num);
		result = 1;
		for (j = 2; j<num; j++)
		{
			if(num%j==0) result = 0;
		}
		printf("%d\n", result); 
	}
	fclose(in_file); 
}
