#include <stdio.h> 
 #include <stdlib.h> 
 
 
 #define MAX 2147483647 
 
 
 int gcd(int n, int m);  
 int find_unit(int a, int b); 
 
 
 int main() 
 { 
 	FILE* in_file;  
 	int test_case; 
 	 
	in_file = fopen("input.txt", "r"); 

 	if(in_file == NULL) { 
 		printf("file open error\n"); 
 		exit(1); 
 	} 
 
 
 	fscanf(in_file, "%d ", &test_case);  
 	while(test_case--) { 
 		int a, b; 
 
 
		fscanf(in_file, "%d %d", &a, &b); 
 
 
		printf("%d\n", find_unit(a, b)); 
	} 
 } 
 
 

 int gcd (int n, int m) 
 { 
	if(m==0) 
 		return n; 
	else 
 		return gcd(m, n%m); 
} 

 int find_unit(int a, int b) 
 { 
 	int boundary = b;
 	int unitFraction=2; 
 	int g; 
 
 
 	 
 	while(boundary <= MAX) 
 	{ 
 		 
 		if(a==1) 
 			return b; 

 
 		 
 		if(a*unitFraction - b < 0) 
 		{ 
 			
 			if(boundary * ++unitFraction <= MAX) 
 				continue; 
 			
 			else 
 				break; 
 		} 
 
 
 		  
 		a = a*unitFraction - b; 
 		b *= unitFraction; 

 
		
		g = gcd(b,a); 

 
		a /= g; 
 		b /= g; 

 
 	
		boundary *= unitFraction++; 
 	} 
 
 
 	return b;  
 }
