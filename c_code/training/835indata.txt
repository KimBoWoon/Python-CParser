

#include <stdio.h>

int makeFibonacci(int num);

void main() {
	FILE *f;
	int numTestCases = 0;
	int i = 0;
	int num = 0;
	
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d", &num);
		printf("%d\n", makeFibonacci(num));
	}
}

int makeFibonacci(int num) {
	if (num == 0)
		return 0;
	else if (num == 1)
		return 1;
	else
		return makeFibonacci(num - 1) + makeFibonacci(num - 2);
}
