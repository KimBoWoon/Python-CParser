

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int palinDrome(char str[],int first, int last){
	
	if(last <= first)	
		return 1;
	else if(str[first] != str[last])	
		return 0;
	else
		palinDrome(str,first+1,last-1);	
}

void main(void){
	FILE *in_file;

	int no_case;
	char str[1000];
	int i;
	
	in_file = fopen("input.txt" , "r");		
	if(in_file == NULL) exit(1);

	fscanf(in_file , "%d", &no_case);

	for(i=0; i<no_case; i++)
	{
		fscanf(in_file , "%s", str);

		printf("%d\n",palinDrome(str,0,strlen(str)-1));
		
	}

	fclose(in_file);
}
