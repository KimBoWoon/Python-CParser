




#include<iostream>
#include<fstream>

int fib(int n);

int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){

		int n ;
		is >> n;

		std:: cout << fib(n) <<std::endl;


	}
}

int fib(int n){


	if ( n <=1 ) 
		return n;

	else
		return fib(n-2) + fib(n-1);


}
