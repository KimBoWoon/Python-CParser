
#include <iostream>
#include <fstream>

using namespace std;

int mcCarthy(int num); 


int main()
{
	ifstream inFile;
	inFile.open("input.txt");

	int numTestCase;	
	inFile >> numTestCase;

	while (numTestCase--)	
	{
		int data;	
		inFile >> data;

		cout << mcCarthy(data) << endl;	
	}

	inFile.close();
	return 0;
}

int mcCarthy(int num)
{
	if (num > 100)	
		return num - 10;

	else  
		return mcCarthy(mcCarthy(num + 11));	
}
