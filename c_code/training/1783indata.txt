
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

bool isOpen = false;
void Parenthesis(char* str, int& left, int& right,int index, int size){
	if (index == size + 1) return;
	else{
		if (str[index] == '('){
			left++;
			isOpen = true;
		}
		else if (str[index] == ')'){
			right++;
			if (left == right)
				isOpen = false;
		}
		Parenthesis(str, left, right, index + 1, size);
	}
}


int main(){

	int caseNum = 0;

	ifstream inputFile("input.txt");

	inputFile >> caseNum;

	for (int i = 0; i < caseNum; i++){
		char str[1001];
		inputFile >> str;

		int left = 0, right = 0;
		Parenthesis(str, left, right, 0, strlen(str) - 1);
		if (left == right && !isOpen)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}

	return 0;
}
