
#include <iostream>
#include <fstream>


int mc(int n);

int main()
{
	std::ifstream infile("input.txt");
	int numCase;

	infile >> numCase;

	for( int i=0; i< numCase; i++)
	{
		int n;
		infile >> n;

		std::cout<<mc(n)<<std::endl;
	}
}

int mc(int n)
{
	if( n > 100 )
		return n-10;
	else
		return mc(mc(n+11));
}
