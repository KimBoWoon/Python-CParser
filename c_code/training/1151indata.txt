
#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream){
		cout << "파일을 열 수 없습니다." << endl;
		return 0;
	}	
	int case_number;
	instream >> case_number;
	for(int i=0; i<case_number; i++)
	{
		int number;
		int result=0; 
		instream >> number;
		for(int i=1; i<45; i++) 
		{
			for(int j=1; j<45; j++) 
			{
				for (int k=1; k<45; k++) 
				{
					if( i*(i+1)/2 + j*(j+1)/2 + k*(k+1)/2 == number) 
						result = 1;
				}
			}
		}
		cout << result << endl;	
	}
}
