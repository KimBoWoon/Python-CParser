


#include <iostream>
#include <cstdlib>
#include <fstream>
using namespace std;
int arr[2];
bool isPrime(int n)
{
	for (int i = 2; i*i <= n; i++)
		if (n%i == 0) return false;
	return true;
}
void judgePrimes(int n)
{
	int differ = n;
	for (int i = 2; i <= n-2; i++)
	{
		if (isPrime(i) && isPrime(n - i))
		{
			int temp = n - i - i;
			if (temp < 0) temp = -temp; 
			if (temp < differ) 
			{
				differ = temp;
				arr[0] = i;
				arr[1] = n - i;
			}
		}
	}
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num = 0;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> num;
		judgePrimes(num);
		cout << arr[0] << " " << arr[1] << endl;
	}
	inStream.close();
	return 0;
}
