  

#include <iostream>
#include <fstream>
using namespace std;
int abs(int a);
int main(){

	ifstream inFile;
	inFile.open("input.txt");
	int no_cases=0; 
	inFile >> no_cases;
	for (int test = 0; test < no_cases; test++){
		int num=0;
		inFile>>num;
		int arr[1000]={2,},brr[1000]={2,},cnt=1;
		for(int i=2; i<num; i++){
			int flag=1;
			for(int j=2; j<i; j++){
				if(i%j==0){
					flag=0;
					break;
				}
			}
			if(flag==1){
				arr[cnt]=i;
				brr[cnt++]=i;
			}
		}
		int crr[2][1000]={0,},l=0;
		for(int i=0; i<cnt; i++){
		
			for(int j=0; j<cnt; j++){
				if((arr[i]+brr[j])==num){
					crr[0][l]=arr[i];
					crr[1][l++]=brr[j];
				}
			}
		}
		int min=abs(crr[1][0]-crr[0][0]);
		int f=0;
		for(int i=0; i<l; i++){
		
			if(min>abs(crr[1][i]-crr[0][i])){
				min=abs(crr[1][i]-crr[0][i]);
				f=i;
			}
		}
		cout<<crr[0][f]<<" "<<crr[1][f]<<endl;
	}
	return 0;
}
int abs(int a){
	if(a<0)
		return -a;
	else
		return a;
}
