#include<stdio.h>

int num(int a, int b)
{
	if (b == 0) return a;
	else return num(b, a%b);
}

int Henry(int a, int b)
{
	int t, gcd;
	if(a==1) return b;
	else{
		t = b/a +1;
		a = a*t-b; b = b*t;
		gcd = num(a, b);
		a /= gcd; b /= gcd;
		Henry(a, b);
	}
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testcase, n;
	int a, b;

	fscanf(inFile, "%d", &testcase);
	for(n=0; n<testcase; ++n)
	{
		fscanf(inFile, "%d %d", &a, &b);
		printf("%d\n", Henry(a, b));
	}
	fclose(inFile);
}
