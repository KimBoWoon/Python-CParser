
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>

using namespace std;

void computeRoom(int h, int w, int n){
	int roomNum=0;
	if( n%h > 0 ){
		roomNum = (n/h)+1;
		roomNum += (n%h)*100;
	}else{
		roomNum = (n/h);
		roomNum += h*100;
	}

	cout<<roomNum<<endl;
}

int main()
{
	ifstream inStream;
	int cases;
	bool flag = true;

	inStream.open("input.txt");
	inStream>>cases;

	while( cases-- ){
		int h, w, n;
		inStream>>h>>w>>n;

		computeRoom(h,w,n);
	}

	return 0;
}
