

#include<iostream>
#include<fstream>
#include<string>
using namespace std;
int main()
{
	int numOfCase = 0; 
	int num = 0;	

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> num;

		for (int j = 2; j <= (num/2)+1; j++){
			if (j > (num / 2)){
				cout << "1" << endl;
				break;
			}
			if (num%j == 0){
				cout << "0" << endl;
				break;
			}

		}
	}
	input.close();
	return 0;
}
