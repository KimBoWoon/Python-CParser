
#include<iostream>
#include<fstream>

using namespace std;

int factorial(int n);

int main(void){
    
    ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {
	    
	    int n,number;
	    inStream >> n;
	    number=factorial(n);
	    cout<<number<<endl;
	}
    
    
    
    
}
int factorial(int n){
    
    if( n <1)
        return 1;
    else 
        return n * factorial(n-1);
}
