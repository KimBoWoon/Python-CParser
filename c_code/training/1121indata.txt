
#include <iostream>
#include <fstream>

using namespace std;

int is_Eureka(int num)  
{
	int i,j,k;	
 		
	for(i=1; ;i++){ 
		for(j=i; ;j++){	
			for(k=j; ; k++)
			{
				if( i*(i+1) + j*(j+1) + k*(k+1) == 2*num ) return 1;  
				else if( i*(i+1) + j*(j+1) + k*(k+1) > 2*num ) break; 
			}
			if(j==k) break; 
		}
		if(i==j) break; 
	}

	return 0;
}

int main()
{
	ifstream file;

	file.open("input.txt");

	if(file.is_open()==false) 
		cout<<"입력파일이 없습니다."<<endl;

	else
	{	
		int case_num;
		file>>case_num;

		for(int i=0; i<case_num; i++)
		{	
			int num;
			file>>num;
			cout<<is_Eureka(num)<<endl;
		}
	}
}
