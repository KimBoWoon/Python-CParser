

#include <stdio.h>
#include <stdlib.h>

void main(void)
{
	FILE *in_file;
	
	int no_cases,data_cases;
	int i, j,k,m,l, data;
	int t[45];
	int n[1001]={0};

	for(m=1; m<45; m++)
	{
		t[m]=m*(m+1)/2;
	}

	for(i=1; i<45;i++){
		for(j=1; j<45; j++)
		{
			for(k=1; k<45; k++)
			{
				data=t[i]+t[j]+t[k];
				if(data<=1000)n[data]=1;
			}
		}

	}
	in_file = fopen("input.txt","r");
	fscanf(in_file,"%d",&no_cases);
	if(in_file==NULL)exit(1);
	for(i=0; i<no_cases; i++)
	{
		fscanf(in_file,"%d",&data_cases);
		printf("%d\n",n[data_cases]);
	}
}
