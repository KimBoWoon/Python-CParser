
#include <iostream>
#include <fstream>

using namespace std;

int fibonacci(int n){

	if ( n <= 1 )
	{
		return n;
	}
	else
		return ( fibonacci(n-1) + fibonacci(n-2) );
}

int main(){

	ifstream inStream;
	inStream.open("input.txt");		

	int num_case;	

	inStream >> num_case;

	for(int i=0; i< num_case ; i++){

		int test;
		inStream >> test;

		cout<< fibonacci(test) << endl;

	}

}
