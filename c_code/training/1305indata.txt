int gcd(int a, int b){
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

	int index = 2;
	while (a != 1){
		double origin = a / b;
		double test = 1.0 / index;
		if (test < origin){
			a = a*index - b;
			b *= index;
			int n = gcd(a, b);
			if (n != 1){
				a /= n;
				b /= n;
			}
		}
		index++;
	}

}
