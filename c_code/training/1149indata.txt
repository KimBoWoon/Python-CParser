int checkEureka(int testnum)
{
	int checkarray[46] = { 0, };
	int *comparearray;
	int count = 0;
	int isbig;
	int tempnum = testnum;
	
	while (true)
	{
		isbig = checkpomular(count+1);
		if (testnum >= isbig)
		{
			checkarray[count] = isbig;
			count++;
		}
		else
			break;
	}

	comparearray = new int[count*count*count];

	for (int i = 0; i<count*count*count; i++)
	{
		comparearray[i] = 0;
	}

	for (int i = 0; i < count; i++)
	{
		for (int j = 0; j < count; j++)
		{
			for (int k = 0; k < count; k++)
			{
				comparearray[checkarray[i] + checkarray[j] + checkarray[k]] = 1;
			}
		}
	}
	int returnnum = comparearray[testnum];

	delete[]comparearray;
	if (returnnum == 1) return 1;
	else return 0;
	
}

int checkpomular(int n)
{
	return (n*n + n) / 2;
}
