int main()

	infile >> numTestCases;
	
	
	while(numTestCases--)
	{

		infile >> a;
		GOLD(a);
		
	}
	
	return 0;
}
void GOLD(int the_num)
{
	int prime_s = 0;
	int prime_f = 0 ;
	
	
	int i  = 0;

	for(i = 2 ; i <= the_num / 2; i++)
	{
	
	if(IsPrime(i)){
		prime_s = the_num - i;
		if(IsPrime(prime_s))
		{
				
			prime_f = i;

				continue;

		}
		

		else {
			prime_s = the_num - prime_f;
		continue;
		}
	}
	else
		continue;
	
	
	}
	
	cout<< prime_f <<" "<<prime_s << endl;

}
bool IsPrime(int num)
{
	int flag = 0;
	if( num == 2 || num == 3)
		return true;
	else{
	for(int i = 2 ; i <= num / 2  ; i ++)
		{if(num % i ==0)
			flag++;
			
		}
			if(flag > 0 )
				return false;
			else
				return true;
	}
	
}
