

#include <iostream>
#include <fstream>

using namespace std;


int gcd(int fst_input, int sec_input)
{
	if (sec_input == 0)
		return fst_input;

	else 
		return gcd(sec_input, fst_input%sec_input);
}
int main()
{
	ifstream inStream;
	int testcase;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "input file open failed";
		return 0;
	}

	inStream >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int first, second;
		inStream >> first >> second;
		cout<<gcd(first, second)<<endl;
	}

	return 0;
}
