
#include <iostream>
#include <fstream>
using namespace std;

int getMcCarthyFunc(int n){
    if(n>100)
        return n-10;
    else
        return getMcCarthyFunc(getMcCarthyFunc(n+11));
}

int main(){
    ifstream inStream;
    inStream.open("input.txt");
    int test_case;
    inStream >> test_case;

    for(int i=0; i<test_case; i++){
        int n;
        inStream >> n;
        cout << getMcCarthyFunc(n) << endl;
    }
}
