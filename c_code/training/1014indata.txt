#include<iostream>
#include<fstream>

using namespace std;
int main()
{
	ifstream inStream;
	int testcase;

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "input file open failed";
			return 0;
	}

	int arr[15];
	int temp;
	bool LOOP = 0;

	inStream >> testcase;


	for (int i = 0; i < testcase; i++)
	{
		int test_num;
		inStream >> test_num;
		int arr_sum = 0;

		for (int j = 0; j < test_num; j++)
		{
			inStream >> arr[j];

		}
			for (int k = 0; k < 1000; k++)
			{

				for (int l = 0; l < test_num; l++)
				{
					if (l==0)
						temp = arr[0];
					if (l == test_num - 1)
					{
						arr[l] = arr[l] - temp;
						if (arr[l] < 0)
							arr[l] = -arr[l];

						
					}
					else
					{
						arr[l] = arr[l] - arr[l + 1];
						if (arr[l] < 0)
							arr[l] = -arr[l];
					}
				}
				if (k == 999)
				{
					for (int l = 0; l < test_num; l++)
					{
						arr_sum += arr[l];

						if (arr_sum == 0)
							LOOP = 0;
						else
							LOOP = 1;
					}
				}
			
		}
		if (LOOP == 1)
			cout << "LOOP" << endl;
		else
			cout << "ZERO" << endl;
	}
	return 0;
}
