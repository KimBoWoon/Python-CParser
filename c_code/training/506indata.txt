#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int factorial(int a)
{
	int result = 1;
	int i = 1;
	for(; i<=a; i++) result*=i;
	return result;
}
void swap (char* str, int a, int b)
{
	char term = str[a];
	str[a]=str[b];
	str[b]=term;
}
void pop(char*str, int a , int b)
{
	swap(str,a,b);
	int i = b;
	for(; i>a+1; i--)	swap(str,i,i-1);
}

void permute(char* str, int size, int k)
{
	int a = 0;
	
	while(1)
	{
		int fac = factorial(size-a-1);
		int l = 0;
		int k1 = k;
		
		while(1)
		{
			if(k1-fac<=0)
				break;	
			l++;
			k1 = k1-fac;
		}
		pop(str,a,l+a);
		a=a+1;
		k=k-l*fac;
		if(a==size) return; 
	}
			
}

void main()
{
	FILE* infile = fopen("input.txt","r");
	int tastcase = 0;
	int i = 0;

	fscanf(infile,"%d",&tastcase);
	for(; i<tastcase; i++)
	{
		char spell[12] ={'a','b','c','d','e','f','g','h','i','j','k','l'};
		int num = 0, k = 0;
		int j=0; 
		fscanf(infile,"%d",&num);
		fscanf(infile,"%d",&k);
		permute(spell,num,k);
		for(j=0; j<num; j++) printf("%c",spell[j]);
		printf("\n");
    }
	fclose(infile);
}
