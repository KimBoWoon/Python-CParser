

#include <iostream>
#include <fstream>

class Hotel
{
private:
	int row, col;
public:
	Hotel(const int& _row, const int& _col):row(_row), col(_col){}
	int roomNum(const int& guestNum)
	{
		int colNum = (guestNum % row == 0 ? row : guestNum % row);
		int rowNum = ((guestNum - 1)/row)+1;
		return colNum*100+rowNum;
	}

};

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail() )
	{
		std::cout << "file open err" << std::endl;
		return 1;
	}

	int caseNum;
	input >> caseNum;

	for (int i = 0; i < caseNum; i++)
	{
		int row, col, guestNum;
		input >> row >> col>> guestNum;
		std::cout << Hotel(row, col).roomNum(guestNum) << std::endl;;
	}

	return 0;
}
