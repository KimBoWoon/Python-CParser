#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int Felin(char str[], int first, int last)
{
	if (last <= first)	
		return 1;
	else if (str[first] != str[last])	
		return 0;
	else	
		return Felin(str, first + 1, last - 1);
}

int main()
{
	int gatsu = 0;
	int judge = 0;
	char munja[1000];
	char bateunmunja[1000];

	ifstream inStream;
	inStream.open("input.txt");


	inStream >> gatsu;

	for (int i = 0; i < gatsu; i++)
	{
		inStream >> bateunmunja;
		cout<< Felin(bateunmunja, 0, strlen(bateunmunja) - 1)<<endl;
	}
	inStream.close();
	return 0;
}
