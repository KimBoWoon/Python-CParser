#include <stdio.h>
#include <stdlib.h>
void Henry(int a, int b){
	int x=2; 
	while((a*x-b)!=0){ 
		if(a*x>b){
			a = (a*x)-b; 
			b = b*x;
			x++;
		}
		x++;
	}
	if((a*x-b)==0){
		printf("%d\n",x);
	}
	else {
		printf("%d\n",b);
	}
}

int main(){
	int testcase,i,a,b;
	FILE *file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &testcase);
	for(i=0; i<testcase; i++)
	{
		fscanf(file,"%d %d",&a,&b);
		Henry(a,b);
	}
}
