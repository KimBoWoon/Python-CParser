
#include <fstream>
#include <iostream>
using namespace std;
int Fibonacci(int n){
	if (n == 0) return 0;
	else if (n == 1) return 1;
	else{
		return Fibonacci(n - 1) + Fibonacci(n - 2);
	}
}


int main(){

	int caseNum = 0;
	ifstream inputFile("input.txt");

	inputFile >> caseNum;
	for (int i = 0; i < caseNum; i++){
		int fiboNum = 0;
		inputFile >> fiboNum;

		cout << Fibonacci(fiboNum) << endl;
	}

	return 0;
}
