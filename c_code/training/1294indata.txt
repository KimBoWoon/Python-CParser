#include <cstdio>

inline bool isPrime(int n) {
	for (int i = 2; i * i <= n; i++) {
		if (n % i == 0) {
			return false;
		}
	}
	return true;
}

int main() {
	FILE * fp = fopen("input.txt", "r");
	int nCase, num; 
	fscanf(fp, "%d", &nCase);

	while (nCase--) {
		fscanf(fp, "%d", &num);
		int halfNum = num / 2;
		for (; halfNum > 1 && !(isPrime(halfNum) && isPrime(num - halfNum)); --halfNum);
		printf("%d %d\n", halfNum, num - halfNum);
	}

	fclose(fp);
	return 0;
}
