

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int gcd(int a, int b);
int comp(int num, int dem, int hen);

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if (instream.fail())
	{
		cerr << "Error" << endl;
		exit(1);
	}

	int no_case;
	instream >> no_case;

	while(no_case--)	
	{
		int numer;	
		int denom;	
		int Henry = 2;

		instream>>numer>>denom;

		while(numer !=1)
		{
			while(comp(numer, denom, Henry))
				Henry++;

			int gcdNum = gcd(denom, Henry);
			numer = ((numer*Henry) - (1*denom)) / gcdNum;
			denom = (denom * Henry)/ gcdNum;

			if(denom % numer == 0) {
				denom /= numer;
				numer = 1;
			}
		}
		cout<<denom<<endl;
	}
}
int gcd(int a, int b)
{
	if(b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int comp(int num, int dem, int hen)
{
	return num*hen < dem;
}
