
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
void cal(int H, int W, int N);
int main()
{
	ifstream inStream;
	int no_cases,H,W,N=0;

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"input file openning failed. \n ";
		exit(1);
	}
	inStream>>no_cases;
	for(int i=0; i<no_cases; i++){
		inStream>>H;
		inStream>>W;
		inStream>>N;
		cal(H,W,N);
	}
	return 0;
}
void cal(int H, int W, int N){
	int i,j=0;
	i=N/H+1 ;
	j=N%H;

	if(i<10&&j!=0)
		cout<<j<<"0"<<i<<endl;
	else if(i>=10&&j!=0)
		cout << j<<i<<endl;
	else if(i<10&&j==0)
		cout<<H<<"0"<<i-1<<endl;
	else if(i>=10&&j==0)
		cout<<H<<i-1<<endl;
}
