#include <stdio.h>
#include <stdlib.h>


int findK (num) {
	int i = 1;
 	while(1)
	{
		if(num<i*(i+1)/2 ) break;
		else i++;
	}
	return i-1;
}



int eureka(int k, int num)
{
	int x = 1;
	int y = 1;
	int z = 1; 
        
	
	for(x=1; x<=k; x++) {
		for(y=1; y<=k; y++) {
			for(z=1; z<=k; z++) {
				if((x*x+x+y*y+y+z*z+z)/2>num) break;
				else if ((x*x+x+y*y+y+z*z+z)/2==num) return 1;
			}
		}
	}
  	return 0;
}
				



void main()
{
	FILE* infile = fopen("input.txt", "r");
	int i = 0;
	int datacase = 0;
	fscanf(infile,"%d",&datacase);
	
	for (; i<datacase; i++)
	{
		int num = 0;
		int k = 0; 
		fscanf(infile, "%d", &num);

		k = findK(num);
		if(eureka(k,num)) printf("1\n");
		else printf("0\n");	

	}
	fclose(infile);
}
