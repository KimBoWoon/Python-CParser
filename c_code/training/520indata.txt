



#include<iostream>
#include<cstdlib>
#include<fstream>
#include<string>

using namespace std;

int Fact(int num);
int newCount = 0;

int main()
{
   int numCases;
   int count = 0;
   int size = 0;
   int checkSize = 0;

   ifstream inStream;
   inStream.open("input.txt");

   if (inStream.fail()) exit(1);

   inStream >> numCases;

   for (int i = 0; i < numCases; i++){
      inStream >> size >> count;
      char alpha[13] = "abcdefghijkl";
      checkSize = size;
      newCount = (--count) / Fact(size - 1);

      for (int k = 0; k < checkSize; k++)
      {
         newCount = (count) / Fact(size - 1);
         cout << alpha[newCount];

         int j = 0;
         for (int t = 0; t < size; t++)
         {
            if (alpha[t] != alpha[newCount])
            {
               alpha[j++] = alpha[t];
            }
         }

         count = (count) % Fact(size - 1);
         size--;
      }

      cout << endl;


   }
}

int Fact(int num)
{
   if (num <= 1) return 1;
   else  return num*Fact(num - 1);
}
