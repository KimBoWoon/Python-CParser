#include<iostream>
#include<fstream>

using namespace std;

int main(void){

	ifstream inStream;
	int numTestCases;
	int number;
	

	inStream.open("input.txt");

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++)
	{
		bool ok = true;
		inStream >> number;
		
		for (int j = 2; j < number; j++)
		{
			if (number %j == 0) {
				ok = false;
				break;
			}
			if (j > number/j) break;
		}
		if (ok == true)
			cout << 1 << endl;
		else 
			cout << 0 << endl;

	}
	return 0;
}
