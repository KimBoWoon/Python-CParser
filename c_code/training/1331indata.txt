#include <stdio.h>
#include <stdlib.h>

void checkHenry(int a, int b);
int gcd(int a, int b);

int num;
void main() {
	int numTestCases = 0;
	int i = 0;
	int a, b;

	FILE *f;
	f = fopen("input.txt", "r");
	if (!fopen)
		exit(1);

	fscanf(f, "%d", &numTestCases);
	for (i = 0; i < numTestCases; i++) {
		num = 2;
		fscanf(f, "%d %d", &a, &b);
		checkHenry(a, b);
	}
}

void checkHenry(int a, int b) {
	int temp = 0;
	if (a == 1) {
		printf("%d\n", b);
		return;
	}
	else {
		if (a * num - b < 0)
			num++;
		else {
			a = a*num - b;
			b = b*num;

			temp = gcd(a, b);
			if (temp > 1) {
				a /= temp;
				b /= temp;
			}
			num++;
		}
		checkHenry(a, b);
	}
}

int gcd(int a, int b) {
	int temp = a;
	if (a < b) {
		a = b;
		b = temp;
	}
	if (b == 0)
		return a;
	else if (b > 0)
		return gcd(b, a % b);

}
