



#include <stdio.h>
#include <stdlib.h>

int Prime(int n)
{
	int divisor;
	int isPrime =1;

	for(divisor=2; divisor*divisor<=n; divisor++)
				{
					if(n%divisor == 0)										
					{
						isPrime = 0;
						break;
					}
				}
	return isPrime;
}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int value;		
		int a,b;
		int c = 0;
		int divisor;

		fscanf(inFile, "%d", &value);									

		for(a=value/2; a<value; a++)
		{
				b = value - a;

				if( Prime(a) == 1 )
					if( Prime(b) ==1 )
						break;
		}
			

		printf("%d %d\n", b, a);
	}

	fclose(inFile);														
}
