

#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>

int gcdd(int num1, int num2);
void Henry(int num1, int num2);
int main(void)
{
	int no_cases, i;
	int a = 0, b = 0;

	FILE *in_file;
	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);  

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d", &a, &b); 
		Henry(a, b); 
	}
	fclose(in_file);
	return 0;
}
void Henry(int num1, int num2)
{
	int x = 2;
	if (num1 == 1)
	{
		printf("%d\n", num2);
		return;
	}

	while (1)
	{
			if (num1*x >= num2)
			{
				num1 = num1*x - num2;
				num2 = num2*x;
				int gcd = gcdd(num1, num2);
				num1 = num1 / gcd;
				num2 = num2 / gcd;
				
				if (num1 == 1)
				{
					printf("%d\n", num2);
					break;
				}
			}
			else
				x++;
	}
}
int gcdd(int num1, int num2) 
{
	if (num2 == 0)
		return num1;
	else if (num2 > 0)
	{
		return gcdd(num2, num1%num2);
	}
}
