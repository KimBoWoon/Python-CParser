#include<iostream>
#include<fstream>

using namespace std;

int result;

void calcFactorial(int data,int dataForCalc){
	if (data < 1){
		result = dataForCalc;
		return;
	}

	calcFactorial(data - 1, dataForCalc*data);

	return;
}

int main(){

	int testCase;
	int data;

	ifstream file;
	file.open("input.txt");

	file >> testCase;
	for (int tCase = 0; tCase < testCase; tCase++){
		file >> data;
		calcFactorial(data,1);
		cout << result << endl;
	}
}
