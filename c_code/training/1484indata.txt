#include<iostream>
#include<fstream>
using namespace std;

class Hotel{
	int cnt;
public:
	Hotel(){
		cnt = 1;
	}
	void HeightFirst(int h, int w, int n);
};

void Hotel::HeightFirst(int h, int w, int n){
	cnt = 1;

	for(int i=1; i<=w; i++){
		for(int j=1; j<=h; j++){
			if( j < 10) 
				if( i < 10){
					if(cnt == n){
						cout << j << '0' << i << endl; 
						return;
					}
					cnt++;
				}
				else{
					if(cnt == n){
						cout << j << i << endl; 
						return;
					}
					cnt++;
				}
			else 
				if( i < 10){
					if(cnt == n){
						cout << j << '0' << i << endl; 
						return;
					}
					cnt++;
				}
				else{
					if(cnt == n){
						cout << j << i << endl; 
						return;
					}
					cnt++;
				}
		}
	}
}

int main(){
	Hotel hotel;

	ifstream inStream;
	int numCases;
	int Height;
	int Width;
	int Num;

	inStream.open("input.txt");

	inStream >> numCases;

	for(int i=0; i<numCases; i++){
		inStream >> Height >> Width >> Num;

		hotel.HeightFirst(Height, Width, Num);
	}

	inStream.close();
}
