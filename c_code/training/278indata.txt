

#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b)
{
	if(b==0)
		return a;
	else
		 return gcd(b, a%b);
}

void fractionadd(int a, int b, int c, int d)
{
	int result[2];
	if(b==d)
	{
		result[0]=a+c; result[1]=b;
		printf("%d %d\n",result[0]/gcd(result[0],result[1]),result[1]/gcd(result[0],result[1]));
	}
	else if(d%b==0)
		return fractionadd(a*(d/b),b*(d/b),c,d);
	else
		return fractionadd(a*d,b*d,c*b,d*b);
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int a,b,c,d;
		int *result;
	
		fscanf(file,"%d %d %d %d", &a,&b,&c,&d); 
		 fractionadd(a,b,c,d);
	}
	fclose(file); 
}
