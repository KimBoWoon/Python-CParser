


#include <stdio.h>

int factorial(int num){
	if(num==0)	return 1;				
	else return num * factorial(num-1);
}

int main(){
	int cases,num,i;

	FILE *infile;
	infile = fopen("input.txt","r");
	fscanf(infile, "%d", &cases);

	for(i=0; i<cases; i++){
		fscanf(infile, "%d", &num);

		printf("%d\n", factorial(num));
	}
}
