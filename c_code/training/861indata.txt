

#include <stdio.h>
#include <stdlib.h>

int fibonacci(int num)
{
	if(num<=1)
		return num;
	else
		return (fibonacci(num-1)+fibonacci(num-2));
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int num,result;
		fscanf(file,"%d",&num); 

		printf("%d\n",fibonacci(num));
	}
	fclose(file); 
}
