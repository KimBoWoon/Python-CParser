

#include <iostream>
#include <fstream>

using namespace std;

int euclid(int n1, int n2);

int main()
{
	ifstream infile;
	int testCase;
	int num1, num2, num3, num4; 
	int result = 0;	
	int result1 = 0;	

	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num1 >> num2 >> num3 >> num4;		
		result = (num2 * num4) / euclid(num2, num4); 
		result1 = num1*(result/num2) + num3*(result/num4);

		cout << result1/euclid(result, result1)<<" "<<result/euclid(result, result1)  << endl;
	}

	infile.close();

	return 0;
}

int euclid(int n1, int n2)
{
	if (n2 == 0) 
		return n1;
	else if (n1 == 1 || n2 == 1) 
		return 1;
	else
		return euclid(n2, n1 % n2);

}
