
#include <stdio.h>
#define MAX 1000000

int isPalindrome(int arr[], int size) {
    int left = 0, right = size-1;
    while(left < right) {
        if(arr[left++] != arr[right--])
            return 0;
    }
    return 1;
}

int palindrome(FILE *fin) {
    int i, k = 0, arr[MAX] = {0};
    int N, n, base = 2;
    
    fscanf(fin, "%d", &N);
    
    for(base=2; base<=64; base++) {
        n = N;
        k = 0;
        while(n > 0) {
            arr[k++] = n%base;
            n /= base;
        }
        if(isPalindrome(arr,k))
            return 1;
    }
    return 0;
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        printf("%d\n", palindrome(fin));          
    
    fclose(fin);            
    
    return 0;           
}
