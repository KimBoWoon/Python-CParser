

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int palindrome(char* word, int first, int last){
	if( last <= first ) return 1;
	else if( word[first] != word[last] ) return 0;
	else palindrome ( word, first+1, last-1 );
}

void main(){
	
	FILE *in_file;
	int no_cases;

	in_file = fopen("input.txt", "r");

	if(in_file == NULL) exit(1);

	fscanf(in_file,"%d", &no_cases);
	while(no_cases--){
		char word[1000];
		fscanf(in_file,"%s",word);
		printf("%d\n",palindrome(word, 0, strlen(word) - 1));
	}
fclose(in_file);
}
