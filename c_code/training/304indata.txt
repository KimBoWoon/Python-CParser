#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_LEN 1000

#define true 1
#define false 0

typedef int boolean;

boolean fraction_sum(unsigned long a, unsigned long b, unsigned long c, unsigned long d);
unsigned long gcd(unsigned long a, unsigned long b);

void main()
{
	FILE* fileDes = NULL;			
	int trialNum;					

	unsigned long a, b, c, d;			
	int i = 0;							

	fileDes = fopen("input.txt", "r");

	if(fileDes == NULL)
		exit(1);

	fscanf(fileDes, "%d", &trialNum);

	for(i = 0; i < trialNum; i++)
	{
		fscanf(fileDes, "%lu %lu %lu %lu", &a, &b, &c, &d);

		fraction_sum(a, b, c, d);
	}

	fclose(fileDes);
}

boolean fraction_sum(unsigned long a, unsigned long b, unsigned long c, unsigned long d)
{


	unsigned long numerator = (a*d) + (b*c);		
	unsigned long demoninator = (b*d);				

	unsigned long fraction_gcd = gcd(demoninator, numerator);

	printf("%lu %lu\n", numerator/fraction_gcd, demoninator/fraction_gcd);

}

unsigned long gcd(unsigned long a, unsigned long b)
{
	if(b == 0)
		return a;
	else
		return gcd(b, a%b);
}
