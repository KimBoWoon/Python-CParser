
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b){						
	if(b==0)
		return a;								
	else
		gcd(b, a%b);						
}

void Henry(int num1, int num2){
	int num, div;
	int max=0;
	int upNum, downNum, g;
	if(num1==1)										
		max=num2;
	else
		max=( (num2/num1) +1);

	upNum=num1*max-1*num2;					
	downNum=num2*max;
	g=gcd(upNum, downNum);						
	upNum/=g;											
	downNum/=g;
	
	if(upNum==0){										
		cout<<max<<endl;
		return;
	}
	else{													
		Henry(upNum, downNum);
	}
}

int main(){
	ifstream infile;
	int i, testCase;
	infile.open("input.txt");
	infile>>testCase;

	for(i=0; i<testCase; i++){
		int div1, div2;

		infile>>div1>>div2;
		
		Henry(div1, div2);
	}
	return 0;
}
