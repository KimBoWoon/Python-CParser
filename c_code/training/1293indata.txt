
#include <cstdio>

using namespace std;

int main(void) {
	FILE *ip = fopen("input.txt", "r");

	int t;

	fscanf(ip, "%d", &t);

	int prime[200] = { 2 }, cnt = 1;

	for (int i = 3; i <= 1000; i++) {
		int j;
		for (j = 0; j < cnt; j++) {
			if (i % prime[j] == 0) {
				break;
			}
		}

		if (j == cnt) prime[cnt++] = i;
	}

	for (int testcase = 0; testcase < t; testcase++) {
		int n;


		fscanf(ip, "%d", &n);

		int i, j;
		int small = 0, big = 1000;

		for (i = 0; i < cnt; i++) {
			for (j = i; j < cnt; j++) {
				if (prime[i] + prime[j] == n) {
					if (big - small > prime[j] - prime[i]) {
						big = prime[j];
						small = prime[i];
					}
				}
			}
		}


		printf("%d %d\n", small, big);
	}

	return 0;
}
