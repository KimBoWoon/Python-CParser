

#include <iostream>
#include <fstream>
using namespace std;
char str[14]={'0','a','b','c','d','e','f','g','h','i','j','k','l','0'};

int strNum=1;

int facTo(int n)
{
	int number=1;
	for(int i=1; i<=n; i++) number*=i;

	return number;
}

int main()
{
	fstream in;
	in.open("input.txt");
	int numcases;
	in>>numcases;
	for(int i=0; i<numcases; i++){
		int n,m;	
		int count=0; 
		in>>n>>m;
		int num=m;
		int test;
		int k=1;
		int countStr=n;
		int extraCase;
		while(countStr!=0){
			while(1)
			{
				test=facTo(n-k);
				count+=test;
				if(count>=num)
				{
				
				   while(str[strNum]=='0'){
						strNum++;
						if(strNum>n)
						{
							strNum=1;
						}
				   }
					cout<<str[strNum];
					str[strNum]='0';
					strNum=1;
					while(str[strNum]=='0')
						strNum++;
				    extraCase=(count-test);	
					countStr--;
					k++;
					num-=extraCase;
					count=0;
					break;
				}
				else{ 
						strNum++;
						if(str[strNum]=='0')
						{
							strNum++;
							 while(str[strNum]=='0'){
									strNum++;
									if(strNum>n)
										strNum=1;

								 }
							
						}
				}
			}
		
		}
		cout<<endl;
		strNum=1;
		
		str[0]='0';
		str[1]='a';
		str[2]='b';
		str[3]='c';
		str[4]='d';
		str[5]='e';
		str[6]='f';
		str[7]='g';
		str[8]='h';
		str[9]='i';
		str[10]='j';
		str[11]='k';
		str[12]='l';
		str[13]='0';
	

	}
}
