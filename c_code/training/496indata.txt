#include<iostream>
#include<fstream>
using namespace std;

int GCD(int n1, int n2){
	if (n2 == 0)	return n1;
	else	return GCD(n2, n1 % n2); 
}
int main(){
	ifstream inStream;
	int numCases;
	int num1;	int num2;

	inStream.open("input.txt");

	inStream >> numCases;

	for (int i = 0; i<numCases; i++){
		inStream >> num1 >> num2;

		cout << GCD(num1, num2) << endl;
	}

	inStream.close();
}
