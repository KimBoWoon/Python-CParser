
#include <iostream>
#include <fstream>

int Fibonacci(int n) {	
	
	if (n == 0) return 0;
	else if (n == 1) return 1;		
	else return Fibonacci(n - 1) + Fibonacci(n - 2);

}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num;
	
	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {

		fin >> num;
		std::cout << Fibonacci(num) << std::endl;
	}
	fin.close();
}
