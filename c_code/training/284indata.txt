

#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b){
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int lcm(int a, int b, int gcd)
{
	return (a / gcd)*(b / gcd)*gcd;
}
int main()
{
	ifstream instream("input.txt");
	int numtestcases, a, b, c, d;
	int gcdnum, result_ja, result_mo;
	instream >> numtestcases;
	for (int i = 0; i < numtestcases; i++){
		instream >> a >> b >> c >> d;
		gcdnum = gcd(b, d);
		result_mo = lcm(b, d, gcdnum);
		result_ja = (result_mo / b)*a + (result_mo / d)*c;
		gcdnum = gcd(result_mo, result_ja);
		result_mo /= gcdnum;
		result_ja /= gcdnum;
		cout << result_ja << ' ' << result_mo << endl;
	}
	instream.close();
}
