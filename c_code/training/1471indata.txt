#include <stdio.h>
#include <stdlib.h>

void main()
{
	FILE *file;
	int N,H,W,G;
	int t;
	int floor,number;
	int i=0;

	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d%d%d", &H,&W,&G);
	
		if(G%H==0)
		{
			floor = H;
			number = G/H;
		}
		else
		{
			floor = G%H;
			number = G/H+1;
		}
		printf("%d%02d\n",  floor,number);
	}
	fclose(file);
}
