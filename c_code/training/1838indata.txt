

#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;

bool check(char* str, int sum, int i, int length);

int main(void)
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {

		char *str = new char[51];
		inStream >> str;
		int length = strlen(str);
		if (length % 2 == 1) cout << "NO" << endl;
		else{
			if (check(str, 0, 0, length) == true) cout <<"YES" << endl;
			else cout << "NO" << endl;
		}
	}
}

bool check(char* str, int sum, int i, int length)
{
	if (str[i] == '(') sum += 1;
	else if (str[i] == ')') sum -= 1;

	if (sum < 0) return false;
	if (i == length - 1){
		if (sum == 0) return true;
		else return false;
	}
	check(str, sum, i + 1, length);
}
