void checkParthesis(char arr[50]) {
	int stat = 1;
	int p = 0;
	while (arr[p] != NULL) {
		if (arr[p] == '(') {
			stat++;
		}
		else if (arr[p] == ')') {
			stat--;
			if (stat == 0) {
				break;
			}
		}
		else {
			break;
		}
		p++;
	}

	if (stat == 1) {
		cout << "YES" << endl;
	}
	else {
		cout << "NO" << endl;
	}
}
