 


#include <stdio.h>
#include <stdlib.h>

int main()
{
    int T;
    FILE *fp;
    fp = fopen("input.txt","r");
    fscanf(fp,"%d",&T);
    while(T--)
    {
        long int value;
        int i,result=0;
        fscanf(fp,"%ld",&value);
        for(i=2;i<=64;i++) 
        {
            int cnt=0, arr[21],j,flag=0;
            long int tmp = value;
            while(tmp >= i) 
            {
                arr[cnt] = tmp%i;
                tmp = tmp / i;
                cnt++;
            }
            arr[cnt] = tmp;
            for(j=0;j<=cnt/2;j++) 
            {
                if(arr[j] != arr[cnt-j])
                {
                    flag=1;
                    break;
                }
            }
            if(flag == 0) 
            {
                result = 1;
                break;
            }
        }
       printf("%d\n",result);
    }
    fclose(fp);
    return 0;
}
