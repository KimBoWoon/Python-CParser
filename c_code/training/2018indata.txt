

#include <iostream>
#include <cstdlib>
#include <fstream>

using namespace std;

int main() {

	ifstream instream;
	instream.open("input.txt");

	if (instream.fail()) {
		cerr << "ERROR" << endl;
		exit(1);
	}

	int no_case;

	instream >> no_case;

	while (no_case--) {
		int num, target, *set;
		instream >> num >> target;

		set = new int[num];
		int res;
		int diff = 999;

		for (int i = 0; i < num; i++)
			instream >> set[i];

		for (int i = 0; i < num - 1; i++) {
			for (int j = i + 1; j < num; j++) {
				int comp = abs((set[i] + set[j]) - target);
				if (diff > comp) {
					diff = comp;
					res=1;
				}
				else if (diff == comp){
					res++;
				}
			}
		}
		cout << res << endl;
		delete[] set;
	}
}
