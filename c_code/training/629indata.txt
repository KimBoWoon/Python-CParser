

#include <iostream>
#include <fstream>
using namespace std;

unsigned int facTo(unsigned int n)
{
	if(n<=0)
	{
		return 1;
	}
	else
	{
		return n*facTo(n-1);
	}
}

int main()
{
	fstream in;
	in.open("input.txt");
	int numcases;
	in>>numcases;
	for(int i=0; i<numcases; i++){
		unsigned int n,m;
		in>>n;
		m=facTo(n);
		cout<<m<<endl;
	}
}
