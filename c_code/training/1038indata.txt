
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>
using namespace std;

void Ducci(int ducci[], int num){
	int j,k, sum=0;
	for(j=0; j<1000; j++){				
		int t=ducci[0];						
		for(k=0; k<num; k++){			
			if(k<num-1)
				ducci[k]=abs(ducci[k]-ducci[k+1]);	
			else if(k==num-1)
				ducci[k]=abs(ducci[k]-t);			
		}
	}

	for(j=0; j<num; j++)				
		sum+=ducci[j];
	if(sum==0)
		cout<<"ZERO"<<endl;
	else
		cout<<"LOOP"<<endl;
}


int main(){
	ifstream infile;
	int i, testCase;
	infile.open("input.txt");
	infile>>testCase;
	for(i=0; i<testCase; i++){
		int j, k, num;
		int ducci[15];
		infile>>num;
		for(j=0; j<num; j++){
			int ducciNum;
			infile>>ducciNum;
			ducci[j]=ducciNum;
		}

		Ducci(ducci, num);

	}
	return 0;

}
