#include <stdio.h>

void henry(int a, int b, int x);

int main() {
	FILE *fp = fopen("input.txt", "r");
	int testCaseNumber = 0, testA = 0, testB = 0;

	fscanf(fp, "%d", &testCaseNumber);
	while (testCaseNumber--) {
		fscanf(fp, "%d %d", &testA, &testB);

		henry(testA, testB, 2);
	}
	fclose(fp);
	return 0;
}

void henry(int a, int b, int x) {
	int localA = a;
	int localB = b;
	int localX = x;
	
	while (1) {
		if ((localA * localX - localB) == 0) {
			printf("%d\n", localX);
			break;
		}
		else {
			if ((localA * localX - localB) > 0) {
				localA = localA * localX - localB;
				localB = localB * localX;
				localX = localX + 1;
			}
			else { 
				localX = localX + 1;
			}
		}
	}
}
