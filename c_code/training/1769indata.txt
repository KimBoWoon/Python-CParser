#include <stdio.h>
#define MAX 20

int palindrome(int arr[], int first, int last);

int main()
{
	FILE *fp = fopen("input.txt", "r");
	int T = 0, number = 0;
	int tmp = 0, q = 2, i = 0;
	int nArr[MAX] = { 0, };

	fscanf(fp, "%d", &T);
	while (T--)
	{
		q = 2, i = 0;
		fscanf(fp, "%d", &number);
		tmp = number;
		while (q < 65)
		{
			while (tmp / q != 0)
			{
				nArr[i++] = tmp % q;
				tmp = tmp / q;
			}
			nArr[i++] = tmp % q;
			if (palindrome(nArr, 0, i - 1))
			{
				printf("1\n");
				q = 66; 
			}
			else
			{
				q++;
				i = 0;
				tmp = number;
			}
		}
		if (q == 65)
		{
			printf("0\n");
		}

	}
	fclose(fp);
	return 0;
}

int palindrome(int arr[], int first, int last)
{
	if (last <= first)
	{
		return 1;
	}
	else if (arr[first] != arr[last])
	{
		return 0;
	}
	else
	{
		return palindrome(arr, first + 1, last - 1);
	}
}
