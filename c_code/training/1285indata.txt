#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int main() {
	int prime_array[1000];
	int index = 0;
	
	for (int j = 2; j <= 1000; j++) {
		bool a = true;
		for (int i = 2; i * i <= j; i++) {
			if (!(j % i)) a = false;
		}
		if(a) prime_array[index++] = j;
	}

	ifstream file;
	file.open("input.txt");

	int cases;
	file >> cases;
	while (cases--) {
		int num;
		file >> num;

		int i, j;
		int x, y;
		for (i = 0; i < index; i++) {
			for (j = i; j < index; j++) {
				if (prime_array[i] + prime_array[j] == num) {
					x = i;	y = j;
					break;
				}
			}
		}

		cout << prime_array[x] << " " << prime_array[y] << endl;
	}

	return 0;
}
