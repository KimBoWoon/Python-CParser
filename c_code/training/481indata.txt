
#include <iostream>    
#include <fstream>
using namespace std;

int gcd(int n1,int n2)
{
	if(n2==0)
		return n1;
	else
		gcd(n2,n1%n2);
}

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >>testcase;

	for(int i=0;i<testcase;i++)
	{
		int num1,num2;
		instream>> num1>>num2 ;
		cout<<gcd(num1,num2)<<endl;
	}
}
