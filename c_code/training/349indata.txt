
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

ifstream inFile("input.txt");
class Map{
private :
	string* map;
	int row;
	int col;
	int* count;
	int c;
public : 
	Map(){
		inFile >> col >> row;
		map = new string[row];
		for (int i = 0; i < row; i++){
			inFile >> map[i];
		}
		count = new int[row*col/2];
		c = 0;
		for (int i = 0; i < row*col/2; i++){
			count[i] = 0;
		}
	}
	void checkMap(){
		for (int y = 1; y < row; y++){
			for (int x = 1; x < col; x++){
				if (map[y][x] == '.'){
					checkNode(x, y);
					for (int i = 0; i < c; i++){
						if (count[i] < count[c]){
							int temp = count[c];
							for (int j = c; j > i; j--){
								count[j] = count[j - 1];
							}
							count[i] = temp;
							break;
						}
					}
					c++;
				}
			}
		}
		
	}
	void printMap(){
		cout << c << endl;
		for (int i = 0; i < c; i++){
			cout << count[i] << " ";
		}
		cout << endl;
	}
	void checkNode(int x, int y){
		if (map[y][x] == '+')
			return;
		if (map[y][x] == '.'){
			count[c]++;
			map[y][x] = '+';
		}
		if (x > 1 && map[y][x-1] == '.'){
			checkNode(x - 1, y);
		}
		if (y > 1 && map[y-1][x] == '.'){
			checkNode(x, y - 1);
		}
		if (x < col - 1 && map[y][x + 1] == '.'){
			checkNode(x + 1, y);
		}
		if (y < row - 1 && map[y+1][x] == '.'){
			checkNode(x, y + 1);
		}
	}
};
int main(){
	int cases;
	inFile >> cases;
	while (cases--){
		Map map;
		map.checkMap();
		map.printMap();
	}
	return 1;
}
