#include <stdio.h>
#include <stdlib.h>

int M_function(int n){

	if(n>100)		return n-10;
	else if(n<=100)	return M_function(M_function(n+11));
}

void main(){

	FILE *in_file;
	int i;
	int num_cases;

	in_file=fopen("input.txt","r");
	if(in_file==NULL) exit(1);

	fscanf(in_file,"%d",&num_cases);

	for(i=0; i<num_cases; i++){

		int value;
		int k;

		fscanf(in_file,"%d",&value);

		k=M_function(value);

		printf("%d\n", k);
	}
	fclose(in_file);
}
