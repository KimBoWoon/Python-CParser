#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int check(char* string);

void main()
{
	FILE *in_file;
	char str[52];
	int data = 0;
	int i;
	int* flags;

	in_file = fopen("input.txt", "r");
	if(in_file == NULL)
		exit(1);

	fscanf(in_file, "%d\n", &data);

	flags = (int*)malloc(data * sizeof(int));

	for (i = 0; i < data; i++)
	{
		fgets(str, 52, in_file);
		if(check(str))
			printf("NO\n");
		else
			printf("YES\n");
	}

	free(flags);
	fclose(in_file);
}

int check(char* string)
{
	int i;
	int count = 0;
	int length = strlen(string) + 1;

	for(i = 0; i < length; i++)
	{
		if(string[i] == '(')
			count++;
		else if(string[i] == ')')
		{
			if(count == 0)
				return 1;
			count--;
		}
	}
	return count;
}
