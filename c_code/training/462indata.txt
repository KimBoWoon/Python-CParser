

#include<iostream>
#include<fstream>

using namespace std;

int gcd(int i, int j);

int main(void)
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {
		int num1, num2;
		inStream >> num1 >> num2;
		cout << gcd(num1, num2) << endl;
	}
	return 0;
}

int gcd(int i, int j){
	if (j == 0) return i;
	else gcd(j, i%j);
}
