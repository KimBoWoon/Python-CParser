

#include <stdio.h>

int mC(int n);

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int inputNum = 0; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d", &inputNum); 

		printf("%d\n", mC(inputNum));

	}

	fclose(f);
	return 0;
}

int mC(int n) {
	if (n > 100) 
		return n - 10;
	else 
	{
		return mC(mC(n+11));
	}
}
