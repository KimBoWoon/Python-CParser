

#include<iostream>
#include<fstream>
using namespace std;

void henry(int a, int b);
int gcd(int a, int b);

int main()
{
	int testCase = 0;
	ifstream file;
	file.open("input.txt");
	file >> testCase;

	int a = 0, b = 0;
	for (int i = 0; i < testCase; i++)
	{
		file >> a >> b; 
		henry(a, b);
	}
	return 0;
}

void henry(int a, int b)
{
	int i = 2;
	double totalHenN = 0; 
	int temp;

	while (1)
	{
		if ((a*i) >= b) 
		{
			if (a == 1) 
			{
				cout << i << endl;
				return;
			}
			a = (a * i) - b;
			b = (b * i);
			temp = gcd(a, b);
			a /= temp;
			b /= temp;
		}

		i++;
	}

}

int gcd(int a, int b)
{
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}
