#include<iostream>
#include<fstream>
using namespace std;
int GCD(int a, int b)
{
	int r;
	do{
		r = a % b;
		a = b;
		b = r;
	} while ( r != 0 );
	
	return a;
}
int main(){
	ifstream inStream;
	int numCases;
	int origin_son;		int origin_mom;
	double sum = 0;
	inStream.open("input.txt");

		inStream >> origin_son >> origin_mom;
	
		int x1[10000] = {0};
		int j=0;
		int result = 0;
		int son = origin_son;
		int mom = origin_mom;

		while(1){
			if(mom % son != 0)
				x1[j] = mom/son + 1;
			else
				x1[j] = mom/son;
		
			int ss = GCD(son,mom);
			son = x1[j] * son - mom;
			if(son == 0){
				break;
			}
			mom = mom * x1[j];

			int gcd = GCD(son,mom);
			son = son/gcd;
			mom = mom/gcd;

			sum = sum + 1/x1[j];
		

					j++;
		}

	}
	inStream.close();
}
