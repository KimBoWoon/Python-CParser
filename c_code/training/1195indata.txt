 

 

#include <stdio.h>
#include <math.h>
int isPrime(int n);

int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test); 
    while(test--) 
    {
        int n,i,j;
        fscanf(fp,"%d",&n);
        for(i=n/2;i>=2;i--) 
        {
            if(isPrime(i) == 1)
                break;
        }
        while(!isPrime(n-i) && 2*i != n) 
        {
            for(j=i-1;j>=2;j--) 
            {
                if(isPrime(j) == 1)
                    break;
            }
            i = j;
        }
        printf("%d %d\n",i,n-i);
    }
    fclose(fp); 
    return 0;
}
int isPrime(int n) 
{
    int flag=0;
    if(n > 2 && n%2 == 0)
        return 0;
    else
    {
        int i;
        for(i=2;i<=sqrt(n);i++)
        {
            if(n%i == 0)
            {
                flag=1;
                break;
            }
        }
        if(flag == 0) return 1;
        else return 0;
    }
}
