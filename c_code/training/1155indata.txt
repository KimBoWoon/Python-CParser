

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if (instream.fail())
	{
		cerr << "Error" << endl;
		exit(1);
	}

	int no_case;
	instream >> no_case;

	int Eureka[51] = { 0 };					

	int resArr[1010] = { 0 };				

	for (int i = 0; i<50; i++)				
	{
		Eureka[i] = (i * (i + 1)) / 2;
	}

	for (int i = 1; i<50; i++)						
		for (int j = 1; j<50; j++)
			for (int k = 1; k<50; k++)
				if ((Eureka[i] + Eureka[j] + Eureka[k]) <= 1000)
					resArr[Eureka[i] + Eureka[j] + Eureka[k]] = 1;

	for (int i = 0; i<no_case; i++)
	{
		int num;
		instream >> num;

		cout << resArr[num] << endl;				
	}
	return 0;
}
