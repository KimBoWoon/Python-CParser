

#include <iostream>
#include <string>
#include <fstream>
using namespace std;
int palindromeRecursive(char *word, int start, int end);

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases = 0;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++)
	{
		char word[1001] = { 0, };
		inFile >> word;
		int end = 0;
		while (word[end] != 0){
			end++;
		}
		cout << palindromeRecursive(word, 0, end - 1) << endl;
	}
}
int palindromeRecursive(char *word, int start, int end){
	if (end <= start) return 1;
	else if (word[start] != word[end]) return 0;
	else return palindromeRecursive(word, start + 1, end - 1);
}
