


#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	fstream in;
	in.open("input.txt");
	int numcases;
	in>>numcases;
	for(int i=0; i<numcases; i++)
	{
		char str[51];
		in>>str;
		int count=0;
		int errorCount=0;
		if(str[0]==')') {cout<<"NO"<<endl; continue;}
		int j=0;
		while(str[j]=='(' || str[j]==')')
		{
			if(str[j]=='(') count++;
			else if(str[j]==')'){
				if(count>0) {count--;}
				else {errorCount++;}
			}
			
			j++;
		}
		if(count==0 && errorCount==0){cout<<"YES"<<endl;}
		else {cout<<"NO"<<endl;}
			
		
	}
}
