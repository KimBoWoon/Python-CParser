



#include<stdio.h>
#include<stdlib.h>
#include<vector>

using namespace std;

#define pb push_back
#define sz(v) ((int)(v).size())


int main()
{
   FILE * inFile;
   int N;
   int testCases;
   int n, m;


   inFile = fopen("input.txt", "r");

   fscanf(inFile, "%d", &testCases);

   for (n = 0; n < testCases; n++)
   {
	   fscanf(inFile, "%d", &N);
	   bool ans = 0;

	   for(int m =2; m<65; m++)
	   {
		   vector <int> arr;
		   for (int v=N; v; v/=m) arr.pb(v%m);
		   int i = sz(arr);
		   bool sw = 0;
		   for (int j=0; j<i; j++)
			   if (arr[j] != arr[i-j-1])
				   sw = 1;
		   if(!sw)
		   {
			   ans = 1;
			   break;
		   }
	   }
	   printf("%d\n", (int)ans);
   }
}
