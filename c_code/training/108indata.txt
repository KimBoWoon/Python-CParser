 

 
#include <stdio.h>

int mccarthy(int value);
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test); 
    while(test--) 
    {
        int value;
        fscanf(fp,"%d",&value);
        printf("%d\n",mccarthy(value)); 
    }
    fclose(fp); 
    return 0;
}
int mccarthy(int value) 
{
    if(value > 100) 
    {
        return value-10;
    }
    else 
    {
        return mccarthy(mccarthy(value+11));
    }
}
