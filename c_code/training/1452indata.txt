
#include <iostream>
#include <fstream>
using namespace std;

int main ()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일을 열 수 없습니다." <<endl;
			return 0;
	}
	int case_number;
	instream >> case_number;
	for(int i=0; i<case_number; i++)
	{
		int H,W,N; 
		instream >> H >> W >> N;
		int roomH,roomW; 
		if(N%H==0)
		{
			roomH = H;
			roomW = N/H;
		}
		else
		{
			roomH = N%H;
			roomW = (N/H)+1;
		}
		if(roomW <10)
			cout << roomH << "0" << roomW << endl; 
		else
			cout << roomH << roomW << endl; 

	}
	return 0;

}
