

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int primenum(int num)
{
	int pnum,result=-1; 
	pnum=num-1;
	while(1) 
	{
		if (pnum==1) 
		{
				result=1;
				return 1;
		}
			else if (num%pnum==0) 
			{
				result=0;
				return 0;
			}
			pnum--; 
	}
}
int goldbach1(int num1,int num2)
{
	int g1;
	if(primenum(num1)&&primenum(num2)){
		g1=num1;
		return g1;}
	else{
		goldbach1(num1-1,num2+1);
	}
}
int goldbach2(int num1,int num2)
{
	int g1;
	if(primenum(num1)&&primenum(num2)){
		g1=num1;
		return g1;}
	else{
		goldbach2(num1+1,num2-1);
	}
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int i, num,num1,num2;
		int g1,g2;
		fscanf(file,"%d", &num); 
		num1=num/2;num2=num1;
		g1=goldbach1(num1,num2);
		g2=goldbach2(num1,num2);
		if(abs(g1-(num-g1))<=abs(g2-(num-g2)))
		{
			printf("%d %d\n",g1,num-g1);
		}
		else
			printf("%d %d\n",g2,num-g2);
	}
	fclose(file); 
}
