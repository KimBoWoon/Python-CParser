#include <stdio.h>
#include <stdlib.h>



int gcd(int a, int b) {
	return (b != 0) ? gcd(b, a % b) : a;
}

void main(void) {
	FILE *in_file;
	int i, no_cases;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i<no_cases; i++) {
		int a,b ;

		fscanf(in_file, "%d %d", &a, &b);

		printf("%ld\n", gcd(a, b));
	}
	fclose(in_file);
}
