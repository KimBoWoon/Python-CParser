

#include <iostream>
#include <cstdlib>
#include <fstream>

int assignRoom(int hotel_height, int hotel_width, int xth);

int main()
{
	std::fstream inputfile; 
	int testcase; 
	int hotel_height, hotel_width, xth;

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> hotel_height>>hotel_width>>xth;
		std::cout << assignRoom(hotel_height,hotel_width,xth) << std::endl;
	}

	return 0;
}

int assignRoom(int hotel_height, int hotel_width, int xth)
{
	int roomnumber;
	int roomheight;

	roomnumber = xth / hotel_height + 1;
	roomheight = (xth % hotel_height) * 100;

	if (roomheight == 0)
	{
		roomheight = hotel_height * 100;
		roomnumber = roomnumber - 1;
	}
	return (roomheight + roomnumber);
}
