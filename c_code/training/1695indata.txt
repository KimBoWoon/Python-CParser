

#include <stdio.h>
#define MAX 100 
long int arr[MAX]; 
int idx; 
void baseConversion(long int x, long int base) 
{
	if (x >= base)
		baseConversion(x / base, base);
	arr[idx++] = x%base; 
	
}
int checkPalindrome(int first, int last) 
{
	if (last <= first) 
		return 1;
	else if (arr[first] != arr[last]) 
		return 0;
	else
		return checkPalindrome(first + 1, last - 1); 
}

int main(int argc, char *argv[])
{
	int i,t, flg = 0;
	long int x, base;
	FILE *fp; 
	fp = fopen("input.txt", "r"); 
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%ld", &x); 
		for (base = 2; base <= 64; base++) 
		{
			baseConversion(x, base); 
			if (checkPalindrome(0, idx - 1)) { 
				printf("1\n"); 
				flg = 1; 
				break;
			}
			else 
			{
				for (i = 0; i < idx; i++) 
					arr[i] = 0;
				idx = 0; flg = 0;
			}
		}

		if (flg == 0 && base == 65) printf("0\n"); 
		
		for (i = 0; i < idx; i++)  
			arr[i] = 0;
		idx = 0; flg = 0;
	}

	fclose(fp); 
	return 0;
}
