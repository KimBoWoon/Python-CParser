 

 
#include <stdio.h>
#define sum(n)((n)*(n+1))/2 


int main()
{
    int test;
    FILE *fp;
    
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    
    while(test--) 
    {
        int value,i,j,k,result=0;
        fscanf(fp,"%d",&value); 
        for(i=1;i<45;i++)
        {
            for(j=1;j<45;j++)
            {
                for(k=1;k<45;k++)
                {
                    if(sum(i)+sum(j)+sum(k) == value)
                    {
                        result = 1; 
                        break;
                    }
                }
                if(result != 0) break;
            }
            if(result != 0) break;
        }
        if(result == 1) printf("1\n");
        else printf("0\n");
        result = 0;
    }
    fclose(fp); 
    return 0;
}
