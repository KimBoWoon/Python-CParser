
#include <stdio.h>
#include <stdlib.h>

#define MAX 500

int arr[MAX], min[MAX][MAX], cost[MAX][MAX];

void getCost(int size) {
   int i, j, k, d;
   int sum, nowCost, tmp1, tmp2;

   
   for (i = 0; i<size; i++)
      min[i][i] = arr[i];

   for (d = 1; d<size; d++)      
      for (i = 0; i<size - d; i++) {
         j = i + d;
         tmp1 = tmp2 = 2e9;
         for (k = i; k<j; k++) {
            sum = min[i][k] + min[k + 1][j];
            nowCost = sum + cost[i][k] + cost[k + 1][j];

            if (tmp2 > nowCost)
            {
               tmp1 = sum;
               tmp2 = nowCost;
            }
         }
         min[i][j] = tmp1;
         cost[i][j] = tmp2;
      }
}

void mergeFile(FILE *fin) {
   int i, size;

   fscanf(fin, "%d", &size);

   for (i = 0; i<size; i++)
      fscanf(fin, "%d", &arr[i]);

   getCost(size);

   printf("%d\n", cost[0][size - 1]);
}

int main() {
   int testCases, cases;
   FILE* fin = fopen("input.txt", "r");    

   fscanf(fin, "%d", &testCases);      

   for (cases = 0; cases<testCases; cases++)
      mergeFile(fin);        

   fclose(fin);            

   return 0;           
}
