void additionFraction(int a, int b, int c, int d) {
	int temp = b;
	int resultA = 0;
	int resultB = 0;

	if (b != d) {
		b = b*d;
		a = a*d;
		d = b;
		c = c*temp;
	}
	resultA = a + c;
	resultB = b;
	temp = gcd(resultA, resultB);
	resultA /= temp;
	resultB /= temp;

	printf("%d %d\n", resultA, resultB);
}

int gcd(int a, int b) {
	int temp = a;
	if (a < b) {
		a = b;
		b = temp;
	}
	if (b == 0)
		return a;
	else if (b > 0)
		return gcd(b, a % b);

}
