


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b) {
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main(void)
{
   ifstream inStream;
   int cases;
   int bj1, bm1, bj2, bm2;
   int bm3, bj3;

   inStream.open("input.txt");
   if(inStream.fail())
   {
      cerr<<"Input file opening failed./n";
      exit(1);
   }

   inStream>>cases;

   for(int i=0; i<cases; i++)
   {
      inStream>> bj1 >> bm1 >> bj2 >> bm2;

	  bm3 = bm1*bm2;
	  bj3= bj1*bm2 + bj2*bm1;

	  cout << bj3/gcd(bj3,bm3) << " " << bm3/gcd(bj3, bm3) << endl;
   }
}
