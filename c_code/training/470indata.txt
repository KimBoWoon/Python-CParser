




#include<iostream>
#include<fstream>

int gcd(int a, int b);

int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){

		int a, b;

		is>>a>>b;

		std::cout<<gcd(a,b)<<std::endl;

	}
}

int gcd(int a, int b){

	if(b == 0)  
		return a;

	else
	 return gcd(b,a%b);


}
