
#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;

void swap(char *a, char*b){
	char temp;
	temp = *a;
	*a = *b;
	*b = temp;
}

void sort(char *str,int start,int end){
	for (int i = start; i < end; i++){
		for (int j = start; j < end; j++){
			if (str[i]<str[j]) swap(str[i], str[j]);
		}
	}
}

int factorial(int t){
	if (t <= 1) return 1;
	return factorial(t - 1) * t;
}

void Permutation(int m,int k){
	char str[12] = {};
	for (int i = 0; i < m; i++) str[i] = 'a' + i;
	for (int i = 0; i < m; i++){
		int t = k / factorial(m - i-1);
		if (k%factorial(m - i-1) == 0) t--;
		swap(str[i],str[i+t]);
		k -= (t*factorial(m - i-1));
		sort(str, i+1, m);
	}
	cout << str << endl;
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;
	int n;
	in >> n;
	for (int i = 0; i < n; i++){
		int m, k; 
		in >> m >> k;
		Permutation(m, k);
	}

	in.close();
	return 0;
}
