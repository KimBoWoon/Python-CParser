#include <iostream>
#include <fstream>
using namespace std;

int fibo(int n);

int main()
{
	ifstream infile;

	infile.open("input.txt");
	
	int numTestCases;
	int result = 0;

	infile >> numTestCases;

	while(numTestCases--)
	{
		infile >> result;
		
		cout<< fibo(result) << endl;
	}

	
	return 0;
}
int fibo(int n)
{
	if(n == 0)
		return 0;
	if(n == 1)
		return 1;
	
	return fibo(n-2) + fibo(n-1);
}
