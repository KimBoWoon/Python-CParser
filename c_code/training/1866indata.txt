#include <iostream>
#include <fstream>
#include <cstdlib>
#include<string>

using namespace std;

void par(char *A, int lp, int rp, int n);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){

		char A[55];
		int len;
	
		inStream >> A;



		par(A, 0, 0, 0);

	

	}

	inStream.close();
	return 0;
}

void par(char *A, int lp, int rp, int n){
	
	if (A[n]!='('&&A[n]!=')'){
		if (lp == rp){
			cout << "YES" << endl;
		}

		else
			cout << "NO" << endl;

	}
	else{
		if (A[n] == '(')
			++lp;

		else if (A[n] == ')'){
			++rp;
		}

		if (rp - lp > 0)
			cout << "NO" << endl;



		else
			par(A, lp, rp, n + 1);
	}
}
