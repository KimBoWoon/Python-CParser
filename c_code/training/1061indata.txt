


#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void Ducci(int *arr,int num)
{
	int **set,j,k,cnt=0;
	int check=0;
	int zero=0;
	set=(int**)malloc(sizeof(int*)*1000);
	for(j=0;j<1000;j++)
			set[j] = (int*)malloc(sizeof(int)*(num));
	
	for(j=0;j<num;j++)
		set[cnt][j] = arr[j];

	while(1)
	{    
		for(k=0;k<num;k++)
		{
			if(set[cnt][k]==0)
				zero++;
		}
		if(zero==num){
			printf("ZERO\n");return;}
		zero=0;

		for(j=0;j<cnt;j++) 
		{
			for(k=0;k<num;k++)
			{
				if(set[j][k]==set[cnt][k])
					check++;
			}	
			if(check==num){
				printf("LOOP\n");return;}
			check=0;
		}

		for(j=0;j<num;j++) 
		{
			if(j==(num-1)){
				set[cnt+1][num-1]=abs(set[cnt][num-1]-set[cnt][0]);
			}
			else{
			set[cnt+1][j]=abs(set[cnt][j]-set[cnt][j+1]);
			}
		}
		cnt++; 
	}
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int j, num;
		int *arr;
		fscanf(file,"%d", &num); 

		arr=(int*)malloc(sizeof(int)*num);
		for(j=0;j<num;j++)
		{
			fscanf(file,"%d",&arr[j]);
		}
		Ducci(arr,num);
	}
	fclose(file); 
}
