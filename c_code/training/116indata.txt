
#include<iostream>
#include<string>
#include<fstream>

using namespace std;

int mcCarthy(int num){
	if (num > 100){
		return num-10;
	}
	else{
		return mcCarthy(mcCarthy(num+11));
	}
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, numBuff = 0;

	f >> numCount;

	for (int i = 0;i<numCount;i++){
		f >> numBuff;
		cout << mcCarthy(numBuff) << endl;
	}
	return 0;
}
