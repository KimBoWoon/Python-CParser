#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

void sosu(int i);

int main()
{
	ifstream infile;

	int nocases;
	int num;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i=0; i<nocases; i++)
	{
		infile >> num;

		sosu(num);
	}

	return 0;
}

void sosu(int i)
{
	int num=1;

	for(int j=2; j*j <= i; j++)
	{
		if(i%j == 0)
		{
			num = 0;
		}
	}

	cout << num << endl;
}
