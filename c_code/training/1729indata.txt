#include <stdio.h>
#define MAX 1001

char s[30];
int CheckPalindrome(int start, int last);
void play(int num, int base, int i);

void main() {
	int numTestCases = 0;
	int input;
	int i = 0, j, k;
	int length;
	FILE *f;
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d", &input);
		for (j = 2; j <= 64; j++) {
			length = 0;
			for (k = 0; k < 30; k++)
				s[k] = -1;
			play(input, j, 0);
			for (k = 0; k < 30; k++) {
				if (s[k] == -1)
					break;
				length++;
			}
			if (CheckPalindrome(0, length-1)) {
				printf("1\n");
				break;
			}
			if (j == 64)
				printf("0\n");
		}
	}
}

int CheckPalindrome(int start, int last) {
	if (start >= last)
		return 1;
	else if (s[start] != s[last])
		return 0;
	else
		return CheckPalindrome(start + 1, last - 1);
}

void play(int num, int base, int i) {
	if (num >= base)
		play(num / base, base, i + 1);
	
	s[i] = num%base + '0';
}
