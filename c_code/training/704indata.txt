
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int pelin(string str, int start){
	int len = str.length();
	if (start != len / 2){
		if (str[start] == str[len - 1 - start])
			return pelin(str, start + 1);
		else
			return 0;
	}
	else
		return 1;
}

int main(){
	ifstream File;
	int numCase = 0;
	string str;
	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> str;
		int ans = pelin(str, 0);
		cout << ans << endl;
	}
	File.close();
	return 0;
}
