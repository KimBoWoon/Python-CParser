


#include<stdio.h>
#include<stdlib.h>

void main(void){
	int no_case=0,h=0,w=0,n=0;
	int i,j;
	int yy=0,xx=0;
	double num=0;
	int yyxx=0;
	FILE *in_file;
	
	in_file = fopen("input.txt","r");  
	if(in_file == NULL) exit(1);

	fscanf(in_file, "%d" , &no_case);  

	for(i=0; i<no_case; i++)
	{
		fscanf(in_file, "%d %d %d",&h,&w,&n); 
		num=(double)n/(double)h; 
		if(n>h && num>1 && n%h!=0)	
			xx=(n/h)+1;
		else if( n<=h)			
			xx=1;
		else if(h==1)			
			xx=n;
		else if(n%h==0)			
			xx=n/h;
		
		if(n%h!=0)				
			yy=(n%h)*100;
		else
			yy=h*100;			
		yyxx = yy + xx;			
		printf("%d\n",yyxx);
	}
	
	fclose(in_file);
}
