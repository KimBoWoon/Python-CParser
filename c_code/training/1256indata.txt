


#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>
int prime(int a);
int Goldbach(int num1, int num2);
int main(void)
{
	FILE *fp;
	int i;
	int numcases;
	int num = 0; 
	int divi;

	fp = fopen("input.txt", "r"); 
	if (fp == NULL) exit(1);

	fscanf(fp, "%d", &numcases);

	for (i = 0; i < numcases; i++)
	{
		fscanf(fp, "%d", &num);
		divi = num/2;
		Goldbach(divi, divi);
	}
	fclose(fp); 
	return 0;
}

int prime(int a) 
{
	int j;

	for (j = 2; j <= a; j++)
	{
		if (a%j == 0)
		{
			if (a == j)    return 1;
			if (a != j)    return 0;
		}
	}
}

int Goldbach(int num1, int num2)
{
	if ((prime(num1) == 1) && (prime(num2) == 1))
	{
		printf("%d %d\n", num1, num2); 
		return;
	}
	else
		 Goldbach(num1 - 1, num2 + 1);  

}
