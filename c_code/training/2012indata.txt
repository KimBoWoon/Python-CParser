

#include<iostream>
#include<fstream>
#include<cmath>
#include<algorithm>
#include<functional>

using namespace std;


int main(void)
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {

		int cases, goal;
		inStream >> cases >> goal;

		int* num = new int[cases];
		int sum = 0;
		int big = 1000;
		bool zero = false;

		for (int j = 0; j < cases; j++){
			inStream >> num[j];
		}
		sort(num, num + cases, less<int>());

		int x = 0;
		int y = cases-1;
		int now = 200000001;
		int gNum;

		while (1){
			if (x == y) break;
			
			gNum = num[x] + num[y] - goal;
			
			if (gNum > 0){
				if (now > double(abs(gNum))){
					now = double(abs(gNum));
					sum = 1;
				}
				else if (now == double(abs(gNum))){
					sum++;
				}
				y--;
			}
			else if (gNum <= 0){
				if (now > double(abs(gNum))){
					now = double(abs(gNum));
					sum = 1;
				}
				else if (now == double(abs(gNum))){
					sum++;
				}
				x++;
			}
		}
		cout << sum << endl;
	}
}
