#include <iostream>
#include <fstream>

int fibo(int inputNum)
{
	if (inputNum <= 1)
		return inputNum;
	else
		return fibo(inputNum - 1) + fibo(inputNum - 2);
}

int main()
{
	std::ifstream file("input.txt");
	int inputNum = 0, testCase=0;
	file >> testCase;
	while (testCase--)
	{
		file >> inputNum;
		std::cout<<fibo(inputNum)<<std::endl;
	}
}
