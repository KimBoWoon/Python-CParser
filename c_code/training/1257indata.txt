


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

bool isPrime(int a);
void Goldbach(int a, int b, int num);

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if (instream.fail())
	{
		cerr << "Error" << endl;
		exit(1);
	}

	int no_case;
	instream >> no_case;

	while(no_case--)	
	{
		int num;
		instream>>num;

		int goldbach1 = num/2;
		int goldbach2 = num - goldbach1;
		
		Goldbach(goldbach1, goldbach2, num);
	}


}
bool isPrime(int a)
{
	int res = 1;
	for(int i=2 ; i<a ; i++)
	{
		if(a % i ==0)
			return 0;
	}
	return 1;
}

void Goldbach(int a, int b, int num)
{
	if(isPrime(a) && isPrime(b))	
		cout<<a<<" "<<b<<endl;
	else
	{
		a--;
		b = num -a;
		Goldbach(a, b, num);
	}

}
