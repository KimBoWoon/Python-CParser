

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int gcd(int a, int b);

int main(void) {
	FILE *input;
	int no_cases;
	int i;
	int a, b;

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d%d", &a, &b);
		printf("%d\n", gcd(a, b));
	}

	fclose(input);
	return 0;
}

int gcd(int a, int b) {
	int temp;
	if (a < b) {
		temp = a;
		a = b;
		b = temp;
	}

	if (b == 0) {
		return a;
	}
	else return gcd(b, a % b);
}
