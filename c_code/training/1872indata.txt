#include<iostream>
#include<fstream>
#include<cstring>
using namespace std;

int left_cnt = 0;
int right_cnt = 0;

void parenthesis(char* arr, int length, int n){
   char left = '(';
   char right = ')';

   if (arr[n] == right)   right_cnt++;
   else if (arr[n] == left)   left_cnt++;

   if (n == length-1){ 
      if (right_cnt - left_cnt == 0) 
         cout << "YES" << endl;

      else
         cout << "NO" << endl;
   }

   else if (left_cnt == 0 && right_cnt > 0) 
      cout << "NO" << endl;

   else if (arr[n] == left){ 
      parenthesis(arr, length, ++n);
   }

   else if (arr[n] == right){ 
      left_cnt--;
      right_cnt--; 
      parenthesis(arr, length, ++n);
   }
}

int main(){
   ifstream inStream;
   int numCases;
   inStream.open("input.txt");

   inStream >> numCases;
   for (int i = 0; i<numCases; i++){
      char symbol[50] = { 0 };

      inStream >> symbol;

      parenthesis(symbol, strlen(symbol), 0);

      left_cnt = 0;
      right_cnt = 0;
   }
   inStream.close();
}
