
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

int factorial(int x){
if(x==0)
	return 1;
else if(x>0){
	return x*factorial(x-1);
}
}
int main(void){
	ifstream infile;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error"<<endl;
		exit(1);
	}
	int num;
	int x;
	infile >> num;
	for(int i=0;i<num;i++){
		infile >> x;
		cout<<factorial(x)<<endl;
	}

	return 0;
}
