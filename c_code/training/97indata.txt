#include <stdio.h>
#include <stdlib.h>

int main(void)
{
	FILE *file;
	int cases,data;
	int divisor,isPrime;

	file=fopen("input.txt","r");

	fscanf(file,"%d",&cases);
	while(cases--){
		fscanf(file,"%d",&data);
		isPrime=1;

		for(divisor=2;divisor*divisor<=data;divisor++){
			if(data%divisor==0){ 
				isPrime=0;
				break;
			}	
		}
		printf("%d\n",isPrime);
	}
	fclose(file);
}
