


#include <stdio.h>
#include <math.h>

void detectPrime(int data){
	int i,cnt=0;
	data = (double)data;
	for(i=2; i<=sqrt(data); i++){
		if(data%i==0)
			cnt++;
	}
	if(cnt==0)
		printf("1\n");
	else
		printf("0\n");
}

int main(){
	int cases,data,i;

	FILE *infile;
	infile = fopen("input.txt","r");
	fscanf(infile, "%d", &cases);
	
	for(i=0; i<cases; i++){
		fscanf(infile, "%d", &data);
		detectPrime(data);
	}

	return 0;
}
