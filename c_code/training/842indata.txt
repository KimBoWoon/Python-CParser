

#include <iostream>
#include <fstream>
using namespace std;
int fibo(int n){
	if (n == 0) return 0;
	if (n < 0) return 1;
	else return fibo(n - 1) + fibo(n-2);

}
int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;
	int value;
	for (int i = 0; i < testcases; i++){
		in >> value;
		cout << fibo(value) << endl;
	}
}
