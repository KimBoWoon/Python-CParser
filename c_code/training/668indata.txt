
#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int fac(int n)
{
	if (n <= 1) return 1;
	else return n*fac(n - 1);
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while(t--)
	{
		int n;
		in >> n;
		cout<<fac(n)<<endl;
	}
	in.close();
}
