void doDucci(int arr1[15], int arr2[15], int count);
void calcNext(int arr1[15], int arr2[15], int count);
int checkZeroSum(int arr[15], int count);

int arrFlag, sum;

int main(void) {

	ifstream dataFile("input.txt");

	int caseNum;
	dataFile >> caseNum;	
	int numNum;
	int arrOri[15] = { 0 };
	int arrAsi[15] = { 0 };
	int i, j;
	for (i = 0; i < caseNum; i++) {

		dataFile >> numNum;

		for (j = 0; j < numNum; j++) {
			dataFile >> arrOri[j];
		}
		
		arrFlag = 0;
		sum = 0;
		doDucci(arrOri, arrAsi, j);		
	}


	return 0;
}

void doDucci(int arr1[15], int arr2[15], int count) {

	for (int c = 0; c < 1000; c++) {		
		if (arrFlag == 0) {
			calcNext(arr1, arr2, count);
			if (checkZeroSum(arr2, count) == 1) {				
				break;
			}
			arrFlag = 1;
		}
		else {
			calcNext(arr2, arr1, count);
			if (checkZeroSum(arr1, count) == 1) {				
				break;
			}
			arrFlag = 0;
		}
	}
	if (checkZeroSum(arr1, count) == 1 || checkZeroSum(arr2, count) == 1) {
		cout << "ZERO" << endl;
	}
	else {
		cout << "LOOP" << endl;
	}

	return;

}

void calcNext(int arr1[15], int arr2[15], int count) {
	for (int k = 0; k < count-1; k++) {
		arr2[k] = abs(arr1[k] - arr1[k + 1]);		
	}
	arr2[count-1] = abs(arr1[count-1] - arr1[0]);
}

int checkZeroSum(int arr[15], int count) {
	sum = 0;
	for (int z = 0; z < count; z++) {
		sum += arr[z];
	}
	if (sum == 0) {
		return 1;
	}
	else {
		return 0;
	}
}
