
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int gcd(int a, int b)
{
	if (b == 0) return a;
	return gcd(b, a%b);
}


void main(){
	FILE *in_file;
	int no_cases;
	int i;
	int a, b, c ,d;
	int g;
	int n, m;

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d %d %d", &a, &b, &c,&d);
		
		g = gcd(a, b);
		a = a / g;
		b = b / g;

		g = gcd(c, d);
		c = c / g;
		d = d / g;

		
		m = b * d;
		n = (a*d) + (c*b);

		printf("%d %d\n", n/gcd(n,m), m/gcd(n,m));
	
		
	}

	fclose(in_file);
}
