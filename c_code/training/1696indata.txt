


#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int nToB(char *_str, int _data, int _b)
{
	int i = 0;

	while(1)
	{
		_str[i++] = _data % _b;
		_data = _data / _b;
		if( _data == 0)
		{
			break;
		}
	}
	return i;
}

char isPalindrome(char *_str, int length)
{
	int head = 0;
	int tail = length - 1;

	while(1)
	{
		if(head >= tail)
		{
			return 1;
		}

		if(0 != (_str[head] - _str[tail]))
		{
			return 0;
		}
		else
		{
			head = head + 1;
			tail = tail - 1;
		}
	}
	return 0;
}

int main(void)
{
	FILE *fp = NULL;
	int testCase;

	fp = fopen("input.txt", "r");
	fscanf(fp, "%d", &testCase);
	while(testCase--)
	{
		int i, inputData, length, result = 0;
		char str[21] = {0, };

		fscanf(fp, "%d", &inputData);

		for(i=64; i>=2; i--)
		{
			length = nToB(str, inputData, i);
			if(1 == isPalindrome(str, length))
			{
				result = 1;
				break;
			}
		}
		printf("%d\n", result);
	}

	return 0;
}
