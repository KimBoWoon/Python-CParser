#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int is_Palindrome(char str[],int first,int last);

int main()
{
	FILE *fp;
	int i,numTestCases;
	fp=fopen("input.txt","r");
	if(fp==NULL)exit(1);

	fscanf(fp,"%d",&numTestCases);
	for(i=0; i<numTestCases; i++){
		char letter[1000];
		fscanf(fp,"%s",letter);

		printf("%d\n",is_Palindrome(letter,0,strlen(letter)-1));
	}

	return 0;
	fclose(fp);
}

int is_Palindrome(char str[],int first,int last)
{
	if(last<=first)
		return 1;
	else if(str[first]!=str[last])
		return 0;
	else
		return is_Palindrome(str,first+1,last-1);
}
