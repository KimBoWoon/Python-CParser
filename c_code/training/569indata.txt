

#include <iostream>
#include <fstream>

using namespace std;

int factorial(int n)
{
	if (n <= 1)
		return 1;
	else
		return n*factorial(n - 1);
}

int main()
{
	ifstream infile;
	int testCase;
	int n, k;
	int strnum=1;
	int numcount = 0;
	int numOfCase=0, kcount=0;
	char str[13];
	infile.open("input.txt");
	
	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> n >> k;
		int kth = k; 
		int j = 0;
		numcount = 0;
		kcount = 0; 
		strnum = 1; 
		for (int a = 0; a < 13; a++) 
		{
			if (a == 0) str[a] = '*';
			str[a] = 96 + a; 
		}

		while (numcount < n)
		{
			while (1)
			{
				numOfCase = factorial(n - j -1);
				kcount += numOfCase;
				if (kth <= kcount)
				{
					while (str[strnum] == '*')
					{
						strnum++; 
						if (strnum > n)
							strnum = 1;
					}
					cout << str[strnum];
					str[strnum] = '*';
					numcount++;
					j++;

					strnum = 1;

					while (str[strnum] == '*')
						strnum++;
					kth -= (kcount - numOfCase);
					
					kcount = 0;
					break;
				}
				else
				{
					strnum++;
					if (str[strnum] == '*')
					{
						strnum++;
						while (str[strnum] == '*')
						{
							strnum++;
							if (strnum > n)
								strnum = 1;
						}
					}
				}

			} 
			
		} 
		cout << endl;
		
	} 


	infile.close();

	return 0;
}
