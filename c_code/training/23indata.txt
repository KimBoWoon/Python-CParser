#include <stdio.h>
#include <stdlib.h>

int primeNumber(int Prime);

int main()
{
	FILE *in_file;
	int i;
	int no_cases;
	int number;

		in_file=fopen("input.txt","r");
		if(in_file==NULL) exit(1);

		fscanf(in_file, "%d", &no_cases);

		for(i=0; i<no_cases; i++){
			fscanf(in_file, "%d",&number);
			
			printf("%d\n",primeNumber(number));
		}
		return 0;

		fclose(in_file);
}

int primeNumber(int Prime)
{
	int i;
	if(Prime==1)
		return 1;
	for(i=2; i<(Prime); i++){
		if(Prime%i==0)
			return 0;
	}
	return 1;
}
