
#include<iostream>
#include<fstream>

using namespace std;
bool primeCheck(int num); 

int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;
	while (cases--){
		int num;
		inFile >> num;
		cout<< primeCheck(num)<<endl;
	}
	return 0;
}

bool primeCheck(int num){
	for (int i = 2; i <= num/2 ; i++){
		if (num%i){
			continue;
		}
		return false;
	}
	return true;
}
