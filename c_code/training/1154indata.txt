#include<stdio.h>
#include<stdlib.h>

int main(void)
{
	FILE *in_file;
	int a, b, c;
	int T,i,j;
	int no_cases,no_data;
	
	T = 44;


	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   

	i = 0;
	while( i< no_cases)
	{
	
		fscanf(in_file, "%d", &no_data);

		for (a = 1; a <= T; a++)
		{
			for (b = 1; b <= T; b++)
			{
				for (c = 1; c <=T; c++)
				{
					if ((((a*(a + 1)) + (b*(b + 1)) + (c*(c + 1))) / 2) == no_data)
					{
						printf("1\n");
						goto one;
				
					}

				}

			}
		}
		printf("0\n");
		i++;
		continue;
		one:
		i++;
	

	}
	fclose(in_file);

	return 0;
}
