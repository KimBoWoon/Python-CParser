
#include <iostream>
#include <fstream>

int n[1000];
int f_n[167];

void getPrimeNumber(int num){
	int a = 0;
	for (int i = 2; i <= num; i++) {
		n[i] = i;
	}

	for (int i = 2; i <= num; i++) {
		if (n[i] == 0) continue;
		else {
			f_n[a] = i;
			a++;
		}
		for (int j = i + i; j <= num; j += i) n[j] = 0;
	}
}

void Goldbach(int num){
	int a = num / 2;
	int i = 0,j = 0;
	
	while (a > f_n[i]) i++;
	j = i;
	
	while (f_n[i] + f_n[j] != num){
		if (f_n[i] + f_n[j] > num){
			j--;
		}
		else if (f_n[i] + f_n[j] < num){
			i++;
		}
	}

	std::cout << f_n[j] << " " << f_n[i] << std::endl;
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num;
	if (fin.is_open() == false)  return -1;
	
	getPrimeNumber(1000);

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {
		fin >> num;
		if (num == 4) std::cout << 2 << " " << 2 << std::endl;
		else Goldbach(num);
	}
	fin.close();
}
