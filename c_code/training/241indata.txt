

#include <iostream>
#include <fstream>


using namespace std;


int gcd(int a,int b) 
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

void sum_fraction(int num1,int deno1,int num2, int deno2) 
{
    if(deno1==deno2)
    {
        int num=num1+num2;
        int gcd_num=gcd(num,deno1);  
        cout<<num/gcd_num<<" "<<deno1/gcd_num<<endl;
    }
    else
        sum_fraction(num1*deno2,deno1*deno2,num2*deno1,deno1*deno2);    
}

int main()
{

    ifstream file;
    
    file.open("input.txt");
    
    if(file.is_open()==false) 
        cout<<"입력파일이 없습니다."<<endl;
    
    else
    {	
        int case_num;
        file>>case_num;
        
        for(int i=0; i<case_num; i++)
        {   
            int num1,deno1,num2,deno2;
            file>>num1>>deno1>>num2>>deno2;
            sum_fraction(num1,deno1,num2,deno2);
        }

    }

        file.close();

}
