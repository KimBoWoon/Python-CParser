
#include <iostream>
#include <fstream>

using namespace std;

int fibo(int n) {
	if (n == 0) return 0;
	else if (n <= 2) return 1;
	else return fibo(n - 1) + fibo(n - 2);
}

int main(void) {
	ifstream in("input.txt");

	int t;

	in >> t;

	for (int testcase = 0; testcase < t; testcase++) {
		int n, result;


		in >> n;

		result = fibo(n);


		cout << result << endl;
	}

	return 0;
}
