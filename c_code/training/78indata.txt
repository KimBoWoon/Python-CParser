#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase;i++)
	{
		int num;
		instream >> num;
		int res=1;
		for(int div=2;div*div<=num;div++) 
		{
			if(num%div==0)
				res= 0;
		}
		cout<<res<<endl;
	}
}
