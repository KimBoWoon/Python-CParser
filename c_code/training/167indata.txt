
#include <iostream>
#include <fstream>

int McCarthy(int n) {

	if (n > 100) return n - 10;
	else return McCarthy(McCarthy(n + 11));
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num;

	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {

		fin >> num;
		std::cout << McCarthy(num) << std::endl;
	}
	fin.close();
}
