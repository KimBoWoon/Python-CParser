int x = 0;
char str[1000000];

int checkPalindrome(char arr[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (arr[first] != arr[last])
		return 0;
	else
		checkPalindrome(arr, first + 1, last - 1);
}

void baseConversion( int n, int base)
{
	static char baseTable[] = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ~!";
	if (n >= base)
		baseConversion(n / base, base);
	str[x++] = baseTable[n%base];
}
