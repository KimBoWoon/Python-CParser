


#include <iostream>
#include <fstream>
#include <cstdlib>
	
using namespace std;

int factorial(int k){
	return k==0? 1: k* factorial(k-1);
}
int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");
	if (instream.fail()) exit(1);

	instream>>no_cases;

	while(no_cases--){
		int k;
		instream>>k;
		cout<< factorial(k)<<endl;
	}
	instream.close();
}
