

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void reduction(int& fc, int& fm);
int gcd(int a, int b);

int main(void) {
	FILE *inFile;
	int caseNum;

	int i;
	int n1,n2,n3,n4;
	int fc;
	int fm;

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &caseNum);

	for (i = 0; i < caseNum; i++) {
		fscanf(inFile, "%d %d %d %d", &n1 , &n2 , &n3 , &n4);
		fc = n1*n4 + n2*n3;
		fm = n2*n4;
		reduction(fc, fm);
		printf("%d %d\n", fc, fm);
	}
	getchar();
	return 0;
}

void reduction(int& fc, int& fm) {
	
	int g = gcd(fc, fm);
	fc = fc / g;
	fm = fm / g;

	return;
	
	

}

int gcd(int a, int b) {
	if (b == 0) {
		return a;
	}
	return gcd(b, a%b);
}
