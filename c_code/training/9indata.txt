#include <iostream>
#include <fstream>


using namespace std;

int main()
{
	std::ifstream inStream; 
	int number, cases;

	inStream.open("input.txt");
	if (inStream.fail())
	{		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;

	for (int i = 0; i < cases; i++)
	{
		inStream >> number;
		if (number < 3) {	
			cout << "1" << endl;
			continue;
		}
		if (number % 2 == 0) {
			cout << "0" << endl;
			continue;
		}
		for (int j = 3; j <= number; j = j + 2)
		{
			if (number == j){
				cout << "1" << endl;
				break;
			}
			if (number%j == 0) {
				cout << "0" << endl;
				break;
			}
			
		}
		
	}
	inStream.close();		
	return 0;
}
