

#include<iostream>
#include<string>
#include<fstream>

using namespace std;

int factorial(int numBuff){
	if (numBuff <= 1){
		return 1;
	}
	else{
		return numBuff*factorial(numBuff-1);
	}	
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, numBuff = 0;

	f >> numCount;

	for (int i = 0;i<numCount;i++){
		f >> numBuff;
		cout << factorial(numBuff) << endl;
	}
	return 0;
}
