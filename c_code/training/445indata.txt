

#include <stdio.h>

int gcd(int a, int b);

void main() {
	int numTestCases;
	int i;
	int a, b;

	FILE *f;
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d %d", &a, &b);
		printf("%d\n", gcd(a, b));
	}
}

int gcd(int a, int b) {
	int temp = a;
	if (a < b) {
		a = b;
		b = temp;
	}
	if (b == 0)
		return a;
	else if(b > 0)
		return gcd(b, a % b);

}
