

#include <iostream>
#include <fstream>
#include <algorithm>
#include <cstdlib>
#include <vector>

using namespace std;

int main()
{
	ifstream infile;
	int testCase;
	int n, k, num;

	infile.open("input.txt");

	infile >> testCase;
	for (int i = 0; i < testCase; i++)
	{
		int count = 1;
		infile >> n >> k;
		vector <int> numbers;
		int g = 0;
		int j = 1, l = n - 1;
		for (int j = 0; j < n; j++)
		{
			infile >> num;
			numbers.push_back(num);

		}
		sort(numbers.begin(), numbers.end());
		
		g = abs(k - (numbers[0] + numbers[l]));
		while (j!=l)
		{
			int temp = numbers[j] + numbers[l];
			if (j == l)
				break;
			temp = abs(k - temp);
			if (g > temp)
			{
				g = temp;
				count = 1;

			}
			else if (g == temp)
			{
				count++;
			}
			if ((numbers[j] + numbers[l]) > k)
				l--;
			else
				j++;

		}
		cout << count << endl;
	}
	infile.close();

	return 0;
}
