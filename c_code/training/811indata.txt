

#include <iostream>
#include <fstream>
#include <cstdlib>
	
using namespace std;

int fibonacci(int k){
	return (k==1||k==0)? k: fibonacci(k-1)+ fibonacci(k-2);
}

int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");
	if (instream.fail()) exit(1);

	instream>>no_cases;

	while(no_cases--){
		int k;
		instream>>k;
		cout<<fibonacci(k)<<endl;
	}
	instream.close();
}
