int prime_num(int num); 
int main(int argc, char* argv[]){

	std:: ifstream is; 
    is.open("input.txt");  

	int times; 
	is >> times;

	for( int i = 0; i < times; i++){
		
		int num; 
		is >> num;

		std::cout<<prime_num(num)<<std::endl; 



	}

}

int prime_num(int num){

	int i = 0;
	for( i = 2; i < num; i++ ){

		if( num%i == 0 )
			break;
		else
			continue;

	}

	if( i == num )
		return 1;
	else
		return 0;



}
