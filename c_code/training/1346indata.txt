
#include <iostream>
#include <fstream>
using namespace std;
int theGreatest(int a, int b){
	if (b == 0)
		return a;
	else
		return theGreatest(b, a%b);
}
void check(int a, int b, int count){
	int gcd=0;
	if (a == 1){
		cout << b << endl;
	}
	else{
		if ((a*count) - b >= 0){
			gcd = theGreatest(a*count - b, b*count);
			a = (a * count - b) / gcd;
			b = (b * count) / gcd;
			check(a, b, count+1);
		}
		else
			check(a, b, count + 1);
	}
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n;
	int a,b;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> a >> b;
		check(a, b, 1);
	}
	return 0;
}
