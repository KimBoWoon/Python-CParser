#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void main(){
	FILE *in_file;
	int i, numcases;
	char str[51] = {'\0'};                                    

	in_file = fopen("input.txt", "r");
	fscanf(in_file,"%d",&numcases);                          

	for(i = 0 ; i < numcases; i++){
		fscanf(in_file, "%s", str);              
		paren(str);                                  
	}
	fclose(in_file);
}



int paren(char *str){
	int  a;
	int left=0;
	int right=0;
	
	for(a = 0; str[a]!='\0'; a++){                  
		if(str[a] == '(' ) 
			left= left +1;
		else if(str[a] == ')' )
			right = right+1;

		if(left < right ) {                  
			printf("NO\n");                   
			return ;}
	}

	if(left == right && left !=0){
		printf("YES\n");
		return; }
	else{
		printf("NO\n");
		return;}
}
