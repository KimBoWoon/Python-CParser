#include<stdio.h>
#include<stdlib.h>

int main()
{
	FILE *in_file;
	int i,j;
	int numTestCases;
	int num;
	int cnt;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL)
		exit(1);

	fscanf(in_file, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++)
	{
		fscanf(in_file, "%d", &num);

		cnt = 0;

		for (j = 2; j < num; j++)
		{
			if ((num%j) == 0)
			{
				cnt++;
			}	
		}

		if (cnt == 0)
			printf("1\n");
		else 
			printf("0\n");
	}
	return 0;
	fclose(in_file);
}
