#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAXSIZE 51

int Parenthesis(char *arr,int i, int count){
	int length=strlen(arr);
	if((arr[0]!='(') || (arr[length-1]!=')')){
		printf("NO\n");
	}
	else{ 
		if(i==length){
			if(count==0){ 
				printf("YES\n");
			}
			else{
				printf("NO\n");
			}
		}
		else if(arr[i]=='('){
			return Parenthesis(arr,i+1,count+1); 
		}
		else if(arr[i]==')'){
			return Parenthesis(arr,i+1,count-1);
		}
		else {
			return Parenthesis(arr,i+1,count); 
		}
	}
}
int main(){
	FILE *file;
	int testcase,i;
	char str[MAXSIZE];
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &testcase);
	for(i=0; i<testcase; i++)
	{
		fscanf(file,"%s", str);
		Parenthesis(str,0,0);
	}
	fclose(file);
 }
