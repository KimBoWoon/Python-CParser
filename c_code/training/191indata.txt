#include <stdio.h>
#include <stdlib.h>
int McCarthy (int number)
{
	if(number>100) return 	number-10;
	else return McCarthy(McCarthy(number+11));
}
void main()
{
	FILE* infile = fopen("input.txt","r");
	int i = 0;
	int datacase = 0;

	fscanf(infile,"%d",&datacase);
	for(; i<datacase; i++)
	{
		int number = 0;
		fscanf(infile,"%d",&number);
		printf("%d\n",McCarthy(number));
	}
	fclose(infile);

}
