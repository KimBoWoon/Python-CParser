#include <iostream>
#include <fstream>
using namespace std;
#define BREAK 1000
inline int result(int num){
	return (num * (num + 1)) / 2;
}

int main() {
	std::ifstream inStream; 
	int number, cases;
	inStream.open("input.txt");
	if (inStream.fail()) {		
		std::cerr << "input file opening failed. \n";
		return -1; 
	}

	inStream >> cases;

	for (int i = 0; i < cases; i++) {
		bool res = false;
		inStream >> number;
		for (int j = 1; j < 46; j++) 
		for (int k = 1; k < 46; k++)
		for (int n = 1; n < 46; n++) {
			if (number == result(j) + result(k) + result(n)){
				res = true;
				cout << res << endl;
				j = k = n = BREAK;
			}
		}
		if (!res) cout << res << endl;
	}
	inStream.close();		
	return 0;
}
