
#include <iostream>
#include <fstream>

int fib(int n);
int main()
{
	std::ifstream infile("input.txt");
	int numCase;

	infile >> numCase;

	for( int i=0; i< numCase; i++)
	{
		int n;
		infile >> n;
		std::cout<<fib(n)<<std::endl;
	}
}
int fib(int n)
{
	if( n <= 1)
		return n;
	else
		return fib(n-1) + fib(n-2);
}
