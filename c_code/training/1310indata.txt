

#include <iostream>
#include <fstream>

using namespace std;
int gcd(int m, int n){
	return (n==0) ? m : gcd(n, m%n);
}
int henry( int a, int b){
	while(a!=1){
		int x= (b/a)+1;
		a = a*x-b;
		b = b*x;
		int g = gcd(a,b);
		a/=g;
		b/=g;

	}
	return b;
}

int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");

	instream>>no_cases;

	while(no_cases--){
		int a,b;
		instream>> a>>b;
		cout<<henry(a,b)<<endl;
	}
	instream.close();
}
