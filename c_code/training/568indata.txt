
#include <iostream> 
#include <fstream>
#include <vector>
using namespace std;
int factorial_num[13]={0,1};
int factorial(int n)
{
	return n*factorial_num[n-1];
}
int main()
{
	int t,n,k;
	for(int i=2; i<=12; i++)
	{
		factorial_num[i]=factorial(i);
	}
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>t;
	for(int i=0; i<t; i++)
	{
		vector<char> v;
		vector<char>::iterator it;
		int div=0,portion=0,rest=0;
		inStream>>n>>k;
		rest=k; 
		if(n==1)
			cout<<'a'<<endl;
		else 
		{
			for(char j='a'; j<'a'+n; j++)
				v.push_back(j);
			div=n-1;
			while(v.size()>2)
			{
				portion=k/factorial_num[div];
				rest=k%factorial_num[div];
				if(rest==0)
				{
					portion=portion-1;
					rest=k-factorial_num[div]*portion;
				}
				it=v.begin()+portion;
				cout<<v[portion];
				k=rest;
				v.erase(it);
				div--;

			}
			if(rest==1) 
				cout<<v[0]<<v[1]<<endl;
			else 
				cout<<v[1]<<v[0]<<endl;
		}
	}
}
