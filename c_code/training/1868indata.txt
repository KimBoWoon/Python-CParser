
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

int parent(char par[], int count){
	int chk=1;
	if(par[0]==')' || par[count-1]=='(')
		chk=0;
	else{
	int l=0;
	int r=0;
	for(int i =0;i<count;i++)
		par[i]==')' ? l++ : r++;
	l==r ? chk=1 : chk=0;
	}
	return chk;
}

int main(void){
	fstream infile;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error"<<endl;
		exit(1);
	}
	
	int c;
	char p[5];
	infile >> c;
	infile.getline(p,5);
	for(int i=0;i<c;i++){
		char par[50]="";
		int count = 0;

		infile.clear();
		infile.getline(par,50);
		
		for(int j=0;j<50;j++)
			if(par[j]=='(' || par[j]==')')
				count++;
		
			if(parent(par,count))
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
