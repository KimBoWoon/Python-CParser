#include <stdio.h>
#include <math.h>

int main(void){
	FILE *fin = fopen("input.txt", "r");
	
	int t;
	int i, j;
	int primes[100001];
	
	fscanf(fin, "%d", &t);
	
	for(i=2; i<=100000; i++){
		int flag=1;
		for(j=2; j*j<=i; j++){
			if(i%j == 0) {
				flag = 0;
				break;
			}
		}
		
		primes[i] = flag;
	}
	
	for(i = 0; i < t; i++){
		int inp;
		fscanf(fin, "%d", &inp);
		printf("%d\n", primes[inp]);
	}
	
	return 0;
}
