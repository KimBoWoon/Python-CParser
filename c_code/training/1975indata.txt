


#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#pragma warning(disable : 4996)
int number[1000000] = { 0 };
int intcmp(const void* v1, const void* v2)
{
	return *(int*)v1 - *(int*)v2;
}
void main()
{
	FILE* inFile;
	int testCases = 0, i = 0, j = 0, k = 0, numberOfNums = 0, numberSum = 0, count = 0, min = 0, temp = 0, left, right, size;
	inFile = fopen("input.txt", "r");
	if (inFile == NULL)
	{
		exit(1);
	}
	fscanf(inFile, "%d", &testCases);
	for (i = 0; i < testCases; i++)
	{
		fscanf(inFile, "%d %d", &numberOfNums, &numberSum);
		right = 0;
		left = numberOfNums - 1;
		for (j = 0; j < numberOfNums; j++)
		{
			fscanf(inFile, "%d", &number[j]);
		}
		qsort(number,numberOfNums, sizeof(number[0]), intcmp);
		min = abs(numberSum - (number[right] + number[left]));
		while (right < left)
		{
			if (min == 0)
				break;
			if (abs(number[right + 1] + number[left] - numberSum) < abs(number[right] + number[left - 1] - numberSum))
			{
				right += 1;
				if (right == left)
					break;
			}
			else if (abs(number[right + 1] + number[left] - numberSum) > abs(number[right] + number[left - 1] - numberSum))
			{
				left -= 1;
				if (right == left)
					break;
			}
			else
			{
				right+=1;
				left-=1;
				if (right == left)
					break;
			}
			if (min > abs(numberSum - (number[right] + number[left])))
			{
				min = abs(numberSum - (number[right] + number[left]));
			}
		}
		right = 0;
		left = numberOfNums - 1;
		while (right < left)
		{
			if ((number[right] + number[left]) < numberSum - min)
				right++;
			else if ((number[right] + number[left]) > numberSum - min)
				left--;
			else
			{
				right++;
				left--;
				count++;
			}
		}
		if (min != 0)
		{
			right = 0;
			left = numberOfNums - 1;
			while (right < left)
			{
				if ((number[right] + number[left]) < numberSum + min)
					right++;
				else if ((number[right] + number[left]) > numberSum + min)
					left--;
				else
				{
					right++;
					left--;
					count++;
				}
			}
		}
		printf("%d\n", count);
		count = 0;
		min = 0;
	}
	fclose(inFile);
}
