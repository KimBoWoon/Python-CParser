


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream inStream;
	int numTestCases;		
	int count;
	int arr[15] = { 0, };	


	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		int sum = 0;				
		int temp;

		inStream >> count;
		for (int j = 0; j < count; j++)
		{
			inStream >> arr[j];
		}

		for (int k = 0; k < 1000; k++)
		{
			temp = arr[0];
			for (int l = 0; l < count - 1; l++)
			{
				arr[l] = abs(arr[l] - arr[l + 1]);
			}
			arr[count - 1] = abs(arr[count - 1] - temp);			
		}

		for (int m = 0; m < count; m++)		
		{
			sum += arr[m];
		}

		if (sum == 0)
		{
			cout << "ZERO" << endl;		
		}
		else
		{
			cout << "LOOP" << endl;
		}
	}

	inStream.close();
	return 0;
}
