void henry(int a, int b,int counter)
{
	if (a == 1){
		cout << b << endl;
		return;
	}
	if ((a*counter - b) > 0){
		a = a*counter - b;
		b = b*counter;
		int gcdNum = gcd(a, b);
		a = a / gcdNum;
		b = b / gcdNum;
		return henry(a, b, ++counter);
	}
	else return henry(a, b, ++counter);
}
int gcd(int numA, int numB)
{
	if (numB == 0) return numA;
	else return gcd(numB, numA%numB);
}
