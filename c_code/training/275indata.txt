

#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>
int gcdd(int num1, int num2);

int main(void)
{
	FILE *in_file;
	int no_cases, n1,n2,n3,n4;
	int i,numer,denom,gcd;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%d %d %d %d", &n1,&n2,&n3,&n4); 
		numer = n2*n4; 
		denom = (n1*n4) + (n2*n3); 
		gcd = gcdd(numer, denom);
		printf("%d %d\n", denom / gcd, numer / gcd); 

	}		
	fclose(in_file); 
}

int gcdd(int num1, int num2) 
{
	if (num2 == 0)
		return num1;
	else if (num2 > 0)
	{
		return gcdd(num2, num1%num2);
	}
}
