#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main()
{
	ifstream inStream;
	int numTestCases;

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int h, w, n;		
		inStream >> h >> w >> n;
		int allRoomNumber = h*w;
		int room = 0;
		int allRoom[9801];
		for (int i = 1; i <= w; i++)
		{
			for (int j = 1; j <= h; j++)
			{
				allRoom[room] = (100 * j) + i;
				room++;
			}
		}
		cout << allRoom[n-1] << endl;

	}
	return 0;
	inStream.close();

}
