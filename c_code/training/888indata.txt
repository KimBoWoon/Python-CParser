#include <iostream>
#include <fstream>
using namespace std;

long fibo(int num){
    if(num == 0) return 0;
    if(num == 1) return 1;
    
    return fibo(num - 1) + fibo(num - 2);
}

int main(){
    ifstream file;
    file.open("input.txt");
    
    int cases;
    file >> cases;
    
    while(cases--){
        int num;
        file >> num;
        cout << fibo(num) << endl;
    }
    
    return 1;
}
