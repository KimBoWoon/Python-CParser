

#include <iostream>
#include <string.h>
#include <fstream>
using namespace std;

int numOfCase[12];
char str[13];

void findString(int n, int idx){
	int len = strlen(str) - 1;

	while (n > 1){
		for (int i = len - idx; i<len; i++){
			if (str[len - idx] < str[i + 1]){
				char tmp = str[len - idx];
				str[len - idx] = str[i + 1];
				str[i + 1] = tmp;
				break;
			}
		}
		n -= numOfCase[idx - 1];
		while (n <= numOfCase[idx - 1]) idx--;
	}
	cout << str << endl;
}

int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;

	for (int i = 1; i <= 12; i++){
		if (i != 1)
			numOfCase[i - 1] = i * numOfCase[i - 2];
		else
			numOfCase[i - 1] = i;
	}

	char BaseAlpha[12] = { 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l' };

	for (int i = 0; i<testcases; i++){
		int alphaLength, number;
		in >> alphaLength>>number;

		for (int j = 0; j < alphaLength; j++){
			str[j] = BaseAlpha[j];
		}

		for (int k = 0; k<12; k++){
			if (number <= numOfCase[k]){
				findString(number, k);
				break;
			}
		}
	}
	return 0;
}
