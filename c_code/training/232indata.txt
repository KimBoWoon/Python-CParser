
#include<stdio.h>

int recursion(int a, int b){
	if (a%b == 0) return b;
	else return recursion(b, a%b);
}

int main(){
	
	FILE *inFile;
	int circle, a, b,c,d,i;
	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d %d %d %d", &a, &b, &c, &d);
		a = a*d + b*c;
		b = b*d;
		c = a > b ? recursion(a, b) : recursion(b, a);
		printf("%d %d\n", a / c, b / c);
	}
	return 0;
}
