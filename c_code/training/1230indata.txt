#include <stdio.h>
#include <stdlib.h>

void goldba(int n);	

void main()
{
	FILE *f;
	int count;
	int i;
	int num;

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);

	for(i=0; i<count; i++)
	{
		fscanf(f, "%d", &num);
		goldba(num);
	}
	fclose(f);
}
void goldba(int n)
{
	int i,x,y;
	int goldnum1Check=1, goldnum2Check=1;	

	for(i=n/2; i>1; i--)		
	{
		for(x=2; x*x<=i; x++)	
		{
			if(i%x==0)		
			{
				goldnum1Check = 0;	
				break;		
			}
		}
		if(goldnum1Check==0)	
		{
			goldnum1Check = 1;	
			continue;			
		}

		for(y=2; y*y<=n-i; y++)	
		{						
			if((n-i)%y==0)		
			{
				goldnum2Check = 0;	
				break;				
			}
		}
		if(goldnum2Check==0)		
		{
			goldnum2Check = 1;		
			continue;				
		}
		else						
		{
			printf("%d %d\n", i, n-i);	
			return;						
		}
	}
}
