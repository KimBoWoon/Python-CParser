
#include <iostream>
#include <fstream>
#include <vector>
#include <cstdlib>
using namespace std;
int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	int t=0;
	inStream>>t;
	for(int i=0; i<t; i++)
	{
		int n=0,value=0,absValue=0,loopOK=0;
		inStream>>n;
		vector<int> v;
		vector<int> v1;
		v.resize(n);
		v1.resize(n);
		for(int j=0; j<n; j++)
		{
			inStream>>value;
			v[j]=value;
		}
		for(int j=1; j<=1000; j++) 
		{
			loopOK=0;
			for(int k=0; k<n-1; k++)
			{
				absValue=abs(v[k]-v[k+1]);
				v1[k]=absValue;
			}
			absValue=abs(v[0]-v[n-1]);
			v1[n-1]=absValue;
			for(int k=0; k<n; k++)
			{
				v[k]=v1[k];
			}
			for(int k=0; k<n; k++)
			{
				if(v1[k]!=0)
				{
					loopOK=1;
					break;
				}
			} 
			if(loopOK==0) break;
		}
		if(loopOK==1)
			cout<<"LOOP"<<endl;
		else 
			cout<<"ZERO"<<endl;
	}
}
