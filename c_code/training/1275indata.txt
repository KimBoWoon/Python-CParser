




#include <stdio.h>
#include <stdlib.h>

int isPrime[1000] = {0};

int main() {
    
    FILE* fp;
    
    int i, j, k, l;
    int num_cases, tmp, num;
    
    fp = fopen("input.txt", "r");
    if(fp == NULL)  exit(1);
    
    fscanf(fp, "%d", &num_cases);
    
    for(i=1; i<=999; i++) {
        
        tmp=0;
        
        for(j=2; j*j<=i; j++) {
            
            if(i%j==0) {
                isPrime[i]=0;
                tmp=1;
                break;
            }
        }
        if(tmp==0)
            isPrime[i]=i;
    }
    
    for(k=0; k<num_cases; k++) {
        
        fscanf(fp, "%d", &num);
        l=0;
        
        while(1) {
            if(isPrime[num/2-l]!=0 && isPrime[num/2+l]!=0) {
                printf("%d %d\n", isPrime[num/2-l], isPrime[num/2+l]);
                break;
            }
            l++;
        }
    }
}
