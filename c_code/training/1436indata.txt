

#include <iostream>
#include <fstream>

using namespace std;

int main(){

	ifstream inStream;
	int numcase = 0;
	
	inStream.open("input.txt");

	inStream >> numcase;

	for(int i = 0 ; i<numcase; i++)
	{
		int height;
		int weight;
		int number;
		int cal;		
		int room_front;	
		int room_rear;	

		inStream >> height;		
		inStream >> weight;		
		inStream >> number;		

		cal = number/height;	

		if(number % height == 0){
			room_front = height;
			room_rear = cal;
		}
		else
		{
			room_front = number - height*cal;
			room_rear = cal + 1;
		}
		cout<< room_front*100 + room_rear <<endl;

	}



	return 0;
}
