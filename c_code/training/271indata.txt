
#include <iostream>
#include <fstream>

int Euclid(int a, int b) {

	if (b == 0) return a;
	else return Euclid(b, a % b);
}

void Fraction(int a, int b, int c, int d) {

	if (b == d){
		int n = Euclid(a + c, b);
		if (n != 0) std::cout << (a + c) / n << " " << b / n << std::endl;
		else std::cout << a + c << " " << std::endl;
	}
	else Fraction(a * d, b * d, b * c, b * d);
}


int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num1, num2, num3, num4;

	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {

		fin >> num1 >> num2 >> num3 >> num4;
		Fraction(num1, num2, num3, num4);
	}
	fin.close();
}
