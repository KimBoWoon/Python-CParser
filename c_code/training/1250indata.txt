

#include<iostream>
#include<fstream>

using namespace std;

bool prime(int k);

int* search(int* arr, int k);

int main(void)
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	int pri[669];
	int n = 0;
	for (int i = 2; i < 5000; i++){

		if (prime(i) == true){
			pri[n++] = i;
		}
	}

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {

		int num;
		inStream >> num;
		bool check = false;
		int* n = new int[2];
		n = search(pri, num);

		for (int j = n[0]; j<=n[1]; j++){
			for (int k = n[0]; k >= 0; k--){
				if (pri[j] + pri[k] == num){
					cout << pri[k] << " " << pri[j] << endl;
					check = true;
					break;
				}
			}
			if (check == true) break;
		}
		
	}
}


bool prime(int k){
	for (int i = 2; i*i <= k; i++){
		if (k%i == 0)
			return false;
	}
	return true;
}

int* search(int* arr,int k){
	bool check = false;
	int brr[2] = { 0, };
	for (int i = 0; i < 669; i++){
		if (arr[i] >= k / 2 && check == false) {
			brr[0] = i;
			check = true;
		}
		if (arr[i] >= k){
			brr[1] = i - 1;
			return brr;
		}
	}
	return 0;
}
