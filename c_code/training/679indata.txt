#include<iostream>
#include<fstream>
using namespace std;

int Factorial(int n){
	if (n == 0) return 1; 
	else return n*Factorial(n - 1); 
}

int main(){
	ifstream inStream;
	int numCases;
	int number;

	inStream.open("input.txt");

	inStream >> numCases;

	for (int i = 0; i<numCases; i++){
		inStream >> number;

		cout << Factorial(number) << endl;
	}

	inStream.close();
}
