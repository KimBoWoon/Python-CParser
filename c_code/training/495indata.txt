

#include <stdio.h>

int gcd(int a, int b);

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int inputA = 0; 
	int inputB = 0;

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d %d", &inputA, &inputB); 

		printf("%d\n", gcd(inputA, inputB));

	}

	fclose(f);
	return 0;
}

int gcd(int a, int b) {
	if (b == 0) 
		return a;
	else 
	{
		return gcd(b, a%b);
	}
}
