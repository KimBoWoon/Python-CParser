


#include <iostream>
#include <fstream>
using namespace std;
int gcd(int a, int b)
{
	if(b==0)
	{
		return a;
	}
	else
	{
		return gcd(b,a%b);
	}
}

int main()
{
   fstream in;
   in.open("input.txt");
   int numcases;
   in >> numcases;
   int a, b; 
   
  
   for (int i = 0; i < numcases; i++)
   {
      in >> a >> b;
	  int gc;
      for(int j=2;;j++)
      {
         if (a == 1)
            break;
         else if (a*j-b>0)
         {
            a = a*j - b;
            b *= j;
            gc = gcd(a, b);
            if ( gc> 1)
            {
               a /= gc;
               b /= gc;
            }
         }
         
      }
      cout << b << endl;

   }

}
