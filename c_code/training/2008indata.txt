
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define MAX_DIFF 100000000

int cmpfunc (const void * a, const void * b) { return ( *(int*)a - *(int*)b ); } 

int findTwoNumbers(int *arr, int target, int size){
    int diff = MAX_DIFF;         
    int left = 0, right = --size;
    int sum, sub, cnt = 0;
    
    while(1){
        if (left >= right) return cnt;      
    
        sum = arr[left] + arr[right];
        sub = abs(target - sum);
    
        if (sub < diff) {       
            diff = sub;
            cnt = 1;
        } else if (sub == diff)     
            cnt++;
    
        
        if (sum < target) 
            left++;
        else if (sum > target)  
            right--;
        else {
            left++;
            right--;
        }
    }
    
    return cnt;
}

void getTwoNumbers(FILE *fin){
    int i, size, targetNumber;
    int *arr = NULL;
    
    fscanf(fin, "%d %d", &size, &targetNumber);
    arr = (int*)malloc(sizeof(int)*size);   
    
    for(i=0; i<size; i++)
        fscanf(fin, "%d", &arr[i]);     
    
    qsort(arr, size, sizeof(int), cmpfunc);     
    
    printf("%d\n", findTwoNumbers(arr, targetNumber, size));    
    
    free(arr);      
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        getTwoNumbers(fin);        
    
    fclose(fin);            
    
    return 0;           
}
