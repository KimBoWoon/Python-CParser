#include <iostream>
#include <fstream>
using namespace std;

int parenthesis(char *ps, int i, int testCase);

int main() {
	ifstream fin;
	fin.open("input.txt");
	
	int testCaseNumber = 0; 
	char PS[51];
	fin >> testCaseNumber;

	while (testCaseNumber--)	{
		int testCase = 0, i = 0; 
		fin >> PS;
		
		testCase = parenthesis(PS, i, testCase);

		testCase == 0 ? cout << "YES" << endl : cout << "NO" << endl;
	}
	return 0;
}

int parenthesis(char *ps, int i, int testCase) {
	if (ps[i] != '(' && ps[i] != ')') {
		return testCase;	
	}
	else if (testCase < 0) {
		return testCase;	
	}
	else {
		switch (ps[i])
		{
		case '(':
			testCase++;
			break;
		case ')':
			testCase--;
			break;
		default:
			break;
		}
		i++;
		return parenthesis(ps, i, testCase);
	}
}
