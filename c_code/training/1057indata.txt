#include <iostream>
#include <fstream>
#include <cmath>
#include <cstdlib>

using namespace std;

void DucciSeq(int num, int* arr);

int main(void)
{
	int count = 0;
	ifstream file;
	file.open("input.txt");
	file >> count;
	
	int num = 0; 
	int arr[15] = { 0 };

	for (int i = 0; i < count; i++)
	{
		file >> num;

		for (int j = 0; j < num; j++) 
			file >> arr[j];

		DucciSeq(num, arr);
	}

	return 0;
}

void DucciSeq(int num, int* arr)
{
	int numOfZero; 
	int temp; 

	for (int i = 0; i < 1000; i++)
	{
		numOfZero = 0;
		temp = arr[0];

		for (int j = 0; j < (num-1); j++)
		{
			arr[j] = abs(arr[j+1] - arr[j]);
			if (arr[j] == 0) numOfZero++;
		}
		arr[num-1] = abs (arr[num - 1] - temp);

		if ((numOfZero == (num-1)) && (arr[num-1]==0)) 
		{
			cout << "ZERO" << endl;
			return;
		}
	}

	cout << "LOOP" << endl;
}
