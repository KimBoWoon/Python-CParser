#include <cstdio>

inline int abs(int num) {
	return (num < 0) ? -num : num;
}

inline bool isZero(int arr[], int num) {
	for (int i = 0; i < num; i++) {
		if (arr[i]) return false;
	}
	return true;
}

bool ducci(int arr[], int num) {
	for (int i = 0; i < 1000; i++) {
		arr[num] = arr[0];
		for (int j = 0; j < num; j++) {
			arr[j] = abs(arr[j] - arr[j + 1]);
		}
		if (isZero(arr, num)) return true;
	}

	return false;
}

int main() {
	FILE * fp = fopen("input.txt", "r");
	int nCase;
	fscanf(fp, "%d", &nCase);

	while (nCase--) {
		int num, arr[16] = { 0 };
		fscanf(fp, "%d", &num);

		for (int i = 0; i < num; i++) {
			fscanf(fp, "%d", arr + i);
		}

		printf("%s\n", ducci(arr, num) ? "ZERO" : "LOOP");
	}

	fclose(fp);
	return 0;
}
