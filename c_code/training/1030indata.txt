#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

int num;
int arr[15];

int ducci(){
	int loop=1000;
	while( loop-- ){
		int flag=1;
		int tmp = arr[0];
		for(int i=0; i<num-1; i++){
			arr[i] = abs(arr[i]-arr[i+1]);
			if( arr[i] != 0 ) flag = 0;
		}
		arr[num-1] = abs(arr[num-1] - tmp);
		if( arr[num-1] != 0 ) flag = 0;
		
		if( flag==1 ) return 0;
	}

	return 1;
}

int main(void){
	FILE* file;
    int cases;

	file = fopen("input.txt", "r");

    fscanf(file, "%d", &cases);

    while(cases--){
		fscanf(file,"%d", &num);
		
		for(int i=0; i<num; i++)
			fscanf(file, "%d", &arr[i]);
        
		if( ducci() )
			puts("LOOP");
		else
			puts("ZERO");
    }
    return 0;
}
