
#include <stdio.h>
#define T_MAX 45
#define T(n) (n)*(n+1)/2

void eureka(FILE* fin){
    int i,j,k,sum,testNum = 0;
    fscanf(fin, "%d", &testNum);

    for(i=1; i<T_MAX && T(i)<testNum; i++)      
    for(j=1; j<T_MAX && T(j)<testNum; j++)      
    for(k=1; k<T_MAX && T(k)<testNum; k++){
        sum = T(i)+T(j)+T(k);   
        
        if(sum == testNum){     
            printf("1\n");
            return;
        }else if(sum > testNum) 
            break;
    }
    printf("0\n");      
    return;
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        eureka(fin);        
    
    fclose(fin);            
    
    return 0;           
}
