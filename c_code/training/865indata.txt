#include <stdio.h>
#include <stdlib.h>

long fibo(int n){
	if(n < 2)      
		return n;
	else
		return (fibo(n-2) + fibo(n-1));

}
void main()
{
	FILE *in_file;
	int i,n;
	int numcases;

	in_file = fopen("input.txt","r"); 

	fscanf(in_file, "%d", &numcases); 

		for(i=0; i < numcases; i++)
		{
			for(n = 0; n <= 30; n++){
				fscanf(in_file, "%d", &n);
				goto a;
			}
			a:
			printf("%d\n", fibo(n));
		}
	fclose(in_file);	
}
