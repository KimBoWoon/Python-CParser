#include <stdio.h>
#include <stdlib.h>

long f(int n){ 
	if(n < 2)
		return 1;
	else
		return n * f(n-1);
}
void main()
{
	FILE *in_file;
	int i, n;
	int numcases;

	in_file = fopen("input.txt","r"); 

	fscanf(in_file, "%d", &numcases); 

		for(i=0; i < numcases; i++)
		{
			for(n = 0; n <= 12; n++){           
				fscanf(in_file, "%d", &n);      
				break;	
			}
			printf("%d\n", f(n));
		}
		fclose(in_file); 
}
