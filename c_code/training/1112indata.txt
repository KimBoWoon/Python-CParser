

#include <stdio.h>
#include <stdlib.h>
void main(void)
{
	FILE *inFile;
	int circle,num, i,j,k; 

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d", &num);
		
		for (i = 1; triangle(i) < num; i++){
			for (j = 1; triangle(j) < num; j++){
				for (k = 1; triangle(k) < num; k++){
					if (triangle(i) + triangle(j) + triangle(k) == num) {
						i = j = k = 45; break;
					}
				}
			}
		}
		if (i >= 45) printf("1\n");
		else printf("0\n");
	}
	fclose(inFile);

}

int triangle(int n){
	return n*(n + 1) / 2;
}
