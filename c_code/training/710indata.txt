#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int checkPalindrome(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if(str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first+1, last-1);
}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;
	char n[1000];

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		fscanf(inFile, "%s", n);

		printf("%d\n", checkPalindrome(n, 0, strlen(n)-1));

	}

	fclose(inFile);
}
