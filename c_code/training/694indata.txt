
#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;
int palindrome(char* str, int m, int n){
	if (m >= n) return 1;
	else{
		if (str[m] == str[n])
			return palindrome(str, m + 1, n - 1);
		else
			return 0;
	}
}


int main(){

	int caseNum = 0;
	
	ifstream inputFile("input.txt");

	inputFile >> caseNum;
	
	for (int i = 0; i < caseNum; i++){
		char str[1001];
		inputFile >> str;
		
		cout << palindrome(str, 0, strlen(str) - 1) << endl;
	}

	return 0;
}
