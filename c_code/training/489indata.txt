



#include <stdio.h>
#include <stdlib.h>


int gcd(int x,int y);

void main(void)
{
	FILE *inFile;
	int i,j;
	int numCases;
	int count =0;
	int a,b;


	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0;i<numCases;i++)
	{
		
		fscanf(inFile, "%d %d",&a,&b );
		printf("%d\n",gcd(a,b));
	}

		

	
	
	fclose(inFile); 

}


int gcd(int x,int y)
{
	int save;
	int k;

	for(k=y;k>0 ; k=save%k)
	{
		save = x;
		x =k;	
	}

		return x;	
}
