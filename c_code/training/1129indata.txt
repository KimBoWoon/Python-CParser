#include <iostream>
#include <fstream>

void find_case(int num, int cases[], int& count);
bool same_matching(int num, const int cases[], const int count);

int main(void){

	std::ifstream inFile("input.txt");

	if (inFile.fail()){
		std::cout << "input file opening fail\n" << std::endl;
		return 0;
	}

	int testCases = 0;
	int num = 0;
	
	inFile >> testCases;

	for (int i = 0; i < testCases; i++){
		inFile >> num;

		int cases[100] = { 0 };
		int count = 0;
		
		find_case(num, cases, count);
		std::cout << same_matching(num, cases, count) << std::endl;

	}
	return 0;
}

void find_case(int num, int cases[], int& count){

	int step = 2;
	cases[0] = 1; 

	int i = 0;
	while (1){
		cases[i + 1] = cases[i] + step;
		i++;
		step++;
		count++;
		if (cases[i] > num){
			break;
		}
	}
}

bool same_matching(int num, const int cases[], const int count){

	for (int i = 0; i < count; i++){
		for (int j = 0; j < count; j++){
			for (int k = 0; k < count; k++){
				if (cases[i] + cases[j] + cases[k] == num){
					return true;
				}
			}
		}
	}
	return false;
}
