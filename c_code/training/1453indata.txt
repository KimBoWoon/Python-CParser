
#include<iostream>
#include<fstream>
using namespace std;

int RoomNum(int h, int s, int n);
bool CheckCon(int h, int s, int n);

int main()
{
	int count = 0;
	ifstream file;
	file.open("input.txt");
	file >> count;

	int H, S, N;
	for (int i = 0; i < count; i++)
	{
		file >> H >> S >> N;
		cout<<RoomNum(H, S, N)<<endl;
	}

	return 0;
}


int RoomNum(int h, int s, int n)
{
	int floor, Rnum; 

	if ((n%h) == 0)
	{
		floor = h; 
		Rnum = n / h;
	}
	else
	{
		floor = n % h;
		Rnum = (n / h)+1;
	}
	return (floor * 100) + Rnum;
}
