#include <stdio.h>
#include <stdlib.h>

int main()
{
   FILE *in_file;
   int i;
   int no_cases,H,W,N;

   in_file=fopen("input.txt","r");
   if(in_file==NULL) exit(1);

   fscanf(in_file, "%d",&no_cases);

   for(i=0; i<no_cases; i++){
      int front,end;
      fscanf(in_file,"%d %d %d", &H,&W,&N);

      if(N%H!=0){
         front=N%H;
         end=N/H+1;
         if(end<10)
            printf("%d0%d\n",front,end);
         else
            printf("%d%d\n",front,end);
      }
      else{
         end=N/H;
         if(end<10)
            printf("%d0%d\n",H,end);
         else
            printf("%d%d\n",H,end);
      }

   }

   fclose(in_file);
   return 0;
}
