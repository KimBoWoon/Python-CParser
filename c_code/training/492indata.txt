
#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;


int gcd(int a, int b){
	if(a==0)
		return b;
	else if(b==0)
		return a;
	else if(a>=b)
		return gcd(b,a%b);
	else if(b>=a)
		return gcd(a,b%a);
}
int main(void){
	ifstream infile;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error"<<endl;
		exit(1);
	}
	int num;
	int x;
	int y;
	infile >> num;
	for(int i=0;i<num;i++){
		infile >> x >> y;
		cout<<gcd(x,y)<<endl;
		
	}

	return 0;
}
