#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;

int main()
{
	ifstream inStream;
	int no_cases,value,time=0,result;

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"Input file openning failed.\n";
		exit(1);
	}
	inStream>>no_cases;
	
	for(int i=0; i<no_cases; i++)
	{
		inStream>>value;

		for(int divisor=2; divisor*divisor<=value; divisor++)
		{
			if(value%divisor==0)
			{
				time++;
				
			}
		}
		if(time==0)
			result=1;
		else
			result=0;

		cout<<result<<endl;

		result=0,time=0;
	}
	return 0;	
}
