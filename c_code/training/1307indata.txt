


#include <iostream>
#include <fstream>
using namespace std;
int henry(int a, int b, int i);
int euclid(int a, int b);
int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++)
	{
		int a, b = 0;
		inFile >> a >> b;
		cout << henry(a, b, 2) << endl;
	}
	return 0;
}

int henry(int a, int b, int i){
	int aNext, bNext, greatest, newA, newB;
	aNext = (a*i - b);
	bNext = b*i; 
	greatest = euclid(aNext, bNext);
	newA = aNext / greatest;
	newB = bNext / greatest;

	if (a == 1){
		return b;
	}
	if (aNext > 0){					
		if (aNext == 1){
			return bNext;
		}
		else return henry(newA, newB, ++i);
	}
	else
		return henry(a, b, ++i);
}

int euclid(int a, int b){
	int AmodB = 0;
	if (b == 0){
		return a;
	}
	else{
		AmodB = a%b;
		return euclid(b, AmodB);
	}
}
