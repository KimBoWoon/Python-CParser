#include <iostream>
#include <fstream>
using namespace std;
int main(){
	FILE *in_file;
	int testcases, num, temp, a[100], i, j, k , length, flag;

	in_file = fopen("input.txt", "r");
	fscanf(in_file, "%d", &testcases);
	for(i=0; i <testcases; i++){
		fscanf(in_file, "%d", &num);
		for(k=2;k<=64; k++){
			temp = num;
			for(j=0;temp>0;j++){
				a[j]=temp % k;
				temp /= k;
			}
			length = j-1;
			flag=1;
			for(j=0; j<(length/2)+1; j++)
			{
				if(a[j]!=a[length-j])
				{
					flag = 0;
					break;
				}
			}
			if(flag==1)
				break;
		}
		cout << flag << '\n';
	}
	fclose(in_file);
}
