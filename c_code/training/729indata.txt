#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;
using std::strlen;

int checkPalindrome(char str[], int first, int last){
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first + 1, last - 1);
}

int main(){

	int testCase;
	char line[1001];

	ifstream file;
	file.open("input.txt");

	file >> testCase;
	for (int tCase = 0; tCase < testCase; tCase++){
		file >> line;
		cout << checkPalindrome(line, 0, std::strlen(line) - 1) << endl;
	}
	return 0;
}
