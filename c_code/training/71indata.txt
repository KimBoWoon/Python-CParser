#include <stdio.h>
#include <stdlib.h>

void main()
{
	FILE *in_file;

	int i;
	int numcases;
	int n, a;

	in_file = fopen("input.txt","r"); 

	fscanf(in_file, "%d", &numcases); 

		for(i=0; i < numcases; i++)
		{	
			fscanf(in_file,"%d", &n);
			a = n-1;
	
			while(1) 
			{
				if (a == 1) 
				{
				printf("1\n");
				break;
				}
					else if (n % a == 0) 
					{
					printf("0\n");
					break;
					}
						a--; 
		}
	}

	fclose(in_file); 
}
