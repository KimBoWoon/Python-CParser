

#include <stdio.h>
#include <stdlib.h>

static fiboresult[31] = { 0 };
int fibo(int n);

void main(void) {
	FILE *input;
	int no_cases;
	int data;
	int i;

	fiboresult[1] = 1;
	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d", &data);
		fiboresult[data] = fibo(data);
		printf("%d\n", fiboresult[data]);

	}

	fclose(input);

}

int fibo(int n) {
	if (n <= 1) {
		return n;
	} else {
		if (fiboresult[n] != 0) {
			return fiboresult[n];
		} else {
			return fibo(n - 1) + fibo(n - 2);
		}
	}
}
