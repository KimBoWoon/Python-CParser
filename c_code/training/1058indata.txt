


#include <iostream>
#include <fstream>
#include <cstdlib>

#define MAX 20

using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if (instream.fail())
	{
		cerr << "Error";
		exit(1);
	}

	int no_cases;
	instream >> no_cases;

	for (int i = 0; i < no_cases; i++)
	{
		int n;
		instream >> n;

		int tuple[MAX];
		
		for (int i = 0; i < n; i++)
		{
			instream >> tuple[i];
		}
		tuple[n] = tuple[0];

		for (int i = 0; i<1000; i++) 
		{  
			for (int j = 0; j < n; j++) 
			{
				tuple[j] = tuple[j] - tuple[j + 1];
			
				if (tuple[j] < 0) tuple[j] = -tuple[j];
			}
			
			tuple[n] = tuple[0]; 
		}
				

		int count = 0;
		for (int k = 0; k < n; k++)
		{
			count += tuple[k];
		}

		if (count == 0)
		{
			cout << "ZERO" << endl;
		}

		else
		{
			cout << "LOOP" <<endl;
		}

	}

	return 0;
}
