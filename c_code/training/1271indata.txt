#include<stdio.h>  
#include<stdlib.h>
#define MAX 1000
void Goldbach(int n)  
{  
    int i,j,flag;  
	int prime[MAX];
	prime[0]=prime[1]=0;
	for(i = 2; i <= n; i++) {
        for(j = 2; j <= i-1; j++) {
            if(i % j == 0) {
				prime[i]=0;
                break;
            }
        }
        if(i == j) {
			prime[i]=i;
        }
	}
	for(i=(n/2)+1; i>1; i--){
		for(j=(n/2); j<n; j++){
			if(prime[i]+prime[j]==n){
				printf("%d %d\n",i,j);
				i=1;
			}
				
		}
	}
	
}
void main()
{	
	int a,testcase,n;
	FILE*file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &testcase);
	for(a=0; a<testcase; a++)
	{
		fscanf(file, "%d", &n);
		Goldbach(n);

	}	
	fclose(file);	
	
}
