
#include <iostream>
#include <fstream>
#include <cmath>
using namespace std;
int cnum(int a, int b){
	int count = 0;
	while (a != 0){
		a /= b;
		count += 1;
	}
	return count;
}
int check(int a[],int number, int cnumber,int count){
	int c = pow(count, cnumber-1);
	for (int i = cnumber - 1; i >= 0; i--){
		a[i] = number / c;
		number %= c;
		c /= count;
	}
	for (int i = 0; i < cnumber / 2; i++){
		if (a[i] != a[cnumber-1-i])
			return 0;
	}
	return 1;
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n,number,cnumber,count;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> number;
		count = 2;
		while (count<=64){
			cnumber = cnum(number, count);
			int* num = new int[cnumber];
			if (check(num, number, cnumber, count)){
				cout << 1 << endl;
				delete num;
				break;
			}
			delete num;
			count += 1;
		}
		if (count > 64)
			cout << 0 << endl;
	}
	return 0;
}
