#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

int palindrome(char a[],int start, int end)
{
	if(start <= end)
	{
		if(strlen(a)==1)
			return 1;
		else if(a[start] == a[end])
			return palindrome(a,start+1, end-1);
		else
			return 0;
	}
	else
		return 1;
}

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일을 열 수 없습니다." << endl;
		return 0;
	}
	int testcase;
	instream >> testcase;
	for(int i=0; i<testcase; i++)
	{
		int start = 0; 
		char a[1000];
		instream >> a;
		int end = strlen(a);
		cout << palindrome(a, start, end-1) << endl;
	}
	return 0;
}
