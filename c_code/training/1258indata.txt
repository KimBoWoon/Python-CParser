




#include<iostream>
#include<fstream>
#include<cstdlib>


bool primeNumber(int num);

int main(int argc, char* argv[]){
	
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;
	
	for(int i = 0; i < times; i++){

		int num;
		is>>num;

		int a = 0;
		int b = 0;
		int c = 0;
		int m_c=1000;
		int j = 2;
		
		while(num-j > 0){
			if(primeNumber(j))
				if(primeNumber(num-j)){
					c = abs(j-(num-j));

					if(m_c>c){
						a = j;
						b = num-j;
						m_c = c;
					}
					
				}
				j++;
		}

		std::cout<<a<<' '<<b<<std::endl;
	}
}

bool primeNumber(int num){
	
	for(int i = 2; i <=num/2; i ++){
		if(num%i == 0)
			return false;
	}
	return true;




}
