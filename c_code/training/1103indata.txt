#include<iostream>
#include<fstream>
using namespace std;
#define G(n) n*(n+1)/2
int main()
{
	ifstream fs;
	fs.open("input.txt");
    int a=1,b=1,c=1,n,k,s[1001]={0};
 
    for(a=1;G(a)+G(b)+G(c)<=1000;a++)
    {
        for(b=1;G(a)+G(b)+G(c)<=1000;b++)
        {
            for(c=1;G(a)+G(b)+G(c)<=1000;c++)
                s[G(a)+G(b)+G(c)]=1;
            c=1;
        }
        b=1;
    }
    fs>>n;
    while(n--)
    {
        fs>>k;
        cout<<s[k]<<"\n";
    }
}
