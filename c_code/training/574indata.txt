#include <stdio.h>
#include <string.h>
#include <stdlib.h>

char arr[12] = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l'};
char permuteList[13];

int factorial(int n)
{
	if(n<=1) return 1;
	return n*factorial(n-1);
}

void popList(int index, int n)
{
	int i;
	if(index == strlen(permuteList)-1) permuteList[index] = NULL;
	else for(i=index; i<n; ++i)	permuteList[i] = permuteList[i+1];
}

void makeResult(int n, int k, int index)
{
	int fac = factorial(n-index);
	int life = k/fac, etc = k%fac;
	int i = 0;

	if(n==index) {
		printf("%c", permuteList[0]);
		return;
	}
	if(etc==0 && life != 0) { life -= 1; etc = fac; } 
	printf("%c", permuteList[life]);
	popList(life, n);
	makeResult(n, etc, index+1);
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testNum, n, k;
	int i, j;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		fscanf(inFile, "%d %d", &n, &k);
		for(j=0; j<n; ++j) permuteList[j] = arr[j];

		makeResult(n, k, 1);
		printf("\n");
	}
	
	fclose(inFile);
}
