 

 
#include <stdio.h>
#include <limits.h>

static long int value; 
void henry(int a, long int b, long int unit); 
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test); 
    while(test--) 
    {
        int a; 
        int i;
        long int b; 
        fscanf(fp,"%d %ld",&a,&b);
        for(i=(int)((double)b/a);i<=b;i++) 
        {
            if(a*i >= b)
            {
                break;
            }
        }
        henry(a,b,i);
        printf("%ld\n",value);
    }    
    fclose(fp);
    return 0;
}

void henry(int a, long int b, long int unit)
{
    int _a = a*unit - b,i;
    long int _b = b*unit;
    if(_a == 0) 
    {
        value = unit;
    }
    else 
    {
        for(i=(int)((double)_b/_a);i<=_b;i++)
        {
            if(_b <= _a*i)
            {
                break;
            }
        }
        henry(_a,_b,i);
    }
}
