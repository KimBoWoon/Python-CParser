#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream inStream;
	int gatsu=0;	

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"wrong\n";
		exit(1);
	}
	inStream>>gatsu;	
	int num = 0;		
	int judge = 0;		
	for(int i=0;i<gatsu;i++)
	{
		inStream>>num;
		judge = 0;
		for(int j=2;j<num;j++)  
		{
			if(num%j==0)
			{
				judge = 1;
				break;
			}
			else
				judge = 0;
		}

		if(judge==1)
		{
			cout<<"0"<<endl;    
		}
		else
			cout<<"1"<<endl;    
	}

	return 0;
}
