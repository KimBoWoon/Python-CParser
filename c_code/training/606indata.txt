#include <stdio.h>
#include <stdlib.h>

int factorial(int );

int main()
{
   FILE *fp;
   int i;
   int no_cases,H,W,N;

   fp=fopen("input.txt","r");
   if(fp==NULL) exit(1);

   fscanf(fp, "%d",&no_cases);

   for(i=0; i<no_cases; i++){
	   int num;
	   fscanf(fp,"%d",&num);

	   printf("%d\n",factorial(num));
   }
   return 0;
   fclose(fp);
}

int factorial(int a)
{
	if(a<=1 )
		return 1;
	else
		return a*factorial(a-1);
}
