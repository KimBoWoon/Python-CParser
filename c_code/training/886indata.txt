

#include <stdio.h>

int fibonacci(int n);

int main(void)
{
	FILE *f = fopen("input.txt", "r");
	if (f == NULL) {
		printf("ERROR\n");
		exit(1);
	}

	int testCase = 0; 
	int inputNum = 0; 

	fscanf(f, "%d", &testCase); 

	while (testCase--) {

		fscanf(f, "%d", &inputNum); 

		printf("%d\n", fibonacci(inputNum));

	}

	fclose(f);
	return 0;
}

int fibonacci(int n) {
	if (n <= 1) 
		return n;
	else 
	{
		return fibonacci(n - 1)+fibonacci(n - 2);
	}
}
