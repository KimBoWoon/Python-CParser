#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int IsPrime(int n)
{
	int count = 0;
	int i;
	for (i = 1; i<n; i++)
	{
		if (n % i == 0)
			count++;
	}
	if (count == 1)
		return 1;
	else
		return 0;
}
int primeSum(int *prime, int num, int data)
{
	int i, j;
	int count = 1;
	int ans[168] = {0};

	for (i = 0; i<num; i++)
	{
		for (j = 0; j<num; j++)
		{
			if (data == (prime[i] + prime[j]))
			{
				ans[count] = prime[i];
				count++;
			}
		}
	}

	count--;
	if (count % 2 == 0)
		printf("%d %d\n", ans[count / 2], data - ans[count / 2]);
	else
		printf("%d %d\n", ans[count / 2 + 1], data - ans[count / 2 + 1]);

	return 0;


}
void main(){

	FILE *in_file;

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);
	
	int j;
	int test;

	fscanf(in_file, "%d", &test);

	for (j = 0; j < test; j++)
	{
		int data;
		int i;
		int prime[168] = { 0 };
		int num = 0;

		fscanf(in_file, "%d", &data);

		for (i = 1; i < data; i++)
		{
			if (IsPrime(i) == 1)
			{
				prime[num] = i;
				num++;
			}
		}

		primeSum(prime, num, data);

	}
	fclose(in_file);
}
