
#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main(){

	ifstream inStream;
	inStream.open("input.txt");

	int num_case;

	inStream>>num_case;

	int cnt1, cnt2;		
	
	for(int i = 0 ; i<num_case; i++)
	{
		int err = 0;

		cnt1 = 0;
		cnt2 = 0;
		
		string mk;			
		inStream >> mk;

		for(int i = 0; i< mk.length(); i++)		
		{

			if(mk.at(i) == '(')
				cnt1++;
			else if(mk.at(i) == ')')
				cnt2++;
			else
				continue;
		}
		if(mk.at(mk.length()-1) =='(' || mk.at(0) == ')')		
			cout<< "NO"<<endl;
		else if(cnt1 == cnt2)
			cout <<"YES"<<endl;
		else
			cout <<"NO"<<endl;
	}

	inStream.close();

	
}
