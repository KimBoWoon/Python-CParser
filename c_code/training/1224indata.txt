

#include <iostream>
#include <fstream>

using namespace std;

int prime(int num)     
{
    for(int j=2; j*j<=num; j++) 
        if(num%j==0) return 0; 

    return 1;
}

int main()
{
    
    ifstream file;
    
    file.open("input.txt");
    
    if(file.is_open()==false) 
        cout<<"입력파일이 없습니다."<<endl;
    
    else
    {
        int case_num;
        file>>case_num;
        
        for(int i=0; i<case_num; i++)
        {
            int n;
            file>>n;

            int prime_num=0;
            
            for(int i=2; i<=n/2; i++)  if(prime(i) & prime(n-i)) prime_num=i;    

            cout<<prime_num<<" "<<(n-prime_num)<<endl;

            
        }
        
    }
    
    file.close();
    
}
