  



#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b); 
int main()
{
	ifstream inFile("input.txt");  
	int no_cases=0;
	inFile>> no_cases;
	for(int l =0; l < no_cases; l++){
		int a=0,b=0,c=0,d=0;
		inFile>>a>>b>>c>>d;
		int gc=gcd(b,d); 
		int bunza=0,bunmo=0;
		int tmpa=a*d/gc; int tmpb=b*d/gc; int tmpc=c*b/gc; int tmpd=d*b/gc;
		bunza= tmpa+tmpc; bunmo=tmpb;
		cout<<bunza/gcd(bunza,bunmo)<<" "<<bunmo/gcd(bunza,bunmo)<<endl;
	}

	return 0;
}

int gcd(int a, int b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
