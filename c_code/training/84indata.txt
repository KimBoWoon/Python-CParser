#include <iostream>
#include <fstream>

using namespace std;
int main()
{
	ifstream infile;
	int num[100000], testCase, j;
	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num[i];
		if (num[i] < 2)
			cout << "0\n"; 
		for (j = 2; j<num[i]; j++)
		{
			if (num[i] % j == 0)
			{
				cout << "0\n"; 
				break;
			}
		}
		if (num[i] == j)
			cout << "1\n"; 
	}
	infile.close();

	return 0;
}
