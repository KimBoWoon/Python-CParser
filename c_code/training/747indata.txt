
#include<iostream>
#include<fstream>
#include<cstring>
#define MAX 1001

using namespace std;

int isPalind(char* str,int start,int end){
	if (start >= end) return 1;
	if (str[start] != str[end]) return 0;
	return isPalind(str, start + 1, end - 1);
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in)return 1;

	int n;
	in >> n;
	for (int i = 0; i < n; i++){
		char str[MAX];
		in >> str;
		cout << isPalind(str, 0, strlen(str) - 1) << endl;
	}

	in.close();
	return 0;
}
