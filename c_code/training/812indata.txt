#include<stdio.h>
#include<stdlib.h>

int fibonacci(int a)
{
	if(a<=1)
		return a;

	else
		return fibonacci(a-1)+fibonacci(a-2);
}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;
	int n;

	inFile = fopen("input.txt", "r");
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);

	for(i=0; i<numCases; i++)
	{
		fscanf(inFile, "%d", &n);

		printf("%d\n", fibonacci(n));

	}

	fclose(inFile);
}
