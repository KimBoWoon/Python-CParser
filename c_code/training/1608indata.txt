#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable:4996)

int matrix[501][501];
int s[501];
int c[501];

int play(int size);


int main() {
	int numTestCases;
	FILE *f;
	int i;
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		int j;
		int result;
		int cases;

		fscanf(f, "%d", &cases);
		for (j = 1; j <= cases; j++) {
			fscanf(f, "%d", &c[j]);
			s[j] = s[j - 1] + c[j];
		}
		result = play(cases);
		printf("%d\n", result);
	}
	return 0;
}

int play(int size) {
	int diagonal, i, j, k;
	int min = 0;
	int temp;

	for (diagonal = 1; diagonal <= size-1; diagonal++) {
		for (i = 1; i <= size - diagonal; i++) {
			j = i + diagonal;
			for (k = i; k < j; k++) {
				temp = matrix[i][k] + matrix[k + 1][j] + s[j] - s[i-1];
				if (k == i)
					matrix[i][j] = temp;

				if (matrix[i][j] >  temp)
					matrix[i][j] = temp;
			}
		}
	}
	return matrix[1][size];
}
