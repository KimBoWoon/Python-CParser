#include <iostream>
#include <fstream>
using namespace std;

int GCD(int a,int b) 
{
	if(b==0)
		return a;
	else
		return GCD(b,a%b);

}
int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일을 열 수 없습니다." << endl;
		return 0;
	}
	int case_number;
	instream >> case_number;
	for(int i=0; i<case_number; i++)
	{
		int sonA,momA,sonB,momB;
		int newMomA,newMomB,newSonA,newSonB;
		int SonResult, MomResult;
		instream >> sonA >> momA >> sonB >> momB;
		newMomA = momA;
		newMomB = momB;
		newSonA = sonA;
		newSonB = sonB;

		if(GCD(momA,sonA)>1){ 
			newMomA /= GCD(sonA,momA);
			newSonA /= GCD(sonA,momA);
		}
		if(GCD(momB,sonB)>1){
			newMomB /= GCD(sonB,momB);
			newSonB /= GCD(sonB,momB);
		}
		
		SonResult = newSonB*newMomA/GCD(newMomA,newMomB) + newSonA*newMomB/GCD(newMomA,newMomB) ; 
		MomResult =	GCD(newMomA,newMomB)*newMomA/GCD(newMomA,newMomB)*newMomB/GCD(newMomA,newMomB); 

		int NewSonResult=SonResult;
		int NewMomResult=MomResult;
		if(GCD(SonResult,MomResult)>1)
		{
			NewSonResult /= GCD(SonResult,MomResult);
			NewMomResult /= GCD(SonResult,MomResult);
		}
		cout << NewSonResult << " " << NewMomResult << endl;

	}
}
