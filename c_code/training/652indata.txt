

#include <stdio.h>
#include <stdlib.h>

int factorial(int num)
{
	if(num<=1)
		return 1;
	else
		return num*factorial(num-1);
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int num,result;
		fscanf(file,"%d",&num); 

		printf("%d\n",factorial(num));
	}
	fclose(file); 
}
