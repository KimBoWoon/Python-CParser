
#include<iostream>
#include<fstream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<vector>
#include<utility>
using namespace std;

int main()
{
	ifstream fs;
	fs.open("input.txt");
    int n;
    fs>>n;
    while(n--)
    {
        int k;
        vector<vector<int> > v;
        vector<pair<int,int> > m;
        m.push_back(make_pair(-1,-1));
        fs>>k;
        v.resize(100001);
        for(int i=0;i<k;i++)
        {
            int x,y;
            fs>>x>>y;
            v[x].push_back(y);
        }
        for(int i=0;i<k;i++)
            sort(v[i].begin(),v[i].end());
        int cury=0;
        for(int i=0,j=0;j<k;i++)
        {
            if(!v[i].size())continue;
            if(abs(v[i][0]-cury)<abs(v[i][v[i].size()-1]-cury))
            {
                int len=v[i].size();
                for(int r=0;r<len;r++)
                {
                    m.push_back(make_pair(i,v[i][r]));
                    j++;
                    cury = v[i][r];
                }
            }
            else
            {
                int len=v[i].size();
                for(int r=len-1;r>=0;r--)
                {
                    m.push_back(make_pair(i,v[i][r]));
                    j++;
                    cury = v[i][r];
                }
            }
        }
        fs>>k;
        for(int i=0;i<k;i++)
        {
            int w;
            fs>>w;
            printf("%d %d\n",m[w].first,m[w].second);
        }
    }
}
