
#include <iostream>
#include <fstream>
#include <cstring>

using namespace std;

bool isPalindrome(char str[], int head, int tail) {
	if (tail <= head) {
		return true;
	}
	else if (str[head] != str[tail]) {
		return false;
	}
	else {
		return isPalindrome(str, head + 1, tail - 1);
	}
}

int main(void) {
	ifstream in("input.txt");

	int t;

	in >> t;

	for (int testcase = 0; testcase < t; testcase++) {
		char str[1002];


		in >> str;

		bool result;
		result = isPalindrome(str, 0, strlen(str) - 1);


		cout << result << endl;
	}

	return 0;
}
