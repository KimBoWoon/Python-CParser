#include <stdio.h>
#include <stdlib.h>
int recursiveEuclidean(int n1, int n2);

void main()
{
	FILE *f;
	int count;	
	int i;		
	int num1, num2;	

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);	
	for(i=0; i<count; i++)
	{
		int result;		
		fscanf(f, "%d %d", &num1, &num2);	
		result = recursiveEuclidean(num1, num2);	
		printf("%d\n", result);	
	}
}
int recursiveEuclidean(int n1, int n2)
{
	int temp;
	if(n2>n1)	
	{
		temp = n1;
		n1 = n2;
		n2 = temp;
	}
	if(n2 == 0)		
		return n1;	
	else
		return recursiveEuclidean(n2, n1%n2);	
}
