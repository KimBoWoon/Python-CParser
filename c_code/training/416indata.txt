
#include<iostream>
#include<fstream>
using namespace std;

int gcd(int a, int b){
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int main(){
	ifstream File;
	int numCase = 0;
	int num1 = 0, num2 = 0;
	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> num1 >> num2;
		int ans = gcd(num1, num2);
		cout << ans << endl;
	}
	File.close();
	return 0;
}
