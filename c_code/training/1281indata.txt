#include<stdio.h>
#include<string.h>

int check_prime(int a)
{
	int num=1, j;

	for(j=2; j*j <= a; j++)
		if(a%j == 0) num = 0;

	return num;
}

void main()
{
	FILE* inFile = fopen("input.txt", "r");
	int testNum, n;
	int i, j;
	int result;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i)
	{
		result = 0;
		fscanf(inFile, "%d", &n);
		if(n==4) { printf("2 2\n"); continue; }
		for(j=3; j<=n/2; j+=2)
		{
			if(!check_prime(j)) continue;
			else if(!check_prime(n-j)) continue;
			else{
				if((n-2*j)>=(n-2*result)) continue;
				result = j;
			}
		}
		printf("%d %d\n", result, n-result);
	}
	fclose(inFile);
}
