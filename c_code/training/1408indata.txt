

#include<iostream>
#include<fstream>
using namespace std;

int main()
{
	ifstream inStream;
	int testcase;   

	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "input file open failed\n";
		return 0;
	}

	inStream >> testcase;

	int height;    
	int width;	   
	int number;    

	for (int i = 0; i < testcase; i++)
	{

		inStream >> height >> width >> number;

		cout << (number - 1) % height + 1;       

		if (((number - 1) / height + 1) > 10)       

			cout << (number - 1) / height + 1<<endl;
		else
			cout << 0 << (number - 1) / height + 1<<endl;

	}
	return 0;
}
