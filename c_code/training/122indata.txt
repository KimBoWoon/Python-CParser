

#include<iostream>
#include<fstream>

using namespace std;

class Mecathy
{
public:
	int mecathy_method(int num)
	{
		if (num > 100)
		{
			return (num - 10);
		}
		else
			return mecathy_method(mecathy_method(num + 11));
	}
};

int main()
{
	ifstream inStream;
	int testcase;
	Mecathy mec;

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "input file open failed";
	}

	inStream >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		int mecnum;
		inStream >> mecnum;

		cout << mec.mecathy_method(mecnum) << endl;
	}

	return 0;
}
