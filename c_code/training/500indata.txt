#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b){
	if (b == 0) 
		return a;
	else		
		return gcd(b, a%b);
}

int main(void)
{
	FILE *file;
	int cases,num1,num2;
	
	file=fopen("input.txt","r");
	fscanf(file,"%d",&cases);

	while(cases--){
		fscanf(file,"%d %d",&num1,&num2);
		printf("%d\n",gcd(num1,num2));
	}
	
	fclose(file);
}
