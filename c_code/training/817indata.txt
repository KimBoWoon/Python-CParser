

#include<stdio.h>
#include<stdlib.h>

int fibonacci(int n){
	
	if(n==0)	
		return 0;
	else if(n==1)
		return 1;
	else		
		return fibonacci(n-1)+fibonacci(n-2);
}

void main(void){
	FILE *in_file;

	int no_case;
	int num;
	int i;
	
	in_file = fopen("input.txt" , "r");		
	if(in_file == NULL) exit(1);

	fscanf(in_file , "%d", &no_case);

	for(i=0; i<no_case; i++)
	{
		fscanf(in_file , "%d", &num);

		printf("%d\n",fibonacci(num));	
	}

	fclose(in_file);
}
