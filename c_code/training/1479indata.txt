#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int H, W;	
		int N;	
		int guestH, guestW; 
		int roomNumber;	

		inStream >> H >> W >> N;
		
		if (N%H == 0){
			guestH = H;
			guestW = N / H;
		}

		else{
			guestH = N%H;
			guestW = (N / H) + 1;
		}

		roomNumber = (guestH * 100) + guestW;
		cout << roomNumber << endl;
	}

	inStream.close();
	return 0;
}
