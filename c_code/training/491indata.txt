#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int num1, num2;

		inStream >> num1 >> num2;

		cout << gcd(num1, num2) << endl;
	}

	inStream.close();
	return 0;
}

int gcd(int a, int b) { 
	if (b == 0) 
		return a; 
	
	else  
		return gcd(b, a%b);
}
