

#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;

void reverse(char *a, int i, int j);
static bool ok = true;

int main()
{
	ifstream inStream;
	inStream.open("input.txt");

	int numTestCases;

	inStream >> numTestCases;

	for (int i = 0; i < numTestCases; i++) {
		ok = true;
		char *origin = new char[1001];
		inStream >> origin;

		reverse(origin, 0, strlen(origin) - 1);
		if (ok == true) cout << "1" << endl;
		else cout << "0" << endl;
		delete origin;
	}
	return 0;
}

void reverse(char *a, int i, int j){
	if (i < j){
		if (a[i] != a[j]){
			ok = false;
			return;
		}
		reverse(a, i + 1, j - 1);
	}
}
