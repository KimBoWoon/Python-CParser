#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error";
		exit(1);
	}
	int no_cases;
	instream>>no_cases;

	for(int i=0 ; i<no_cases ; i++)
	{
		int data;
		instream>>data;

		int res =1;

		for(int j=2 ; j<data ; j++)
		{
			if(data % j ==0)
			{
				res = 0;
				cout<<res<<endl;
				break;
			}
		}
		if(res ==1)
			cout<<res<<endl;
	}
	return 0;
}
