#include <iostream>
#include <fstream>

using namespace std;

int main(){
	ifstream fin("input.txt");

	int nCase;
	fin >> nCase;

	for (int ci = 0; ci < nCase; ++ci){
		int n;
		fin >> n;
		bool r = true;
		for (int i = 2; i <= n / 2; ++i)
		{
			if (n % i == 0){
				r = false;
				break;
			}
		}

		cout << (r ? 1 : 0) << endl;
	}
}
