
#include <iostream>
#include <fstream>
using namespace std;
int factorial(int n){
	if (n <= 1)
		return 1;
	else
		return factorial(n - 1)*n;
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n,number;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> number;
		cout << factorial(number) << endl;
	}
	return 0;
}
