#include <fstream>

using namespace std;

int Factorial(int num)
{
	if (num == 0)
		return 1;
	else
		return num*Factorial(num - 1);
}

void set(char *arrABC, int size)
{
	for (int i = 0; i<size; i++)
		arrABC[i] = 97 + i;
}

void modify(char *arr, int size, int delNum)
{
	for (int i = 0; i<size; i++)
		if (delNum == i)
		{
			cout << arr[delNum];
			for (int j = delNum; j<size - 1; j++)
				arr[j] = arr[j + 1];
			break;
		}
}
void permutation(int charNum, int num)
{
	int number = charNum;
	int temp = charNum - 1;
	int *ABC = new int[charNum];

	char *alphabet = new char[charNum];
	set(alphabet, charNum);

	num -= 1;
	for (int i = temp; i >= 1; i--)
	{
		ABC[temp - i] = num / Factorial(i);
		num %= Factorial(i);
	}
	ABC[temp] = 0; 

	
	
	for (int i = 0; i<charNum; i++)
	{
		modify(alphabet, number--, ABC[i]);
	}
	
	cout << endl;

	delete[] alphabet;
	delete[] ABC;
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		int charNum, num;
		instream >> charNum >> num;
		permutation(charNum, num);
	}

	instream.close();
	return 0;
}
