#include<fstream>
#include<iostream>

using namespace std;

int main()
{
	ifstream file("input.txt");
	int testCase;
	int floor, room, guest, roomNum;

	file >> testCase;
	for (int i = 0; i < testCase; i++){
		file >> floor >> room >> guest;

		roomNum = guest / floor;
		if (guest % floor == 0){
			cout << floor;
		}
		else{
			cout << guest % floor;
			roomNum += 1;
		}

		if (roomNum < 10)
			cout << "0" << roomNum << endl;
		else {
			cout << roomNum << endl;
		}
	}

	file.close();
	return 0;
}
