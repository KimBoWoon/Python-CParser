
#include <iostream>
#include <fstream>

using namespace std;
int theGreatest(int a, int b){
	if (b == 0)
		return a;
	else
		return theGreatest(b, a%b);
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n,number1,number2;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> number1 >> number2;
		cout << theGreatest(number1, number2) << endl;
	}
}
