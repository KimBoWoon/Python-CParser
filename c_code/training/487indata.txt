

#include <iostream>
#include <fstream>

using namespace std;

int euclid(int num1, int num2);

int main()
{
	ifstream infile;
	int testCase;
	int num1, num2; 

	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num1 >> num2;

		cout << euclid(num1,num2) << endl; 
	}

	infile.close();

	return 0;
}

int euclid(int n1, int n2)
{
	if (n2 == 0) 
		return n1;
	else if (n1 == 1 || n2 == 1) 
		return 1;
	else
		return euclid(n2, n1 % n2);

}
