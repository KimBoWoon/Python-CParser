
#include<iostream>
#include<string>
#include<fstream>

using namespace std;



int euclid(int a, int b){
	int r;
	do{
		r = a%b;
		a = b;
		b = r;
	} while (r != 0);
	return a;
}

int fraction(int lcm, int *numBuff){
	int firstLcmLess = lcm/numBuff[1];
	int secondLcmLess = lcm / numBuff[3];
	int sumBuff = (numBuff[0] * firstLcmLess) + (numBuff[2] * secondLcmLess);

	return sumBuff;
}

int main(){
	ifstream f;
	f.open("input.txt");
	int numCount = 0, gcd = 0, numBuff[4];
	f >> numCount;

	for (int i = 0;i<numCount;i++){
		for (int j = 0;j<4;j++){
			f >> numBuff[j];
		}
		int lcm = (numBuff[1] * numBuff[3]) / euclid(numBuff[1], numBuff[3]);
		int upperNum = fraction(lcm, numBuff);
		if (upperNum > lcm){
			gcd = euclid(upperNum, lcm);
		}
		else{
			gcd = euclid(lcm, upperNum);
		}
		cout << upperNum/gcd << " " << lcm/gcd << endl;
	}
	return 0;
}
