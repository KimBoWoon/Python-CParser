
#include <iostream>
#include <fstream>
using namespace std;

void gcd(unsigned int a, unsigned int b){
	if(0 == b) cout<<a<<endl; 
	else if(0 == a) cout<<b<<endl;
	else gcd(b,a%b); 
}

int main(){
	ifstream in;
	in.open("input.txt");

	short t;
	in>>t;

	while(t--){
		unsigned int a,b;
		in>>a>>b;
		gcd(a,b);
	}
	in.close();
}
