#include <iostream> 
#include <fstream>
#define MAX 13

using namespace std;

char digits[MAX] = "abcdefghijkl";
int roomAcount[MAX] = {};

void initRoom(){
	for (int i = 0; i < MAX; i++){
		roomAcount[i] = 0;
	}
}

int getFact(int a){
	if (a == 1 || a == 0)
		return 1;
	return a * getFact(a-1);
}

int getRoomNum(int cnt,int length){
	int i = 0;
	int i_cnt = 0;
	int i_break = 0;

	while (i_break <= MAX){
		if (roomAcount[i] == 0){
			if (i_cnt == cnt){
				return i;
			}
			i_cnt++;
		}
		if (i + 1 == length)
			i = 0;
		else i++;
		i_break++;
	}
	return 0;
}

int calcPermuation(int number, int length, int realLength){
	int a, cnt;
	int roomNum;

	a = number;
	cnt = 0;
	while (1){
		if (a <= getFact(length))
			break;
		a = a - getFact(length);
		cnt++;
	}
	roomNum = getRoomNum(cnt, realLength);
	roomAcount[roomNum] = 1;
	cout << digits[ roomNum ];

	return a;
}
int main()
{
	int testCase, length,number;
	
	
	ifstream file;
	file.open("input.txt");

	file >> testCase;
	for (int tCase = 0; tCase < testCase; tCase++){
		file >> length  >> number;
		initRoom();

		for (int i = 0; i < length; i++){
			number = calcPermuation(number,length-i-1,length);			
		}cout << endl;
	}
	return 0;
}
