

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int checkPrime(int data);

void main(void) {
	FILE *input;
	int no_cases;
	int i,j;
	int data;

	input = fopen("input.txt", "r");
	if (input == NULL) exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d", &data);
		printf("%d\n", checkPrime(data));
	}

	fclose(input);
}

int checkPrime(int data) {
	int j;
	for (j = 2; j <= (int) sqrt(data); j++) if (data % j == 0) return 0;
	return 1;
}
