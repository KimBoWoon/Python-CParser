
#include <iostream>
#include <fstream>
using namespace std;

unsigned int gcd(unsigned int a, unsigned int b)
{                  
	if(b == 0) return a; 
	else gcd(b, a%b);    
}

int main(){
	ifstream in;
	in.open("input.txt");

	unsigned short t;
	in>>t;

	while(t--)
	{
		unsigned short  a, b, c, d;
		unsigned int up, bottom, G;
		in>>a>>b>>c>>d;

		up = a*d + c*b;                  
		bottom = b*d;                    
		G = gcd(up, bottom);
		up /= G;                         
		bottom /= G;

		cout<<up<<" "<<bottom<<endl;          
	}
	in.close();
}
