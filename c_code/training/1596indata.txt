#include <iostream>
#include <cstdlib>
#include <fstream>
#define MAX_NUM 500
using namespace std;

int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num = 0, temp = 0, min = 0;
	int *arr;
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int dms[MAX_NUM][MAX_NUM] = {0,};
		inStream >> num; 
		arr = (int*)malloc(sizeof(int)*num); 
		for (int j = 0; j < num; j++) inStream >> arr[j]; 
		for (int j = 0, k = 0; j < num - 1; j++)
		{
			k = j + 1;
			dms[j][k] = arr[j] + arr[k];
		}
		for (int j = 0,q = 0; q < num - 2; q++)
		{
			for (int k = q + 2, n = j; k < num; k++, n++)
			{
				temp = 0;
				for (int l = n, m = l + 1; l < k; l++, m++)
				{
					temp += arr[l]; 
					min = dms[n][l] + dms[m][k];
					if (l == n)
						dms[n][k] = min;
					else if (dms[n][k] > min)
						dms[n][k] = min;
				}
				temp += arr[k]; 
				dms[n][k] += temp; 
			}
		}
		cout << dms[0][num-1] << endl;

		delete arr; 
	}
	inStream.close();
	return 0;
}
