






#include<iostream>
#include<fstream>
using namespace std;

int arr[51] = { 0 };
void Paren(char str[], int index)
{
	if (str[0] == ')') { 
		arr[0] = -1;
		return;
	}
	if(str[index+1]== 0 && str[index]=='(') 
	{
		arr[index] = -2;
		return;
	}
	
	if (str[index] == '(') { 
		arr[index] = 1;
		Paren(str, index + 1);
	}
	else if (str[index] == ')') { 
		arr[index] = -1;
		Paren(str, index + 1);
	}
	else
		return;

	


}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int sum = 0;
	char pa[51] = { 0 };
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> pa;
		Paren(pa, 0);

		for (int i = 0; i < 51; i++) 
			sum += arr[i];

		if (sum == 0)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;

		for (int i = 0; i < 51; i++)
			arr[i]=0;
		sum = 0;
	}
	return 0;
}
