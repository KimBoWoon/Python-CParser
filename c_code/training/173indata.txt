

#include <stdio.h>
#include <stdlib.h>

int McCarthy(int num)
{
	if(num>100)
		return (num-10);
	else
		return McCarthy(McCarthy(num+11));
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int num;
		fscanf(file,"%d",&num); 

		printf("%d\n",McCarthy(num));
	}
	fclose(file); 
}
