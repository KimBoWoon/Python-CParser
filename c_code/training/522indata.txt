

#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)

int fact(int n);

void main()
{
	FILE* inFile;
	int x,y,i,l, numTestCases, StringLength, numbers;
	char string[13] = "abcdefghijkl";
	char temp[13];
	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &numTestCases);
	for (i = 0; i < numTestCases; i++)
	{
		fscanf(inFile, "%d %d", &StringLength, &numbers);
		strcpy(temp, string);
		x = numbers - 1;
		while (StringLength)
		{
			printf("%c", temp[x / fact(StringLength - 1)]);
			for (y = 0; y < StringLength; y++)
			{
				if (temp[y] != temp[x / fact(StringLength - 1)])
				{
					temp[l++] = temp[y];
				}
			}
			l = 0;
			x %= fact(StringLength - 1);
			StringLength--;
		}
		printf("\n");
	}
}

int fact(int n)
{
	if (n <= 1)
	{
		return 1;
	}
	else
	{
		return n*fact(n - 1);
	}
}
