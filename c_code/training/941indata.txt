#include <iostream>
#include <algorithm>
#include <cstdio>
#include <memory.h>

using namespace std;

int arr[1001];
int arr2[1001][1001];

int maxScore(const int l, const int r)
{
	if (r < l) return 0;
	if (l == r) return arr[l];

	int &ret = arr2[l][r];
	if (ret != -1) return ret;
	ret = 0;

	ret = max(ret, arr[l] + min(maxScore(l + 2, r), maxScore(l + 1, r - 1)));
	ret = max(ret, arr[r] + min(maxScore(l + 1, r - 1), maxScore(l, r - 2)));

	return ret;
}

int main()
{
	FILE *file;
	file = fopen("input.txt", "r");
	int t;
	fscanf(file,"%d", &t);
	while (t--)
	{
		memset(arr2, -1, sizeof(arr2));
		int n; 
		fscanf(file,"%d", &n);
		for (int i = 0; i<n; ++i) fscanf(file,"%d", &arr[i]);
		printf("%d\n", maxScore(0, n - 1));
	}
	return 0;
}
