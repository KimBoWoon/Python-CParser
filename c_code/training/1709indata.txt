

#include <iostream>
#include <fstream>
#include <vector>

using namespace std;
bool palindrome(vector <int> a){
	for( int i=0,j=a.size()-1; i<j; i++,j--)
		if( a[i]!=a[j]) return false;
	return true;
}

bool isPalindrome(int n){
	for( int i = 64;i>=2;i--){
		vector <int> arr;
		int cpy_n=n;
		while (cpy_n!=0){
			arr.push_back(cpy_n%i);
			cpy_n/=i;
		}
		if( palindrome(arr)) return true;
	}
	return false;
}

int main(){
	ifstream instream;
	instream.open("input.txt");

	int no_cases;
	instream>>no_cases;

	while(no_cases--){
		int n;
		instream>>n;
		cout<<isPalindrome(n)<<endl;
	}
	instream.close();
}
