

#include <stdio.h>
#define AbsSubtract(a,b) (a >= b ? a - b : b - a) 
int checkDucciSquence(int *arr, int n) 
{
	int i, total = 0, t, j = 0;
	while (j < 342) 
	{
		t = arr[0]; 
		for (i = 0; i < n; i++){
			if (i == n - 1) arr[i] = AbsSubtract(arr[i], t);
			arr[i] = AbsSubtract(arr[i], arr[i + 1]);
		}
		j++;
	}
	total = 0; 
	for (i = 0; i < n; i++) total += arr[i]; 
	if (!total) return 0; 
	else return 1; 
}

int main()
{
	int t, n, arr[15] = { 0 }, i;
	FILE *fp;
	if ((fp = fopen("input.txt", "r")) == NULL) return 0; 
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%d", &n); 
		for (i = 0; i < n; i++) fscanf(fp, "%d", arr + i); 
		if (!checkDucciSquence(arr, n)) printf("ZERO\n"); 
		else printf("LOOP\n");

	}
	fclose(fp); 
	return 1;
}
