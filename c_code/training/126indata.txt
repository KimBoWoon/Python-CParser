


#include <iostream>
#include <fstream>
#include<cstdlib>
using namespace std;

int mac(int n); 

int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int x;
		inStream >> x;
		cout << mac(x) << endl; 

	}
	inStream.close();
	return 0;
}

int mac(int n){ 
	if (n>100) 
		return n - 10; 
	else 
		return mac(mac(n + 11));
}
