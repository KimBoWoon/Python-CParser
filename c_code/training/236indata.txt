


#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int GDC(int val1, int val2);

int main()
{
	ifstream inStream;
	int numTestCases;		
	int value1, value2, value3, value4;		

	inStream.open("input.txt");

	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> value1 >> value2 >> value3 >> value4;
		int bun = value1 * value4 + value2 * value3;		
		int mo = value2 * value4;							
		int gdc = GDC(bun, mo);
		cout << bun / gdc << " " << mo / gdc << endl;
	}

	inStream.close();
	return 0;
}

int GDC(int value1, int value2)
{
	if (value2 == 0)
	{
		return value1;
	}
	else
	{
		return GDC(value2, value1 % value2);
	}
}
