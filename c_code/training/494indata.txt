#include <stdio.h>

int gcd(int a, int b);

int main()
{
	FILE* fp = fopen("input.txt", "r");
	int testCaseNumber = 0;
	int i = 0;
	int a = 0;
	int b = 0;
	int result = 0;

	if (fp == NULL)
	{
		printf("file error\n");
	}

	fscanf(fp, "%d", &testCaseNumber);

	for (i = 0; i < testCaseNumber; i++)
	{
		fscanf(fp, "%d %d", &a, &b);
		result = gcd(a, b);
		printf("%d\n", result);
	}

	fclose(fp);
	return 0;
}

int gcd(int a, int b)
{
	if (b == 0)
	{	
		return a;
	}
	else
	{	
		return gcd(b, a % b);
	}
}
