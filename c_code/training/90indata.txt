
#include<iostream>
#include<fstream>
#include<cstdlib>

using namespace std;

int main(){
	ifstream infile;
	
	int num;
	int x;

	infile.open("input.txt");
	if(infile.fail()){
		cerr << "input file error" <<endl;
		exit(1);
	}

	infile >> num;
	for(int i=0;i<num;i++){
		infile >> x;

		int chk=1;
		for(int j=2;j<x;j++){
			if(x%j == 0)
				chk=0;
		}
		cout << chk << endl;
	}
	return 0;
}
