
#include <iostream>
#include <fstream>
using namespace std;
int checkzero(int str[], int count){
	int checkZero = 0;
	for (int i = 0; i < count; i++){
		if (str[i] == 0)
			checkZero++;
	}
	if (checkZero == count){
		cout << "ZERO" << endl;
		return 0;
	}
	return 1;
}
int checkloop(int str[], int check[],int count){
	int checkcount = 1;
	for (int i = 0; i < count; i++){
		if (str[i] != check[i]){
			checkcount--;
			break;
		}
	}
	if (checkcount == 1){
		cout << "LOOP" << endl;
		return 1;
	}
	else
		return 0;
}
int main(){
	ifstream in;
	in.open("input.txt");
	int n, count,fcase[15],check,first,cnt,checkcase[15];
	in >> n;
	for (int i = 0; i < n; i++){
		in >> count;
		for (int j = 0; j < count; j++){
			in >> fcase[j];
		}
		cnt=check = 1;
		while (check==1){
			first = fcase[0];
			for (int j = 0; j < count; j++){
				if (j == count - 1){
					if (fcase[j] >= first)
						fcase[j] = (fcase[j] - first);
					else
						fcase[j] = first - fcase[j];
				}
				else{
					if (fcase[j] >= fcase[j + 1])
						fcase[j] = (fcase[j] - fcase[j + 1]);
					else
						fcase[j] = fcase[j + 1] - fcase[j];
				}
				if (cnt == 1000){
					checkcase[j] = fcase[j];
				}
			}
			if (cnt>1000){
				if (checkloop(fcase, checkcase, count))
					break;
			}
			check = checkzero(fcase, count);
			cnt++;
		}
	}
	return 0;
}
