


#include <stdio.h>
#include <stdlib.h>

int fib(int n)
{
	if( n<= 1 )
		return n;
	else
		return fib(n-2) + fib(n-1);
}

void main(void)
{
	FILE *inFile;
	int i;
	int numCases;

	inFile = fopen("input.txt", "r");									
	if(inFile == NULL) exit(1);

	fscanf(inFile, "%d", &numCases);									

	for(i=0; i<numCases; i++)
	{
		int N = 0;														

		fscanf(inFile, "%d", &N);										

		printf("%d\n",fib(N));
	}

	fclose(inFile);														
}
