#include<iostream>
#include<fstream>
using namespace std;
int euclid(int val1,int val2);
int main(){
	ifstream fin;
	fin.open("input.txt");
	int num, val1,val2;
	fin >> num;
	for (int cnt = 0; cnt < num;cnt++){
		fin >> val1 >> val2;
		cout<<euclid(val1, val2)<<endl;
	}
}
int euclid(int val1,int val2){
	if (val2 == 0) return val1;		
	else return euclid(val2, val1%val2);		
}
