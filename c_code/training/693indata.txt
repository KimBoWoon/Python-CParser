 

 
#include <stdio.h>
#include <string.h>

int palindrome(char str[],int i,int j); 
int main()
{
    FILE *fp;
    int test;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    while(test--) 
    {
        char str[1000];
        fscanf(fp,"%s",str); 
        printf("%d\n",palindrome(str,0,strlen(str)-1)); 
    }
    fclose(fp); 
    return 0;
}
int palindrome(char str[],int i, int j) 
{
    if(j-i <= 0) 
    {
        return 1;
    }
    else 
    {
        if(str[i] == str[j]) 
        {
            return palindrome(str,++i,--j);
        }
        else 
        {
            return 0;    
        }
    }
}
