



#include <stdio.h>

int T, A[45], C[1001];
int main() {
	int i = 0, j = 0, x = 0, n;
	FILE* fp;

	if ((fp = fopen("input.txt", "rt")) == NULL) {
		return -1;
	}

	for (i = 1; i<45; i++) 
		A[i] = i*(i + 1) / 2;

	for (i = 1; i<45; i++) {
		for (j = i; j<45; j++) {
			for (x = j; x<45; x++) {
				if (A[i] + A[j] + A[x] < 1001) { 
					C[A[i] + A[j] + A[x]] = 1; 
				}
			}
		}
	}

	fscanf(fp, "%d", &T);
	for (i = 0; i<T; i++) {
		fscanf(fp, "%d", &n);
		printf("%d\n", C[n]); 
	}
	return 0;
}
