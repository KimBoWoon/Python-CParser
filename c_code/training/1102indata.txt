
#include<iostream>
#include<fstream>
#include<cstdio>
using namespace std;

int main(){
	ifstream File;
	int numCase = 0;
	int num = 0;
	bool flag = false;

	File.open("input.txt");
	File >> numCase;
	int arr[44] = { 0, };
	arr[0] = 1;
	for (int i = 1; i < 44; i++){
		arr[i] = arr[i - 1] + (i + 1);
	}

	for (int iter = 0; iter < numCase; iter++){
		File >> num;
		flag = false;
		for (int i = 0; i < 44; i++){
			if (flag) break;		
			for (int j = 0; j < 44; j++){
				if (flag) break;
				for (int k = 0; k < 44; k++){
					if (num == arr[i] + arr[j] + arr[k]){
						flag = true;
						break;
					}
				}
			}
		}
		if (flag)
			cout << '1' << endl;
		else
			cout << '0' << endl;
	}
	File.close();
	return 0;
}
