
#include <fstream>
#include <iostream>
using namespace std;

int factorial(int n){
	if (n == 0) return 1;
	else return n * factorial(n - 1);
}

int main(){

	int caseNum = 0;

	ifstream inputFile("input.txt");
	inputFile >> caseNum;

	for (int i = 0; i < caseNum; i++){
		int testNum = 0;
		inputFile >> testNum;

		cout << factorial(testNum) << endl;
	}

	return 0;
}
