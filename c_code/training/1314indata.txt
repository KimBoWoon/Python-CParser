

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;
int gcd(int a, int b); 
int T(int a, int b); 
int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int x, y;
		inStream >> x >> y;
		cout << T(x, y) << endl; 
	}
	inStream.close();
	return 0;
}

int T(int a, int b)
{
	int Gcd; 
	int dif = -1, i = 2, x =a, y = b; 
	while (dif != 0) 
	{
		dif = x*i - y;
		if (dif < 0)
			i++; 
		else if (dif > 0)
		{
			x = x*i - y; 
			y = y*i; 
			i++; 
			Gcd = gcd(x, y); 
			x = x / Gcd;
			y = y / Gcd;
		}
	}
	return i; 

}


int gcd(int a, int b) 
{
	if (a >= b) 
	{
		if (b == 0) 
			return a;
		else 
			return gcd(b, a%b);
	}
	else 
	{
		if (a == 0)
			return b;
		else
			return gcd(a, b%a);
	}
}
