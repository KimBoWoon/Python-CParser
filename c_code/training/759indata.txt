

#include <cstring>
#include <iostream>
#include <fstream>

int palindrome(char * str,int start, int end)
{
	if(start>=end)
		return 1;
	else if(str[start] != str[end])
		return 0;
	else
		return palindrome(str,start+1,end-1);
}

int main(int argc,char* argv[])
{
	std::ifstream file;
	file.open("input.txt");

	int test;
	file >>test;

	for(int i=0; i<test; i++)
	{
		char str[1000];
		file >> str;
		std::cout << palindrome(str, 0 , strlen(str)-1) << std::endl;
	}
	return 0;
}
