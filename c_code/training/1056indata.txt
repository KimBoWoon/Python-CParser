
#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일을 열 수 없습니다" << endl;
		return 0;
	}
	int testcase;
	instream >> testcase;
	for(int z=0; z<testcase; z++)
	{
		int tuple=0;
		instream >> tuple;
		int *arr = new int [tuple]; 
		int *temp = new int [tuple];

		for(int i=0; i<tuple; i++) 
		{
			instream >> arr[i];
		}
		int n=1000;
		
		while(n) 
		{
			int res=0;
			for(int i=0; i<tuple; i++) 
			{
				if(i==tuple-1)
				{
					temp[i] = abs(arr[tuple-1]-arr[0]); 
				}
				else
				{
					temp[i] = abs(arr[i]-arr[i+1]); 
				}
				res += temp[i];
			}
			if(res==0) 
			{
				cout << "ZERO " << endl; 
				break;
			}
			
			for(int i=0; i<tuple; i++) 
			{
				arr[i]=temp[i];
			}
			
			n--;
		}
		if(n==0)
			cout << "LOOP" << endl; 
	}
}
