
#include<iostream>
#include<fstream>

using namespace std;

int gcd(int a, int b){
	if (b == 0) return a;
	else gcd(b, a%b);
}

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;

	int n;
	in >> n;
	for (int i = 0; i < n; i++){
		int a, b;
		in >> a >> b;
		cout << gcd(a, b) << endl;
	}

	in.close();
	return 0;
}
