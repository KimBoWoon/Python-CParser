

#include <stdio.h>
int gcd(int n, int m) 
{
	if (!m) return n; 
	else return gcd(m, n%m); 
} 
int findHenry(int a, int b) 
{
	int boundary = b;
	int denominator = 2;
	int g;
	while (boundary <= 0xffffffff)
	{
		if (a == 1) return b; 

		if (a*denominator - b < 0) 
		{
			if (boundary * ++denominator <= 0xffffffff) continue;
			else break;
		}
		a = a*denominator - b; 
		b *= denominator;  
		g = gcd(b, a); 
		a /= g; 
		b /= g; 
		boundary *= denominator++; 
	}
	return b; 
}
int main()
{
	FILE* fp;
	int t;
	int a, b;
	if ((fp = fopen("input.txt", "r")) == NULL) return 0; 
	fscanf(fp, "%d ", &t); 
	while (t--) { 
		fscanf(fp, "%d %d", &a, &b); 
		printf("%d\n", findHenry(a, b)); 
	}
	fclose(fp); 
}
