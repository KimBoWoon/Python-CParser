





#include <iostream>
#include <fstream>

bool IsPrime(int testCase);	

int main(){
	std::ifstream inFile;	
	inFile.open("input.txt");

	int testCaseNum;	

	inFile>>testCaseNum;

	for(int i = 0; i < testCaseNum; i++){
		int testCase;	
		inFile>>testCase;	

		std::cout<<IsPrime(testCase)<<std::endl;
	}

}

bool IsPrime(int testCase){
	int i;
	for(i = 2; i < testCase; i++){
		if(testCase % i == 0)
			break;
	}
	if(i == testCase)
		return true;
	else
		return false;
}
