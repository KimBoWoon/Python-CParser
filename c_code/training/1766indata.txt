#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;
int isPalindrome(int *A, int x, int y);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int num;

		inStream >> num;
		
		int ans = 0;
		for (int j = 2; j <= 64; j++){
			int n = num;
			int A[21];
			int a = 0;
			while (n >= j){
				A[a] = n%j;
				n /= j;
				++a;
			}
			A[a] = n;

			if (isPalindrome(A, 0, a)){
				ans = 1;
				break;
			}
		}
		
		cout << ans << endl;
	}

	inStream.close();
	return 0;
}

int isPalindrome(int *A, int x, int y){
	if (y <= x)return 1;
	if (A[x] != A[y])return 0;
	return isPalindrome(A, x + 1, y - 1);
}
