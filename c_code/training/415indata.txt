#include <stdio.h>
#include <string.h> 
#include <stdlib.h>

int gcd(int a, int b);

void main(void)
 {
	 FILE *in_file;
	int no_cases; 
	int i,m,n;
	
	
	in_file = fopen("input.txt","r");
	if(in_file == NULL) exit(1);
	fscanf(in_file,"%d",&no_cases);
	for(i=0; i<no_cases; i++){
		fscanf(in_file,"%d%d",&m,&n);
		printf("%d\n", gcd(m,n));
		
	}
}

int gcd(int a, int b)
{
	if (b == 0) return a;
	else return gcd(b, a%b);
}
