

#include <iostream>
#include <cstdlib>
#include <fstream>

int fib(int n);

int main()
{
	std::fstream inputfile; 
	int testcase; 
	int n;

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> n;
		std::cout << fib(n) << std::endl;
	}

	return 0;
}

int fib(int n)
{
	if (n <= 1) return n;
	else return fib(n - 1) + fib(n - 2);
}
