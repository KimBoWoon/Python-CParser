

#include <stdio.h> 
int main()
{
	FILE *fp;
	int h, n, w, t, x, y; 
	fp = fopen("input.txt", "r"); 
	if (fp == NULL)
	{
		return 0;
	}
	fscanf(fp, "%d", &t); 
	while (t--)
	{
		fscanf(fp, "%d %d %d", &h, &w, &n); 
		if (n%h == 0) 
		{
			x = h; 
			y = n / h; 
		}
		else
		{
			x = n%h;
			y = n / h + 1;
		}
		printf("%d%02d\n", x, y); 
	}
	fclose(fp); 
	return 1;
}
