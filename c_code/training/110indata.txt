#include <iostream>
#include <fstream>

using namespace std;

int fn(int n){
	if (n > 100)
		return n - 10;
	else
		return fn(fn(n+11));
}

int main(){
	ifstream fin("input.txt");

	int nCase;
	fin >> nCase;

	for (int ci = 0; ci < nCase; ++ci)
	{
		int n;
		fin >> n;
		cout << fn(n) << endl;
	}
}
