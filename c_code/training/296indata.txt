#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a, int b);

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int s1, m1, s2, m2;

		inStream >> s1 >> m1 >> s2 >> m2;

		int s = (s1*m2) + (s2*m1);
		int m = m1*m2;
	

		cout << s / gcd(s, m) << " " << m / gcd(s, m) << endl;

	}

	inStream.close();
	return 0;
}

int gcd(int a, int b) {
	if (b == 0) 
		return a;

	else 
		return gcd(b, a%b);
}
