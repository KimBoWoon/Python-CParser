
#include <iostream>
#include <fstream>
using namespace std;

int main(){
	ifstream in;
	in.open("input.txt");
	int n, h, w,guestcount,count;
	in >> n;
	for (int i = 0; i < n; i++){
		in >> h >> w >> guestcount;
		count = guestcount;
		for (int y = 1; y <= w; y++){
			for (int x = 1; x <= h; x++){
				count--;
				if (count == 0){
					if (y < 10){
						cout << x << "0" << y << endl;
					}
					else
						cout << x << y << endl;
					break;
				}
			}
		}
	}
}
