




#include <stdio.h>
#include <stdlib.h>

using namespace std;
 
int T, N;
int A[1003], S[1003];
int D[1003][1003];
 
int dy(int s, int e);
int max(int a, int b);

 
int main()
{
	FILE *inFile;
	inFile = fopen("input.txt","r"); 
	if(inFile ==NULL) exit(1);

	for (fscanf(inFile,"%d", &T);T--;){
        fscanf(inFile,"%d", &N);
        for (int i=1;i<=N;i++) 
		{
			fscanf(inFile,"%d", A+i);
			S[i] = S[i-1] + A[i];
		}

		for (int i=1;i<=N;i++) 
		{
			for (int j=i;j<=N;j++) 
			{
				D[i][j] = -1;
				
			}
		}
		printf("%d\n", dy(1, N));
    }
}


int dy(int s, int e)
{
    if (s > e) 
		return 0;
    if (s == e) 
		return A[s];

    int &ret = D[s][e];

    if (ret >= 0) 
		return ret;

    ret = max(S[e] - S[s-1] - dy(s+1, e), S[e] - S[s-1] - dy(s, e-1));
    return ret;
}

int max(int a, int b)
{
	if(a>=b)
		return a;
	else
		return b;
}
