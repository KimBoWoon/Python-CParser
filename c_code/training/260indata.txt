

#include<iostream>
#include<fstream>
using namespace std;

int uclid(int num1, int num2) 
{
	if(num2==0)	
		return num1;
	else		
		return uclid(num2,num1%num2);
}

void fraction(int s1,int m1, int s2, int m2) 
{
	int newSon = s1*m2+s2*m1;
	int newMa = m1*m2;
	int gcd = uclid(newSon,newMa);
	cout<<newSon/gcd<<" "<<newMa/gcd<<endl;
}

int main()
{
	fstream instream("input.txt",ios::in);
	int testcase;
	instream>>testcase;

	while(testcase--)
	{
		int son1,son2,ma1,ma2;
		instream>>son1>>ma1>>son2>>ma2;
		fraction(son1,ma1,son2,ma2);
	}

	return 0;
}
