#include<cmath>
#include<fstream>
using namespace std;


int reFac(int num)
{
	if (num <= 1)
		return 1;
	else         
		return num*reFac(num - 1);
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int facNum;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> facNum;
		cout << reFac(facNum) << endl; 
	}
	return 0;
}
