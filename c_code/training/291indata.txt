#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int gcd(int a,int b)
{
	if (b ==0) return a;
	else return gcd(b, a%b);
}
int main()
{
	ifstream in;
	in.open("input.txt");
	int t;
	in >> t;
	while(t--)
	{
		int a,b,c,d;
		int K;
		int res;
		int r1, r2;
		in >> a>>b>>c>>d;
		K = b*d / gcd(b, d);
		res = (K/b*a) + (K/d*c);
		r1=res/gcd(res, K);
		r2 = K / gcd(res, K);
		cout<<r1<<" "<<r2<<endl;
	}
	in.close();
}
