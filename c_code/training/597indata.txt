
#include<iostream>
#include<fstream>
using namespace std;

int factorial(int n){
	if (n == 1 || n == 0)
		return 1;
	else
		return n*factorial(n - 1);
}

int main(){
	ifstream File;
	int numCase = 0;
	int num = 0;
	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> num;
		int ans = factorial(num);
		cout << ans << endl;
	}
	File.close();
	return 0;
}
