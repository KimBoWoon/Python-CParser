#include<stdio.h>
#include<stdlib.h>

int fib(int n)
{
    if (n <= 1) 
        return n;
    else 
        return fib(n-1)+fib(n-2);
}

int main()
{
    FILE *in_file;
    int i;
    int no_cases;
    
    in_file = fopen("input.txt","r");
    if(in_file == NULL) exit(1);
    
    fscanf(in_file,"%d",&no_cases);
    
    for(i=0; i<no_cases; i++)
    {
        int n;
        
        fscanf(in_file, "%d",&n);
        fib(n);
        
        printf("%d \n",fib(n));
    }
    fclose(in_file);
}
