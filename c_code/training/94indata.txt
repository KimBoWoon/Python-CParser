#include <iostream>
#include <fstream>

class FileClass{
private:
	std::ifstream file;

public:
	FileClass(const char* name){
		file.open(name);
	}

	void closeFile(){
		file.close();
	}

	int getItem(){
		int temp;
		file >> temp;
		return temp;
	}
};

class PrimeNum{
public:
	bool run(const int num){
		if (num == 2)	return true;
		for (int i = 2; i * i <= num; i++){
			if (!(num % i)) return false;
		}
		return true;
	}
};

int main(){
	FileClass fileclass("input.txt");

	PrimeNum primenum;

	int cases = fileclass.getItem();

	while (cases--){
		if (primenum.run(fileclass.getItem()))	std::cout << 1 << std::endl;
		else									std::cout << 0 << std::endl;
	}

}
