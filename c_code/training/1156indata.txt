




#include<iostream>
#include<fstream>


int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;

	for(int i = 0; i < times; i++){

		int k;
		is>>k; 

		int eureka = 0; 

		for(int a = 1; a <= 45; a++){

			for(int b = 1; b <= 45; b++){

				for(int c = 1; c <= 45; c++){


					if(2*k == a*a + a + b*b + b + c*c + c){ 

						eureka = 1;
						b+=100;
						a+=100;
						break;

					}
					
				}
	
			}

		}
		
		std::cout<<eureka<<std::endl; 

	}

	return 0;
}
