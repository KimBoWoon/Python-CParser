

#include <iostream>
#include <fstream>
using namespace std;

void DucciSequence(int input[], int n)
{
	int result[15];
	int cnt = 0;
	int isZero=0;
	
	while(1){
	for(int i=0; i<n; i++){

		if(i==n-1) {result[i] = input[i] - input[0];}
		else{result[i] = input[i] - input[i+1];}
		
		if(result[i] < 0) result[i] = -result[i];
	
		}
		for(int i =0; i<n; i++)input[i] = result[i];

		
		cnt++;
		if(cnt > 1000) {cout << "LOOP"<<endl; break;}

		for(int i = 0; i<n; i++){
			
			if(result[i] == 0 ) isZero++;
		}
		if(isZero == n) {cout <<"ZERO"<<endl; break;}
		isZero = 0;
	}
}

int main()
{
	int numTestCases;
	int testcase;
	int input[15];
	ifstream inFile;
	inFile.open("input.txt");
	inFile >> numTestCases; 

	if(inFile.is_open() == false) {cout << "파일을 열 수 없습니다" <<endl;}

		for(int i = 0 ; i < numTestCases ; i++) {
			inFile >> testcase;
			for(int j = 0; j < testcase; j++){
				inFile >> input[j];
			}
			DucciSequence(input, testcase);
	}

}
