




#include<iostream>
#include<fstream>


int main(int argc, char* argv[]){
	
	std:: ifstream is;
	is.open("input.txt"); 
	
	int times; 
	is >> times;
	
	for( int i = 0; i < times; i++){

		int h,w,n; 
		is >> h >> w >> n;

		if(n%h==0){

			std::cout<<h;
		
			if(n/h+1<10)
			std::cout<<'0';

			std::cout<<n/h<<std::endl;
		}

		else{

			std::cout<<n%h;
		
			if(n/h+1<10)
				std::cout<<'0';

			std::cout<<n/h+1<<std::endl;
		}


	}




}
