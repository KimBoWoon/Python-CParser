#include <iostream>
#include <fstream>
#include <cstring>

int McCarthy(int n)
{
	if (n > 100)
		return n - 10;
	else
		return McCarthy(McCarthy(n + 11));
}

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail())
	{
		std::cerr << "file open err" ;
		return 1;
	}

	int caseNum;
	input >> caseNum;
	for (int i = 0; i < caseNum; i++)
	{
		int n;
		input >> n;
		std::cout << McCarthy(n) << std::endl;
	}
	
	return 0;
}
