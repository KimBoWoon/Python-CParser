

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int checkPalindrome(char data[], int first, int last);

int main(void) {
	FILE *input;
	int no_cases;
	int i;
	char data[1000];

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%s", &data);
		printf("%d\n", checkPalindrome(data,0,strlen(data)-1));

	}

	fclose(input);
	return 0;
}

int checkPalindrome(char data[], int first, int last) {
	if (last <= first)
		return 1;
	else if (data[first] != data[last])
		return 0;
	else
		return checkPalindrome(data, first + 1, last - 1);
}
