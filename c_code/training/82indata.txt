#include <iostream>
#include <fstream>
#include <cstdlib>
#define N 100000
using namespace std;
bool check[N+1];
bool prime[N+1];
int main()
{
	int t,num;
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>t;
    for(int i=2; i<=N; i++)
    {
        if(!check[i]) 
			prime[i]=true;
		else
			continue;
        for(int j=i; j<=N; j+=i)
            check[j]=true;
    }
	for(int i=0; i<t; i++)
	{
		inStream>>num;
		cout<<prime[num]<<endl;
	}
    return 0;
}
