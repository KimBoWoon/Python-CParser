
#include <iostream>
#include <fstream>
#include <algorithm>
#include <vector>

std::vector<int> v;


int abs(int a)
{
	if ( a < 0 )return -a;
	else return a;
}

int main()
{
	std::ifstream infile("input.txt");
	
	int numCase;
	int min = 2e9;
	int count =0;

	infile >> numCase;

	for( int i=0; i< numCase; i++)
	{
		int n;
		int k;
		infile>> n >> k;

		for(int k=0; k<n; k++)
		{
			int a;
			infile >> a;
			v.push_back(a);	
		}

	

		
		sort(v.begin(),v.end());

		
		int h = 0;
		int t = v.size()-1;
		

		while(true)
		{

			int sum;
			int sum1;
			
			sum = v[h];
			sum1 = v[t];


			sum += sum1;


			int dis = abs(sum -k );

			

			if( sum < k )
			{ 
				if( min > dis )
				{
					min = dis;
				
					count = 1;
				}
				else if ( min == dis )
					count++;
				h++;
			}
			else
			{
				if( min > dis )
				{
					min = dis;

					count = 1;
				}
				else if ( min == dis )
					count++;

				t--;
			}


			if( h == t )break;
			
		}


		std::cout<<count<<std::endl;
		
		min = 2e9;
		count = 0;
		v.clear();

	}

}
