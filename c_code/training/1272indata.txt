
#include <iostream>
#include <fstream>
using namespace std;

int prime[1001]={0,0,2};

int main()
{
	ifstream inStream;
	inStream.open("input.txt");
	int t,tmp,num;
	inStream>>t;
	for(int i=3; i<=999; i+=2) 
	{
		tmp=0;
		for(int j=3; j*j<=i; j+=2)
		{
			if(i%j==0)
			{
				prime[i]=0;
				tmp=1;
				break;
			}
		}
		if(tmp==0) prime[i]=i;
	}
	for(int i=0; i<t; i++)
	{
		inStream>>num;
		int j=0;
		while(1)
		{
			if(prime[num/2-j]!=0 && prime[num/2+j]!=0) 
			{
				cout<<prime[num/2-j]<<' '<<prime[num/2+j]<<endl;
				break;
			}
			j++;
		}
	}
}
