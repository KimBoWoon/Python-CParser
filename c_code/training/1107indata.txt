

#include <stdio.h>
#include <stdlib.h>

int A[45],D[1001];

void main(){
	
	FILE *in_file;
	
	int no_cases,knum;
	int i,j,k;
	for (i=1;i<45;i++) 
		A[i] = i*(i+1)/2;
	for (i=1;i<45;i++){
		for (j=i;j<45;j++)
			for (k=j;k<45;k++)
				if (A[i]+A[j]+A[k] < 1001)
					D[A[i]+A[j]+A[k]] =	1;
	}
	in_file = fopen("input.txt", "r");
	if(in_file == NULL) exit(1);
	fscanf(in_file,"%d", &no_cases);
	while(no_cases--){
		fscanf(in_file, "%d",  &knum);	
		printf("%d\n", D[knum]);
	}
	fclose(in_file);
}
