#include <iostream>
#include <fstream>
using namespace std;

int tri[46];
bool check[1001];
int ans[1001];

int main()
{
	int n,data,tmp;
	ifstream inStream;
	inStream.open("input.txt");
	inStream>>n;
	for(int i=1; i<=45; i++)
		tri[i]=(i*(i+1))/2;
	for(int t=0; t<n; t++)
	{
		inStream>>data;
		if(check[data])
		{
			cout<<ans[data]<<endl;
			continue;
		}
		for(int i=1; i<=45; i++)
			for(int j=i; j<=45; j++)
				for(int k=j; k<=45; k++)
				{
					tmp=tri[i]+tri[j]+tri[k];
					if(tmp==data)
					{
						check[data]=true;
						ans[data]=1;
						break;
					}
				}
				cout<<ans[data]<<endl;
	}
}
