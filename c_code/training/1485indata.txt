#include <iostream>
#include <fstream>
using namespace std;


int main(){
	ifstream file;
	file.open("input.txt");

	int cases;

	file >> cases;

	while (cases--){
		int height, width, in_room;

		file >> height >> width >> in_room;

		int guest_floor = in_room % height;
		int guest_room = in_room / height + ((in_room % height == 0) ? 0 : 1);
		
		cout << ((guest_floor == 0) ? height : guest_floor);
		if (guest_room < 10)
			cout << "0";
		cout << guest_room << endl;
	}


	return 0;
}
