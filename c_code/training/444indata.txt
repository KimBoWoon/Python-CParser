#include <iostream>
#include <fstream>

int gcd(int num1, int num2);

int main(void){

	std::ifstream inFile("input.txt");
	if (inFile.fail()){
		std::cout << "not file" << std::endl;
		return 0;
	}

	int testCases = 0;
	inFile >> testCases;

	int num1 = 0;
	int num2 = 0;

	for (int i = 0; i < testCases; i++){
		inFile >> num1 >> num2;

		std::cout << gcd(num1, num2) << std::endl;
	}
	return 0;
}

int gcd(int num1, int num2){

	if (num1 < 0 || num2 < 0){
		return -1;
	}

	if (num2 == 0){
		int temp = num1;
		num1 = num2;
		num2 = temp;
	}

	int r = 0;
	r = num1 % num2;
	num1 = num2;
	num2 = r;

	if (r == 0){
		return num1;
	}
	else{
		return gcd(num1, num2);
	}
}
