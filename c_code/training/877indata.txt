

#include <iostream>
#include <fstream>

using namespace std;

int fibonacci(int num);

int main()
{
	ifstream infile;
	int testCase;
	int num; 

	infile.open("input.txt");
	if (infile.fail())
	{
		cerr << "Input file opening failed. \n";
	}

	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> num;

		cout << fibonacci(num) << endl; 
	}

	infile.close();

	return 0;
}

int fibonacci(int n)
{
	if (n == 0)		
		return 0;
	else if (n == 1) 
		return 1;
	else
		return fibonacci(n - 2) + fibonacci(n - 1);

	
}
