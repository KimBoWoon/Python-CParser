#include <iostream>
#include <fstream>

int fac(int inputNum);

int main()
{
	std::ifstream file("input.txt");
	int testCase = 0, inputNum = 0;

	file >> testCase;

	while (testCase--)
	{
		file >> inputNum;
		std::cout << fac(inputNum)<<std::endl;
	}
}

int fac(int inputNum)
{
	if (inputNum <= 1)
		return 1;
	else
		return inputNum*fac(inputNum - 1);
}
