  



#include <iostream>
#include <fstream>
using namespace std;
int factorial(int n);
int main()
{
	ifstream inFile("input.txt");  
	int no_cases=0,num=0;
	inFile>> no_cases;
	for(int l =0; l < no_cases; l++){
		inFile>>num; 
		cout<<factorial(num)<<endl;
	}

	return 0;
}

int factorial(int n)
{
	if(n==0) 
		return 1;
	if(n==1)
		return 1;
	else
	{
		return n*factorial(n-1);
	}
}
