
#include <iostream>
#include <cstdlib>
#include <fstream>

int mac(int n);

int main()
{
	std::fstream inputfile;
	int testcase;
	int a;

	inputfile.open("input.txt");
	if (inputfile.fail())
	{
		std::cerr << "Input file opening failed \n";
		exit(1);
	}
	inputfile >> testcase;

	for (int i = 0; i < testcase; i++)
	{
		inputfile >> a;
		std::cout << mac(a) << std::endl;
	}

	return 0;
}

int mac(int a)
{
	if (a >100)
		return a-10;
	else 
		return mac(mac(a+11));
}
