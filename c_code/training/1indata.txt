

#include <stdio.h>
#define criterionSquare(a,b) (b*b <= a ? 1:0) 

int main()
{
	unsigned int a, b, t;
	FILE *fp;
	fp = fopen("input.txt", "r"); 
	fscanf(fp, "%d", &t); 

	while (t--){
		fscanf(fp, "%d", &a);
		b = 2; 
		while (criterionSquare(a, b)){
			if (!(a - (a / b)*b)){ 
				printf("0\n"); 
				break; 
			}b++;
		}
		if (!criterionSquare(a, b)) printf("1\n"); 
	}
	fclose(fp);
	return 1;
}
