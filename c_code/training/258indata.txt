

#include <iostream>
#include <fstream>
using namespace std;

int gcd(int a, int b){
	if (b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int main(void){
	ifstream in;
	in.open("input.txt");
	int testcases;
	in >> testcases;
	for (int i = 0; i < testcases; i++){
		int a, b, c, d;
		int resultJa, resultMo;
		int gcdValue;
		in >> a >> b >> c >> d;
		resultJa = a*d + b*c;
		resultMo = b*d;
		gcdValue = gcd(resultJa, resultMo);
		cout << resultJa / gcdValue << ' ' << resultMo / gcdValue << endl;
	}
}
