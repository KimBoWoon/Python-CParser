#include <stdio.h>

void fractional(int a, int b, int c, int d);
int gcd(int a, int b);

int main()
{
	FILE* fp = fopen("input.txt", "r");
	int testCaseNumber = 0;
	int i = 0;
	int a = 0;
	int b = 0;
	int c = 0;
	int d = 0;
	int r = 0;
	int s = 0;

	if (fp == NULL)
	{
		printf("file error\n");
	}

	fscanf(fp, "%d", &testCaseNumber);

	for (i = 0; i < testCaseNumber; i++)
	{
		fscanf(fp, "%d %d %d %d", &a, &b, &c, &d);
		fractional(a, b, c, d);
	}

	fclose(fp);
	return 0;
}

void fractional(int a, int b, int c, int d)
{
	int dividend = (a * d) + (b * c);
	int divisor = b * d;
	int gcdDD = gcd(dividend, divisor);
	
	printf("%d %d\n", dividend / gcdDD, divisor / gcdDD);
}

int gcd(int a, int b)
{
	if (b == 0)
	{	
		return a;
	}
	else
	{	
		return gcd(b, a % b);
	}
}
