





#include <stdio.h>
#include <string.h>

#define MAX 1000

int checkPalindrome(char *str, int first, int last)
    {
	if(last<=first)	
	    return 1;	
	else if(str[first]!=str[last])	
	    return 0;	
	else	
	    return checkPalindrome(str,first+1,last-1);	
    }

int main()
    {
	int cases,i;
	char data[MAX];

	FILE *in_file;
	in_file = fopen("input.txt","r");
	fscanf(in_file,"%d",&cases);

	for(i=0;i<cases;i++)
	{
		fscanf(in_file,"%s", data);
		printf("%d\n", checkPalindrome(data,0,strlen(data)-1));
	}

	fclose(in_file);
	return 0;
}
