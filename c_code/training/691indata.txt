

#include <stdio.h>
int palindrome(char *string, int first, int last) 
{
	if (last <= first) return 1;  
	else if (string[first] != string[last]) return 0; 
	else return palindrome(string, first + 1, last - 1); 
}
int main()
{
	int t , i;
	FILE *fp;
	char buff[1000];
	if ((fp = fopen("input.txt", "r")) == NULL) return 0; 
	fscanf(fp, "%d", &t); 
	while (t--) 
	{
		fscanf(fp, "%s", buff); 
		i = 0;
		while (buff[i] != NULL) i++; 
		printf("%d\n", palindrome(buff, 0, i - 1)); 
	}
	fclose(fp); 
	return 1;
}
