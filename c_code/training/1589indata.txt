

#include <stdio.h>
#include <stdlib.h>
int d[501][501];
int file[501], fileSum[501];
int merge(int number)
{
	int i=0, j=0,k=1;
	int temp1, min=0;
	for(i=1; i<=number-1; i++)
	{
		d[i][i+k]=file[i]+file[i+1];
	}
	for(k=1; k<=number-1; k++)
	{
		for(i=1; i<=number-k; i++)
		{
			for(j=i; j<i+k; j++)
			{
				temp1=d[i][j]+d[j+1][i+k]+fileSum[i+k]-fileSum[i-1];
				if(min==0 || min>=temp1)
					min=temp1;
			}
			d[i][i+k]=min;
			min=0;
		}
	}
	return d[1][number];
}
void main()
{
	FILE* inFile;
	int i=0, testCase=0, numberOfFile, j,k, min;
	inFile=fopen("input.txt", "r");
	if(inFile==NULL)
	{
		exit(1);
	}
	fscanf(inFile, "%d", &testCase);
	for(i=0; i<testCase; i++)
	{
		fscanf(inFile, "%d", &numberOfFile);
		for(j=1; j<=numberOfFile; j++)
		{
			fscanf(inFile, "%d", &file[j]);
			fileSum[j]=file[j]+fileSum[j-1];
			for(k=1; k<=numberOfFile; k++)
			{
				d[j][k]=0;
			}
		};
		printf("%d\n", merge(numberOfFile));
	}
}
