 

 
#include <stdio.h>

long fibonaci(int value); 
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    while(test--) 
    {
        int value;
        fscanf(fp,"%d",&value); 
        printf("%ld\n",fibonaci(value)); 
    }
    fclose(fp); 
    return 0;
}
long fibonaci(int value)
{
    if(value <= 1) 
    {
        return value;
    }
    else 
    {
        return fibonaci(value-1) + fibonaci(value-2);    
    }
}
