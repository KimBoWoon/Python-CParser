
 
#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;

int main()
{
	freopen("input.txt", "r", stdin);
	setbuf(stdout, NULL);

	int numTestcases;
	int testcases;
	
	scanf("%d", &numTestcases);

	for(int i = 1; i <= numTestcases; ++i) 
	{
	   
	   scanf("%d", &testcases);

	   for(int base=2; base<=64; base++)
	   {
	   	vector<int> vec;
	   	int temp=testcases;
	   	int flag = 1;

	   while(temp!=0)
	   {
	   vec.push_back(temp%base);
	   temp/=base;
	   }

	   	for(int i=0; i<vec.size()/2; i++) 
		{
		if(vec.at(i)!=vec.at(vec.size()-1-i))
		{ flag=false; break; }
		}

	   if(flag)
	   { cout << 1 << endl; break; }
	   if(base==64) cout << 0 << endl; 
	   }
	       
	}
	return 0;
}
