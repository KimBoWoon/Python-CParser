


#include <stdio.h>
#include <string.h>

#define MAX 1000

int Palindrome(char *str, int first, int last){
	if(last<=first)	return 1;	
	else if(str[first]!=str[last])	return 0;	
	else	return Palindrome(str,first+1,last-1);	
}

int main(){
	int cases,i;
	char data[MAX];

	FILE *infile;
	infile = fopen("input.txt","r");
	fscanf(infile,"%d",&cases);

	for(i=0;i<cases;i++){
		fscanf(infile,"%s", data);
		printf("%d\n", Palindrome(data,0,strlen(data)-1));
	}

	fclose(infile);
	return 0;
}
