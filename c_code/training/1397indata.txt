
#include <iostream>
#include <fstream>

using namespace std;

void perform(ifstream &inFile);			
int arrangeRoom(int height, int order);	

int main()
{
	int numTestCase;	
	ifstream inFile;

	inFile.open("input.txt");
	inFile >> numTestCase;

	for (int i = 0; i < numTestCase; i++)	
	{
		perform(inFile);	
	}

	inFile.close();

	return 0;
}

void perform(ifstream &inFile)
{
	int height,	
		weight,	
		order,	
		roomNumber;	

	inFile >> height >> weight >> order;

	roomNumber = arrangeRoom(height, order);	

	cout << roomNumber << endl;	
}

int arrangeRoom(int height, int order)
{
	int roomNumber = 0;	
	
	

	if (order % height)	
	{
		roomNumber = order % height * 100;	
		roomNumber += order / height + 1;	
	}

	else  
	{
		roomNumber = height * 100;		
		roomNumber += order / height;	
	}
	
	return roomNumber;	
}
