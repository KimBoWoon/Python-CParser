
#include <stdio.h>
#include <stdlib.h>

void cal(FILE *infile, int save);

int main()
{
	int nocases;
	int save;
	int i;

	FILE *infile;

	infile = fopen("input.txt", "r");

	fscanf(infile, "%d", &nocases);

	for(i = 0; i < nocases; i++)
	{
		fscanf(infile, "%d", &save);
		cal(infile,save);
	}

	return 0;
}

void cal(FILE *infile, int save)
{
	int abc[1000][15];
	int count = 0;
	int A = 0;
	int B = 0;
	int C = 0;
	int calsave = 0;
	int i;
	int j;

	for(i = 0; i<save; i++)
	{
		fscanf(infile, "%d", &abc[0][i]);
	}

	while(1)
	{
		if(A < save-1)
		{
			abc[count+1][A] = abc[count][A] > abc[count][A+1] ? (abc[count][A]-abc[count][A+1])%10 : (abc[count][A+1]-abc[count][A])%10;
			A ++;
		}

		else if(A == save-1)
		{
			abc[count+1][A] = abc[count][A] > abc[count][0] ? (abc[count][A]-abc[count][0])%10 : (abc[count][0] - abc[count][A])%10;

			for(i = 0; i<save; i++)
			{
				if(abc[count+1][i] == 0)
				{
					B ++;
				}
			}

			if(B == save)
			{
				printf("ZERO\n");
				break;
			}

			else
			{
				B = 0;
			}

			for(i = 0; i < count; i++)
			{
				for(j = 0; j < save; j++)
				{
					if(abc[i][j] == abc[count+1][j])
					{
						C ++;
					}
				}

				if(C == save)
				{
					printf("LOOP\n");
					save = 1;
					break;
				}

				else
					C = 0;
			}

			if(save == 1)
			{
				break;
			}

			A = 0;
			count++;
		}
	}
}
