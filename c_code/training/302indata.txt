#include<iostream>
#include<fstream>
using namespace std;

int GCD(int a, int b)
{
	int r;
	do{
		r = a % b;
		a = b;
		b = r;
	} while ( r != 0 );
	
	return a;
}

int main(){
	ifstream inStream;
	int numCases;
	int son1;	int son2;	int mom1;	int mom2;

	inStream.open("input.txt");

	inStream >> numCases;

	for(int i=0; i<numCases; i++){
		inStream >> son1 >> mom1 >> son2 >> mom2;
		
		int res_son = son1*mom2 + mom1*son2;
		int res_mom = mom1*mom2;

		int gcd = GCD(res_son,res_mom);

		cout << res_son/gcd << " " << res_mom/gcd << endl;
		
	}
	inStream.close();
}
