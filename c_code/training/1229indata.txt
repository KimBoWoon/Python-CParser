#include<iostream>
#include<fstream>

#define MAX 1000

using namespace std;

int decimal[MAX];
int n;

int calcDecimal(int a)
{
	int num = 1;
	int j;

	for (j = 2; j*j <= a; j++)
	{
		if (a%j == 0)
		{
			num = 0;
		}
	}

	return num;
}

void makeCoulum(){
	int cnt = 0;
	for (int i = 1; i <= MAX; i++){
		if ( calcDecimal(i) )
			decimal[cnt ++ ] = i;
	}
	n = cnt;
}

int sub(int a, int b){
	return a > b ? a - b : b - a;
}

void goldBach(int data){
	int min = MAX;
	int a, b;

	for (int i = 0; i < n; i++){
		for (int j = 0; j < n; j++){
			if (decimal[i] + decimal[j] == data && sub(decimal[i],decimal[j]) <= min){
				min = sub(decimal[i], decimal[j]);
				a = i; b = j;
			}
		}
	}
	cout << decimal[b] << " " << decimal[a] << endl;
}

int main(){

	int T, a;
	ifstream file;
	file.open("input.txt");


	makeCoulum();

	file >> T;
	while (T--){
		file >> a;
		goldBach(a);
	}

	return 0;
}
