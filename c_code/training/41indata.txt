
#include <stdio.h>
#include <stdlib.h>

#define MIN 2
#define MAX 100001

int *createPrime();	
void freeInt(int *input);	
void checkPrime(int number, int *arr);	


void main() {
	FILE *f;
	int numTestCases = 0;
	int number = 0;
	int i = 0;
	int *arr;

	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (; i < numTestCases; i++) {
		fscanf(f, "%d", &number);
		arr = createPrime();
		checkPrime(number, arr);
		freeInt(arr);
	}
	fclose(f);
}

int *createPrime() {
	int i = 0;
	int j = 0;
	int *arr;
	arr = (int *)malloc(sizeof(int)*MAX);

	for (i = 0; i < MAX; i++) {
		arr[i] = 1;
	}

	for (i = MIN; i < MAX; i++) {
		if (arr[i] == 0)
			continue;
		for (j = i + i; j <= MAX - 1; j += i)
			arr[j] = 0;
	}	
	return arr;
}

void freeInt(int *input) {
	if (input != NULL)
		free(input);
	
	return;
}

void checkPrime(int number, int *arr) {
	if (arr[number] == 1)
		printf("1\n");
	else
		printf("0\n");

	return;
}
