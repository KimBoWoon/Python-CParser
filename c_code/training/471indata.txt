

#include <stdio.h>
#include <stdlib.h>

int Euclid(int num1, int num2)
{
	if(num2==0)
		return num1;
	else
		return Euclid(num2,num1%num2);
}

void main(void)
{
	FILE *file;
	int i, test;

	file=fopen("input.txt","r"); 
	if(file==NULL) exit(1);
	
	fscanf(file,"%d", &test); 

	for(i=0; i<test; i++)
	{
		int num1,num2;
		fscanf(file,"%d %d",&num1,&num2); 

		printf("%d\n",Euclid(num1,num2));
	}
	fclose(file); 
}
