#include <iostream>
#include <fstream>
#include <cstring>

bool isItPalindrome(char text[], int start, int end)
{
	if (strlen(text) <= 1 || start > end)
		return true;

	if (text[start] == text[end])
		return isItPalindrome(text, start + 1, end - 1);
	else
		return false;
}

int main()
{
	std::ifstream input;
	input.open("input.txt");
	if (input.fail())
	{
		std::cerr << "file open err" ;
		return 1;
	}

	int caseNum;
	input >> caseNum;
	for (int i = 0; i < caseNum; i++)
	{
		char text[1000];
		input >> text;
		std::cout << isItPalindrome(text, 0, strlen(text) - 1) << std::endl;
	}
	return 0;
}
