#include <stdio.h>
#include <stdlib.h>

int Fibonacci(int num);
int ans;
void main()
{
	FILE *file;
	int i=0;
	int t, n;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d", &n);
		printf("%d\n", Fibonacci(n));
	}
	fclose(file);
}
int Fibonacci(int num) 
{
	if(num <=1) 
		ans=num;
	else
		ans =  Fibonacci(num-1) + Fibonacci(num-2); 

	return ans;
}
