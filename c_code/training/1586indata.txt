

#include<iostream>
#include<fstream>
#include<cmath>
#include<cstdlib>
using namespace std;



int merge(int* a, int* sum, int** dp, int size){
	for (int i = size; i >= 1; i--){
		for (int j = 1; j <= i; j++){
			int end = size - i + j;
			if (j == end){
				dp[j][j] = 0;
			}
			else if (j == end - 1){
				dp[j][j + 1] = a[j] + a[j + 1];
			}
			else{
				int min = 1000000000;
				for (int k = j; k < end; k++){
					int temp = dp[j][k] + dp[k + 1][end] + sum[end] - sum[j - 1];
					if (temp < min)
						min = temp;
				}
				dp[j][end] = min;
			}
			
		}
	}
	return dp[1][size];
}

int main(){
	ifstream File;
	int numCase = 0;
	int size = 0;
	int* a, *b;
	int** dp;

	File.open("input.txt");
	File >> numCase;
	for (int iter = 0; iter < numCase; iter++){
		File >> size;
		a = new int[size+1]();
		b = new int[size+1]();
		dp = new int*[size+1];
		for (int i = 0; i <= size; i++){
			dp[i] = new int[size+1]();
		}
		for (int i = 1; i <= size; i++){
			File >> a[i];
			b[i] = b[i - 1] + a[i];
		}
		int res = merge(a, b, dp, size);
		 
		cout << res << endl;

		for (int i = 0; i <= size; i++){
			delete[] dp[i];
		}
		delete dp;
	}
	File.close();
	return 0;
}
