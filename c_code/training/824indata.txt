
#include <iostream>
#include <fstream>
using namespace std;

int fibonacci(short n){
	if(1 >= n) return n; 
	else return fibonacci(n-1) + fibonacci(n-2); 
}

int main(){
	ifstream in;
	in.open("input.txt");

	short n,t;
	in>>t;

	while(t--){
		in>>n;
		if(1 >= n) cout<<n<<endl; 
		else cout<<fibonacci(n)<<endl;
	}
	in.close();
}
