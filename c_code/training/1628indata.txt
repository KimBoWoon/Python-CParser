#include <iostream>
#include <fstream>

using namespace std;

int A[500][500], M[500];

int merge(int n, const int d[]);

int main() {

   ifstream instream;
   instream.open("input.txt");

   int numTestCases;

   instream >> numTestCases;

   for(int i=0; i<numTestCases; i++) {

      int numData;
      int *arr;

      instream >> numData;

      arr = new int[numData];

      for(int j=1; j<=numData; j++) {

         instream >> arr[j];
         M[j] = M[j-1] + arr[j];
      }

      cout << merge(numData, arr) << endl;
   }
}

int merge(int n, const int arr[]) {

   int l, r;

   for(int i=0; i<n; i++) {
      for(int j=i; j>=0; j--) {
         if(i != j) {

            A[j][i] = 9999999999;

            for(int k=0; k<i-j; k++) {

               if(A[j][i] > M[i+1]-M[j]+A[j][j+k]+A[j+k+1][i]) {

                  A[j][i] = M[i+1]-M[j]+A[j][j+k]+A[j+k+1][i];
               }
            }
         }
      }
   }

   return A[0][n-1];
}
