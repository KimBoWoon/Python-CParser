#include<stdio.h>
#include<stdlib.h>

int f(int n)
{
    if(n <=1)
        return 1;
    else
        return n*f(n-1);
}

int main()
{
    FILE *in_file;
    int i;
    int no_cases;
    
    in_file = fopen("input.txt","r");
    if(in_file == NULL) exit(1);
    
    fscanf(in_file,"%d",&no_cases);
    
    for(i=0; i<no_cases; i++)
    {
        int n;
        
        fscanf(in_file, "%d",&n);
        f(n);
        
        printf("%d \n",f(n));
    }
    fclose(in_file);
}
