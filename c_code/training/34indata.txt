  



#include <iostream>
#include <fstream>
using namespace std;

int main()
{
	 ifstream inFile("input.txt");  

	int no_cases=0;
	
	inFile>> no_cases;
	for(int k =0; k < no_cases; k++)
	{
		int num=0;
		inFile >> num;

		bool flag=1;
		for(int i=2; i<num; i++)
		{
			if(num%i==0)
			{
				flag=0;
				break;
			}
		}
		if(flag==1)
			cout<<"1"<<endl;
		else
			cout<<"0"<<endl;

		
	}
	inFile.close();
	return 0;
}
