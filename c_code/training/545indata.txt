

#include<iostream>
#include<fstream>

#define INF 100

using namespace std;
int factorial(int n){
   
   if (n < 1) return 1;

   else return n*factorial(n - 1);

}

void sort(char *arr, int count){

   char temp;
   temp = arr[count];
   for (int i = count; i < INF; i++){
      arr[i] = arr[i + 1];
   }
   arr[11] = temp;
}

int main(void)
{
   int num_case;
   ifstream inStream;
   inStream.open("input.txt");
   inStream >> num_case;

   for (int i = 0; i < num_case; i++) {
      char arr[INF] = { 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l' };
      int num, cases, exp;
      inStream >> num >> cases;
      num = num-1;
      int count = factorial(num);
      cases = cases - 1;
      while (num >= 1){
         exp = cases / count;  
         cases %= count;
         cout << arr[exp];
         sort(arr, exp);
         count /= num;
         num--;
      }
      cout << arr[0]<< endl;
      
   }
   return 0;
}
