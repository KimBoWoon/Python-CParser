
#include <iostream>
#include <fstream>
using namespace std;


int getTVlaue(int n)	
{
	return n*(n+1)/2;
}

int main()
{
 fstream in;
 in.open("input.txt");

 int numcases;
 int testValue;
 int T1,T2,T3;
 int ureka=0;
 in>>numcases;

 for(int i=0; i<numcases; i++)		
 {
	 in>>testValue;
	 for(int n1=1; n1<=90; n1++)	
	 {
		 T1=getTVlaue(n1);
		 for(int n2=1; n2<=90; n2++)
		 {
			 T2=getTVlaue(n2);
			 for(int n3=1; n3<=90; n3++)
			 {
				 T3=getTVlaue(n3);
				 if(testValue==(T1+T2+T3))	
				 {
					 ureka=1;
					 break;

				 }
			 }
			 if(ureka)
				 break;
		 }
		 if(ureka)
			 break;
	 }
	 cout<<ureka<<endl;
	 ureka=0;
 }
}
