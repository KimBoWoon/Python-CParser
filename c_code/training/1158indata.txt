#include<stdio.h>
#include<stdlib.h>

void main(){

	FILE *file;

	int numCases;
	int i,j,m,n;
	int T1[45];     
	int T2[1001] = {0};   
	int num;

	file = fopen("input.txt","r");
	if(file == NULL) exit(1);

	for(i=1; i<45; i++){
		T1[i] = i*(i+1)/2;
	}

	for(i=1; i<45; i++){
		for(j=1; j<45; j++){
			for(m=1; m<45; m++){

				if((T1[i]+T1[j]+T1[m])<1001) T2[(T1[i]+T1[j]+T1[m])] =1;   
			}
		}
	}

	fscanf(file, "%d", &numCases);

	for(i=0; i<numCases; i++){
		fscanf(file, "%d", &num);

		printf("%d\n", T2[num]);
	}

	fclose(file);
}
