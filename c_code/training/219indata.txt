

#include <stdio.h>
#include <stdlib.h>

void sumFraction(int a, int b, int c, int d);
void reducePrint(int a, int b);

int main(void) {
	FILE *input;
	int no_cases;
	int i;
	int a, b, c, d;

	input = fopen("input.txt", "r");
	if (input == NULL)
		exit(1);

	fscanf(input, "%d", &no_cases);

	for (i = 0; i < no_cases; i++) {
		fscanf(input, "%d%d%d%d", &a, &b, &c, &d);
		sumFraction(a, b, c, d);
	}

	fclose(input);
	return 0;
}

void sumFraction(int a, int b, int c, int d) {
	if (b == d) {
		reducePrint(a + c, d);
	} else {
		reducePrint(a * d + c * b, b * d);
	}
}

void reducePrint(int a, int b) {
	int no_gcd;
	no_gcd = gcd(a, b);
	printf("%d %d\n", a / no_gcd, b / no_gcd);

}

int gcd(int a, int b) {
	int temp;
	if (a < b) {
		temp = a;
		a = b;
		b = temp;
	}

	if (b == 0) {
		return a;
	} else
		return gcd(b, a % b);
}
