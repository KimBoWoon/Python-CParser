

#include <stdio.h>
#include <stdlib.h>

int prime(int num){
	
	int flag = 1;
	int i;
	for( i = 2;i*i<=num;i++){
		if(num%i==0){
			flag =0; 
			break;
		}
	}	return flag;
}
void main(){
	
	FILE *in_file;
	int no_cases;

	in_file = fopen("input.txt", "r");

	if(in_file == NULL) exit(1);

	fscanf(in_file,"%d", &no_cases);

	while(no_cases--){
		int num;
		fscanf(in_file, "%d", &num);
		printf("%d\n",prime(num));
	}
	fclose(in_file);
}
