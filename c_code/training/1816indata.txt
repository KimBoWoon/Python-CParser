

#include <iostream>
#include <fstream>
#include <string>

void Vps(std::string s);
bool isVps = false;

int main()
{
	std::ifstream infile("input.txt");

	int numCases ;
	
	infile >> numCases;
	
	std::string s;

	for(int i=0; i< numCases; i++)
	{

		infile >> s;

		isVps = false;

		Vps(s);

		if( isVps)
			std::cout<<"YES"<<std::endl;
		else
			std::cout<<"NO"<<std::endl;
	}
}


void Vps(std::string s)
{
	if( s.at(0) == ')' || s.length()%2 == 1 )
	{
		isVps = false;
	}
	else
	{
		for( int i=1; i< s.length() ; i++ )
		{
			if( s.at(i) == ')' )
			{
				if( s.length() == 2 )
				{
					isVps = true;
					break;
				}
				s.erase(i,1);
				s.erase(0,1);
				Vps(s);
				break;
			}	
			else if ( s.at(i) == '(' && s.length() == 2)
			{
				isVps =false;
			}
		}
	}
}
