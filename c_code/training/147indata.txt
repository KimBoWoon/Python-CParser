

#include <stdio.h>

int McCarthy(int num);

void main() {
	int numTestCases;
	int i;
	int num;

	FILE *f;
	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d", &num);
		printf("%d\n", McCarthy(num));
	}
}

int McCarthy(int num) {
	if (num > 100)
		return num - 10;
	else
		return McCarthy(McCarthy(num + 11));
}
