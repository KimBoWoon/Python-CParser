 

 

#include <stdio.h>
#include <string.h>

void parenthesis(char* str,int i,int cnt);  
static int value; 
int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    while(test--) 
    {
        char str[50];
        fscanf(fp,"%s",str); 
        parenthesis(str,0,0);
        if(value == 0) printf("NO\n");
        else printf("YES\n");
    }
    fclose(fp); 
    return 0;
}
void parenthesis(char* str,int i,int cnt)
{
    if(i<0) 
    {
        value = 0;
    }
    else if(strlen(str) == cnt) 
    {
        if(i == 0)
            value = 1;
        else 
            value = 0;
    }
    else 
    {
        switch(str[cnt])
        {
            case '(':
                parenthesis(str,i+1,cnt+1);
            break;
            case ')':
                parenthesis(str,i-1,cnt+1);
            break;
        }
    }
}
