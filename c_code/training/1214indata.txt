void twoNumber(int a[], int num)
{
	int i,j,k=0,l,result,p,q,r,index;
	int minusIndex[168][2];

	int temp;
	int minus[168];
	for(l=0; l<168; l++)
		minus[l]=1000;
	for(i=0; i<168; i++){
		for(j=0; j<168; j++){
			if(num==a[i]+a[j]){
				minusIndex[k][0]=a[i];
				minusIndex[k][1]=a[j];
				minus[k++]=abs(a[i]-a[j]);
			}
		}
	}
	for(p=0; p<168; p++){
		if(minus[p]<min){
			min=minus[p];
			index=p;
		}
	}
	printf("%d %d\n",minusIndex[index][0],minusIndex[index][1]);
