
#include <fstream>
#include <iostream>
using namespace std;

int gcd(int a, int b){
	if (b == 0) return a;
	else return gcd(b, a % b);
}


int main(){

	int caseNum = 0;
	ifstream inputFile("input.txt");

	inputFile >> caseNum;
	for (int i = 0; i < caseNum; i++){
		int testNum1 = 0, testNum2 = 0, testNum3 = 0, testNum4 = 0;
		inputFile >> testNum1 >> testNum2 >> testNum3 >> testNum4;

		testNum1 *= testNum4;
		testNum3 *= testNum2;
		cout << (testNum1 + testNum3) / gcd((testNum1 + testNum3), (testNum2 * testNum4)) << " " << (testNum2 * testNum4) / gcd((testNum1 + testNum3), (testNum2 * testNum4)) << endl;
	}

	return 0;
}
