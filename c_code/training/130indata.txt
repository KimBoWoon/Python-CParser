

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int mac(int a);

void main(void) {
	FILE *inFile;
	int caseNum;

	int i;
	int n1;

	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &caseNum);

	for (i = 0; i < caseNum; i++) {
		fscanf(inFile, "%d", &n1);
		printf("%d\n", mac(n1));
	}
}

int mac(int a) {
	if (a > 100) {
		return a - 10;
	}
	else {
		return mac(mac(a + 11));
	}
}
