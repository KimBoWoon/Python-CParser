

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;
int itIs(int x); 
int itZero(int x, int y); 
int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	inStream >> numTestCases;
	
	int count = 0;
	int prime[168];
	for (int i = 0; i < 1000; i++) 
	{		
		int a = itIs(i);
		if (a == 1)
		{
			prime[count] = i;
			count++;
		}
	}
	for (int i = 0; i < numTestCases; i++)
	{
		int x, y, that, sum, dif; 
		sum = 0, dif = 1000;
		inStream >> that;
		for (int i = 0; i < 168; i++)
		{
			for (int j = 0; j < 168; j++)
			{
				if (j >= i) 
				{
					sum = prime[i] + prime[j];
					if (sum == that) 
					{
						int temp = itZero(prime[j], prime[i]);
						if (temp < dif) 
						{
							x = prime[i]; y = prime[j]; 
							dif = temp;
						}
					}
				}
			}
		}
		cout << x << " " << y << endl; 
	}
	inStream.close();
	return 0;
}

int itIs(int x) 
{
	int i;
	if (x < 2)
	{
		return 0; 
	}
	for (i = 2; i < x; i++)
	{
		if (x % i == 0)
			return 0; 
	}
	return 1; 
}

int itZero(int x, int y) 
{
	int dif = 0;
	dif = x - y;
	if (dif < 0){
		dif = dif*-1;
	}
	return dif;
}
