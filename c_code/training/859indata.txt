

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int Fibonacci(int a);		

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error"<<endl;
		exit(1);
	}

	int no_case;
	instream>>no_case;

	for(int i=0; i<no_case; i++)		
	{
		int num;
		instream>>num;

		cout<<Fibonacci(num)<<endl;		
	}
}

int Fibonacci(int a)		
{
	if(a <= 1)
		return a;
	else
		return Fibonacci(a-1)+Fibonacci(a-2);
}
