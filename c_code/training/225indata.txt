


#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)
int uclid(int num1, int num2); 
void main(void)
{
	FILE* inFile;
	int testCase = 0, numUp1=0, numDown1=0, numUp2=0, numDown2=0;
	int sumUp = 0, sumDown = 0,gcd=0; 
	int i = 0;
	inFile = fopen("input.txt", "r"); 
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase); 

	for (i = 0; i < testCase; i++)
	{
		fscanf(inFile, "%d %d %d %d", &numUp1,&numDown1, &numUp2, &numDown2); 
		if (numDown1 == numDown2) 
		{
			sumDown = numDown1;
			sumUp = numUp1 + numUp2;
		}
		else if (numDown1%numDown2 == 0) 
		{
			sumDown = numDown1;
			sumUp = numUp1 + numUp2*(numDown1 / numDown2);
		}
		else if (numDown2%numDown1 == 0)
		{
			sumDown = numDown2;
			sumUp = numUp1*(numDown2 / numDown1) + numUp2;
		}
		else 
		{
			sumDown = numDown1*numDown2; 
			sumUp = numUp1*numDown2 + numUp2*numDown1; 
		}
		gcd = uclid(sumDown, sumUp); 
		sumDown /= gcd; 
		sumUp /= gcd;
		printf("%d %d\n", sumUp, sumDown); 
		sumUp = 0; sumDown = 0;
	}
	fclose(inFile);
}
int uclid(int num1, int num2)
{
	if (num2 == 0) 
	{
		return num1;
	}
	else 
	{
		return uclid(num2, num1%num2);
	}
}
