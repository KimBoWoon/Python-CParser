

#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cmath>
using namespace std;

int main ()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream>> testcase;

	for(int i=0;i<testcase;i++)
	{
		int number,count =0;
		instream >> number;
		int *primenum = new int[number];
		for(int j=2;j<number;j++){
			int res=1;
			for(int div=2;div*div<=j;div++)  
			{
				if(j%div==0){               
					res=0;
				}
			}
			if(res==1) 
			{
				primenum[count]=j;     
				count++;			
			}
		}


		int p1=0,p2=0,p3=0,p4=0;
		for(int b=0;b<count;b++)
		{
			for(int c=b;c<count;c++)
			{


				if((primenum[b]+primenum[c])==number) 
				{
					p1=primenum[b];p2=primenum[c];
				}
			}
		}
		cout<<p1<<" "<<p2<<endl;
	}
}
