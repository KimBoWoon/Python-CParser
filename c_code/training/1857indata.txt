

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int parenthesis(char arr[], int size)
{
	int i;
	int a=0, b=0;
	int result = 1;
	
	for(i=0; i<size; i++) 
	{
		if(arr[i]=='(')
			a++;
		else
			b++;
	}

	if(a != b || arr[0]==')') 
	{
		result = 0;
		return printf("NO\n");
	}
	else 
	{
		result = 1;
		a=0, b=0;
		for(i=0; i<size; i++) 
		{	
			if(i!=0 && a==b && arr[i]==')')
			{
				result = 0;
				return printf("NO\n");
			}
			if(arr[i]=='(')
				a++;
			else
				b++;
		}
	}
	if(result == 1) 
		return printf("YES\n");
}

void main()
{
	FILE *file;
	int i, j;
	int test;
	char arr[100]={0};

	file = fopen("input.txt", "r"); 
	fscanf(file, "%d", &test); 
	for(i=0; i<test; i++)
	{
		fscanf(file, "%s", &arr);
		parenthesis(arr, strlen(arr));
	}
	fclose(file); 
}
