

#pragma warning(disable:4996)
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int checkPalindrome(char str[], int first, int last);

int main(void)
{
	FILE *in_file;
	int no_cases;
	char line[1000];
	int i;

	in_file = fopen("input.txt", "r");
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);   

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%s", &line);
		printf("%d \n", checkPalindrome(line, 0, strlen(line) - 1));
	}
	fclose(in_file);
}
int checkPalindrome(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkPalindrome(str, first + 1, last - 1);
}
