#include <stdio.h>
#include <stdlib.h>

int main(void)
{
   FILE *fp;
   int numcases,n,i;

   fp = fopen("input.txt", "r");
   if (fp == NULL) exit(1);

   fscanf(fp, "%d", &numcases);
   
   for (i = 0; i < numcases; i++)
   {
      int j,k,num,str[100],length,cnt,z;
      fscanf(fp, "%d", &n);
    
      for(j=2;j<=64;j++)
      {
		  num=n;
		  k = 0;
          while(num > 0)
		  {
			  str[k]=num%j;
              num/=j;
			  k++;
         }
		  str[k] = num;
          length = k;
          cnt=1;
		  z = 0;
          while(z < length)
		  {
			  if(str[z]!=str[length-1])
              {
				  cnt=0;break;
              }
			  z++;
              length--;
         }
         if(cnt==1)
         {
               printf("1\n");break;
         }
         else if(j==64){
            printf("0\n");
         }
      }
   }
   fclose(fp); 

   return 0;
}
