 

 
#include <stdio.h>
#include <math.h>

int main()
{
    int value=0,flag=1;
    unsigned int input,i;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&value); 
    while(value--)
    {
        fscanf(fp,"%u",&input); 

        if((input %2 == 0 && input != 2) || input == 1) 
        {
            flag = 0;
        }
        if(input == 2 || input == 3) 
        {
            flag = 1;
        }
        for(i=2;i<=sqrt(input);i++) 
        {
            if(input % i == 0)
            {
                flag = 0;
                break;
            }
        }
        printf("%d\n",flag);
        flag = 1;
    }
    fclose(fp); 
    return 0;
}
