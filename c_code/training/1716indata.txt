#include <iostream>
#include <fstream>
#include <cstring>
#include <cstdlib>
#include <string>
using namespace std;

int sw = 0;
int check_pal(char * str, int first, int last)
{
	if (first >= last)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return check_pal(str, first + 1, last - 1);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num = 0, sh = 0, rem = 0, len = 0;
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		int arr[21];
		sw = 0;
		for (int j = 0; j < 21; j++)
			arr[j] = -1;
		inStream >> num;
		for (int j = 2; j <= 64; j++)
		{
			sw = 0;
			sh = num;
			for (int k = 0;sh > 0;k++)
			{
				rem = sh % j;
				sh /= j;
				arr[k] = rem;
				len = k; 
			}
			for (int k = 0;k <= len;k++)
			{
				if (k >= len - k)
					break;
				else if (arr[k] == arr[len - k])
					sw++;
				else if (arr[k] != arr[len - k])
					break;
			}
			if (sw == (len+1)/2) break; 
		}
		if (sw == (len+1)/2)
			cout << 1 << endl; 
		else
			cout << 0 << endl;
	}
	return 0;
}
