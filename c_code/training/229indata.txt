
#include <stdio.h>
#include <stdlib.h>

unsigned int gcd(unsigned int x, unsigned int y);  

void main(void)
{
	FILE *file;
	int i, nocase;
	file=fopen("input.txt", "r");
	if (file == NULL)
		exit(1);
	fscanf(file, "%d", &nocase);
	for (i = 0; i < nocase; i++)
	{
		unsigned int n1, n2, m1, m2; 
		unsigned int num, mother; 
		unsigned int gcdTwomother;
		unsigned int gcdNumMother; 

		fscanf(file, "%u %u %u %u", &n1, &m1, &n2, &m2);
		gcdTwomother = gcd(m1, m2);
		mother = m1*m2 / gcdTwomother; 
		num = (n1*m2 / gcdTwomother) + (n2*m1 / gcdTwomother); 
		gcdNumMother = gcd(num, mother);  
		printf("%u %u\n", num / gcdNumMother, mother / gcdNumMother);  
		
	}
	fclose(file);
}


unsigned int gcd(unsigned int x, unsigned int y)
{
	if (x == 0) return y;
	if (y == 0) return x;
	return ((x >= y) ? (gcd(y, x%y)) : (gcd(x, y%x)));
}
