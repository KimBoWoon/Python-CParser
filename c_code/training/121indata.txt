


#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable : 4996)
int mccarthy(int number); 

void main(void)
{
	FILE* inFile;
	int testCase = 0, number = 0;
	int i = 0;
	inFile = fopen("input.txt", "r"); 
	if (inFile == NULL) exit(1);
	fscanf(inFile, "%d", &testCase); 

	for (i = 0; i < testCase; i++)
	{
		fscanf(inFile, "%d", &number);
		printf("%d\n", mccarthy(number));
	}
	fclose(inFile);
}

int mccarthy(int number)
{
	if (number>100) 
	{
		return number-10;
	}
	else 
	{
		return mccarthy(mccarthy(number+11));
	}
}
