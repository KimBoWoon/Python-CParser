
#include <iostream>
#include <fstream>

int Euclid(int a ,int b) {

	if (b == 0) return a;
	else return Euclid(b, a % b);
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase, num1, num2;

	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	for (int i = 0; i < numcase; i++) {

		fin >> num1 >> num2;
		std::cout << Euclid(num1,num2) << std::endl;
	}
	fin.close();
}
