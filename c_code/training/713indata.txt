


#include <iostream>
#include <fstream>
#include<cstdlib>
#include<string>
using namespace std;
int palin(string str, int first, int last); 

int main()
{
	ifstream inStream;
	int numTestCases;
	inStream.open("input.txt"); 
	if (inStream.fail())
	{
		cerr << "Input file opening failed. \n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		string set;
		inStream >> set;
		cout << palin(set, 0, set.size()-1) << endl; 
	}
	inStream.close();
	return 0;
}

int palin(string str, int first, int last)
{
	if (last <= first) 
		return 1;
	else if (str[first] != str[last]) 
		return 0;
	else 
		return palin(str, first + 1, last - 1);
}
