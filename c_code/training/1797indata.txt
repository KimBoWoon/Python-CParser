#include<iostream>
#include<fstream>
#include<cstring>

using namespace std;

void recursive(char ps[50], int left_count, int right_count, int site, int end);

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if (instream.fail())
	{
		cerr << "input file open failed";
		return 0;
	}

	int testcase;
	instream >> testcase;
	char ps[50];
	bool ps_ck = 0;
	bool last_ck = 1;
	for (int i = 0; i < testcase; i++)
	{
		instream >> ps;
		int left_count = 0;
		int right_count = 0;
		recursive(ps, left_count, right_count, 0, strlen(ps));
	}
	return 0;
}

void recursive(char ps[50], int left_count, int right_count, int site, int end)
{

	

	if (site == end)
	{
		if (ps[0] == ')')
		{
			cout << "NO" << endl;
			return;
		}

		if ((left_count == right_count)&&ps[end-1]!='(')
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	else
	{
		if (ps[site] == '(')
			left_count++;
		else if (ps[site] == ')')
			right_count++;

		recursive(ps, left_count, right_count, site + 1, end);
	}
}
