
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>

int gcd(int n1, int n2) { 
	if (n2 == 0)
		return n1;
	else if (n1 == 0)
		return n2;

	else
		return gcd(n2, n1%n2);
}

int henry(int n1,int n2){
	int i=2;
	int gcd_n=0;
	
	if(n1==1)
		return n2;
		
	while(1){
	
		if(n1*i-n2>=0){

			n1=n1*i-n2;
			n2=n2*i;

			gcd_n=gcd(n1,n2);
			n1=n1/gcd_n;
			n2=n2/gcd_n;
			if(n1==1){
				return n2;
				break;
			}
			else
				i++;
		}
		else
			i++;
	}
}
int main(){
	
	FILE *fp;
	int numcases;
	int i;
	int result=0;

	fp=fopen("input.txt","r");
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);


	for(i=0;i<numcases;i++){
		int n1,n2;

		fscanf(fp,"%d %d",&n1,&n2);

		result=henry(n1,n2);
		printf("%d\n",result);
	}
	fclose(fp);
}
