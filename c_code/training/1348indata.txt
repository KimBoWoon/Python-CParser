






#include<iostream>
#include<fstream>
using namespace std;


int Henry(int a, int b)
{
	if ( b%a == 0) 
		return b/a;
	else {
		Henry(a*(b/a+1)-b, b*(b/a+1));
	}
}

int main()
{
	fstream instream("input.txt", ios::in);
	int testcase;
	int a, b;
	instream >> testcase;

	for (int i = 0; i<testcase; i++)
	{
		instream >> a >> b;

		cout <<Henry(a,b)<< endl;
	}
	return 0;
}
