
#include <iostream>
#include <fstream>
#include <algorithm>
#include <cstdlib>
using namespace std;

int main()
{
	ifstream instream;
	instream.open("input.txt");

	int testcase;
	instream >> testcase;

	for(int i=0;i<testcase ;i++)
	{ 
		int n,k;
		instream >> n>>k;

		int *arr;
		arr = new int [n];
		for(int j=0;j<n;j++)
		{
			instream>> arr[j];
		}
		sort(arr,arr+n);


		int start=0;
		int end=n-1;
		int min=2000000000;  
		int result=0;
		while(start<end)
		{
			if(min>abs(k-arr[start]-arr[end]))
			{	
				min=abs(k-arr[start]-arr[end]);
				result=1;
			}
			else if(min  == abs(k-arr[start]-arr[end]))
			{result++;}
			
			if(arr[start] + arr[end] <k)
				start ++;
			else
				end--;
		}
		
		cout<<result <<endl;
			
	}
}
