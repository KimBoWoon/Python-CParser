#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int T, N;
int i, j, k;
int a[50],ck[1010];

void preprocess(){
	for (i = 1; i < 50; ++i){
		a[i] = (i * (i + 1)) / 2;
	}

	for (i = 1; i < 50; ++i){
		for (j = i; j < 50; ++j){
			for (k = j; k < 50; ++k){
				if (a[i] + a[j] + a[k] <= 1000){
					ck[a[i] + a[j] + a[k]] = 1;
				}
			}
		}
	}
}


int main(){
	FILE *in_file;
	in_file = fopen("input.txt", "r");
	if (in_file == NULL) 
		exit(1);

	preprocess();

	fscanf(in_file,"%d", &T);

	while (T--){
		fscanf(in_file, "%d", &N);
		if (ck[N]) puts("1");
		else puts("0");
	}

	fclose(in_file);
}
