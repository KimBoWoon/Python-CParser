#include <stdio.h>

int T;
int A[45], D[1001]; 
int main()
{
	FILE* fp;  
	int i, j, k;
	fp = fopen("input.txt", "r"); 
	for (i = 1; i<45; i++)
		A[i] = i*(i + 1) / 2; 
	for (i = 1; i<45; i++)
		for (j = i; j<45; j++)
			for (k = j; k<45; k++)
				if (A[i] + A[j] + A[k] < 1001)   
					D[A[i] + A[j] + A[k]] = 1;   

	fscanf(fp, "%d", &T);  
	for (i = 0; i<T; i++)
	{
		fscanf(fp, "%d", &k);  
		printf("%d\n", D[k]); 
	}
	return 0;

}
