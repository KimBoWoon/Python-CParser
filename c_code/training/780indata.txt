#include <stdio.h>
#include <string.h>

int palindrome(char str[], int first, int last);

int main()
{
	FILE* fp = fopen("input.txt", "r");
	int testCaseNumber = 0;
	int i = 0;
	char str[1000] = { 0, };
	int first = 0;
	int last = 0;
	int result = 0;

	if (fp == NULL)
	{
		printf("file error\n");
	}

	fscanf(fp, "%d", &testCaseNumber);

	for (i = 0; i < testCaseNumber; i++)
	{
		fscanf(fp, "%s", str);
		result = palindrome(str, 0, strlen(str) - 1);
		printf("%d\n", result);
	}

	fclose(fp);
	return 0;
}

int palindrome(char str[], int first, int last)
{
	if (last <= first)
	{
		return 1;
	}
	else if (str[first] != str[last])
	{
		return 0;
	}
	else
	{
		return palindrome(str, first + 1, last - 1);
	}
}
