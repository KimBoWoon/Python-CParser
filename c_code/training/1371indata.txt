

#include <iostream>
#include <fstream>
#include <string.h>
#include <cmath>
using namespace std;

int euclid(int n1, int n2);

int main()
{
	ifstream infile;
	int testCase;
	int a, b; 
	int tmp=0;
	int result1 = 0, result2 = 0;
	infile.open("input.txt");
	
	infile >> testCase;

	for (int i = 0; i < testCase; i++)
	{
		infile >> a >> b;

		result1 = a;
		result2 = b;
		int j = 2;
		while (1)
		{
			if (result1 == 1)
				break;
			
			if (result1*j > result2)
			{
				result1 = result1*j - result2;
				result2 *= j;
				tmp = euclid(result1, result2);

				if ( tmp > 1)
				{

					result1 /= tmp;
					result2 /= tmp;
				}
			}
			j++;
		}

		cout << result2 << endl;

	}
	infile.close();

	return 0;
}

int euclid(int n1, int n2) 
{
	if (n2 == 0) 
		return n1;
	else if (n1 == 1 || n2 == 1) 
		return 1;
	else
		return euclid(n2, n1 % n2);

}
