

#include <iostream>
#include <fstream>
#include <cstdlib>
	
using namespace std;
int euclid(int m, int n){
	return (n==0) ? m : euclid(n, m%n);
}
int main(){
	ifstream instream;
	int no_cases;

	instream.open("input.txt");
	if (instream.fail()) exit(1);
			
	instream>>no_cases;

	while(no_cases--){
		int m,n;
		instream>>m>>n;

		int gcd = (m>n) ? euclid(m,n): euclid(n,m);
		cout<<gcd<<endl;
	}
	instream.close();
}
