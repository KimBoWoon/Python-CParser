

#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

bool IsVPS(char* paren, int len, int idx);

int main()
{
	int testCase = 0;
	ifstream file;
	file.open("input.txt");
	file >> testCase;

	char paren[50] = {0}; 
	for (int i = 0; i < testCase; i++)
	{
		file >> paren;
		
		if (IsVPS(paren, strlen(paren),0))
			cout << "YES" << endl;
		else
			cout << "NO" << endl;

	}
	return 0;
}

bool IsVPS(char* paren, int len, int idx)
{
	int flag = 0;
	
	for (int idx = 0; idx < len; idx++)
	{
		if (paren[idx] == '(')
			flag++;
		else 
			flag--;

		if (flag < 0) 
			return false;

		if (idx == (len-1)) 
		{
			if (flag == 0)
				return true;
			else
				return false;
		}
	}
}
