
#include<stdio.h>

int recursion(char str[], int left, int right){
	if (left == right || left>right) printf("1\n");
	else if (str[left] == str[right]) recursion(str, left+1, right-1);
	else printf("0\n");
}

int main(){

	FILE *inFile;
	int circle;
	char str[1001];
	inFile = fopen("input.txt", "r");
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%s", &str);
		recursion(str, 0, strlen(str)-1);
	}
	return 0;
}
