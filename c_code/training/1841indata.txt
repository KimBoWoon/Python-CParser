
#include <iostream>
#include <fstream>
#include <string>

char P[50];

int MAX_SIZE;
int result;

void Parenthesis(int num) {
	
	if (num == MAX_SIZE) return;

	if (P[num] == '(') result--;
	else if( result < 0 ) result++;
	else result = 51;

	Parenthesis(num + 1);
}

int main() {

	std::ifstream fin;

	fin.open("input.txt");
	int numcase;
	std::string str;
	if (fin.is_open() == false)  return -1;

	fin >> numcase;
	while (numcase--) {
		
		fin >> str;
		MAX_SIZE = str.length();
		result = 0;
		for (int i = 0; i < MAX_SIZE; i++) {
			P[i] = str.at(i);
		}		
		Parenthesis(0);
		if (result == 0) std::cout << "YES" << std::endl;
		else std::cout << "NO" << std::endl;
	}
	fin.close();
}
