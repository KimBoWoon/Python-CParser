
#pragma warning(disable:4996)
#include <stdio.h>
#include <stdlib.h>

int McC(int n){ 
	if(n>100) 
		return n-10;

	else
		return McC(McC(n+11));
}

int main(){
	FILE *fp;
	int i;
	int numcases;

	fp=fopen("input.txt","r"); 
	if(fp==NULL) exit(1);

	fscanf(fp,"%d",&numcases);
	
	for(i=0;i<numcases;i++){
		int n;
		int result=0;
		int i;

		fscanf(fp,"%d",&n);
		result=McC(n);

		printf("%d\n",result);
		
	}
	fclose(fp);
}
