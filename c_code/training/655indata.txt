#include<iostream>
#include<fstream>
#include<cstdlib>
using namespace std;
int factorial(int num);
int main()
{
	ifstream inStream;
	int no_cases,num,result;

	inStream.open("input.txt");
	if(inStream.fail())
	{
		cerr<<"Input file openning failed.\n";
		exit(1);
	}
	inStream>>no_cases;
	
	for(int i=0; i<no_cases; i++)
	{
		inStream>>num;
		result=factorial(num);
		cout<<result<<endl;
	}	
}
int factorial(int num)
{
	if(num<=1)
		return 1;
	else
		return (num*factorial(num-1));
}
