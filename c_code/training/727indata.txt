#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int pal(int s,int e, char* str){
    if( s >= e ) return 1;
    else if( str[s] != str[e] ) return 0;
    else return pal(s+1, e-1, str);
}

int main(void){
    FILE* file;
    file = fopen("input.txt","r");
    
    int cases;
    fscanf(file, "%d", &cases);
    while(cases--){
        char str[1001];
        fscanf(file, "%s",  str);
        printf("%d\n", pal(0, strlen(str)-1, str));
         }
    return 0;
}
