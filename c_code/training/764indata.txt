
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int checkP(char str[], int first, int last)
{
	if (last <= first)
		return 1;
	else if (str[first] != str[last])
		return 0;
	else
		return checkP(str, first + 1, last - 1);
}
void main(){
	FILE *in_file;
	int no_cases, no_data;
	int i;
	char str[1000];

	in_file = fopen("input.txt", "r"); 
	if (in_file == NULL) exit(1);

	fscanf(in_file, "%d", &no_cases);

	for (i = 0; i < no_cases; i++)
	{
		fscanf(in_file, "%s", str);
	
		printf("%d\n", checkP(str,0,strlen(str)-1));
		
	}

	fclose(in_file);
}
