

#include <stdio.h>
#include <stdlib.h>

int main()
{
   FILE *inFile;
   int numcases;
   int t, h, w, n;


   inFile = fopen("input.txt", "r");
   if (inFile == NULL) exit(1);

   fscanf(inFile, "%d", &numcases);

   for (t = 0; t < numcases; t++)
   {
      fscanf(inFile, "%d %d %d", &h, &w, &n);
      printf("%d%02d\n", (n - 1) % h + 1, (n - 1) / h + 1);
   }

   fclose(inFile);

   return 0;
}
