
#include<fstream>
#include<iostream>

using namespace std;

int main(){
	ifstream in;
	in.open("input.txt");
	if (!in) return 1;
	int n = 0;
	in >> n;
	for (int i = 0; i < n; i++){
		int temp = 0, flag = 0;
		in >> temp;
		for (int j = 2; j <= temp / 2; j++){
			if (temp%j == 0 && j != temp){
				flag = 1;
				break;
			}
		}
		if (!flag) cout << 1 << endl;
		else cout << 0 << endl;
	}
	in.close();
	return 0;
}
