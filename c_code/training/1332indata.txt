#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void swap(long long *a, long long *b)
{
	long long temp = *a;
	*a = *b;
	*b = temp;
}

long long GCD(long long a, long long b)			
{
	if (a <= b)
		swap(&a, &b);
	if (b == 0)
		return a;
	GCD(b, a%b);
}

void Henry(long long up, long long down)
{
	long long i = 1;
	long long result = 0;						
	long long modify_up, modify_down;			

	while (i++)
	{
		if (down <= (up * i))					
		{
			result = i;							

			if (down == (up * i))				
			{
				printf("%llu\n", result);
				break;
			}
			modify_down = down * i / GCD(down, i);								
			modify_up = (up * (modify_down / down)) - (1 * (modify_down / i));  

			up = modify_up / GCD(modify_up, modify_down);						
			down = modify_down / GCD(modify_up, modify_down);

			Henry(up, down);													
			break;
		}
		else																	
		{
			if ((down % up) == 0)				
				i = (down / up) - 1;
			else
				i = (down / up);
		}
	}
}

int main()
{
	int numcase;
	long long up, down;

	FILE * in_file;
	in_file = fopen("input.txt", "r");

	fscanf(in_file, "%d", &numcase);

	while (numcase--)
	{
		fscanf(in_file, "%llu %llu", &up, &down);
		Henry(up, down);
	}
}
