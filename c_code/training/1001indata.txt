 

 

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int test;
    FILE *fp;
    fp = fopen("input.txt","r"); 
    fscanf(fp,"%d",&test);
    while(test--) 
    {
        int n,i,*value,tmp,j,flag=0,sum=0;    
        fscanf(fp,"%d",&n);
        value = (int*)malloc(sizeof(int)*n);
        for(i=0;i<n;i++) 
        {
            fscanf(fp,"%d",&value[i]);
        }
        
        for(j=0;j<1000;j++) 
        {
            tmp = value[0];
            for(i=0;i<n-1;i++) 
            {
                value[i] = value[i] - value[i+1];
                if(value[i] < 0) value[i] = -value[i];
            }
            value[n-1] = value[n-1] - tmp;
            if(value[n-1] < 0) value[n-1] = -value[n-1];
        }
        for(i=0;i<n;i++) 
        {
            sum += value[i];
        }
        if(sum == 0) 
        {
            printf("ZERO\n");
        }
        else 
        {
            printf("LOOP\n");
        }
        free(value);
    }
    fclose(fp); 
    return 0;
}
