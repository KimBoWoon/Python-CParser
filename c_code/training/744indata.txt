
#include <stdio.h>
#include <string.h>
#define MAX_LENGTH 1000

int palindrome(char inStr[], int first, int last){
    if(inStr[first] == inStr[last])
        return palindrome(inStr, first+1, last-1);
    else if(first >= last)
        return 1;
    return 0;
}

void getPalindrome(FILE *fin){
    char inStr[MAX_LENGTH] = "\0";
    fscanf(fin, "%s", inStr);
    
    printf("%d\n", palindrome(inStr, 0, strlen(inStr)-1));
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        getPalindrome(fin);        
    
    fclose(fin);            
    
    return 0;           
}
