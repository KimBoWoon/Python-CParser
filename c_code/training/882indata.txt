#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int cal(int num); 

int main()
{
	ifstream infile;

	int nocases;
	int num;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i = 0; i<nocases; i++)
	{
		infile >> num;

		cout << cal(num) << endl;
	}
}

int cal(int num)
{
	if(num <= 1)
	{
		return num;
	}

	else
		return cal(num-1)+cal(num-2);
}
