
#include <stdio.h>
#include <string.h>

int isValid(char str[], int length, int index, int cnt){
    if(cnt < 0) return 0;       
    if(str[index] == '(') return isValid(str, length, ++index, ++cnt);
    else if(str[index] == ')') return isValid(str, length, ++index, --cnt);
    if(index == length && cnt == 0) return 1;   
    return 0;       
}

void vps(FILE *fin){
    char str[51] = "";
    
    fscanf(fin, "%s", str);
    
    if(isValid(str, strlen(str), 0, 0))
        printf("YES\n");
    else
        printf("NO\n");
}

int main(){
    int testCases, cases;
    FILE* fin = fopen("input.txt", "r");    
    
    fscanf(fin, "%d", &testCases);      
    
    for(cases=0; cases<testCases; cases++)
        vps(fin);      
    
    fclose(fin);            
    
    return 0;           
}
