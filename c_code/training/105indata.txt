#include<stdio.h>

int McC(int n)
{
	if(n>100)		return n-10;
	return McC(McC(n+11));
}

void main()
{
	FILE*inFile = fopen("input.txt", "r");
	int testNum, num;
	int i;

	fscanf(inFile, "%d", &testNum);
	for(i=0; i<testNum; ++i){
		fscanf(inFile, "%d", &num);
		printf("%d\n", McC(num));
	}

	fclose(inFile);
}
