#include <iostream>
#include <fstream>
#include <cstdlib>
#include <vector>
#include <algorithm>

using namespace std;

int main() {

	ifstream instream;
	int numTestCases;

	instream.open("input.txt");
	if(instream.fail()) {
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	instream >> numTestCases;

	for(int i=0; i<numTestCases; i++) {

		int num;
		instream >> num;

		bool tmp = 0;

		for(int b=2; b<=64; b++) {

			vector<int> V;
			int data = num;

			while(data > 0) {
				V.push_back(data%b);
				data /= b;
			}

			bool val = 1;

			for(int j=0; j<V.size(); j++) {
				if(V[j] != V[(int)V.size()-1-j]) {
					val = 0;
					break;
				}
			} 

			if(val) {
				tmp = 1;
				break;
			}
		}
		if(tmp)
			cout << 1 << endl;
		else
			cout << 0 << endl;
	}
	instream.close();
}
