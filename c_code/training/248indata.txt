#include <stdio.h>
#include <stdlib.h>

int gcd(int n1, int n2);	
int lcm(int n1, int n2, int gcd);	
void fractionAddition(int n1_numer, int n1_deno, int n2_numer, int n2_deno, int n_lcm);
void main()
{
	FILE *f;
	int count;
	int i;
	int num1_denominator, num1_numerator, num2_denominator, num2_numerator;

	f = fopen("input.txt", "r");
	if(f==NULL) exit(1);

	fscanf(f, "%d", &count);

	for(i=0; i<count; i++)
	{
		int gcd_num;
		int lcm_num;
		fscanf(f, "%d %d %d %d", &num1_numerator, &num1_denominator, &num2_numerator, &num2_denominator);
		gcd_num = gcd(num1_denominator, num2_denominator);	
		lcm_num = lcm(num1_denominator, num2_denominator, gcd_num);	
		fractionAddition(num1_numerator, num1_denominator,num2_numerator, num2_denominator, lcm_num);
	}
	fclose(f);
}
int gcd(int n1, int n2)
{
	int temp;
	if(n2>n1)	
	{
		temp = n1;
		n1 = n2;
		n2 = temp;
	}
	if(n2 == 0)		
		return n1;	
	else
		return gcd(n2, n1%n2);	
}
int lcm(int n1, int n2, int gcd)
{
	return n1*n2/gcd;	
}
void fractionAddition(int n1_numer, int n1_deno, int n2_numer, int n2_deno, int n_lcm)
{
	int n1, n2, gcd_num;
	int numerator, denominator;
	n1 = n_lcm/n1_deno*n1_numer;	
	n2 = n_lcm/n2_deno*n2_numer;

	numerator = n1+n2;		
	denominator = n_lcm;	

	gcd_num = gcd(numerator, denominator);	
	if(gcd_num != 1)		
	{
		numerator /= gcd_num;				
		denominator /= gcd_num;
	}

	printf("%d %d\n", numerator, denominator);	
}
