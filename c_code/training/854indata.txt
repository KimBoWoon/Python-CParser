#include <iostream>
#include <fstream>
#include <cstdlib>

int fibonacci(int n);
int main()
{
	std::fstream infile;
	int testcase;
	infile.open("input.txt");
	if (infile.fail())
	{
		std::cerr << "file open error";
		exit(1);
	}
	infile >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		int testnum;
		infile >> testnum;
		std::cout << fibonacci(testnum) << std::endl;
	}
	return 0;
}
int fibonacci(int n)
{
	if (n <= 1) return n;
	else
		return fibonacci(n - 1) + fibonacci(n - 2);
}
