
#include <iostream>
#include <fstream>
#define MAXLEN 1000
using namespace std;

int abs(int a, int b){
	return a>b ? (a-b) : (b-a);
}

int allElemZero(int arr[MAXLEN], int size){
	for(int i=0; i<size; i++)
		if(arr[i]!=0)
			return 0;
	return 1;
}


void DucciSequence(int arr[MAXLEN], int size, int cnt){
	if(cnt == MAXLEN){
		cout << "LOOP" << endl;
		return;
	}
	else if(allElemZero(arr, size)==1){
		cout << "ZERO" << endl;
		return;
	}

	int* temp = new int[size];
	for(int i=0; i<size-1; i++)
		temp[i] = abs(arr[i], arr[i+1]);
	temp[size-1] = abs(arr[size-1], arr[0]);
	for(int i=0; i<size; i++)
		arr[i] = temp[i];
	delete[] temp;
	return DucciSequence(arr, size, cnt+1);
}

int main(){
	ifstream f;
	f.open("input.txt", ifstream::in);
	int numcase;
	f >> numcase;
	for(int i=0; i<numcase; i++){
		int arr[MAXLEN];
		int size;
		f >> size;
		for(int j=0; j<size; j++)
			f >> arr[j];
		DucciSequence(arr, size, 0);
	}

	return 0;
}
