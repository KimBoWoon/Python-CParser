#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

void cal(int a, int b, int c); 

int main()
{
	ifstream infile;

	int nocases;
	int num1, num2, num3;

	infile.open("input.txt");

	if(infile.fail())
	{
		cerr << "파일열기 오류" << endl;
		exit(1);
	}

	infile >> nocases;

	for(int i=0; i<nocases; i++)
	{
		infile >> num1;
		infile >> num2;
		infile >> num3;

		cal(num1,num2,num3);
	}
	
	return 0;
}

void cal(int a, int b, int c)
{
	int fnum = 1;
	int snum = 1;
	int savenum = 0;

	while(1)
	{
		if(c <= a)
		{
			fnum = c;
			break;
		}

		if(c > a)
		{
			snum ++;
			c = c-a;
		}
	}

	savenum = fnum * 100 + snum;

	cout << savenum << endl;
}
