#include <iostream>
#include <fstream>
using namespace std;

int mac(int n)
{
	if(n>100)
		return n-10;
	else
		return mac(mac(n+11));

}

int main()
{
	ifstream instream;
	instream.open("input.txt");
	if(!instream)
	{
		cout << "파일을 열 수 없습니다." << endl;
		return 0;
	}
	int test_case;
	instream >> test_case;
	for(int i=0; i<test_case; i++)
	{
		int num;
		instream >> num;
		cout << mac(num) << endl;
	}
}
