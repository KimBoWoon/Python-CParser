#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int main(){
	ifstream inStream;
	int numTestCases;	

	inStream.open("input.txt");
	if (inStream.fail()){
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;	

	for (int i = 0; i < numTestCases; i++){
		int num;

		inStream >> num;

		if (num == 2)	
			cout << "1" << endl;

		else if (num % 2 == 0)	
			cout << "0"<<endl;

		else{	
			int check = 0;
			
			for (int j = 3; j < num; j++)	
				if (num % j == 0)
					++check;

			if (check == 0)	
				cout << "1" << endl;

			else 
				cout << "0" << endl;
		}
	}

	inStream.close();
	return 0;
}
