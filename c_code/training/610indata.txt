


#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

int fac(int n)
{
	if (n <= 1) 
		return 1;
	else 
		return n * fac(n-1);
}
int main(void)
{
	ifstream inStream;
	int numTestCases = 0;
	int num = 0;
	inStream.open("input.txt");
	if (inStream.fail())
	{
		cerr << "Input file opening failed.\n";
		exit(1);
	}

	inStream >> numTestCases;
	for (int i = 0; i < numTestCases; i++)
	{
		inStream >> num;
		cout << fac(num) << endl;
	}
	return 0;
}
