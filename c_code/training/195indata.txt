#include <stdio.h>

int mccarthy(int n);

int main()
{
	FILE* fp = fopen("input.txt", "r");
	int testCaseNumber = 0;
	int i = 0;
	int n = 0;
	int result = 0;

	if (fp == NULL)
	{
		printf("file error\n");
	}

	fscanf(fp, "%d", &testCaseNumber);

	for (i = 0; i < testCaseNumber; i++)
	{
		fscanf(fp, "%d", &n);
		result = mccarthy(n);
		printf("%d\n", result);
	}

	fclose(fp);
	return 0;
}

int mccarthy(int n)
{
	if (n > 100)
	{
		return n - 10;
	}
	else
	{	
		return mccarthy(mccarthy(n + 11));
	}
}
