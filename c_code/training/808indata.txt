

#include <iostream>
#include <fstream>
using namespace std;
int fibonacciRecursive(int num);

int main(){
	ifstream inFile("input.txt");
	if (inFile.fail()){
		return 0;
	}
	int testCases = 0;
	inFile >> testCases;
	for (int i = 0; i < testCases; i++)
	{
		int num = 0;
		inFile >> num;
		cout << fibonacciRecursive(num) << endl;
	}
}
int fibonacciRecursive(int num){

	if (num == 0){
		return 0;
	}
	else if (num == 1){
		return 1;
	}
	else {
		return fibonacciRecursive(num-1) + fibonacciRecursive(num - 2);
	}
}
