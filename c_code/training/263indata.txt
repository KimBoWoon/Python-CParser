




#include <iostream>
#include <fstream>


int fraction(int x, int y);

int main() {
	std::ifstream inFile;
	int testCaseNum;

	inFile.open("input.txt");

	inFile >> testCaseNum;

	for(int i = 0; i < testCaseNum; i++) {
		int x, y;
		int n, m;
		int den = 0, num = 0, gcd = 0;

		inFile >> x >> y >> n >> m;

		if (y != m) {
			den = y * m;
			num = (x * m) + (y * n);
		}
		else {
			den = y;
			num = x + n;
		}

		gcd = fraction(num, den);
		std::cout << num / gcd << ' ' << den / gcd << std::endl;
	}
}
int fraction(int x, int y) {
	if (y == 0)
		return x;
	else
		return fraction(y, x % y);
}
