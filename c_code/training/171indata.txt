

#include <iostream>
#include <fstream>
#include <cstdlib>

using namespace std;

int Maccarthy(int a);		

int main()
{
	ifstream instream;
	instream.open("input.txt");

	if(instream.fail())
	{
		cerr<<"Error"<<endl;
		exit(1);
	}

	int no_case;
	instream>>no_case;

	for(int i=0; i<no_case; i++)		
	{
		int num;
		instream>>num;

		cout<<Maccarthy(num)<<endl;		
	}
}

int Maccarthy(int a)		
{
	if(a > 100)
		return a-10;
	else
		return Maccarthy(Maccarthy(a+11));
}
