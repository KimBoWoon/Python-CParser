#include <iostream>
#include <fstream>
#include <cstdlib>
using namespace std;

bool isPrime(int n); 
void gold(int data); 
int num1 = 0, num2 = 0; 
int result = 1000; 

int main()
{
	ifstream instream;    
	int numtestcases;
	instream.open("input.txt");
	if (instream.fail()) exit(1);
	instream >> numtestcases;
	while (numtestcases--)
	{
		int num;
		instream >> num;
		num1 = 0; num2 = 0; result = 1000;  
		gold(num);                     
		cout << endl;
	}
	instream.close();
}
bool isPrime(int num)       
{
	if (num == 1)          
		return false;
	else
	{
		for (int i = 2; i < num; i++)  
			if (num%i == 0)
				return false;
	}
	return true;                         
}
void gold(int data)
{
	for (int i = 2; i < data; i++)              
	{
		if (isPrime(i) && isPrime(data - i))     
		{
			if (abs((data - i) - i) < result)         
				{
					num1 = data - i;                
					num2 = i;                       
					result = abs((data - i) - i);     
				}		
		}
	}
	cout << num2 << " " << num1;  
}
