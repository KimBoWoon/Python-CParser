
#define TRUE 1
#define FALSE !TRUE

#include <stdio.h>
#include <stdlib.h>

int checkNumber(int num);	

void main() {
	FILE *f;
	int numTestCases = 0;
	int i = 0;
	int num = 0;

	f = fopen("input.txt", "r");
	fscanf(f, "%d", &numTestCases);

	for (i = 0; i < numTestCases; i++) {
		fscanf(f, "%d", &num);
		printf("%d\n", checkNumber(num));
	}

	fclose(f);
}

int checkNumber(int num) {
	int i = 0, j = 0, k = 0;
	int result = 0;

	for (i = 1; i < 46; i++) {
		for (j = 1; j < 46; j++) {
			for (k = 1; k < 46; k++) {
				result = i*(i + 1) / 2 + j*(j + 1) / 2 + k*(k + 1) / 2;
				if (result == num)
					return TRUE;
			}
		}
	}
	return FALSE;
}
