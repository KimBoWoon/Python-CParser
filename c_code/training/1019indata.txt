

#include<stdio.h>
#define ABS(X) (X<0)?(-X):(X)
int count, num, i,sum;

void check(int numbers[], int next[]){
	if (count > 1000) { printf("LOOP\n"); return; }
	for (i = 0; i < num; i++) {
		next[i] = numbers[i] - numbers[(i == num - 1) ? (0) : (i + 1)];
		next[i] = ABS(next[i]);
		sum += next[i];
	}
	if (sum == 0) { printf("ZERO\n"); return; }
	count++; sum = 0;
	check(next, numbers);
}

int main(){

	FILE *inFile;
	int circle;
	int numbers[16],numbers2[16];
	inFile = fopen("input.txt", "r"); 
	fscanf(inFile, "%d", &circle);
	while (circle--)
	{
		fscanf(inFile, "%d", &num);
		for (i = 0; i < num; i++) fscanf(inFile, "%d", &numbers[i]);
		count = 0; sum = 0;
		check(numbers, numbers2);
	}
	return 0;
}
