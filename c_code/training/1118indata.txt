

#include<iostream>
#include<fstream>
#include<string>
using namespace std;
int main()
{
	int numOfCase = 0;	
	int inputNum = 0;	
	int triNum[45];	
	for (int i = 1; i < 45; i++){
		triNum[i] = (i)*(i + 1) / 2;
	}

	ifstream input("input.txt");
	if (input.fail()){
		cout << "파일을 여는 데 실패했습니다." << endl;
	}
	input >> numOfCase;

	for (int i = 0; i < numOfCase; i++){
		input >> inputNum;

		for (int j = 1; j < 45; j++){	
			if (triNum[j] > inputNum){
				cout << "0" << endl;
				break;
			}
			for (int k = 1; k < 45; k++){
				if (triNum[j] + triNum[k]> inputNum){
					break;
				}
				for (int p = 1; p < 45; p++){
					if (triNum[j] + triNum[k] + triNum[p] == inputNum){ 
						cout << "1" << endl;
						j = 45; 
					}
					else if (triNum[j] + triNum[k] + triNum[p] > inputNum){ 
						break;
					}
				}
			}
		}
	}
	input.close();
	return 0;
}
