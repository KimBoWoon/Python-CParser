
#include<iostream>
#include<fstream>

using namespace std;

bool primeCheck(int num); 

int main(){
	ifstream inFile("input.txt");
	int cases;
	inFile >> cases;
	while (cases--){
		int num;
		inFile >> num;
		for (int i = num/2; i >0 ; i--){
			if (primeCheck(i)){
				if (primeCheck(num - i)){
					cout << i << " " << num - i << endl;
					break;
				}
			}
		}	
	}
	return 1;
}

bool primeCheck(int num){
	for (int i = 2; i*i <= num; i++){
		if (num%i){
			continue;
		}
		return false;
	}
	return true;
}
