#include <iostream>
#include <fstream>
using namespace std;

int fact(int n);

int main()
{
	ifstream infile;

	infile.open("input.txt");
	
	int numTestCases;
	int result = 0;

	infile >> numTestCases;

	while(numTestCases--)
	{
		infile >> result;
		cout << fact(result);
		cout << endl;
	}

	
	return 0;
}
int fact(int n)
{
	if( n > 0 )
		return n * fact(n-1);
	else
		return 1;
}
