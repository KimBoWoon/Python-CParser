#include <stdio.h>
#include <stdlib.h>

int GCD(int num1, int num2);

void main()
{
	FILE *file;
	int i=0;
	int t, n1,n2;
	file = fopen("input.txt","r");
	if(file ==NULL)
		exit(1);
	fscanf(file, "%d", &t);
	for(i=0; i<t; i++)
	{
		fscanf(file,"%d%d", &n1,&n2);
		printf("%d\n", GCD(n1,n2));
	}
	fclose(file);
}
int GCD(int num1, int num2) 
{
	int ans=0;
	while(num2!=0){
		ans = num1%num2;
		num1 =num2;
		num2 = ans;
	};
	return num1;
}
