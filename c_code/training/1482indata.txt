#include <stdio.h>

int getRoomNumber(int x, int y);

int main()
{
	FILE* stream = fopen("input.txt", "r");
	int testCaseNumber = 0;
	int i = 0;
	int h = 0;
	int w = 0;
	int n = 0;

	if (stream == NULL)
	{
		printf("file error\n");
	}

	fscanf(stream, "%d", &testCaseNumber);

	for (i = 0; i < testCaseNumber; i++)
	{
		fscanf(stream, "%d %d %d", &h, &w, &n);
		w = getRoomNumber(h, n);
		printf("%d\n", w);
	}

	fclose(stream);
	return 0;
}

int getRoomNumber(int x, int y)
{
	int h = x; 
	int n = y; 
	int floor = 0;
	int nth = 0;
	int roomNumber = 0;

	if ((n % h) == 0)
	{
		floor = h * 100;
		nth = n / h;
	}
	else
	{
		floor = (n % h) * 100;
		nth = (n / h) + 1;
	}

	roomNumber = floor + nth;
	return roomNumber;
}
